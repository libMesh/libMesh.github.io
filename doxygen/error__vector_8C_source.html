44 ErrorVectorReal ErrorVector::minimum() const
45 {
46  LOG_SCOPE ("minimum()", "ErrorVector");
47 
48  const dof_id_type n = cast_int<dof_id_type>(this->size());
49  ErrorVectorReal min = std::numeric_limits<ErrorVectorReal>::max();
50 
51  for (dof_id_type i=0; i<n; i++)
52  {
53  // Only positive (or zero) values in the error vector
54  libmesh_assert_greater_equal ((*this)[i], 0.);
55  if (this->is_active_elem(i))
56  min = std::min (min, (*this)[i]);
57  }
58 
59  // ErrorVectors are for positive values
60  libmesh_assert_greater_equal (min, 0.);
61 
62  return min;
63 }

67 Real ErrorVector::mean() const
68 {
69  LOG_SCOPE ("mean()", "ErrorVector");
70 
71  const dof_id_type n = cast_int<dof_id_type>(this->size());
72 
73  Real the_mean = 0;
74  dof_id_type nnz = 0;
75 
76  for (dof_id_type i=0; i<n; i++)
77  if (this->is_active_elem(i))
78  {
79  the_mean += ( static_cast<Real>((*this)[i]) - the_mean ) / (nnz + 1);
80 
81  nnz++;
82  }
83 
84  return the_mean;

90 Real ErrorVector::median()
91 {
92  const dof_id_type n = cast_int<dof_id_type>(this->size());
93 
94  if (n == 0)
95  return 0.;
96 
97 
98  // Build a StatisticsVector<ErrorVectorReal> containing
99  // only our active entries and take its mean
100  StatisticsVector<ErrorVectorReal> sv;
101 
102  sv.reserve (n);
103 
104  for (dof_id_type i=0; i<n; i++)
105  if (this->is_active_elem(i))
106  sv.push_back((*this)[i]);
107 
108  return sv.median();
109 }

124 Real ErrorVector::variance(const Real mean_in) const
125 {
126  const dof_id_type n = cast_int<dof_id_type>(this->size());
127 
128  LOG_SCOPE ("variance()", "ErrorVector");
129 
130  Real the_variance = 0;
131  dof_id_type nnz = 0;
132 
133  for (dof_id_type i=0; i<n; i++)
134  if (this->is_active_elem(i))
135  {
136  const Real delta = ( static_cast<Real>((*this)[i]) - mean_in );
137  the_variance += (delta * delta - the_variance) / (nnz + 1);
138 
139  nnz++;
140  }
141 
142  return the_variance;
143 }

208 void ErrorVector::plot_error(const std::string & filename,
209  const MeshBase & oldmesh) const
210 {
211  UniquePtr<MeshBase> meshptr = oldmesh.clone();
212  MeshBase & mesh = *meshptr;
213 
214  // The all_first_order routine will prepare_for_use(), which would
215  // break our ordering if elements get changed.
216  mesh.allow_renumbering(false);
217  mesh.all_first_order();
218 
219 #ifdef LIBMESH_ENABLE_AMR
220  // We don't want p elevation when plotting a single constant value
221  // per element
222  {
223  MeshBase::element_iterator el =
224  mesh.elements_begin();
225  const MeshBase::element_iterator end_el =
226  mesh.elements_end();
227 
228  for ( ; el != end_el; ++el)
229  {
230  Elem * elem = *el;
231  elem->set_p_refinement_flag(Elem::DO_NOTHING);
232  elem->set_p_level(0);
233  }
234  }
235 #endif // LIBMESH_ENABLE_AMR
236 
237  EquationSystems temp_es (mesh);
238  ExplicitSystem & error_system
239  = temp_es.add_system<ExplicitSystem> ("Error");
240  error_system.add_variable("error", CONSTANT, MONOMIAL);
241  temp_es.init();
242 
243  const DofMap & error_dof_map = error_system.get_dof_map();
244 
245  MeshBase::const_element_iterator el =
246  mesh.active_local_elements_begin();
247  const MeshBase::const_element_iterator end_el =
248  mesh.active_local_elements_end();
249  std::vector<dof_id_type> dof_indices;
250 
251  for ( ; el != end_el; ++el)
252  {
253  const Elem * elem = *el;
254 
255  error_dof_map.dof_indices(elem, dof_indices);
256 
257  const dof_id_type elem_id = elem->id();
258 
259  //0 for the monomial basis
260  const dof_id_type solution_index = dof_indices[0];
261 
262  // libMesh::out << "elem_number=" << elem_number << std::endl;
263  libmesh_assert_less (elem_id, (*this).size());
264 
265  // We may have zero error values in special circumstances
266  // libmesh_assert_greater ((*this)[elem_id], 0.);
267  error_system.solution->set(solution_index, (*this)[elem_id]);
268  }
269 
270  error_system.solution->close();
271 
272  // We may have to renumber if the original numbering was not
273  // contiguous. Since this is just a temporary mesh, that's probably
274  // fine.
275  if (mesh.max_elem_id() != mesh.n_elem() ||
276  mesh.max_node_id() != mesh.n_nodes())
277  {
278  mesh.allow_renumbering(true);
279  mesh.renumber_nodes_and_elements();
280  }
281 
282  if (filename.rfind(".gmv") < filename.size())
283  {
284  GMVIO(mesh).write_discontinuous_gmv(filename,
285  temp_es, false);
286  }
287  else if (filename.rfind(".plt") < filename.size())
288  {
289  TecplotIO (mesh).write_equation_systems
290  (filename, temp_es);
291  }
292 #ifdef LIBMESH_HAVE_EXODUS_API
293  else if ((filename.rfind(".exo") < filename.size()) ||
294  (filename.rfind(".e") < filename.size()))
295  {
296  ExodusII_IO io(mesh);
297  io.write(filename);
298  io.write_element_data(temp_es);
299  }
300 #endif
301  else if (filename.rfind(".xda") < filename.size())
302  {
303  XdrIO(mesh).write("mesh-"+filename);
304  temp_es.write("soln-"+filename,WRITE,
305  EquationSystems::WRITE_DATA |
306  EquationSystems::WRITE_ADDITIONAL_DATA);
307  }
308  else if (filename.rfind(".xdr") < filename.size())
309  {
310  XdrIO(mesh,true).write("mesh-"+filename);
311  temp_es.write("soln-"+filename,ENCODE,
312  EquationSystems::WRITE_DATA |
313  EquationSystems::WRITE_ADDITIONAL_DATA);
314  }
315  else
316  {
317  libmesh_here();
318  libMesh::err << "Warning: ErrorVector::plot_error currently only"
319  << " supports .gmv, .plt, .xdr/.xda, and .exo/.e (if enabled) output;" << std::endl;
320  libMesh::err << "Could not recognize filename: " << filename
321  << std::endl;
322  }
323 }
