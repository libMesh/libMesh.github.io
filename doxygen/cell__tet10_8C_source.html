130 bool Tet10::has_affine_map() const
131 {
132  // Make sure edges are straight
133  if (!this->point(4).relative_fuzzy_equals
134  ((this->point(0) + this->point(1))/2))
135  return false;
136  if (!this->point(5).relative_fuzzy_equals
137  ((this->point(1) + this->point(2))/2))
138  return false;
139  if (!this->point(6).relative_fuzzy_equals
140  ((this->point(2) + this->point(0))/2))
141  return false;
142  if (!this->point(7).relative_fuzzy_equals
143  ((this->point(3) + this->point(0))/2))
144  return false;
145  if (!this->point(8).relative_fuzzy_equals
146  ((this->point(3) + this->point(1))/2))
147  return false;
148  if (!this->point(9).relative_fuzzy_equals
149  ((this->point(3) + this->point(2))/2))
150  return false;
151  return true;
152 }

167 std::unique_ptr<Elem> Tet10::build_side_ptr (const unsigned int i,
168  bool proxy)
169 {
170  libmesh_assert_less (i, this->n_sides());
171 
172  if (proxy)
173  return libmesh_make_unique<Side<Tri6,Tet10>>(this,i);
174 
175  else
176  {
177  std::unique_ptr<Elem> face = libmesh_make_unique<Tri6>();
178  face->subdomain_id() = this->subdomain_id();
179 
180  for (unsigned n=0; n<face->n_nodes(); ++n)
181  face->set_node(n) = this->node_ptr(Tet10::side_nodes_map[i][n]);
182 
183  return face;
184  }
185 }

626 float Tet10::embedding_matrix (const unsigned int i,
627  const unsigned int j,
628  const unsigned int k) const
629 {
630  // Choose an optimal diagonal, if one has not already been selected
631  this->choose_diagonal();
632 
633  // Permuted j and k indices
634  unsigned int
635  jp=j,
636  kp=k;
637 
638  if ((i>3) && (this->_diagonal_selection!=DIAG_02_13))
639  {
640  // Just the enum value cast to an unsigned int...
641  const unsigned ds = static_cast<unsigned int>(this->_diagonal_selection); // == 1 or 2
642 
643  // Instead of doing a lot of arithmetic, use these
644  // straightforward arrays for the permutations. Note that 3 ->
645  // 3, and the first array consists of "forward" permutations of
646  // the sets {0,1,2}, {4,5,6}, and {7,8,9} while the second array
647  // consists of "reverse" permutations of the same sets.
648  const unsigned int perms[2][10] =
649  {
650  {1, 2, 0, 3, 5, 6, 4, 8, 9, 7},
651  {2, 0, 1, 3, 6, 4, 5, 9, 7, 8}
652  };
653 
654  // Permute j
655  jp = perms[ds-1][j];
656  // if (jp<3)
657  // jp = (jp+ds)%3;
658  // else if (jp>3)
659  // jp = (jp-1+ds)%3 + 1 + 3*((jp-1)/3);
660 
661  // Permute k
662  kp = perms[ds-1][k];
663  // if (kp<3)
664  // kp = (kp+ds)%3;
665  // else if (kp>3)
666  // kp = (kp-1+ds)%3 + 1 + 3*((kp-1)/3);
667  }
668 
669  // Debugging:
670  // libMesh::err << "Selected diagonal " << _diagonal_selection << std::endl;
671  // libMesh::err << "j=" << j << std::endl;
672  // libMesh::err << "k=" << k << std::endl;
673  // libMesh::err << "jp=" << jp << std::endl;
674  // libMesh::err << "kp=" << kp << std::endl;
675 
676  // Call embedding matrix with permuted indices
677  return this->_embedding_matrix[i][jp][kp];
678 }

684 Real Tet10::volume () const
685 {
686  // Make copies of our points. It makes the subsequent calculations a bit
687  // shorter and avoids dereferencing the same pointer multiple times.
688  Point
689  x0 = point(0), x1 = point(1), x2 = point(2), x3 = point(3), x4 = point(4),
690  x5 = point(5), x6 = point(6), x7 = point(7), x8 = point(8), x9 = point(9);
691 
692  // The constant components of the dx/dxi vector, linear in xi, eta, zeta.
693  // These were copied directly from the output of a Python script.
694  Point dx_dxi[4] =
695  {
696  -3*x0 - x1 + 4*x4, // constant
697  4*x0 - 4*x4 - 4*x7 + 4*x8, // zeta
698  4*x0 - 4*x4 + 4*x5 - 4*x6, // eta
699  4*x0 + 4*x1 - 8*x4 // xi
700  };
701 
702  // The constant components of the dx/deta vector, linear in xi, eta, zeta.
703  // These were copied directly from the output of a Python script.
704  Point dx_deta[4] =
705  {
706  -3*x0 - x2 + 4*x6, // constant
707  4*x0 - 4*x6 - 4*x7 + 4*x9, // zeta
708  4*x0 + 4*x2 - 8*x6, // eta
709  4*x0 - 4*x4 + 4*x5 - 4*x6 // xi
710  };
711 
712  // The constant components of the dx/dzeta vector, linear in xi, eta, zeta.
713  // These were copied directly from the output of a Python script.
714  Point dx_dzeta[4] =
715  {
716  -3*x0 - x3 + 4*x7, // constant
717  4*x0 + 4*x3 - 8*x7, // zeta
718  4*x0 - 4*x6 - 4*x7 + 4*x9, // eta
719  4*x0 - 4*x4 - 4*x7 + 4*x8 // xi
720  };
721 
722  // 2x2x2 conical quadrature rule. Note: there is also a five point
723  // rule for tets with a negative weight which would be cheaper, but
724  // we'll use this one to preclude any possible issues with
725  // cancellation error.
726  const int N = 8;
727  static const Real w[N] =
728  {
729  3.6979856358852914509238091810505e-02L,
730  1.6027040598476613723156741868689e-02L,
731  2.1157006454524061178256145400082e-02L,
732  9.1694299214797439226823542540576e-03L,
733  3.6979856358852914509238091810505e-02L,
734  1.6027040598476613723156741868689e-02L,
735  2.1157006454524061178256145400082e-02L,
736  9.1694299214797439226823542540576e-03L
737  };
738 
739  static const Real xi[N] =
740  {
741  1.2251482265544137786674043037115e-01L,
742  5.4415184401122528879992623629551e-01L,
743  1.2251482265544137786674043037115e-01L,
744  5.4415184401122528879992623629551e-01L,
745  1.2251482265544137786674043037115e-01L,
746  5.4415184401122528879992623629551e-01L,
747  1.2251482265544137786674043037115e-01L,
748  5.4415184401122528879992623629551e-01L
749  };
750 
751  static const Real eta[N] =
752  {
753  1.3605497680284601717109468420738e-01L,
754  7.0679724159396903069267439165167e-02L,
755  5.6593316507280088053551297149570e-01L,
756  2.9399880063162286589079157179842e-01L,
757  1.3605497680284601717109468420738e-01L,
758  7.0679724159396903069267439165167e-02L,
759  5.6593316507280088053551297149570e-01L,
760  2.9399880063162286589079157179842e-01L
761  };
762 
763  static const Real zeta[N] =
764  {
765  1.5668263733681830907933725249176e-01L,
766  8.1395667014670255076709592007207e-02L,
767  6.5838687060044409936029672711329e-02L,
768  3.4202793236766414300604458388142e-02L,
769  5.8474756320489429588282763292971e-01L,
770  3.0377276481470755305409673253211e-01L,
771  2.4571332521171333166171692542182e-01L,
772  1.2764656212038543100867773351792e-01L
773  };
774 
775  Real vol = 0.;
776  for (int q=0; q<N; ++q)
777  {
778  // Compute dx_dxi, dx_deta, dx_dzeta at the current quadrature point.
779  Point
780  dx_dxi_q = dx_dxi[0] + zeta[q]*dx_dxi[1] + eta[q]*dx_dxi[2] + xi[q]*dx_dxi[3],
781  dx_deta_q = dx_deta[0] + zeta[q]*dx_deta[1] + eta[q]*dx_deta[2] + xi[q]*dx_deta[3],
782  dx_dzeta_q = dx_dzeta[0] + zeta[q]*dx_dzeta[1] + eta[q]*dx_dzeta[2] + xi[q]*dx_dzeta[3];
783 
784  // Compute scalar triple product, multiply by weight, and accumulate volume.
785  vol += w[q] * triple_product(dx_dxi_q, dx_deta_q, dx_dzeta_q);
786  }
787 
788  return vol;
789 }
