130 bool Tet10::has_affine_map() const
131 {
132  // Make sure edges are straight
133  if (!this->point(4).relative_fuzzy_equals
134  ((this->point(0) + this->point(1))/2))
135  return false;
136  if (!this->point(5).relative_fuzzy_equals
137  ((this->point(1) + this->point(2))/2))
138  return false;
139  if (!this->point(6).relative_fuzzy_equals
140  ((this->point(2) + this->point(0))/2))
141  return false;
142  if (!this->point(7).relative_fuzzy_equals
143  ((this->point(3) + this->point(0))/2))
144  return false;
145  if (!this->point(8).relative_fuzzy_equals
146  ((this->point(3) + this->point(1))/2))
147  return false;
148  if (!this->point(9).relative_fuzzy_equals
149  ((this->point(3) + this->point(2))/2))
150  return false;
151  return true;
152 }

156 UniquePtr<Elem> Tet10::build_side (const unsigned int i,
157  bool proxy) const
158 {
159  libmesh_assert_less (i, this->n_sides());
160 
161  if (proxy)
162  return UniquePtr<Elem>(new Side<Tri6,Tet10>(this,i));
163 
164  else
165  {
166  Elem* face = new Tri6;
167  face->subdomain_id() = this->subdomain_id();
168 
169  for (unsigned n=0; n<face->n_nodes(); ++n)
170  face->set_node(n) = this->get_node(Tet10::side_nodes_map[i][n]);
171 
172  return UniquePtr<Elem>(face);
173  }
174 
175  libmesh_error_msg("We'll never get here!");
176  return UniquePtr<Elem>();
177 }

621 float Tet10::embedding_matrix (const unsigned int i,
622  const unsigned int j,
623  const unsigned int k) const
624 {
625  // Choose an optimal diagonal, if one has not already been selected
626  this->choose_diagonal();
627 
628  // Permuted j and k indices
629  unsigned int
630  jp=j,
631  kp=k;
632 
633  if ((i>3) && (this->_diagonal_selection!=DIAG_02_13))
634  {
635  // Just the enum value cast to an unsigned int...
636  const unsigned ds = static_cast<unsigned int>(this->_diagonal_selection); // == 1 or 2
637 
638  // Instead of doing a lot of arithmetic, use these
639  // straightforward arrays for the permutations. Note that 3 ->
640  // 3, and the first array consists of "forward" permutations of
641  // the sets {0,1,2}, {4,5,6}, and {7,8,9} while the second array
642  // consists of "reverse" permutations of the same sets.
643  const unsigned int perms[2][10] =
644  {
645  {1, 2, 0, 3, 5, 6, 4, 8, 9, 7},
646  {2, 0, 1, 3, 6, 4, 5, 9, 7, 8}
647  };
648 
649  // Permute j
650  jp = perms[ds-1][j];
651  // if (jp<3)
652  // jp = (jp+ds)%3;
653  // else if (jp>3)
654  // jp = (jp-1+ds)%3 + 1 + 3*((jp-1)/3);
655 
656  // Permute k
657  kp = perms[ds-1][k];
658  // if (kp<3)
659  // kp = (kp+ds)%3;
660  // else if (kp>3)
661  // kp = (kp-1+ds)%3 + 1 + 3*((kp-1)/3);
662  }
663 
664  // Debugging:
665  // libMesh::err << "Selected diagonal " << _diagonal_selection << std::endl;
666  // libMesh::err << "j=" << j << std::endl;
667  // libMesh::err << "k=" << k << std::endl;
668  // libMesh::err << "jp=" << jp << std::endl;
669  // libMesh::err << "kp=" << kp << std::endl;
670 
671  // Call embedding matrx with permuted indices
672  return this->_embedding_matrix[i][jp][kp];
