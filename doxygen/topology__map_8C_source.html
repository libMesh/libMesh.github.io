54 void TopologyMap::add_node(const Node & mid_node,
55  const std::vector<std::pair<dof_id_type, dof_id_type> > & bracketing_nodes)
56 {
57  const dof_id_type mid_node_id = mid_node.id();
58 
59  libmesh_assert_not_equal_to(mid_node_id, DofObject::invalid_id);
60 
61  for (unsigned int i=0; i != bracketing_nodes.size(); ++i)
62  {
63  const dof_id_type id1 = bracketing_nodes[i].first;
64  const dof_id_type id2 = bracketing_nodes[i].second;
65  const dof_id_type lower_id = std::min(id1, id2);
66  const dof_id_type upper_id = std::max(id1, id2);
67 
68  // We should never be inserting inconsistent data
69 #ifndef NDEBUG
70  map_type::iterator it =
71  _map.find(std::make_pair(lower_id, upper_id));
72 
73  if (it != _map.end())
74  libmesh_assert_equal_to (it->second, mid_node_id);
75 #endif
76 
77  this->_map.insert(std::make_pair(std::make_pair(lower_id, upper_id),
78  mid_node_id));
79 
80  }
81 }

84 dof_id_type TopologyMap::find(const std::vector<std::pair<dof_id_type, dof_id_type> > & bracketing_nodes) const
85 {
86  dof_id_type new_node_id = DofObject::invalid_id;
87 
88  for (unsigned int i = 0; i != bracketing_nodes.size(); ++i)
89  {
90  const dof_id_type lower_id = std::min(bracketing_nodes[i].first,
91  bracketing_nodes[i].second);
92  const dof_id_type upper_id = std::max(bracketing_nodes[i].first,
93  bracketing_nodes[i].second);
94 
95  const dof_id_type possible_new_node_id =
96  this->find(lower_id, upper_id);
97 
98  if (possible_new_node_id != DofObject::invalid_id)
99  {
100  // If we found a node already, but we're still here, it's to
101  // debug map consistency: we'd better always find the same
102  // node
103  if (new_node_id != DofObject::invalid_id)
104  libmesh_assert_equal_to (new_node_id, possible_new_node_id);
105 
106  new_node_id = possible_new_node_id;
107  }
108 
109  // If we're not debugging map consistency then we can quit as
110  // soon as we find a node
111 #ifdef NDEBUG
112  if (new_node_id != DofObject::invalid_id)
113  break;
114 #endif
115  }
116  return new_node_id;
117 }

120 dof_id_type TopologyMap::find(dof_id_type bracket_node1,
121  dof_id_type bracket_node2) const
122 {
123  const dof_id_type lower_id = std::min(bracket_node1, bracket_node2);
124  const dof_id_type upper_id = std::max(bracket_node1, bracket_node2);
125 
126  map_type::const_iterator it =
127  _map.find(std::make_pair(lower_id, upper_id));
128 
129  if (it == _map.end())
130  return DofObject::invalid_id;
131 
132  libmesh_assert_not_equal_to (it->second, DofObject::invalid_id);
133 
134  return it->second;
135 }

141 void TopologyMap::fill(const MeshBase & mesh)
142 {
143  // Populate the nodes map
144  MeshBase::const_element_iterator
145  it = mesh.elements_begin(),
146  end = mesh.elements_end();
147  for (; it != end; ++it)
148  {
149  const Elem * elem = *it;
150 
151  // We only need to add nodes which might be added during mesh
152  // refinement; this means they need to be child nodes.
153  if (!elem->has_children())
154  continue;
155 
156  for (unsigned int c = 0; c != elem->n_children(); ++c)
157  {
158  if (elem->child_ptr(c)->is_remote())
159  continue;
160 
161  for (unsigned int n = 0; n != elem->n_nodes_in_child(c); ++n)
162  {
163  const std::vector<std::pair<dof_id_type, dof_id_type> >
164  bracketing_nodes = elem->bracketing_nodes(c,n);
165 
166  this->add_node(elem->child_ptr(c)->node_ref(n),
167  bracketing_nodes);
168  }
169  }
170  }
171 }
