177 Real Edge4::volume () const
178 {
179  // Make copies of our points. It makes the subsequent calculations a bit
180  // shorter and avoids dereferencing the same pointer multiple times.
181  Point
182  x0 = point(0),
183  x1 = point(1),
184  x2 = point(2),
185  x3 = point(3);
186 
187  // Construct constant data vectors.
188  // \vec{x}_{\xi} = \vec{a1}*xi**2 + \vec{b1}*xi + \vec{c1}
189  // This is copy-pasted directly from the output of a Python script.
190  Point
191  a1 = -27*x0/16 + 27*x1/16 + 81*x2/16 - 81*x3/16,
192  b1 = 9*x0/8 + 9*x1/8 - 9*x2/8 - 9*x3/8,
193  c1 = x0/16 - x1/16 - 27*x2/16 + 27*x3/16;
194 
195  // 4 point quadrature, 7th-order accurate
196  const unsigned int N = 4;
197  const Real q[N] = {-std::sqrt(525 + 70*std::sqrt(30.)) / 35,
198  -std::sqrt(525 - 70*std::sqrt(30.)) / 35,
199  std::sqrt(525 - 70*std::sqrt(30.)) / 35,
200  std::sqrt(525 + 70*std::sqrt(30.)) / 35};
201  const Real w[N] = {(18 - std::sqrt(30.)) / 36,
202  (18 + std::sqrt(30.)) / 36,
203  (18 + std::sqrt(30.)) / 36,
204  (18 - std::sqrt(30.)) / 36};
205 
206  Real vol=0.;
207  for (unsigned int i=0; i<N; ++i)
208  vol += w[i] * (q[i]*q[i]*a1 + q[i]*b1 + c1).norm();
209 
210  return vol;
211 }
