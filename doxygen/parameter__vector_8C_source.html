83 void ParameterVector::resize(unsigned int s)
84 {
85  libmesh_assert(!_is_shallow_copy);
86 
87  const std::size_t old_size = this->_params.size();
88 
89  // If we're shrinking the vector, we don't want to leak memory.
90  // Note that we're using < in these for loops, not !=
91  // We don't know a priori if we're shrinking or growing
92  for (unsigned int i=s; i < old_size; ++i)
93  delete _params[i];
94 
95  this->_params.resize(s);
96 
97  for (unsigned int i=old_size; i < s; ++i)
98  this->_params[i] =
99  new ParameterPointer<Number>(libmesh_nullptr);
100 }
