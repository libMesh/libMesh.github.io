99 std::unique_ptr<Elem> InfHex::side_ptr (const unsigned int i)
100 {
101  libmesh_assert_less (i, this->n_sides());
102 
103  // Return value
104  std::unique_ptr<Elem> face;
105 
106  // Think of a unit cube: (-1,1) x (-1,1) x (-1,1),
107  // with (in general) the normals pointing outwards
108  switch (i)
109  {
110  // the face at z = -1
111  // the base, where the infinite element couples to conventional
112  // elements
113  case 0:
114  {
115  // Oops, here we are, claiming the normal of the face
116  // elements point outwards -- and this is the exception:
117  // For the side built from the base face,
118  // the normal is pointing _into_ the element!
119  // Why is that? - In agreement with build_side_ptr(),
120  // which in turn _has_ to build the face in this
121  // way as to enable the cool way \p InfFE re-uses \p FE.
122  face = libmesh_make_unique<Quad4>();
123  break;
124  }
125 
126  // These faces connect to other infinite elements.
127  case 1: // the face at y = -1
128  case 2: // the face at x = 1
129  case 3: // the face at y = 1
130  case 4: // the face at x = -1
131  {
132  face = libmesh_make_unique<InfQuad4>();
133  break;
134  }
135 
136  default:
137  libmesh_error_msg("Invalid side i = " << i);
138  }
139 
140  // Set the nodes
141  for (unsigned n=0; n<face->n_nodes(); ++n)
142  face->set_node(n) = this->node_ptr(InfHex8::side_nodes_map[i][n]);
143 
144  return face;
145 }

173 Real InfHex::quality (const ElemQuality q) const
174 {
175  switch (q)
176  {
177 
187  case DIAGONAL:
188  {
189  // Diagonal between node 0 and node 2
190  const Real d02 = this->length(0,2);
191 
192  // Diagonal between node 1 and node 3
193  const Real d13 = this->length(1,3);
194 
195  // Find the biggest and smallest diagonals
196  const Real min = std::min(d02, d13);
197  const Real max = std::max(d02, d13);
198 
199  libmesh_assert_not_equal_to (max, 0.0);
200 
201  return min / max;
202 
203  break;
204  }
205 
213  case TAPER:
214  {
215 
219  const Real d01 = this->length(0,1);
220  const Real d12 = this->length(1,2);
221  const Real d23 = this->length(2,3);
222  const Real d03 = this->length(0,3);
223 
224  std::vector<Real> edge_ratios(2);
225 
226  // Bottom
227  edge_ratios[8] = std::min(d01, d23) / std::max(d01, d23);
228  edge_ratios[9] = std::min(d03, d12) / std::max(d03, d12);
229 
230  return *(std::min_element(edge_ratios.begin(), edge_ratios.end())) ;
231 
232  break;
233  }
234 
235 
243  case STRETCH:
244  {
248  const Real sqrt3 = 1.73205080756888;
249 
253  const Real d02 = this->length(0,2);
254  const Real d13 = this->length(1,3);
255  const Real max_diag = std::max(d02, d13);
256 
257  libmesh_assert_not_equal_to ( max_diag, 0.0 );
258 
262  std::vector<Real> edges(4);
263  edges[0] = this->length(0,1);
264  edges[1] = this->length(1,2);
265  edges[2] = this->length(2,3);
266  edges[3] = this->length(0,3);
267 
268  const Real min_edge = *(std::min_element(edges.begin(), edges.end()));
269  return sqrt3 * min_edge / max_diag ;
270  }
271 
272 
277  default:
278  return Elem::quality(q);
279  }
280 }

393 bool InfHex::contains_point (const Point & p, Real tol) const
394 {
395  // For infinite elements with linear base interpolation:
396  // make use of the fact that infinite elements do not
397  // live inside the envelope. Use a fast scheme to
398  // check whether point \p p is inside or outside
399  // our relevant part of the envelope. Note that
400  // this is not exclusive: only when the distance is less,
401  // we are safe. Otherwise, we cannot say anything. The
402  // envelope may be non-spherical, the physical point may lie
403  // inside the envelope, outside the envelope, or even inside
404  // this infinite element. Therefore if this fails,
405  // fall back to the FEInterface::inverse_map()
406  const Point my_origin (this->origin());
407 
408  // determine the minimal distance of the base from the origin
409  // Use norm_sq() instead of norm(), it is faster
410  Point pt0_o(this->point(0) - my_origin);
411  Point pt1_o(this->point(1) - my_origin);
412  Point pt2_o(this->point(2) - my_origin);
413  Point pt3_o(this->point(3) - my_origin);
414  const Real min_distance_sq = std::min(pt0_o.norm_sq(),
415  std::min(pt1_o.norm_sq(),
416  std::min(pt2_o.norm_sq(),
417  pt3_o.norm_sq())));
418 
419  // work with 1% allowable deviation. We can still fall
420  // back to the InfFE::inverse_map()
421  const Real conservative_p_dist_sq = 1.01 * (Point(p - my_origin).norm_sq());
422 
423 
424 
425  if (conservative_p_dist_sq < min_distance_sq)
426  {
427  // the physical point is definitely not contained in the element
428  return false;
429  }
430 
431  // this captures the case that the point is not (almost) in the direction of the element.:
432  // first, project the problem onto the unit sphere:
433  Point p_o(p - my_origin);
434  pt0_o /= pt0_o.norm();
435  pt1_o /= pt1_o.norm();
436  pt2_o /= pt2_o.norm();
437  pt3_o /= pt3_o.norm();
438  p_o /= p_o.norm();
439 
440 
441  // now, check if it is in the projected face; using that the diagonal contains
442  // the largest distance between points in it
443  Real max_h = std::max((pt0_o - pt2_o).norm_sq(),
444  (pt1_o - pt2_o).norm_sq())*1.01;
445 
446  if ((p_o - pt0_o).norm_sq() > max_h ||
447  (p_o - pt1_o).norm_sq() > max_h ||
448  (p_o - pt2_o).norm_sq() > max_h ||
449  (p_o - pt3_o).norm_sq() > max_h )
450  {
451  // the physical point is definitely not contained in the element
452  return false;
453  }
454 
455  // Declare a basic FEType. Will use default in the base,
456  // and something else (not important) in radial direction.
457  FEType fe_type(default_order());
458 
459  const Point mapped_point = FEInterface::inverse_map(dim(),
460  fe_type,
461  this,
462  p,
463  tol,
464  false);
465 
466  return FEInterface::on_reference_element(mapped_point, this->type(), tol);
467 }
