29 void CentroidPartitioner::partition_range(MeshBase & mesh,
30  MeshBase::element_iterator it,
31  MeshBase::element_iterator end,
32  unsigned int n)
33 {
34  // Check for an easy return
35  if (n == 1)
36  {
37  this->single_partition_range (it, end);
38  return;
39  }
40 
41  // We don't yet support distributed meshes with this Partitioner
42  if (!mesh.is_serial())
43  libmesh_not_implemented();
44 
45  // Compute the element centroids. Note: we used to skip this step
46  // if the number of elements was unchanged from the last call, but
47  // that doesn't account for elements that have moved a lot since the
48  // last time the Partitioner was called...
49  this->compute_centroids (it, end);
50 
51  switch (this->sort_method())
52  {
53  case X:
54  {
55  std::sort(_elem_centroids.begin(),
56  _elem_centroids.end(),
57  CentroidPartitioner::sort_x);
58 
59  break;
60  }
61 
62 
63  case Y:
64  {
65  std::sort(_elem_centroids.begin(),
66  _elem_centroids.end(),
67  CentroidPartitioner::sort_y);
68 
69  break;
70 
71  }
72 
73 
74  case Z:
75  {
76  std::sort(_elem_centroids.begin(),
77  _elem_centroids.end(),
78  CentroidPartitioner::sort_z);
79 
80  break;
81  }
82 
83 
84  case RADIAL:
85  {
86  std::sort(_elem_centroids.begin(),
87  _elem_centroids.end(),
88  CentroidPartitioner::sort_radial);
89 
90  break;
91  }
92  default:
93  libmesh_error_msg("Unknown sort method: " << this->sort_method());
94  }
95 
96  // Make sure the user has not handed us an
97  // invalid number of partitions.
98  libmesh_assert_greater (n, 0);
99 
100  // Compute target_size, the approximate number of elements on each processor.
101  const dof_id_type target_size = _elem_centroids.size() / n;
102 
103  for (dof_id_type i=0; i<_elem_centroids.size(); i++)
104  {
105  Elem * elem = _elem_centroids[i].second;
106 
107  // FIXME: All "extra" elements go on the last processor... this
108  // could probably be improved.
109  elem->processor_id() =
110  std::min (cast_int<processor_id_type>(i / target_size),
111  cast_int<processor_id_type>(n-1));
112  }
113 }

128 void CentroidPartitioner::compute_centroids (MeshBase::element_iterator it,
129  MeshBase::element_iterator end)
130 {
131  _elem_centroids.clear();
132 
133  for (; it != end; ++it)
134  {
135  Elem * elem = *it;
136 
137  _elem_centroids.push_back(std::make_pair(elem->centroid(), elem));
138  }
