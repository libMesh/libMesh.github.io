31 void MeshTools::Subdivision::find_one_ring(const Tri3Subdivision * elem,
32  std::vector<const Node *> & nodes)
33 {
34  libmesh_assert(elem->is_subdivision_updated());
35  libmesh_assert(elem->get_ordered_node(0));
36 
37  unsigned int valence = elem->get_ordered_valence(0);
38  nodes.resize(valence + 6);
39 
40  // The first three vertices in the patch are the ones from the element triangle
41  nodes[0] = elem->get_ordered_node(0);
42  nodes[1] = elem->get_ordered_node(1);
43  nodes[valence] = elem->get_ordered_node(2);
44 
45  const unsigned int nn0 = elem->local_node_number(nodes[0]->id());
46 
47  const Tri3Subdivision * nb = dynamic_cast<const Tri3Subdivision *>(elem->neighbor_ptr(nn0));
48  libmesh_assert(nb);
49 
50  unsigned int j, i = 1;
51 
52  do
53  {
54  ++i;
55  j = nb->local_node_number(nodes[0]->id());
56  nodes[i] = nb->node_ptr(next[j]);
57  nb = static_cast<const Tri3Subdivision *>(nb->neighbor_ptr(j));
58  } while (nb != elem);
59 
60  /* for nodes connected with N (= valence[0]) */
61  nb = static_cast<const Tri3Subdivision *>(elem->neighbor_ptr(next[nn0]));
62  j = nb->local_node_number(nodes[1]->id());
63  nodes[valence+1] = nb->node_ptr(next[j]);
64 
65  nb = static_cast<const Tri3Subdivision *>(nb->neighbor_ptr(next[j]));
66  j = nb->local_node_number(nodes[valence+1]->id());
67  nodes[valence+4] = nb->node_ptr(next[j]);
68 
69  nb = static_cast<const Tri3Subdivision *>(nb->neighbor_ptr(next[j]));
70  j = nb->local_node_number(nodes[valence+4]->id());
71  nodes[valence+5] = nb->node_ptr(next[j]);
72 
73  /* for nodes connected with 1 */
74  nb = static_cast<const Tri3Subdivision *>(elem->neighbor_ptr(next[nn0]));
75  j = nb->local_node_number(nodes[1]->id());
76  // nodes[valence+1] has been determined already
77 
78  nb = static_cast<const Tri3Subdivision *>(nb->neighbor_ptr(j));
79  j = nb->local_node_number(nodes[1]->id());
80  nodes[valence+2] = nb->node_ptr(next[j]);
81 
82  nb = static_cast<const Tri3Subdivision *>(nb->neighbor_ptr(j));
83  j = nb->local_node_number(nodes[1]->id());
84  nodes[valence+3] = nb->node_ptr(next[j]);
85 
86  return;
87 }

90 void MeshTools::Subdivision::all_subdivision(MeshBase & mesh)
91 {
92  std::vector<Elem *> new_elements;
93  new_elements.reserve(mesh.n_elem());
94  const bool mesh_has_boundary_data =
95  (mesh.get_boundary_info().n_boundary_ids() > 0);
96 
97  std::vector<Elem *> new_boundary_elements;
98  std::vector<short int> new_boundary_sides;
99  std::vector<boundary_id_type> new_boundary_ids;
100 
101  // Container to catch ids handed back from BoundaryInfo
102  std::vector<boundary_id_type> ids;
103 
104  MeshBase::const_element_iterator el = mesh.elements_begin();
105  const MeshBase::const_element_iterator end_el = mesh.elements_end();
106  for (; el != end_el; ++el)
107  {
108  const Elem * elem = *el;
109  libmesh_assert_equal_to(elem->type(), TRI3);
110 
111  Elem * tri = new Tri3Subdivision;
112  tri->set_id(elem->id());
113  tri->subdomain_id() = elem->subdomain_id();
114  tri->set_node(0) = (*el)->node_ptr(0);
115  tri->set_node(1) = (*el)->node_ptr(1);
116  tri->set_node(2) = (*el)->node_ptr(2);
117 
118  if (mesh_has_boundary_data)
119  {
120  for (unsigned short side = 0; side < elem->n_sides(); ++side)
121  {
122  mesh.get_boundary_info().boundary_ids(elem, side, ids);
123 
124  for (std::size_t id=0; id<ids.size(); ++id)
125  {
126  // add the boundary id to the list of new boundary ids
127  new_boundary_ids.push_back(ids[id]);
128  new_boundary_elements.push_back(tri);
129  new_boundary_sides.push_back(side);
130  }
131  }
132 
133  // remove the original element from the BoundaryInfo structure
134  mesh.get_boundary_info().remove(elem);
135  }
136 
137  new_elements.push_back(tri);
138  mesh.insert_elem(tri);
139  }
140  mesh.prepare_for_use();
141 
142  if (mesh_has_boundary_data)
143  {
144  // If the old mesh had boundary data, the new mesh better have some too.
145  libmesh_assert_greater(new_boundary_elements.size(), 0);
146 
147  // We should also be sure that the lengths of the new boundary data vectors
148  // are all the same.
149  libmesh_assert_equal_to(new_boundary_sides.size(), new_boundary_elements.size());
150  libmesh_assert_equal_to(new_boundary_sides.size(), new_boundary_ids.size());
151 
152  // Add the new boundary info to the mesh.
153  for (std::size_t s = 0; s < new_boundary_elements.size(); ++s)
154  mesh.get_boundary_info().add_side(new_boundary_elements[s],
155  new_boundary_sides[s],
156  new_boundary_ids[s]);
157  }
158 
159  mesh.prepare_for_use();
160 }

163 void MeshTools::Subdivision::prepare_subdivision_mesh(MeshBase & mesh, bool ghosted)
164 {
165  mesh.prepare_for_use();
166 
167  // convert all mesh elements to subdivision elements
168  all_subdivision(mesh);
169 
170  if (!ghosted)
171  {
172  // add the ghost elements for the boundaries
173  add_boundary_ghosts(mesh);
174  }
175  else
176  {
177  // This assumes that the mesh already has the ghosts. Only tagging them is required here.
178  tag_boundary_ghosts(mesh);
179  }
180 
181  mesh.prepare_for_use();
182 
183  std::vector<std::vector<const Elem *> > nodes_to_elem_map;
184  MeshTools::build_nodes_to_elem_map(mesh, nodes_to_elem_map);
185 
186  // compute the node valences
187  MeshBase::const_node_iterator nd = mesh.nodes_begin();
188  const MeshBase::const_node_iterator end_nd = mesh.nodes_end();
189  for (; nd != end_nd; ++nd)
190  {
191  Node * node = *nd;
192  std::vector<const Node *> neighbors;
193  MeshTools::find_nodal_neighbors(mesh, *node, nodes_to_elem_map, neighbors);
194  const unsigned int valence =
195  cast_int<unsigned int>(neighbors.size());
196  libmesh_assert_greater(valence, 1);
197  node->set_valence(valence);
198  }
199 
200  MeshBase::const_element_iterator el = mesh.elements_begin();
201  const MeshBase::const_element_iterator end_el = mesh.elements_end();
202  for (; el != end_el; ++el)
203  {
204  Tri3Subdivision * elem = dynamic_cast<Tri3Subdivision *>(*el);
205  libmesh_assert(elem);
206  if (!elem->is_ghost())
207  elem->prepare_subdivision_properties();
208  }

212 void MeshTools::Subdivision::tag_boundary_ghosts(MeshBase & mesh)
213 {
214  MeshBase::element_iterator el = mesh.elements_begin();
215  const MeshBase::element_iterator end_el = mesh.elements_end();
216  for (; el != end_el; ++el)
217  {
218  Elem * elem = *el;
219  libmesh_assert_equal_to(elem->type(), TRI3SUBDIVISION);
220 
221  Tri3Subdivision * sd_elem = static_cast<Tri3Subdivision *>(elem);
222  for (unsigned int i = 0; i < elem->n_sides(); ++i)
223  {
224  if (elem->neighbor_ptr(i) == libmesh_nullptr)
225  {
226  sd_elem->set_ghost(true);
227  // set all other neighbors to ghosts as well
228  if (elem->neighbor_ptr(next[i]))
229  {
230  Tri3Subdivision * nb = static_cast<Tri3Subdivision *>(elem->neighbor_ptr(next[i]));
231  nb->set_ghost(true);
232  }
233  if (elem->neighbor_ptr(prev[i]))
234  {
235  Tri3Subdivision * nb = static_cast<Tri3Subdivision *>(elem->neighbor_ptr(prev[i]));
236  nb->set_ghost(true);
237  }
238  }
239  }
240  }

244 void MeshTools::Subdivision::add_boundary_ghosts(MeshBase & mesh)
245 {
246  static const Real tol = 1e-5;
247 
248  // add the mirrored ghost elements (without using iterators, because the mesh is modified in the course)
249  std::vector<Tri3Subdivision *> ghost_elems;
250  std::vector<Node *> ghost_nodes;
251  const unsigned int n_elem = mesh.n_elem();
252  for (unsigned int eid = 0; eid < n_elem; ++eid)
253  {
254  Elem * elem = mesh.elem_ptr(eid);
255  libmesh_assert_equal_to(elem->type(), TRI3SUBDIVISION);
256 
257  // If the triangle happens to be in a corner (two boundary
258  // edges), we perform a counter-clockwise loop by mirroring the
259  // previous triangle until we come back to the original
260  // triangle. This prevents degenerated triangles in the mesh
261  // corners and guarantees that the node in the middle of the
262  // loop is of valence=6.
263  for (unsigned int i = 0; i < elem->n_sides(); ++i)
264  {
265  libmesh_assert_not_equal_to(elem->neighbor_ptr(i), elem);
266 
267  if (elem->neighbor_ptr(i) == libmesh_nullptr &&
268  elem->neighbor_ptr(next[i]) == libmesh_nullptr)
269  {
270  Elem * nelem = elem;
271  unsigned int k = i;
272  for (unsigned int l=0;l<4;l++)
273  {
274  // this is the vertex to be mirrored
275  Point point = nelem->point(k) + nelem->point(next[k]) - nelem->point(prev[k]);
276 
277  // Check if the proposed vertex doesn't coincide
278  // with one of the existing vertices. This is
279  // necessary because for some triangulations, it can
280  // happen that two mirrored ghost vertices coincide,
281  // which would then lead to a zero size ghost
282  // element below.
283  Node * node = libmesh_nullptr;
284  for (std::size_t j = 0; j < ghost_nodes.size(); ++j)
285  {
286  if ((*ghost_nodes[j] - point).norm() < tol * (elem->point(k) - point).norm())
287  {
288  node = ghost_nodes[j];
289  break;
290  }
291  }
292 
293  // add the new vertex only if no other is nearby
294  if (node == libmesh_nullptr)
295  {
296  node = mesh.add_point(point);
297  ghost_nodes.push_back(node);
298  }
299 
300  Tri3Subdivision * newelem = new Tri3Subdivision();
301 
302  // add the first new ghost element to the list just as in the non-corner case
303  if (l == 0)
304  ghost_elems.push_back(newelem);
305 
306  newelem->set_node(0) = nelem->node_ptr(next[k]);
307  newelem->set_node(1) = nelem->node_ptr(k);
308  newelem->set_node(2) = node;
309  newelem->set_neighbor(0, nelem);
310  newelem->set_ghost(true);
311  if (l>0)
312  newelem->set_neighbor(2, libmesh_nullptr);
313  nelem->set_neighbor(k, newelem);
314 
315  mesh.add_elem(newelem);
316  mesh.get_boundary_info().add_node(nelem->node_ptr(k), 1);
317  mesh.get_boundary_info().add_node(nelem->node_ptr(next[k]), 1);
318  mesh.get_boundary_info().add_node(nelem->node_ptr(prev[k]), 1);
319  mesh.get_boundary_info().add_node(node, 1);
320 
321  nelem = newelem;
322  k = 2 ;
323  }
324 
325  Tri3Subdivision * newelem = new Tri3Subdivision();
326 
327  newelem->set_node(0) = elem->node_ptr(next[i]);
328  newelem->set_node(1) = nelem->node_ptr(2);
329  newelem->set_node(2) = elem->node_ptr(prev[i]);
330  newelem->set_neighbor(0, nelem);
331  nelem->set_neighbor(2, newelem);
332  newelem->set_ghost(true);
333  newelem->set_neighbor(2, elem);
334  elem->set_neighbor(next[i],newelem);
335 
336  mesh.add_elem(newelem);
337 
338  break;
339  }
340  }
341 
342  for (unsigned int i = 0; i < elem->n_sides(); ++i)
343  {
344  libmesh_assert_not_equal_to(elem->neighbor_ptr(i), elem);
345  if (elem->neighbor_ptr(i) == libmesh_nullptr)
346  {
347  // this is the vertex to be mirrored
348  Point point = elem->point(i) + elem->point(next[i]) - elem->point(prev[i]);
349 
350  // Check if the proposed vertex doesn't coincide with
351  // one of the existing vertices. This is necessary
352  // because for some triangulations, it can happen that
353  // two mirrored ghost vertices coincide, which would
354  // then lead to a zero size ghost element below.
355  Node * node = libmesh_nullptr;
356  for (std::size_t j = 0; j < ghost_nodes.size(); ++j)
357  {
358  if ((*ghost_nodes[j] - point).norm() < tol * (elem->point(i) - point).norm())
359  {
360  node = ghost_nodes[j];
361  break;
362  }
363  }
364 
365  // add the new vertex only if no other is nearby
366  if (node == libmesh_nullptr)
367  {
368  node = mesh.add_point(point);
369  ghost_nodes.push_back(node);
370  }
371 
372  Tri3Subdivision * newelem = new Tri3Subdivision();
373  ghost_elems.push_back(newelem);
374 
375  newelem->set_node(0) = elem->node_ptr(next[i]);
376  newelem->set_node(1) = elem->node_ptr(i);
377  newelem->set_node(2) = node;
378  newelem->set_neighbor(0, elem);
379  newelem->set_ghost(true);
380  elem->set_neighbor(i, newelem);
381 
382  mesh.add_elem(newelem);
383  mesh.get_boundary_info().add_node(elem->node_ptr(i), 1);
384  mesh.get_boundary_info().add_node(elem->node_ptr(next[i]), 1);
385  mesh.get_boundary_info().add_node(elem->node_ptr(prev[i]), 1);
386  mesh.get_boundary_info().add_node(node, 1);
387  }
388  }
389  }
390 
391  // add the missing ghost elements (connecting new ghost nodes)
392  std::vector<Tri3Subdivision *> missing_ghost_elems;
393  std::vector<Tri3Subdivision *>::iterator ghost_el = ghost_elems.begin();
394  const std::vector<Tri3Subdivision *>::iterator end_ghost_el = ghost_elems.end();
395  for (; ghost_el != end_ghost_el; ++ghost_el)
396  {
397  Tri3Subdivision * elem = *ghost_el;
398  libmesh_assert(elem->is_ghost());
399 
400  for (unsigned int i = 0; i < elem->n_sides(); ++i)
401  {
402  if (elem->neighbor_ptr(i) == libmesh_nullptr &&
403  elem->neighbor_ptr(prev[i]) != libmesh_nullptr)
404  {
405  // go around counter-clockwise
406  Tri3Subdivision * nb1 = static_cast<Tri3Subdivision *>(elem->neighbor_ptr(prev[i]));
407  Tri3Subdivision * nb2 = nb1;
408  unsigned int j = i;
409  unsigned int n_nb = 0;
410  while (nb1 != libmesh_nullptr && nb1->id() != elem->id())
411  {
412  j = nb1->local_node_number(elem->node_id(i));
413  nb2 = nb1;
414  nb1 = static_cast<Tri3Subdivision *>(nb1->neighbor_ptr(prev[j]));
415  libmesh_assert(nb1 == libmesh_nullptr || nb1->id() != nb2->id());
416  n_nb++;
417  }
418 
419  libmesh_assert_not_equal_to(nb2->id(), elem->id());
420 
421  // Above, we merged coinciding ghost vertices. Therefore, we need
422  // to exclude the case where there is no ghost element to add between
423  // these two (identical) ghost nodes.
424  if (elem->node_ptr(next[i])->id() == nb2->node_ptr(prev[j])->id())
425  break;
426 
427  // If the number of already present neighbors is less than 4, we add another extra element
428  // so that the node in the middle of the loop ends up being of valence=6.
429  // This case usually happens when the middle node corresponds to a corner of the original mesh,
430  // and the extra element below prevents degenerated triangles in the mesh corners.
431  if (n_nb < 4)
432  {
433  // this is the vertex to be mirrored
434  Point point = nb2->point(j) + nb2->point(prev[j]) - nb2->point(next[j]);
435 
436  // Check if the proposed vertex doesn't coincide with one of the existing vertices.
437  // This is necessary because for some triangulations, it can happen that two mirrored
438  // ghost vertices coincide, which would then lead to a zero size ghost element below.
439  Node * node = libmesh_nullptr;
440  for (std::size_t k = 0; k < ghost_nodes.size(); ++k)
441  {
442  if ((*ghost_nodes[k] - point).norm() < tol * (nb2->point(j) - point).norm())
443  {
444  node = ghost_nodes[k];
445  break;
446  }
447  }
448 
449  // add the new vertex only if no other is nearby
450  if (node == libmesh_nullptr)
451  {
452  node = mesh.add_point(point);
453  ghost_nodes.push_back(node);
454  }
455 
456  Tri3Subdivision * newelem = new Tri3Subdivision();
457 
458  newelem->set_node(0) = nb2->node_ptr(j);
459  newelem->set_node(1) = nb2->node_ptr(prev[j]);
460  newelem->set_node(2) = node;
461  newelem->set_neighbor(0, nb2);
462  newelem->set_neighbor(1, libmesh_nullptr);
463  newelem->set_ghost(true);
464  nb2->set_neighbor(prev[j], newelem);
465 
466  mesh.add_elem(newelem);
467  mesh.get_boundary_info().add_node(nb2->node_ptr(j), 1);
468  mesh.get_boundary_info().add_node(nb2->node_ptr(prev[j]), 1);
469  mesh.get_boundary_info().add_node(node, 1);
470 
471  nb2 = newelem;
472  j = nb2->local_node_number(elem->node_id(i));
473  }
474 
475  Tri3Subdivision * newelem = new Tri3Subdivision();
476  newelem->set_node(0) = elem->node_ptr(next[i]);
477  newelem->set_node(1) = elem->node_ptr(i);
478  newelem->set_node(2) = nb2->node_ptr(prev[j]);
479  newelem->set_neighbor(0, elem);
480  newelem->set_neighbor(1, nb2);
481  newelem->set_neighbor(2, libmesh_nullptr);
482  newelem->set_ghost(true);
483 
484  elem->set_neighbor(i, newelem);
485  nb2->set_neighbor(prev[j], newelem);
486 
487  missing_ghost_elems.push_back(newelem);
488  break;
489  }
490  } // end side loop
491  } // end ghost element loop
492 
493  // add the missing ghost elements to the mesh
494  std::vector<Tri3Subdivision *>::iterator missing_el = missing_ghost_elems.begin();
495  const std::vector<Tri3Subdivision *>::iterator end_missing_el = missing_ghost_elems.end();
496  for (; missing_el != end_missing_el; ++missing_el)
497  mesh.add_elem(*missing_el);
498 }
