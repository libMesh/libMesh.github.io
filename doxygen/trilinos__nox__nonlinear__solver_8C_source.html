103 bool Problem_Interface::computeF(const Epetra_Vector & x,
104  Epetra_Vector & r,
105  NOX::Epetra::Interface::Required::FillType /*fillType*/)
106 {
107  START_LOG("residual()", "TrilinosNoxNonlinearSolver");
108 
109  NonlinearImplicitSystem & sys = _solver->system();
110 
111  EpetraVector<Number> X_global(*const_cast<Epetra_Vector *>(&x), sys.comm()), R(r, sys.comm());
112  EpetraVector<Number> & X_sys = *cast_ptr<EpetraVector<Number> *>(sys.solution.get());
113  EpetraVector<Number> & R_sys = *cast_ptr<EpetraVector<Number> *>(sys.rhs);
114 
115  // Use the systems update() to get a good local version of the parallel solution
116  X_global.swap(X_sys);
117  R.swap(R_sys);
118 
119  sys.get_dof_map().enforce_constraints_exactly(sys);
120  sys.update();
121 
122  // Swap back
123  X_global.swap(X_sys);
124  R.swap(R_sys);
125 
126  R.zero();
127 
128  //-----------------------------------------------------------------------------
129  // if the user has provided both function pointers and objects only the pointer
130  // will be used, so catch that as an error
131 
132  if (_solver->residual && _solver->residual_object)
133  libmesh_error_msg("ERROR: cannot specifiy both a function and object to compute the Residual!");
134 
135  if (_solver->matvec && _solver->residual_and_jacobian_object)
136  libmesh_error_msg("ERROR: cannot specifiy both a function and object to compute the combined Residual & Jacobian!");
137 
138  if (_solver->residual != libmesh_nullptr)
139  _solver->residual(*sys.current_local_solution.get(), R, sys);
140 
141  else if (_solver->residual_object != libmesh_nullptr)
142  _solver->residual_object->residual(*sys.current_local_solution.get(), R, sys);
143 
144  else if (_solver->matvec != libmesh_nullptr)
145  _solver->matvec(*sys.current_local_solution.get(), &R, libmesh_nullptr, sys);
146 
147  else if (_solver->residual_and_jacobian_object != libmesh_nullptr)
148  _solver->residual_and_jacobian_object->residual_and_jacobian (*sys.current_local_solution.get(), &R, libmesh_nullptr, sys);
149 
150  else
151  return false;
152 
153  R.close();
154  X_global.close();
155 
156  STOP_LOG("residual()", "TrilinosNoxNonlinearSolver");
157 
158  return true;
159 }

305 template <typename T>
306 std::pair<unsigned int, Real>
307 NoxNonlinearSolver<T>::solve (SparseMatrix<T> & /* jac_in */, // System Jacobian Matrix
308  NumericVector<T> & x_in, // Solution vector
309  NumericVector<T> & /* r_in */, // Residual vector
310  const double, // Stopping tolerance
311  const unsigned int)
312 {
313  this->init ();
314 
315  if (this->user_presolve)
316  this->user_presolve(this->system());
317 
318  EpetraVector<T> * x_epetra = cast_ptr<EpetraVector<T> *>(&x_in);
319  // Creating a Teuchos::RCP as they do in NOX examples does not work here - we get some invalid memory references
320  // thus we make a local copy
321  NOX::Epetra::Vector x(*x_epetra->vec());
322 
323  Teuchos::RCP<Teuchos::ParameterList> nlParamsPtr = Teuchos::rcp(new Teuchos::ParameterList);
324  Teuchos::ParameterList & nlParams = *(nlParamsPtr.get());
325  nlParams.set("Nonlinear Solver", "Line Search Based");
326 
327  //print params
328  Teuchos::ParameterList & printParams = nlParams.sublist("Printing");
329  printParams.set("Output Precision", 3);
330  printParams.set("Output Processor", 0);
331  printParams.set("Output Information",
332  NOX::Utils::OuterIteration +
333  NOX::Utils::OuterIterationStatusTest +
334  NOX::Utils::InnerIteration +
335  NOX::Utils::LinearSolverDetails +
336  NOX::Utils::Parameters +
337  NOX::Utils::Details +
338  NOX::Utils::Warning);
339 
340  Teuchos::ParameterList & dirParams = nlParams.sublist("Direction");
341  dirParams.set("Method", "Newton");
342  Teuchos::ParameterList & newtonParams = dirParams.sublist("Newton");
343  newtonParams.set("Forcing Term Method", "Constant");
344 
345  Teuchos::ParameterList & lsParams = newtonParams.sublist("Linear Solver");
346  lsParams.set("Aztec Solver", "GMRES");
347  lsParams.set("Max Iterations", static_cast<int>(this->max_linear_iterations));
348  lsParams.set("Tolerance", this->initial_linear_tolerance);
349  lsParams.set("Output Frequency", 1);
350  lsParams.set("Size of Krylov Subspace", 1000);
351 
352  //create linear system
353  Teuchos::RCP<NOX::Epetra::Interface::Required> iReq(_interface);
354  Teuchos::RCP<NOX::Epetra::LinearSystemAztecOO> linSys;
355  Teuchos::RCP<Epetra_Operator> pc;
356 
357  if (this->jacobian || this->jacobian_object || this->residual_and_jacobian_object)
358  {
359  if(this->_preconditioner)
360  {
361  // PJNFK
362  lsParams.set("Preconditioner", "User Defined");
363 
364  TrilinosPreconditioner<Number> * trilinos_pc =
365  cast_ptr<TrilinosPreconditioner<Number> *>(this->_preconditioner);
366  pc = Teuchos::rcp(trilinos_pc);
367 
368  Teuchos::RCP<NOX::Epetra::Interface::Preconditioner> iPrec(_interface);
369  linSys = Teuchos::rcp(new NOX::Epetra::LinearSystemAztecOO(printParams, lsParams, iReq, iPrec, pc, x));
370  }
371  else
372  {
373  lsParams.set("Preconditioner", "None");
374  // lsParams.set("Preconditioner", "Ifpack");
375  // lsParams.set("Preconditioner", "AztecOO");
376 
377  // full jacobian
378  NonlinearImplicitSystem & sys = _interface->_solver->system();
379  EpetraMatrix<Number> & jacSys = *cast_ptr<EpetraMatrix<Number> *>(sys.matrix);
380  Teuchos::RCP<Epetra_RowMatrix> jacMat = Teuchos::rcp(jacSys.mat());
381 
382  Teuchos::RCP<NOX::Epetra::Interface::Jacobian> iJac(_interface);
383  linSys = Teuchos::rcp(new NOX::Epetra::LinearSystemAztecOO(printParams, lsParams, iReq, iJac, jacMat, x));
384  }
385  }
386  else
387  {
388  // matrix free
389  Teuchos::RCP<NOX::Epetra::MatrixFree> MF = Teuchos::rcp(new NOX::Epetra::MatrixFree(printParams, iReq, x));
390 
391  Teuchos::RCP<NOX::Epetra::Interface::Jacobian> iJac(MF);
392  linSys = Teuchos::rcp(new NOX::Epetra::LinearSystemAztecOO(printParams, lsParams, iReq, iJac, MF, x));
393  }
394 
395  //create group
396  Teuchos::RCP<NOX::Epetra::Group> grpPtr = Teuchos::rcp(new NOX::Epetra::Group(printParams, iReq, x, linSys));
397  NOX::Epetra::Group & grp = *(grpPtr.get());
398 
399  Teuchos::RCP<NOX::StatusTest::NormF> absresid =
400  Teuchos::rcp(new NOX::StatusTest::NormF(this->absolute_residual_tolerance, NOX::StatusTest::NormF::Unscaled));
401  Teuchos::RCP<NOX::StatusTest::NormF> relresid =
402  Teuchos::rcp(new NOX::StatusTest::NormF(grp, this->relative_residual_tolerance));
403  Teuchos::RCP<NOX::StatusTest::MaxIters> maxiters =
404  Teuchos::rcp(new NOX::StatusTest::MaxIters(this->max_nonlinear_iterations));
405  Teuchos::RCP<NOX::StatusTest::FiniteValue> finiteval =
406  Teuchos::rcp(new NOX::StatusTest::FiniteValue());
407  Teuchos::RCP<NOX::StatusTest::NormUpdate> normupdate =
408  Teuchos::rcp(new NOX::StatusTest::NormUpdate(this->absolute_step_tolerance));
409  Teuchos::RCP<NOX::StatusTest::Combo> combo =
410  Teuchos::rcp(new NOX::StatusTest::Combo(NOX::StatusTest::Combo::OR));
411  combo->addStatusTest(absresid);
412  combo->addStatusTest(relresid);
413  combo->addStatusTest(maxiters);
414  combo->addStatusTest(finiteval);
415  combo->addStatusTest(normupdate);
416 
417  Teuchos::RCP<Teuchos::ParameterList> finalPars = nlParamsPtr;
418 
419  Teuchos::RCP<NOX::Solver::Generic> solver = NOX::Solver::buildSolver(grpPtr, combo, nlParamsPtr);
420  NOX::StatusTest::StatusType status = solver->solve();
421  this->converged = (status == NOX::StatusTest::Converged);
422 
423  const NOX::Epetra::Group & finalGroup = dynamic_cast<const NOX::Epetra::Group &>(solver->getSolutionGroup());
424  const NOX::Epetra::Vector & noxFinalSln = dynamic_cast<const NOX::Epetra::Vector &>(finalGroup.getX());
425 
426  *x_epetra->vec() = noxFinalSln.getEpetraVector();
427  x_in.close();
428 
429  Real residual_norm = finalGroup.getNormF();
430  unsigned int total_iters = solver->getNumIterations();
431  _n_linear_iterations = finalPars->sublist("Direction").sublist("Newton").sublist("Linear Solver").sublist("Output").get("Total Number of Linear Iterations", -1);
432 
433  // do not let Trilinos to deallocate what we allocated
434  pc.release();
435  iReq.release();
436 
437  return std::make_pair(total_iters, residual_norm);
438 }
