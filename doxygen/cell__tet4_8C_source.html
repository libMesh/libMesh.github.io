162  const IOPackage iop,
163  std::vector<dof_id_type> & conn) const
164 {

167  libmesh_assert_not_equal_to (iop, INVALID_IO_PACKAGE);
168

170  switch (iop)
171  {
172  case TECPLOT:
173  {
174  conn.resize(8);
175  conn[0] = this->node(0)+1;
176  conn[1] = this->node(1)+1;
177  conn[2] = this->node(2)+1;
178  conn[3] = this->node(2)+1;
179  conn[4] = this->node(3)+1;
180  conn[5] = this->node(3)+1;
181  conn[6] = this->node(3)+1;
182  conn[7] = this->node(3)+1;
183  return;
184  }
185 
186  case VTK:
187  {
188  conn.resize(4);
189  conn[0] = this->node(0);
190  conn[1] = this->node(1);
191  conn[2] = this->node(2);
192  conn[3] = this->node(3);
193  return;
194  }
195 
196  default:
197  libmesh_error_msg("Unsupported IO package " << iop);
198  }

286 Real Tet4::volume () const
287 {
288  // The volume of a tetrahedron is 1/6 the box product formed
289  // by its base and apex vectors
290  Point a = point(3) - point(0);
291 
292  // b is the vector pointing from 0 to 1
293  Point b = point(1) - point(0);
294 
295  // c is the vector pointing from 0 to 2
296  Point c = point(2) - point(0);
297 
298  return triple_product(a, b, c) / 6.;
299 }

304 std::pair<Real, Real> Tet4::min_and_max_angle() const
305 {
306  Point n[4];
307 
308  // Compute the outward normal vectors on each face
309  n[0] = (this->point(2) - this->point(0)).cross(this->point(1) - this->point(0));
310  n[1] = (this->point(1) - this->point(0)).cross(this->point(3) - this->point(0));
311  n[2] = (this->point(2) - this->point(1)).cross(this->point(3) - this->point(1));
312  n[3] = (this->point(0) - this->point(2)).cross(this->point(3) - this->point(2));
313 
314  Real dihedral_angles[6]; // 01, 02, 03, 12, 13, 23
315 
316  // Compute dihedral angles
317  for (unsigned int k=0,i=0; i<4; ++i)
318  for (unsigned int j=i+1; j<4; ++j,k+=1)
319  dihedral_angles[k] = std::acos(n[i]*n[j] / n[i].norm() / n[j].norm()); // return value is between 0 and PI
320 
321  // Return max/min dihedral angles
322  return std::make_pair(*std::min_element(dihedral_angles, dihedral_angles+6),
323  *std::max_element(dihedral_angles, dihedral_angles+6));
324 
325 }

330 float Tet4::embedding_matrix (const unsigned int i,
331  const unsigned int j,
332  const unsigned int k) const
333 {
334  // Choose an optimal diagonal, if one has not already been selected
335  this->choose_diagonal();
336 
337  // Permuted j and k indices
338  unsigned int
339  jp=j,
340  kp=k;
341 
342  if ((i>3) && (this->_diagonal_selection!=DIAG_02_13))
343  {
344  // Just the enum value cast to an unsigned int...
345  const unsigned ds = static_cast<unsigned int>(this->_diagonal_selection);
346 
347  // Permute j, k:
348  // ds==1 ds==2
349  // 0 -> 1 0 -> 2
350  // 1 -> 2 1 -> 0
351  // 2 -> 0 2 -> 1
352  if (jp != 3)
353  jp = (jp+ds)%3;
354 
355  if (kp != 3)
356  kp = (kp+ds)%3;
357  }
358 
359  // Debugging
360  // libMesh::err << "Selected diagonal " << _diagonal_selection << std::endl;
361  // libMesh::err << "j=" << j << std::endl;
362  // libMesh::err << "k=" << k << std::endl;
363  // libMesh::err << "jp=" << jp << std::endl;
364  // libMesh::err << "kp=" << kp << std::endl;
365 
366  // Call embedding matrx with permuted indices
367  return this->_embedding_matrix[i][jp][kp];
368 }
