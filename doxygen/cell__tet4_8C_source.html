284 Real Tet4::volume () const
285 {
286  // The volume of a tetrahedron is 1/6 the box product formed
287  // by its base and apex vectors
288  Point a = point(3) - point(0);
289 
290  // b is the vector pointing from 0 to 1
291  Point b = point(1) - point(0);
292 
293  // c is the vector pointing from 0 to 2
294  Point c = point(2) - point(0);
295 
296  return triple_product(a, b, c) / 6.;
297 }

302 std::pair<Real, Real> Tet4::min_and_max_angle() const
303 {
304  Point n[4];
305 
306  // Compute the outward normal vectors on each face
307  n[0] = (this->point(2) - this->point(0)).cross(this->point(1) - this->point(0));
308  n[1] = (this->point(1) - this->point(0)).cross(this->point(3) - this->point(0));
309  n[2] = (this->point(2) - this->point(1)).cross(this->point(3) - this->point(1));
310  n[3] = (this->point(0) - this->point(2)).cross(this->point(3) - this->point(2));
311 
312  Real dihedral_angles[6]; // 01, 02, 03, 12, 13, 23
313 
314  // Compute dihedral angles
315  for (unsigned int k=0,i=0; i<4; ++i)
316  for (unsigned int j=i+1; j<4; ++j,k+=1)
317  dihedral_angles[k] = std::acos(n[i]*n[j] / n[i].norm() / n[j].norm()); // return value is between 0 and PI
318 
319  // Return max/min dihedral angles
320  return std::make_pair(*std::min_element(dihedral_angles, dihedral_angles+6),
321  *std::max_element(dihedral_angles, dihedral_angles+6));
322 
323 }

337 bool Tet4::contains_point (const Point & p, Real tol) const
338 {
339  // See the response by Tony Noe on this thread.
340  // http://bbs.dartmouth.edu/~fangq/MATH/download/source/Point_in_Tetrahedron.htm
341  Point
342  col1 = point(1) - point(0),
343  col2 = point(2) - point(0),
344  col3 = point(3) - point(0);
345 
346  Point r;
347 
348  libmesh_try
349  {
350  RealTensorValue(col1(0), col2(0), col3(0),
351  col1(1), col2(1), col3(1),
352  col1(2), col2(2), col3(2)).solve(p - point(0), r);
353  }
354  libmesh_catch (ConvergenceFailure &)
355  {
356  // The Jacobian was singular, therefore the Tet had
357  // zero volume. In this degenerate case, it is not
358  // possible for the Tet to contain any points.
359  return false;
360  }
361 
362  // The point p is inside the tetrahedron if r1 + r2 + r3 < 1 and
363  // 0 <= ri <= 1 for i = 1,2,3.
364  return
365  r(0) > -tol &&
366  r(1) > -tol &&
367  r(2) > -tol &&
368  r(0) + r(1) + r(2) < 1.0 + tol;
369 }

374 float Tet4::embedding_matrix (const unsigned int i,
375  const unsigned int j,
376  const unsigned int k) const
377 {
378  // Choose an optimal diagonal, if one has not already been selected
379  this->choose_diagonal();
380 
381  // Permuted j and k indices
382  unsigned int
383  jp=j,
384  kp=k;
385 
386  if ((i>3) && (this->_diagonal_selection!=DIAG_02_13))
387  {
388  // Just the enum value cast to an unsigned int...
389  const unsigned ds = static_cast<unsigned int>(this->_diagonal_selection);
390 
391  // Permute j, k:
392  // ds==1 ds==2
393  // 0 -> 1 0 -> 2
394  // 1 -> 2 1 -> 0
395  // 2 -> 0 2 -> 1
396  if (jp != 3)
397  jp = (jp+ds)%3;
398 
399  if (kp != 3)
400  kp = (kp+ds)%3;
401  }
402 
403  // Debugging
404  // libMesh::err << "Selected diagonal " << _diagonal_selection << std::endl;
405  // libMesh::err << "j=" << j << std::endl;
406  // libMesh::err << "k=" << k << std::endl;
407  // libMesh::err << "jp=" << jp << std::endl;
408  // libMesh::err << "kp=" << kp << std::endl;
409 
410  // Call embedding matrix with permuted indices
411  return this->_embedding_matrix[i][jp][kp];
412 }
