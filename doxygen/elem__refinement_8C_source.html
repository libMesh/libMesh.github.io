41 void Elem::refine (MeshRefinement & mesh_refinement)
42 {
43  libmesh_assert_equal_to (this->refinement_flag(), Elem::REFINE);
44  libmesh_assert (this->active());
45 
46  // Create my children if necessary
47  if (!_children)
48  {
49  _children = new Elem *[this->n_children()];
50 
51  unsigned int parent_p_level = this->p_level();
52  for (unsigned int c=0; c<this->n_children(); c++)
53  {
54  _children[c] = Elem::build(this->type(), this).release();
55  Elem * current_child = this->child_ptr(c);
56 
57  current_child->set_refinement_flag(Elem::JUST_REFINED);
58  current_child->set_p_level(parent_p_level);
59  current_child->set_p_refinement_flag(this->p_refinement_flag());
60 
61  for (unsigned int nc=0; nc<current_child->n_nodes(); nc++)
62  {
63  Node * node =
64  mesh_refinement.add_node(*this, c, nc,
65  current_child->processor_id());
66  node->set_n_systems (this->n_systems());
67  current_child->set_node(nc) = node;
68  }
69 
70  mesh_refinement.add_elem (current_child);
71  current_child->set_n_systems(this->n_systems());
72  }
73  }
74  else
75  {
76  unsigned int parent_p_level = this->p_level();
77  for (unsigned int c=0; c<this->n_children(); c++)
78  {
79  Elem * current_child = this->child_ptr(c);
80  libmesh_assert(current_child->subactive());
81  current_child->set_refinement_flag(Elem::JUST_REFINED);
82  current_child->set_p_level(parent_p_level);
83  current_child->set_p_refinement_flag(this->p_refinement_flag());
84  }
85  }
86 
87  // Un-set my refinement flag now
88  this->set_refinement_flag(Elem::INACTIVE);
89 
90  // Leave the p refinement flag set - we will need that later to get
91  // projection operations correct
92  // this->set_p_refinement_flag(Elem::INACTIVE);
93 
94  for (unsigned int c=0; c<this->n_children(); c++)
95  {
96  libmesh_assert_equal_to (this->child_ptr(c)->parent(), this);
97  libmesh_assert(this->child_ptr(c)->active());
98  }
99  libmesh_assert (this->ancestor());
100 }

104 void Elem::coarsen()
105 {
106  libmesh_assert_equal_to (this->refinement_flag(), Elem::COARSEN_INACTIVE);
107  libmesh_assert (!this->active());
108 
109  // We no longer delete children until MeshRefinement::contract()
110  // delete [] _children;
111  // _children = libmesh_nullptr;
112 
113  unsigned int parent_p_level = 0;
114 
115  // re-compute hanging node nodal locations
116  for (unsigned int c=0; c<this->n_children(); c++)
117  {
118  Elem * mychild = this->child_ptr(c);
119  if (mychild == remote_elem)
120  continue;
121  for (unsigned int nc=0; nc<mychild->n_nodes(); nc++)
122  {
123  Point new_pos;
124  bool calculated_new_pos = false;
125 
126  for (unsigned int n=0; n<this->n_nodes(); n++)
127  {
128  // The value from the embedding matrix
129  const float em_val = this->embedding_matrix(c,nc,n);
130 
131  // The node location is somewhere between existing vertices
132  if ((em_val != 0.) && (em_val != 1.))
133  {
134  new_pos.add_scaled (this->point(n), em_val);
135  calculated_new_pos = true;
136  }
137  }
138 
139  if(calculated_new_pos)
140  {
141  //Move the existing node back into it's original location
142  for(unsigned int i=0; i<LIBMESH_DIM; i++)
143  {
144  Point & child_node = mychild->point(nc);
145  child_node(i)=new_pos(i);
146  }
147  }
148  }
149  }
150 
151  for (unsigned int c=0; c<this->n_children(); c++)
152  {
153  Elem * mychild = this->child_ptr(c);
154  if (mychild == remote_elem)
155  continue;
156  libmesh_assert_equal_to (mychild->refinement_flag(), Elem::COARSEN);
157  mychild->set_refinement_flag(Elem::INACTIVE);
158  if (mychild->p_level() > parent_p_level)
159  parent_p_level = mychild->p_level();
160  }
161 
162  this->set_refinement_flag(Elem::JUST_COARSENED);
163  this->set_p_level(parent_p_level);
164 
165  libmesh_assert (this->active());
166 }

170 void Elem::contract()
171 {
172  // Subactive elements get deleted entirely, not contracted
173  libmesh_assert (this->active());
174 
175  // Active contracted elements no longer can have children
176  delete [] _children;
177  _children = libmesh_nullptr;
178 
179  if (this->refinement_flag() == Elem::JUST_COARSENED)
180  this->set_refinement_flag(Elem::DO_NOTHING);
