38 // Method for 2D, 3D -- see inf_fe_1D.C for a 1D version of this
39 template <unsigned int Dim, FEFamily T_radial, InfMapType T_base>
40 void InfFE<Dim,T_radial,T_base>::reinit(const Elem* inf_elem,
41  const unsigned int s,
42  const Real tolerance,
43  const std::vector<Point>* const pts,
44  const std::vector<Real>* const weights)
45 {
46  if (weights != NULL)
47  libmesh_not_implemented_msg("ERROR: User-specified weights for infinite elements are not implemented!");
48 
49  if (pts != NULL)
50  libmesh_not_implemented_msg("ERROR: User-specified points for infinite elements are not implemented!");
51 
52  // We don't do this for 1D elements!
53  libmesh_assert_not_equal_to (Dim, 1);
54 
55  libmesh_assert(inf_elem);
56  libmesh_assert(qrule);
57 
58  // Don't do this for the base
59  libmesh_assert_not_equal_to (s, 0);
60 
61  // Build the side of interest
62  const UniquePtr<Elem> side(inf_elem->build_side(s));
63 
64  // set the element type
65  elem_type = inf_elem->type();
66 
67  // eventually initialize radial quadrature rule
68  bool radial_qrule_initialized = false;
69 
70  if (current_fe_type.radial_order != fe_type.radial_order)
71  {
72  current_fe_type.radial_order = fe_type.radial_order;
73  radial_qrule->init(EDGE2, inf_elem->p_level());
74  radial_qrule_initialized = true;
75  }
76 
77  // Initialize the face shape functions
78  if (this->get_type() != inf_elem->type() ||
79  base_fe->shapes_need_reinit() ||
80  radial_qrule_initialized)
81  this->init_face_shape_functions (qrule->get_points(), side.get());
82 
83 
84  // compute the face map
85  this->_fe_map->compute_face_map(this->dim, _total_qrule_weights, side.get());
86 
87  // make a copy of the Jacobian for integration
88  const std::vector<Real> JxW_int(this->_fe_map->get_JxW());
89 
90  // Find where the integration points are located on the
91  // full element.
92  std::vector<Point> qp; this->inverse_map (inf_elem, this->_fe_map->get_xyz(),
93  qp, tolerance);
94 
95  // compute the shape function and derivative values
96  // at the points qp
97  this->reinit (inf_elem, &qp);
98 
99  // copy back old data
100  this->_fe_map->get_JxW() = JxW_int;
101 
102 }

126 template <unsigned int Dim, FEFamily T_radial, InfMapType T_base>
127 void InfFE<Dim,T_radial,T_base>::init_face_shape_functions(const std::vector<Point>&,
128  const Elem* inf_side)
129 {
130  libmesh_assert(inf_side);
131 
132  // Currently, this makes only sense in 3-D!
133  libmesh_assert_equal_to (Dim, 3);
134 
135  // Initialiize the radial shape functions
136  this->init_radial_shape_functions(inf_side);
137 
138  // Initialize the base shape functions
139  this->update_base_elem(inf_side);
140 
141  // Initialize the base quadratur rule
142  base_qrule->init(base_elem->type(), inf_side->p_level());
143 
144  // base_fe still corresponds to the (dim-1)-dimensional base of the InfFE object,
145  // so update the fe_base.
146  {
147  libmesh_assert_equal_to (Dim, 3);
148 
149  UniquePtr<FEBase> ap_fb(FEBase::build(Dim-2, this->fe_type));
150 
151  delete base_fe;
152  base_fe = ap_fb.release();
153  base_fe->attach_quadrature_rule(base_qrule);
154  }
155 
156  // initialize the shape functions on the base
157  base_fe->init_base_shape_functions(base_fe->qrule->get_points(),
158  base_elem);
159 
160  // the number of quadrature points
161  const unsigned int n_radial_qp =
162  cast_int<unsigned int>(som.size());
163  const unsigned int n_base_qp = base_qrule->n_points();
164  const unsigned int n_total_qp = n_radial_qp * n_base_qp;
165 
166  // the quadratur weigths
167  _total_qrule_weights.resize(n_total_qp);
168 
169  // now inite the shapes for boundary work
170  {
171 
172  // The element type and order to use in the base map
173  const Order base_mapping_order ( base_elem->default_order() );
174  const ElemType base_mapping_elem_type ( base_elem->type() );
175 
176  // the number of mapping shape functions
177  // (Lagrange shape functions are used for mapping in the base)
178  const unsigned int n_radial_mapping_sf =
179  cast_int<unsigned int>(radial_map.size());
180  const unsigned int n_base_mapping_shape_functions = Base::n_base_mapping_sf(base_mapping_elem_type,
181  base_mapping_order);
182 
183  const unsigned int n_total_mapping_shape_functions =
184  n_radial_mapping_sf * n_base_mapping_shape_functions;
185 
186 
187  // initialize the node and shape numbering maps
188  {
189  _radial_node_index.resize (n_total_mapping_shape_functions);
190  _base_node_index.resize (n_total_mapping_shape_functions);
191 
192  const ElemType inf_face_elem_type (inf_side->type());
193 
194  // fill the node index map
195  for (unsigned int n=0; n<n_total_mapping_shape_functions; n++)
196  {
197  compute_node_indices (inf_face_elem_type,
198  n,
199  _base_node_index[n],
200  _radial_node_index[n]);
201 
202  libmesh_assert_less (_base_node_index[n], n_base_mapping_shape_functions);
203  libmesh_assert_less (_radial_node_index[n], n_radial_mapping_sf);
204  }
205 
206  }
207 
208  // rezise map data fields
209  {
210  std::vector<std::vector<Real> >& psi_map = this->_fe_map->get_psi();
211  std::vector<std::vector<Real> >& dpsidxi_map = this->_fe_map->get_dpsidxi();
212  std::vector<std::vector<Real> >& d2psidxi2_map = this->_fe_map->get_d2psidxi2();
213  psi_map.resize (n_total_mapping_shape_functions);
214  dpsidxi_map.resize (n_total_mapping_shape_functions);
215  d2psidxi2_map.resize (n_total_mapping_shape_functions);
216 
217  // if (Dim == 3)
218  {
219  std::vector<std::vector<Real> >& dpsideta_map = this->_fe_map->get_dpsideta();
220  std::vector<std::vector<Real> >& d2psidxideta_map = this->_fe_map->get_d2psidxideta();
221  std::vector<std::vector<Real> >& d2psideta2_map = this->_fe_map->get_d2psideta2();
222  dpsideta_map.resize (n_total_mapping_shape_functions);
223  d2psidxideta_map.resize (n_total_mapping_shape_functions);
224  d2psideta2_map.resize (n_total_mapping_shape_functions);
225  }
226 
227  for (unsigned int i=0; i<n_total_mapping_shape_functions; i++)
228  {
229  psi_map[i].resize (n_total_qp);
230  dpsidxi_map[i].resize (n_total_qp);
231  d2psidxi2_map[i].resize (n_total_qp);
232 
233  // if (Dim == 3)
234  {
235  std::vector<std::vector<Real> >& dpsideta_map = this->_fe_map->get_dpsideta();
236  std::vector<std::vector<Real> >& d2psidxideta_map = this->_fe_map->get_d2psidxideta();
237  std::vector<std::vector<Real> >& d2psideta2_map = this->_fe_map->get_d2psideta2();
238  dpsideta_map[i].resize (n_total_qp);
239  d2psidxideta_map[i].resize (n_total_qp);
240  d2psideta2_map[i].resize (n_total_qp);
241  }
242  }
243  }
244 
245 
246  // compute shape maps
247  {
248  const std::vector<std::vector<Real> >& S_map = (base_fe->get_fe_map()).get_phi_map();
249  const std::vector<std::vector<Real> >& Ss_map = (base_fe->get_fe_map()).get_dphidxi_map();
250 
251  std::vector<std::vector<Real> >& psi_map = this->_fe_map->get_psi();
252  std::vector<std::vector<Real> >& dpsidxi_map = this->_fe_map->get_dpsidxi();
253  std::vector<std::vector<Real> >& dpsideta_map = this->_fe_map->get_dpsideta();
254 
255  for (unsigned int rp=0; rp<n_radial_qp; rp++) // over radial qp's
256  for (unsigned int bp=0; bp<n_base_qp; bp++) // over base qp's
257  for (unsigned int ti=0; ti<n_total_mapping_shape_functions; ti++) // over all mapping shapes
258  {
259  // let the index vectors take care of selecting the appropriate base/radial mapping shape
260  const unsigned int bi = _base_node_index [ti];
261  const unsigned int ri = _radial_node_index[ti];
262  psi_map [ti][bp+rp*n_base_qp] = S_map [bi][bp] * radial_map [ri][rp];
263  dpsidxi_map [ti][bp+rp*n_base_qp] = Ss_map[bi][bp] * radial_map [ri][rp];
264  dpsideta_map [ti][bp+rp*n_base_qp] = S_map [bi][bp] * dradialdv_map[ri][rp];
265 
266  // second derivatives are not implemented for infinite elements
267  // d2psidxi2_map [ti][bp+rp*n_base_qp] = 0.;
268  // d2psidxideta_map [ti][bp+rp*n_base_qp] = 0.;
269  // d2psideta2_map [ti][bp+rp*n_base_qp] = 0.;
270  }
271 
272  }
273 
274  }
275 
276  // quadrature rule weights
277  {
278  const std::vector<Real>& radial_qw = radial_qrule->get_weights();
279  const std::vector<Real>& base_qw = base_qrule->get_weights();
280 
281  libmesh_assert_equal_to (radial_qw.size(), n_radial_qp);
282  libmesh_assert_equal_to (base_qw.size(), n_base_qp);
283 
284  for (unsigned int rp=0; rp<n_radial_qp; rp++)
285  for (unsigned int bp=0; bp<n_base_qp; bp++)
286  {
287  _total_qrule_weights[ bp+rp*n_base_qp ] = radial_qw[rp] * base_qw[bp];
288  }
289  }
