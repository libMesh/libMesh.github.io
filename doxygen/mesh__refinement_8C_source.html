102 Node * MeshRefinement::add_node(Elem & parent,
103  unsigned int child,
104  unsigned int node,
105  processor_id_type proc_id)
106 {
107  LOG_SCOPE("add_node()", "MeshRefinement");
108 
109  unsigned int parent_n = parent.as_parent_node(child, node);
110 
111  if (parent_n != libMesh::invalid_uint)
112  return parent.node_ptr(parent_n);
113 
114  const std::vector<std::pair<dof_id_type, dof_id_type> >
115  bracketing_nodes = parent.bracketing_nodes(child, node);
116 
117  // If we're not a parent node, we *must* be bracketed by at least
118  // one pair of parent nodes
119  libmesh_assert(bracketing_nodes.size());
120 
121  const dof_id_type new_node_id =
122  _new_nodes_map.find(bracketing_nodes);
123 
124  // Return the node if it already exists
125  if (new_node_id != DofObject::invalid_id)
126  return _mesh.node_ptr(new_node_id);
127 
128  // Otherwise we need to add a new node, with a default id and the
129  // requested processor_id. Figure out where to add the point:
130 
131  Point p; // defaults to 0,0,0
132 
133  for (unsigned int n=0; n != parent.n_nodes(); ++n)
134  {
135  // The value from the embedding matrix
136  const float em_val = parent.embedding_matrix(child,node,n);
137 
138  if (em_val != 0.)
139  {
140  p.add_scaled (parent.point(n), em_val);
141 
142  // If we'd already found the node we shouldn't be here
143  libmesh_assert_not_equal_to (em_val, 1);
144  }
145  }
146 
147  Node * new_node = _mesh.add_point (p, DofObject::invalid_id, proc_id);
148 
149  libmesh_assert(new_node);
150 
151  // Add the node to the map.
152  _new_nodes_map.add_node(*new_node, bracketing_nodes);
153 
154  // Return the address of the new node
155  return new_node;
156 }

190 void MeshRefinement::create_parent_error_vector(const ErrorVector & error_per_cell,
191  ErrorVector & error_per_parent,
192  Real & parent_error_min,
193  Real & parent_error_max)
194 {
195  // This function must be run on all processors at once
196  parallel_object_only();
197 
198  // Make sure the input error vector is valid
199 #ifdef DEBUG
200  for (std::size_t i=0; i != error_per_cell.size(); ++i)
201  {
202  libmesh_assert_greater_equal (error_per_cell[i], 0);
203  // isnan() isn't standard C++ yet
204 #ifdef isnan
205  libmesh_assert(!isnan(error_per_cell[i]));
206 #endif
207  }
208 
209  // Use a reference to std::vector to avoid confusing
210  // this->comm().verify
211  std::vector<ErrorVectorReal> & epc = error_per_parent;
212  libmesh_assert(this->comm().verify(epc));
213 #endif // #ifdef DEBUG
214 
215  // error values on uncoarsenable elements will be left at -1
216  error_per_parent.clear();
217  error_per_parent.resize(error_per_cell.size(), 0.0);
218 
219  {
220  // Find which elements are uncoarsenable
221  MeshBase::element_iterator elem_it = _mesh.active_local_elements_begin();
222  const MeshBase::element_iterator elem_end = _mesh.active_local_elements_end();
223  for (; elem_it != elem_end; ++elem_it)
224  {
225  Elem * elem = *elem_it;
226  Elem * parent = elem->parent();
227 
228  // Active elements are uncoarsenable
229  error_per_parent[elem->id()] = -1.0;
230 
231  // Grandparents and up are uncoarsenable
232  while (parent)
233  {
234  parent = parent->parent();
235  if (parent)
236  {
237  const dof_id_type parentid = parent->id();
238  libmesh_assert_less (parentid, error_per_parent.size());
239  error_per_parent[parentid] = -1.0;
240  }
241  }
242  }
243 
244  // Sync between processors.
245  // Use a reference to std::vector to avoid confusing
246  // this->comm().min
247  std::vector<ErrorVectorReal> & epp = error_per_parent;
248  this->comm().min(epp);
249  }
250 
251  // The parent's error is defined as the square root of the
252  // sum of the children's errors squared, so errors that are
253  // Hilbert norms remain Hilbert norms.
254  //
255  // Because the children may be on different processors, we
256  // calculate local contributions to the parents' errors squared
257  // first, then sum across processors and take the square roots
258  // second.
259  {
260  MeshBase::element_iterator elem_it = _mesh.active_local_elements_begin();
261  const MeshBase::element_iterator elem_end = _mesh.active_local_elements_end();
262 
263  for (; elem_it != elem_end; ++elem_it)
264  {
265  Elem * elem = *elem_it;
266  Elem * parent = elem->parent();
267 
268  // Calculate each contribution to parent cells
269  if (parent)
270  {
271  const dof_id_type parentid = parent->id();
272  libmesh_assert_less (parentid, error_per_parent.size());
273 
274  // If the parent has grandchildren we won't be able to
275  // coarsen it, so forget it. Otherwise, add this child's
276  // contribution to the sum of the squared child errors
277  if (error_per_parent[parentid] != -1.0)
278  error_per_parent[parentid] += (error_per_cell[elem->id()] *
279  error_per_cell[elem->id()]);
280  }
281  }
282  }
283 
284  // Sum the vector across all processors
285  this->comm().sum(static_cast<std::vector<ErrorVectorReal> &>(error_per_parent));
286 
287  // Calculate the min and max as we loop
288  parent_error_min = std::numeric_limits<double>::max();
289  parent_error_max = 0.;
290 
291  for (std::size_t i = 0; i != error_per_parent.size(); ++i)
292  {
293  // If this element isn't a coarsenable parent with error, we
294  // have nothing to do. Just flag it as -1 and move on
295  // Note that this->comm().sum might have left uncoarsenable
296  // elements with error_per_parent=-n_proc, so reset it to
297  // error_per_parent=-1
298  if (error_per_parent[i] < 0.)
299  {
300  error_per_parent[i] = -1.;
301  continue;
302  }
303 
304  // The error estimator might have already given us an
305  // estimate on the coarsenable parent elements; if so then
306  // we want to retain that estimate
307  if (error_per_cell[i])
308  {
309  error_per_parent[i] = error_per_cell[i];
310  continue;
311  }
312  // if not, then e_parent = sqrt(sum(e_child^2))
313  else
314  error_per_parent[i] = std::sqrt(error_per_parent[i]);
315 
316  parent_error_min = std::min (parent_error_min,
317  error_per_parent[i]);
318  parent_error_max = std::max (parent_error_max,
319  error_per_parent[i]);
320  }
321 }

332 bool MeshRefinement::test_level_one (bool libmesh_dbg_var(libmesh_assert_pass))
333 {
334  // This function must be run on all processors at once
335  parallel_object_only();
336 
337  // We may need a PointLocator for topological_neighbor() tests
338  // later, which we need to make sure gets constructed on all
339  // processors at once.
340  UniquePtr<PointLocatorBase> point_locator;
341 
342 #ifdef LIBMESH_ENABLE_PERIODIC
343  bool has_periodic_boundaries =
344  _periodic_boundaries && !_periodic_boundaries->empty();
345  libmesh_assert(this->comm().verify(has_periodic_boundaries));
346 
347  if (has_periodic_boundaries)
348  point_locator = _mesh.sub_point_locator();
349 #endif
350 
351  MeshBase::element_iterator elem_it = _mesh.active_local_elements_begin();
352  const MeshBase::element_iterator elem_end = _mesh.active_local_elements_end();
353 
354  bool failure = false;
355 
356 #ifndef NDEBUG
357  Elem * failed_elem = libmesh_nullptr;
358  Elem * failed_neighbor = libmesh_nullptr;
359 #endif // !NDEBUG
360 
361  for ( ; elem_it != elem_end && !failure; ++elem_it)
362  {
363  // Pointer to the element
364  Elem * elem = *elem_it;
365 
366  for (unsigned int n=0; n<elem->n_neighbors(); n++)
367  {
368  Elem * neighbor =
369  topological_neighbor(elem, point_locator.get(), n);
370 
371  if (!neighbor || !neighbor->active() ||
372  neighbor == remote_elem)
373  continue;
374 
375  if ((neighbor->level() + 1 < elem->level()) ||
376  (neighbor->p_level() + 1 < elem->p_level()) ||
377  (neighbor->p_level() > elem->p_level() + 1))
378  {
379  failure = true;
380 #ifndef NDEBUG
381  failed_elem = elem;
382  failed_neighbor = neighbor;
383 #endif // !NDEBUG
384  break;
385  }
386  }
387  }
388 
389  // If any processor failed, we failed globally
390  this->comm().max(failure);
391 
392  if (failure)
393  {
394  // We didn't pass the level one test, so libmesh_assert that
395  // we're allowed not to
396 #ifndef NDEBUG
397  if (libmesh_assert_pass)
398  {
399  libMesh::out << "MeshRefinement Level one failure, element: "
400  << *failed_elem
401  << std::endl;
402  libMesh::out << "MeshRefinement Level one failure, neighbor: "
403  << *failed_neighbor
404  << std::endl;
405  }
406 #endif // !NDEBUG
407  libmesh_assert(!libmesh_assert_pass);
408  return false;
409  }
410  return true;
411 }

471 bool MeshRefinement::refine_and_coarsen_elements ()
472 {
473  // This function must be run on all processors at once
474  parallel_object_only();
475 
476  // We can't yet turn a non-level-one mesh into a level-one mesh
477  if (_face_level_mismatch_limit)
478  libmesh_assert(test_level_one(true));
479 
480  // Possibly clean up the refinement flags from
481  // a previous step
482  MeshBase::element_iterator elem_it = _mesh.elements_begin();
483  const MeshBase::element_iterator elem_end = _mesh.elements_end();
484 
485  for ( ; elem_it != elem_end; ++elem_it)
486  {
487  // Pointer to the element
488  Elem * elem = *elem_it;
489 
490  // Set refinement flag to INACTIVE if the
491  // element isn't active
492  if ( !elem->active())
493  {
494  elem->set_refinement_flag(Elem::INACTIVE);
495  elem->set_p_refinement_flag(Elem::INACTIVE);
496  }
497 
498  // This might be left over from the last step
499  if (elem->refinement_flag() == Elem::JUST_REFINED)
500  elem->set_refinement_flag(Elem::DO_NOTHING);
501  }
502 
503  // Parallel consistency has to come first, or coarsening
504  // along processor boundaries might occasionally be falsely
505  // prevented
506  bool flags_were_consistent = this->make_flags_parallel_consistent();
507 
508  // In theory, we should be able to remove the above call, which can
509  // be expensive and should be unnecessary. In practice, doing
510  // consistent flagging in parallel is hard, it's impossible to
511  // verify at the library level if it's being done by user code, and
512  // we don't want to abort large parallel runs in opt mode... but we
513  // do want to warn that they should be fixed.
514  if (!flags_were_consistent)
515  {
516  libMesh::out << "Refinement flags were not consistent between processors!\n"
517  << "Correcting and continuing.";
518  }
519 
520  // Smooth refinement and coarsening flags
521  _smooth_flags(true, true);
522 
523  // First coarsen the flagged elements.
524  const bool coarsening_changed_mesh =
525  this->_coarsen_elements ();
526 
527  // First coarsen the flagged elements.
528  // FIXME: test_level_one now tests consistency across periodic
529  // boundaries, which requires a point_locator, which just got
530  // invalidated by _coarsen_elements() and hasn't yet been cleared by
531  // prepare_for_use().
532 
533  // libmesh_assert(this->make_coarsening_compatible());
534  // libmesh_assert(this->make_refinement_compatible());
535 
536  // FIXME: This won't pass unless we add a redundant find_neighbors()
537  // call or replace find_neighbors() with on-the-fly neighbor updating
538  // libmesh_assert(!this->eliminate_unrefined_patches());
539 
540  // We can't contract the mesh ourselves anymore - a System might
541  // need to restrict old coefficient vectors first
542  // _mesh.contract();
543 
544  // First coarsen the flagged elements.
545  // Now refine the flagged elements. This will
546  // take up some space, maybe more than what was freed.
547  const bool refining_changed_mesh =
548  this->_refine_elements();
549 
550  // First coarsen the flagged elements.
551  // Finally, the new mesh needs to be prepared for use
552  if (coarsening_changed_mesh || refining_changed_mesh)
553  {
554 #ifdef DEBUG
555  _mesh.libmesh_assert_valid_parallel_ids();
556 #endif
557 
558  _mesh.prepare_for_use (/*skip_renumber =*/false);
559 
560  if (_face_level_mismatch_limit)
561  libmesh_assert(test_level_one(true));
562  libmesh_assert(test_unflagged(true));
563  libmesh_assert(this->make_coarsening_compatible());
564  libmesh_assert(this->make_refinement_compatible());
565  // FIXME: This won't pass unless we add a redundant find_neighbors()
566  // call or replace find_neighbors() with on-the-fly neighbor updating
567  // libmesh_assert(!this->eliminate_unrefined_patches());
568 
569  return true;
570  }
571  else
572  {
573  if (_face_level_mismatch_limit)
574  libmesh_assert(test_level_one(true));
575  libmesh_assert(test_unflagged(true));
576  libmesh_assert(this->make_coarsening_compatible());
577  libmesh_assert(this->make_refinement_compatible());
578  }
579 
580  // Otherwise there was no change in the mesh,
581  // let the user know. Also, there is no need
582  // to prepare the mesh for use since it did not change.
583  return false;
584

708  // along processor boundaries might occasionally be falsely
709  // prevented
710  bool flags_were_consistent = this->make_flags_parallel_consistent();
711 
712  // In theory, we should be able to remove the above call, which can
713  // be expensive and should be unnecessary. In practice, doing
714  // consistent flagging in parallel is hard, it's impossible to
715  // verify at the library level if it's being done by user code, and
716  // we don't want to abort large parallel runs in opt mode... but we
717  // do want to warn that they should be fixed.
718  libmesh_assert(flags_were_consistent);
719  if (!flags_were_consistent)
720  {
721  libMesh::out << "Refinement flags were not consistent between processors!\n"
722  << "Correcting and continuing.";
723  }
724 
725  // Smooth refinement flags
726  _smooth_flags(true, false);
727 
728  // Now refine the flagged elements. This will
729  // take up some space, maybe more than what was freed.
730  const bool mesh_changed =
731  this->_refine_elements();
732 
733  if (_face_level_mismatch_limit)
734  libmesh_assert(test_level_one(true));
735  libmesh_assert(this->make_refinement_compatible());
736  // FIXME: This won't pass unless we add a redundant find_neighbors()

738  // libmesh_assert(!this->eliminate_unrefined_patches());
739 
740  // Finally, the new mesh needs to be prepared for use
741  if (mesh_changed)

743 
744  return mesh_changed;

849 {
850  // This function must be run on all processors at once
851  parallel_object_only();
852 
853  // We may need a PointLocator for topological_neighbor() tests
854  // later, which we need to make sure gets constructed on all
855  // processors at once.
856  UniquePtr<PointLocatorBase> point_locator;
857 
858 #ifdef LIBMESH_ENABLE_PERIODIC
859  bool has_periodic_boundaries =
860  _periodic_boundaries && !_periodic_boundaries->empty();
861  libmesh_assert(this->comm().verify(has_periodic_boundaries));
862 
863  if (has_periodic_boundaries)
864  point_locator = _mesh.sub_point_locator();
865 #endif
866 
867  LOG_SCOPE ("make_coarsening_compatible()", "MeshRefinement");
868 
869  // Unless we encounter a specific situation level-one
870  // will be satisfied after executing this loop just once
871  bool level_one_satisfied = true;
872 
873 
874  // Unless we encounter a specific situation we will be compatible
875  // with any selected refinement flags
876  bool compatible_with_refinement = true;
877 
878 
879  // find the maximum h and p levels in the mesh
880  unsigned int max_level = 0;
881  unsigned int max_p_level = 0;
882 
883  {
884  // First we look at all the active level-0 elements. Since it doesn't make
885  // sense to coarsen them we must un-set their coarsen flags if
886  // they are set.
887  MeshBase::element_iterator el = _mesh.active_elements_begin();
888  const MeshBase::element_iterator end_el = _mesh.active_elements_end();
889 
890  for (; el != end_el; ++el)
891  {
892  Elem * elem = *el;
893  max_level = std::max(max_level, elem->level());
894  max_p_level =
895  std::max(max_p_level,
896  static_cast<unsigned int>(elem->p_level()));
897 
898  if ((elem->level() == 0) &&
899  (elem->refinement_flag() == Elem::COARSEN))
900  elem->set_refinement_flag(Elem::DO_NOTHING);
901 
902  if ((elem->p_level() == 0) &&
903  (elem->p_refinement_flag() == Elem::COARSEN))
904  elem->set_p_refinement_flag(Elem::DO_NOTHING);
905  }
906  }
907 
908  // if there are no refined elements on this processor then
909  // there is no work for us to do
910  if (max_level == 0 && max_p_level == 0)
911  {
912  // But we still have to check with other processors
913  this->comm().min(compatible_with_refinement);
914 
915  return compatible_with_refinement;
916  }
917 
918 
919 
920  // Loop over all the active elements. If an element is marked
921  // for coarsening we better check its neighbors. If ANY of these neighbors
922  // are marked for refinement AND are at the same level then there is a
923  // conflict. By convention refinement wins, so we un-mark the element for
924  // coarsening. Level-one would be violated in this case so we need to re-run
925  // the loop.
926  if (_face_level_mismatch_limit)
927  {
928 
929  repeat:
930  level_one_satisfied = true;
931 
932  do
933  {
934  level_one_satisfied = true;
935 
936  MeshBase::element_iterator el = _mesh.active_elements_begin();
937  const MeshBase::element_iterator end_el = _mesh.active_elements_end();
938 
939  for (; el != end_el; ++el)
940  {
941  Elem * elem = *el;
942  bool my_flag_changed = false;
943 
944  if (elem->refinement_flag() == Elem::COARSEN) // If the element is active and
945  // the coarsen flag is set
946  {
947  const unsigned int my_level = elem->level();
948 
949  for (unsigned int n=0; n<elem->n_neighbors(); n++)
950  {
951  const Elem * neighbor =
952  topological_neighbor(elem, point_locator.get(), n);
953 
954  if (neighbor != libmesh_nullptr && // I have a
955  neighbor != remote_elem) // neighbor here
956  {
957  if (neighbor->active()) // and it is active
958  {
959  if ((neighbor->level() == my_level) &&
960  (neighbor->refinement_flag() == Elem::REFINE)) // the neighbor is at my level
961  // and wants to be refined
962  {
963  elem->set_refinement_flag(Elem::DO_NOTHING);
964  my_flag_changed = true;
965  break;
966  }
967  }
968  else // I have a neighbor and it is not active. That means it has children.
969  { // While it _may_ be possible to coarsen us if all the children of
970  // that element want to be coarsened, it is impossible to know at this
971  // stage. Forget about it for the moment... This can be handled in
972  // two steps.
973  elem->set_refinement_flag(Elem::DO_NOTHING);
974  my_flag_changed = true;
975  break;
976  }
977  }
978  }
979  }
980  if (elem->p_refinement_flag() == Elem::COARSEN) // If
981  // the element is active and the order reduction flag is set
982  {
983  const unsigned int my_p_level = elem->p_level();
984 
985  for (unsigned int n=0; n<elem->n_neighbors(); n++)
986  {
987  const Elem * neighbor =
988  topological_neighbor(elem, point_locator.get(), n);
989 
990  if (neighbor != libmesh_nullptr && // I have a
991  neighbor != remote_elem) // neighbor here
992  {
993  if (neighbor->active()) // and it is active
994  {
995  if ((neighbor->p_level() > my_p_level &&
996  neighbor->p_refinement_flag() != Elem::COARSEN)
997  || (neighbor->p_level() == my_p_level &&
998  neighbor->p_refinement_flag() == Elem::REFINE))
999  {
1000  elem->set_p_refinement_flag(Elem::DO_NOTHING);
1001  my_flag_changed = true;
1002  break;
1003  }
1004  }
1005  else // I have a neighbor and it is not active.
1006  { // We need to find which of its children
1007  // have me as a neighbor, and maintain
1008  // level one p compatibility with them.
1009  // Because we currently have level one h
1010  // compatibility, we don't need to check
1011  // grandchildren
1012 
1013  libmesh_assert(neighbor->has_children());
1014  for (unsigned int c=0; c!=neighbor->n_children(); c++)
1015  {
1016  const Elem * subneighbor = neighbor->child_ptr(c);
1017  if (subneighbor != remote_elem &&
1018  subneighbor->active() &&
1019  has_topological_neighbor(subneighbor, point_locator.get(), elem))
1020  if ((subneighbor->p_level() > my_p_level &&
1021  subneighbor->p_refinement_flag() != Elem::COARSEN)
1022  || (subneighbor->p_level() == my_p_level &&
1023  subneighbor->p_refinement_flag() == Elem::REFINE))
1024  {
1025  elem->set_p_refinement_flag(Elem::DO_NOTHING);
1026  my_flag_changed = true;
1027  break;
1028  }
1029  }
1030  if (my_flag_changed)
1031  break;
1032  }
1033  }
1034  }
1035  }
1036 
1037  // If the current element's flag changed, we hadn't
1038  // satisfied the level one rule.
1039  if (my_flag_changed)
1040  level_one_satisfied = false;
1041 
1042  // Additionally, if it has non-local neighbors, and
1043  // we're not in serial, then we'll eventually have to
1044  // return compatible_with_refinement = false, because
1045  // our change has to propagate to neighboring
1046  // processors.
1047  if (my_flag_changed && !_mesh.is_serial())
1048  for (unsigned int n=0; n != elem->n_neighbors(); ++n)
1049  {
1050  Elem * neigh =
1051  topological_neighbor(elem, point_locator.get(), n);
1052 
1053  if (!neigh)
1054  continue;
1055  if (neigh == remote_elem ||
1056  neigh->processor_id() !=
1057  this->processor_id())
1058  {
1059  compatible_with_refinement = false;
1060  break;
1061  }
1062  // FIXME - for non-level one meshes we should
1063  // test all descendants
1064  if (neigh->has_children())
1065  for (unsigned int c=0; c != neigh->n_children(); ++c)
1066  if (neigh->child_ptr(c) == remote_elem ||
1067  neigh->child_ptr(c)->processor_id() !=
1068  this->processor_id())
1069  {
1070  compatible_with_refinement = false;
1071  break;
1072  }
1073  }
1074  }
1075  }
1076  while (!level_one_satisfied);
1077 
1078  } // end if (_face_level_mismatch_limit)
1079 
1080 
1081  // Next we look at all of the ancestor cells.
1082  // If there is a parent cell with all of its children
1083  // wanting to be unrefined then the element is a candidate
1084  // for unrefinement. If all the children don't
1085  // all want to be unrefined then ALL of them need to have their
1086  // unrefinement flags cleared.
1087  for (int level=(max_level); level >= 0; level--)
1088  {
1089  MeshBase::element_iterator el = _mesh.level_elements_begin(level);
1090  const MeshBase::element_iterator end_el = _mesh.level_elements_end(level);
1091  for (; el != end_el; ++el)
1092  {
1093  Elem * elem = *el;
1094  if (elem->ancestor())
1095  {
1096 
1097  // right now the element hasn't been disqualified
1098  // as a candidate for unrefinement
1099  bool is_a_candidate = true;
1100  bool found_remote_child = false;
1101 
1102  for (unsigned int c=0; c<elem->n_children(); c++)
1103  {
1104  Elem * child = elem->child_ptr(c);
1105  if (child == remote_elem)
1106  found_remote_child = true;
1107  else if ((child->refinement_flag() != Elem::COARSEN) ||
1108  !child->active() )
1109  is_a_candidate = false;
1110  }
1111 
1112  if (!is_a_candidate && !found_remote_child)
1113  {
1114  elem->set_refinement_flag(Elem::INACTIVE);
1115 
1116  for (unsigned int c=0; c<elem->n_children(); c++)
1117  {
1118  Elem * child = elem->child_ptr(c);
1119  if (child == remote_elem)
1120  continue;
1121  if (child->refinement_flag() == Elem::COARSEN)
1122  {
1123  level_one_satisfied = false;
1124  child->set_refinement_flag(Elem::DO_NOTHING);
1125  }
1126  }
1127  }
1128  }
1129  }
1130  }
1131 
1132  if (!level_one_satisfied && _face_level_mismatch_limit) goto repeat;
1133 
1134 
1135  // If all the children of a parent are set to be coarsened
1136  // then flag the parent so that they can kill their kids...
1137  MeshBase::element_iterator all_el = _mesh.elements_begin();
1138  const MeshBase::element_iterator all_el_end = _mesh.elements_end();
1139  for (; all_el != all_el_end; ++all_el)
1140  {
1141  Elem * elem = *all_el;
1142  if (elem->ancestor())
1143  {
1144 
1145  // Presume all the children are local and flagged for
1146  // coarsening and then look for a contradiction
1147  bool all_children_flagged_for_coarsening = true;
1148  bool found_remote_child = false;
1149 
1150  for (unsigned int c=0; c<elem->n_children(); c++)
1151  {
1152  Elem * child = elem->child_ptr(c);
1153  if (child == remote_elem)
1154  found_remote_child = true;
1155  else if (child->refinement_flag() != Elem::COARSEN)
1156  all_children_flagged_for_coarsening = false;
1157  }
1158 
1159  if (!found_remote_child &&
1160  all_children_flagged_for_coarsening)
1161  elem->set_refinement_flag(Elem::COARSEN_INACTIVE);
1162  else if (!found_remote_child)
1163  elem->set_refinement_flag(Elem::INACTIVE);
1164  }
1165  }
1166 
1167  // If one processor finds an incompatibility, we're globally
1168  // incompatible
1169  this->comm().min(compatible_with_refinement);
1170 
1171  return compatible_with_refinement;
1172 }
