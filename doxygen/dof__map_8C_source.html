54 AutoPtr<SparsityPattern::Build> DofMap::build_sparsity
55 (const MeshBase& mesh) const
56 {
57  libmesh_assert (mesh.is_prepared());
58 
59  START_LOG("build_sparsity()", "DofMap");
60 
61  // Compute the sparsity structure of the global matrix. This can be
62  // fed into a PetscMatrix to allocate exacly the number of nonzeros
63  // necessary to store the matrix. This algorithm should be linear
64  // in the (# of elements)*(# nodes per element)
65 
66  // We can be more efficient in the threaded sparsity pattern assembly
67  // if we don't need the exact pattern. For some sparse matrix formats
68  // a good upper bound will suffice.
69 
70  // See if we need to include sparsity pattern entries for coupling
71  // between neighbor dofs
72  bool implicit_neighbor_dofs = this->use_coupled_neighbor_dofs(mesh);
73 
74  // We can compute the sparsity pattern in parallel on multiple
75  // threads. The goal is for each thread to compute the full sparsity
76  // pattern for a subset of elements. These sparsity patterns can
77  // be efficiently merged in the SparsityPattern::Build::join()
78  // method, especially if there is not too much overlap between them.
79  // Even better, if the full sparsity pattern is not needed then
80  // the number of nonzeros per row can be estimated from the
81  // sparsity patterns created on each thread.
82  AutoPtr<SparsityPattern::Build> sp
83  (new SparsityPattern::Build (mesh,
84  *this,
85  this->_dof_coupling,
86  implicit_neighbor_dofs,
87  need_full_sparsity_pattern));
88 
89  Threads::parallel_reduce (ConstElemRange (mesh.active_local_elements_begin(),
90  mesh.active_local_elements_end()), *sp);
91 
92  sp->parallel_sync();
93 
94 #ifndef NDEBUG
95  // Avoid declaring these variables unless asserts are enabled.
96  const processor_id_type proc_id = mesh.processor_id();
97  const dof_id_type n_dofs_on_proc = this->n_dofs_on_processor(proc_id);
98 #endif
99  libmesh_assert_equal_to (sp->sparsity_pattern.size(), n_dofs_on_proc);
100 
101  STOP_LOG("build_sparsity()", "DofMap");
102 
103  // Check to see if we have any extra stuff to add to the sparsity_pattern
104  if (_extra_sparsity_function)
105  {
106  if (_augment_sparsity_pattern)
107  {
108  libmesh_here();
109  libMesh::out << "WARNING: You have specified both an extra sparsity function and object.\n"
110  << " Are you sure this is what you meant to do??"
111  << std::endl;
112  }
113 
114  _extra_sparsity_function
115  (sp->sparsity_pattern, sp->n_nz,
116  sp->n_oz, _extra_sparsity_context);
117  }
118 
119  if (_augment_sparsity_pattern)
120  _augment_sparsity_pattern->augment_sparsity_pattern
121  (sp->sparsity_pattern, sp->n_nz, sp->n_oz);
122 
123  return sp;
124 }

227 void DofMap::attach_matrix (SparseMatrix<Number>& matrix)
228 {
229  parallel_object_only();
230 
231  // We shouldn't be trying to re-attach the same matrices repeatedly
232  libmesh_assert (std::find(_matrices.begin(), _matrices.end(),
233  &matrix) == _matrices.end());
234 
235  _matrices.push_back(&matrix);
236 
237  matrix.attach_dof_map (*this);
238 
239  // If we've already computed sparsity, then it's too late
240  // to wait for "compute_sparsity" to help with sparse matrix
241  // initialization, and we need to handle this matrix individually
242  bool computed_sparsity_already =
243  ((_n_nz && !_n_nz->empty()) ||
244  (_n_oz && !_n_oz->empty()));
245  this->comm().max(computed_sparsity_already);
246  if (computed_sparsity_already &&
247  matrix.need_full_sparsity_pattern())
248  {
249  // We'd better have already computed the full sparsity pattern
250  // if we need it here
251  libmesh_assert(need_full_sparsity_pattern);
252  libmesh_assert(_sp.get());
253 
254  matrix.update_sparsity_pattern (_sp->sparsity_pattern);
255  }
256 
257  if (matrix.need_full_sparsity_pattern())
258  need_full_sparsity_pattern = true;
259 }

285 template <typename iterator_type>
286 void DofMap::set_nonlocal_dof_objects(iterator_type objects_begin,
287  iterator_type objects_end,
288  MeshBase &mesh,
289  dofobject_accessor objects)
290 {
291  // This function must be run on all processors at once
292  parallel_object_only();
293 
294  // First, iterate over local objects to find out how many
295  // are on each processor
296  std::vector<dof_id_type>
297  ghost_objects_from_proc(this->n_processors(), 0);
298 
299  iterator_type it = objects_begin;
300 
301  for (; it != objects_end; ++it)
302  {
303  DofObject *obj = *it;
304 
305  if (obj)
306  {
307  processor_id_type obj_procid = obj->processor_id();
308  // We'd better be completely partitioned by now
309  libmesh_assert_not_equal_to (obj_procid, DofObject::invalid_processor_id);
310  ghost_objects_from_proc[obj_procid]++;
311  }
312  }
313 
314  std::vector<dof_id_type> objects_on_proc(this->n_processors(), 0);
315  this->comm().allgather(ghost_objects_from_proc[this->processor_id()],
316  objects_on_proc);
317 
318 #ifdef DEBUG
319  for (processor_id_type p=0; p != this->n_processors(); ++p)
320  libmesh_assert_less_equal (ghost_objects_from_proc[p], objects_on_proc[p]);
321 #endif
322 
323  // Request sets to send to each processor
324  std::vector<std::vector<dof_id_type> >
325  requested_ids(this->n_processors());
326 
327  // We know how many of our objects live on each processor, so
328  // reserve() space for requests from each.
329  for (processor_id_type p=0; p != this->n_processors(); ++p)
330  if (p != this->processor_id())
331  requested_ids[p].reserve(ghost_objects_from_proc[p]);
332 
333  for (it = objects_begin; it != objects_end; ++it)
334  {
335  DofObject *obj = *it;
336  if (obj->processor_id() != DofObject::invalid_processor_id)
337  requested_ids[obj->processor_id()].push_back(obj->id());
338  }
339 #ifdef DEBUG
340  for (processor_id_type p=0; p != this->n_processors(); ++p)
341  libmesh_assert_equal_to (requested_ids[p].size(), ghost_objects_from_proc[p]);
342 #endif
343 
344  // Next set ghost object n_comps from other processors
345  for (processor_id_type p=1; p != this->n_processors(); ++p)
346  {
347  // Trade my requests with processor procup and procdown
348  processor_id_type procup =
349  cast_int<processor_id_type>((this->processor_id() + p) %
350  this->n_processors());
351  processor_id_type procdown =
352  cast_int<processor_id_type>((this->n_processors() +
353  this->processor_id() - p) %
354  this->n_processors());
355  std::vector<dof_id_type> request_to_fill;
356  this->comm().send_receive(procup, requested_ids[procup],
357  procdown, request_to_fill);
358 
359  // Fill those requests
360  const unsigned int
361  sys_num = this->sys_number(),
362  n_var_groups = this->n_variable_groups();
363 
364  std::vector<dof_id_type> ghost_data
365  (request_to_fill.size() * 2 * n_var_groups);
366 
367  for (std::size_t i=0; i != request_to_fill.size(); ++i)
368  {
369  DofObject *requested = (this->*objects)(mesh, request_to_fill[i]);
370  libmesh_assert(requested);
371  libmesh_assert_equal_to (requested->processor_id(), this->processor_id());
372  libmesh_assert_equal_to (requested->n_var_groups(sys_num), n_var_groups);
373  for (unsigned int vg=0; vg != n_var_groups; ++vg)
374  {
375  unsigned int n_comp_g =
376  requested->n_comp_group(sys_num, vg);
377  ghost_data[i*2*n_var_groups+vg] = n_comp_g;
378  dof_id_type my_first_dof = n_comp_g ?
379  requested->vg_dof_base(sys_num, vg) : 0;
380  libmesh_assert_not_equal_to (my_first_dof, DofObject::invalid_id);
381  ghost_data[i*2*n_var_groups+n_var_groups+vg] = my_first_dof;
382  }
383  }
384 
385  // Trade back the results
386  std::vector<dof_id_type> filled_request;
387  this->comm().send_receive(procdown, ghost_data,
388  procup, filled_request);
389 
390  // And copy the id changes we've now been informed of
391  libmesh_assert_equal_to (filled_request.size(),
392  requested_ids[procup].size() * 2 * n_var_groups);
393  for (std::size_t i=0; i != requested_ids[procup].size(); ++i)
394  {
395  DofObject *requested = (this->*objects)(mesh, requested_ids[procup][i]);
396  libmesh_assert(requested);
397  libmesh_assert_equal_to (requested->processor_id(), procup);
398  for (unsigned int vg=0; vg != n_var_groups; ++vg)
399  {
400  unsigned int n_comp_g =
401  cast_int<unsigned int>(filled_request[i*2*n_var_groups+vg]);
402  requested->set_n_comp_group(sys_num, vg, n_comp_g);
403  if (n_comp_g)
404  {
405  dof_id_type my_first_dof =
406  filled_request[i*2*n_var_groups+n_var_groups+vg];
407  libmesh_assert_not_equal_to (my_first_dof, DofObject::invalid_id);
408  requested->set_vg_dof_base
409  (sys_num, vg, my_first_dof);
410  }
411  }
412  }
413  }
414 
415 #ifdef DEBUG
416  // Double check for invalid dofs
417  for (it = objects_begin; it != objects_end; ++it)
418  {
419  DofObject *obj = *it;
420  libmesh_assert (obj);
421  unsigned int num_variables = obj->n_vars(this->sys_number());
422  for (unsigned int v=0; v != num_variables; ++v)
423  {
424  unsigned int n_comp =
425  obj->n_comp(this->sys_number(), v);
426  dof_id_type my_first_dof = n_comp ?
427  obj->dof_number(this->sys_number(), v, 0) : 0;
428  libmesh_assert_not_equal_to (my_first_dof, DofObject::invalid_id);
429  }
430  }
431 #endif
432 }

436 void DofMap::reinit(MeshBase& mesh)
437 {
438  libmesh_assert (mesh.is_prepared());
439 
440  START_LOG("reinit()", "DofMap");
441 
442  const unsigned int
443  sys_num = this->sys_number(),
444  n_var_groups = this->n_variable_groups();
445 
446  // The DofObjects need to know how many variable groups we have, and
447  // how many variables there are in each group.
448  std::vector<unsigned int> n_vars_per_group; n_vars_per_group.reserve (n_var_groups);
449 
450  for (unsigned int vg=0; vg<n_var_groups; vg++)
451  n_vars_per_group.push_back (this->variable_group(vg).n_variables());
452 
453 #ifdef LIBMESH_ENABLE_AMR
454 
455  //------------------------------------------------------------
456  // Clear the old_dof_objects for all the nodes
457  // and elements so that we can overwrite them
458  {
459  MeshBase::node_iterator node_it = mesh.nodes_begin();
460  const MeshBase::node_iterator node_end = mesh.nodes_end();
461 
462  for ( ; node_it != node_end; ++node_it)
463  {
464  (*node_it)->clear_old_dof_object();
465  libmesh_assert (!(*node_it)->old_dof_object);
466  }
467 
468  MeshBase::element_iterator elem_it = mesh.elements_begin();
469  const MeshBase::element_iterator elem_end = mesh.elements_end();
470 
471  for ( ; elem_it != elem_end; ++elem_it)
472  {
473  (*elem_it)->clear_old_dof_object();
474  libmesh_assert (!(*elem_it)->old_dof_object);
475  }
476  }
477 
478 
479  //------------------------------------------------------------
480  // Set the old_dof_objects for the elements that
481  // weren't just created, if these old dof objects
482  // had variables
483  {
484  MeshBase::element_iterator elem_it = mesh.elements_begin();
485  const MeshBase::element_iterator elem_end = mesh.elements_end();
486 
487  for ( ; elem_it != elem_end; ++elem_it)
488  {
489  Elem* elem = *elem_it;
490 
491  // Skip the elements that were just refined
492  if (elem->refinement_flag() == Elem::JUST_REFINED) continue;
493 
494  for (unsigned int n=0; n<elem->n_nodes(); n++)
495  {
496  Node* node = elem->get_node(n);
497 
498  if (node->old_dof_object == NULL)
499  if (node->has_dofs(sys_num))
500  node->set_old_dof_object();
501  }
502 
503  libmesh_assert (!elem->old_dof_object);
504 
505  if (elem->has_dofs(sys_num))
506  elem->set_old_dof_object();
507  }
508  }
509 
510 #endif // #ifdef LIBMESH_ENABLE_AMR
511 
512 
513  //------------------------------------------------------------
514  // Then set the number of variables for each \p DofObject
515  // equal to n_variables() for this system. This will
516  // handle new \p DofObjects that may have just been created
517  {
518  // All the nodes
519  MeshBase::node_iterator node_it = mesh.nodes_begin();
520  const MeshBase::node_iterator node_end = mesh.nodes_end();
521 
522  for ( ; node_it != node_end; ++node_it)
523  (*node_it)->set_n_vars_per_group(sys_num, n_vars_per_group);
524 
525  // All the elements
526  MeshBase::element_iterator elem_it = mesh.elements_begin();
527  const MeshBase::element_iterator elem_end = mesh.elements_end();
528 
529  for ( ; elem_it != elem_end; ++elem_it)
530  (*elem_it)->set_n_vars_per_group(sys_num, n_vars_per_group);
531  }
532 
533 
534  // Zero _n_SCALAR_dofs, it will be updated below.
535  this->_n_SCALAR_dofs = 0;
536 
537  //------------------------------------------------------------
538  // Next allocate space for the DOF indices
539  for (unsigned int vg=0; vg<n_var_groups; vg++)
540  {
541  const VariableGroup &vg_description = this->variable_group(vg);
542 
543  const unsigned int n_var_in_group = vg_description.n_variables();
544  const FEType& base_fe_type = vg_description.type();
545 
546  // Don't need to loop over elements for a SCALAR variable
547  // Just increment _n_SCALAR_dofs
548  if(base_fe_type.family == SCALAR)
549  {
550  this->_n_SCALAR_dofs += base_fe_type.order*n_var_in_group;
551  continue;
552  }
553 
554  // This should be constant even on p-refined elements
555  const bool extra_hanging_dofs =
556  FEInterface::extra_hanging_dofs(base_fe_type);
557 
558  // For all the active elements
559  MeshBase::element_iterator elem_it = mesh.active_elements_begin();
560  const MeshBase::element_iterator elem_end = mesh.active_elements_end();
561 
562  // Count vertex degrees of freedom first
563  for ( ; elem_it != elem_end; ++elem_it)
564  {
565  Elem* elem = *elem_it;
566  libmesh_assert(elem);
567 
568  // Skip the numbering if this variable is
569  // not active on this element's subdomain
570  if (!vg_description.active_on_subdomain(elem->subdomain_id()))
571  continue;
572 
573  const ElemType type = elem->type();
574  const unsigned int dim = elem->dim();
575 
576  FEType fe_type = base_fe_type;
577 
578 #ifdef LIBMESH_ENABLE_AMR
579  // Make sure we haven't done more p refinement than we can
580  // handle
581  if (elem->p_level() + base_fe_type.order >
582  FEInterface::max_order(base_fe_type, type))
583  {
584 # ifdef DEBUG
585  if (FEInterface::max_order(base_fe_type,type) < static_cast<unsigned int>(base_fe_type.order))
586  libmesh_error_msg("ERROR: Finite element " \
587  << Utility::enum_to_string(base_fe_type.family) \
588  << " on geometric element " \
589  << Utility::enum_to_string(type) \
590  << "\nonly supports FEInterface::max_order = " \
591  << FEInterface::max_order(base_fe_type,type) \
592  << ", not fe_type.order = " \
593  << base_fe_type.order);
594 
595  libMesh::err
596  << "WARNING: Finite element "
597  << Utility::enum_to_string(base_fe_type.family)
598  << " on geometric element "
599  << Utility::enum_to_string(type) << std::endl
600  << "could not be p refined past FEInterface::max_order = "
601  << FEInterface::max_order(base_fe_type,type)
602  << std::endl;
603 # endif
604  elem->set_p_level(FEInterface::max_order(base_fe_type,type)
605  - base_fe_type.order);
606  }
607 #endif
608 
609  fe_type.order = static_cast<Order>(fe_type.order +
610  elem->p_level());
611 
612  // Allocate the vertex DOFs
613  for (unsigned int n=0; n<elem->n_nodes(); n++)
614  {
615  Node* node = elem->get_node(n);
616 
617  if (elem->is_vertex(n))
618  {
619  const unsigned int old_node_dofs =
620  node->n_comp_group(sys_num, vg);
621 
622  const unsigned int vertex_dofs =
623  std::max(FEInterface::n_dofs_at_node(dim, fe_type,
624  type, n),
625  old_node_dofs);
626 
627  // Some discontinuous FEs have no vertex dofs
628  if (vertex_dofs > old_node_dofs)
629  {
630  node->set_n_comp_group(sys_num, vg,
631  vertex_dofs);
632 
633  // Abusing dof_number to set a "this is a
634  // vertex" flag
635  node->set_vg_dof_base(sys_num, vg,
636  vertex_dofs);

653  elem_it = mesh.active_elements_begin();
654 
655  for ( ; elem_it != elem_end; ++elem_it)
656  {
657  Elem* elem = *elem_it;
658  libmesh_assert(elem);
659 
660  // Skip the numbering if this variable is
661  // not active on this element's subdomain
662  if (!vg_description.active_on_subdomain(elem->subdomain_id()))
663  continue;
664 
665  const ElemType type = elem->type();
666  const unsigned int dim = elem->dim();
667 
668  FEType fe_type = base_fe_type;
669  fe_type.order = static_cast<Order>(fe_type.order +
670  elem->p_level());

673  for (unsigned int n=0; n<elem->n_nodes(); n++)
674  {
675  Node* node = elem->get_node(n);
676 
677  const unsigned int old_node_dofs =
678  node->n_comp_group(sys_num, vg);
679 
680  const unsigned int vertex_dofs = old_node_dofs?
681  cast_int<unsigned int>(node->vg_dof_base (sys_num,vg)):0;
682 
683  const unsigned int new_node_dofs =
684  FEInterface::n_dofs_at_node(dim, fe_type, type, n);
685 
686  // We've already allocated vertex DOFs
687  if (elem->is_vertex(n))
688  {
689  libmesh_assert_greater_equal (old_node_dofs, vertex_dofs);
690  // //if (vertex_dofs < new_node_dofs)
691  // libMesh::out << "sys_num,vg,old_node_dofs,vertex_dofs,new_node_dofs="
692  // << sys_num << ","
693  // << vg << ","
694  // << old_node_dofs << ","
695  // << vertex_dofs << ","
696  // << new_node_dofs << '\n',
697  // node->debug_buffer();
698 
699  libmesh_assert_greater_equal (vertex_dofs, new_node_dofs);
700  }
701  // We need to allocate the rest
702  else
703  {
704  // If this has no dofs yet, it needs no vertex
705  // dofs, so we just give it edge or face dofs
706  if (!old_node_dofs)
707  {
708  node->set_n_comp_group(sys_num, vg,
709  new_node_dofs);
710  // Abusing dof_number to set a "this has no
711  // vertex dofs" flag
712  if (new_node_dofs)
713  node->set_vg_dof_base(sys_num, vg,
714  0);
715  }
716 
717  // If this has dofs, but has no vertex dofs,
718  // it may still need more edge or face dofs if
719  // we're p-refined.
720  else if (vertex_dofs == 0)
721  {
722  if (new_node_dofs > old_node_dofs)
723  {
724  node->set_n_comp_group(sys_num, vg,
725  new_node_dofs);
726 
727  node->set_vg_dof_base(sys_num, vg,
728  vertex_dofs);
729  }
730  }
731  // If this is another element's vertex,
732  // add more (non-overlapping) edge/face dofs if
733  // necessary
734  else if (extra_hanging_dofs)
735  {
736  if (new_node_dofs > old_node_dofs - vertex_dofs)
737  {
738  node->set_n_comp_group(sys_num, vg,
739  vertex_dofs + new_node_dofs);
740 
741  node->set_vg_dof_base(sys_num, vg,
742  vertex_dofs);
743  }
744  }
745  // If this is another element's vertex, add any
746  // (overlapping) edge/face dofs if necessary
747  else
748  {
749  libmesh_assert_greater_equal (old_node_dofs, vertex_dofs);
750  if (new_node_dofs > old_node_dofs)
751  {
752  node->set_n_comp_group(sys_num, vg,
753  new_node_dofs);
754 
755  node->set_vg_dof_base (sys_num, vg,
756  vertex_dofs);
757  }
758  }
759  }
760  }
761  // Allocate the element DOFs
762  const unsigned int dofs_per_elem =
763  FEInterface::n_dofs_per_elem(dim, fe_type,
764  type);
765 
766  elem->set_n_comp_group(sys_num, vg, dofs_per_elem);
767 
768  }
769  } // end loop over variable groups
770 
771  // Calling DofMap::reinit() by itself makes little sense,
772  // so we won't bother with nonlocal DofObjects.
773  // Those will be fixed by distribute_dofs
774 
775  //------------------------------------------------------------
776  // Finally, clear all the current DOF indices
777  // (distribute_dofs expects them cleared!)
778  this->invalidate_dofs(mesh);
779 
780  STOP_LOG("reinit()", "DofMap");
781 }

1408  // possible that the neighbors are of different subdomains and

1411  if(this->variable(v).type().family == SCALAR &&

1418  {

1422  dof_index >= this->end_dof())
1423  _send_list.push_back(dof_index);

1426

1448  _send_list.push_back(dof_index);

1458  }
1459  }

1462  for (unsigned int s=0; s<elem->n_neighbors(); s++)

1465  family.clear();

1471  else
1472 #endif

1477  if (family[i]->processor_id() != this->processor_id())
1478  {
1479  // Get the DOF indices for this neighboring element
1480  this->dof_indices (family[i], di);
1481 
1482  // Insert the remote DOF indices into the send list
1483  for (std::size_t j=0; j != di.size(); ++j)
1484  if (di[j] < this->first_dof() ||
1485  di[j] >= this->end_dof())
1486  _send_list.push_back(di[j]);
1487  }
1488  }
1489  }
1490 
1491  // Now loop over all non_local active elements and add any missing
1492  // nodal-only neighbors. This will also get any dofs from nonlocal
1493  // nodes on local elements, because every nonlocal node exists on a
1494  // nonlocal nodal neighbor element.
1495  MeshBase::const_element_iterator elem_it
1496  = mesh.active_elements_begin();
1497  const MeshBase::const_element_iterator elem_end
1498  = mesh.active_elements_end();
1499 
1500  for ( ; elem_it != elem_end; ++elem_it)
1501  {
1502  const Elem* elem = *elem_it;
1503 
1504  // If this is one of our elements, we've already added it
1505  if (elem->processor_id() == this->processor_id())
1506  continue;
1507 
1508  // Do we need to add the element DOFs?
1509  bool add_elem_dofs = false;
1510 
1511  // Check all the nodes of the element to see if it
1512  // shares a node with us
1513  for (unsigned int n=0; n!=elem->n_nodes(); n++)
1514  if (node_on_processor[elem->node(n)])
1515  add_elem_dofs = true;
1516 
1517  // Add the element degrees of freedom if it shares at
1518  // least one node.
1519  if (add_elem_dofs)
1520  {
1521  // Get the DOF indices for this neighboring element
1522  this->dof_indices (elem, di);
1523 
1524  // Insert the remote DOF indices into the send list
1525  for (std::size_t j=0; j != di.size(); ++j)
1526  if (di[j] < this->first_dof() ||
1527  di[j] >= this->end_dof())
1528  _send_list.push_back(di[j]);
1529  }
1530  }
1531
