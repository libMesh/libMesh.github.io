556 #define __FUNCT__ "DMLibMeshParseDecompositionDescriptor_Private"
557 static PetscErrorCode DMLibMeshParseDecompositionDescriptor_Private(DM dm, const char* ddesc, PetscInt* dtype, PetscInt* dcount, PetscInt** dsizes, char ****dlists)
558 {
559  PetscFunctionBegin;
560  PetscErrorCode ierr;
561  PetscBool eq;
562  char *s0, *s, *ss;
563  struct token *llfirst = PETSC_NULL, *lllast = PETSC_NULL, *tok;
564  PetscInt stcount = 0, brcount = 0, d, i;
565  size_t len0, count;
566 
567  /*
568  Parse the decomposition descriptor.
569  Decomposition names could be of one of two forms:
570  var:v1,v2;v3,v4;v4,v5;
571  block:b1,b2;b3,b4;b4,b5;
572  resulting in an overlapping decomposition that groups
573  variables (v1,v2), (v3,v4), (v4,v5) or
574  blocks (b1,b2), (b3,b4), (b4,b5).
575  */
576  /* Copy the descriptor so that we can manipulate it in place. */
577  ierr = PetscStrallocpy(ddesc,&s0); CHKERRQ(ierr);
578  ierr = PetscStrlen(s0, &len0) ; CHKERRQ(ierr);
579  ierr = PetscStrstr(s0,":",&ss); CHKERRQ(ierr);
580  if(!ss) {
581  ss = s0+len0;
582  }
583  else {
584  *ss = 0;
585  }
586  ierr = PetscStrcmp(s0,"var",&eq); CHKERRQ(ierr);
587  if(eq) {
588  *dtype=DMLIBMESH_FIELD_DECOMPOSITION;
589  }
590  else {
591  ierr = PetscStrcmp(s0,"block",&eq);CHKERRQ(ierr);
592  if(!eq)
593  SETERRQ1(((PetscObject)dm)->comm, PETSC_ERR_ARG_WRONG, "Could not determine decomposition type from descriptor: %s\n", ddesc); CHKERRQ(ierr);
594  *dtype=DMLIBMESH_DOMAIN_DECOMPOSITION;
595  }
596  ierr = PetscStrlen(s0,&count); CHKERRQ(ierr);
597  while(count < len0) {
598  struct token *st, *br;
599  ++ss; ++count;
600  s=ss;
601  while(*ss && *ss != ',' && *ss != ';') {
602  ++ss; ++count;
603  }
604  st = PETSC_NULL; br = PETSC_NULL;
605  if(*ss) {
606  /*
607  Found a separator, or a break.
608  Add an appropriate token to the list.
609  A token separator ',' produces no token.
610  */
611  if(*ss == ';') {
612  /* Create a break token: a token with a null string. */
613 #if PETSC_RELEASE_LESS_THAN(3,5,0)
614  ierr = PetscNew(struct token,&br);CHKERRQ(ierr);
615 #else
616  ierr = PetscNew(&br);CHKERRQ(ierr);
617 #endif
618  }
619  *ss = 0;
620  if(s != ss) {
621  /* A nonempty string. */
622 #if PETSC_RELEASE_LESS_THAN(3,5,0)
623  ierr = PetscNew(struct token, &st);CHKERRQ(ierr);
624 #else
625  ierr = PetscNew(&st);CHKERRQ(ierr);
626 #endif
627  st->s = s; /* The string will be properly copied below. */
628  }
629  /* Add the new tokens to the list. */
630  if(st) {
631  if(!lllast) {
632  llfirst = lllast = st;
633  }
634  else {
635  lllast->next = st; lllast = st;
636  }
637  }
638  if(br) {
639  if(!lllast) {
640  llfirst = lllast = br;
641  }
642  else {
643  lllast->next = br; lllast = br;
644  }
645  }
646  }
647  }
648  /* The result of parsing is in the linked list ll. */
649  /* Count up the strings and the breaks. */
650  tok = llfirst;
651  while(tok) {
652  if(tok->s)
653  ++stcount;
654  else
655  ++brcount;
656  tok = tok->next;
657  }
658  /* Allocate the space for the output. */
659  *dcount = brcount;
660  ierr = PetscMalloc(*dcount*sizeof(PetscInt), dsizes); CHKERRQ(ierr);
661  ierr = PetscMalloc(*dcount*sizeof(char**), dlists); CHKERRQ(ierr);
662  for(d = 0; d < *dcount; ++d) (*dsizes)[d] = 0;
663  tok = llfirst; d = 0;
664  while(tok) {
665  if(tok->s)
666  ++(*dsizes)[d];
667  else
668  ++d;
669  tok = tok->next;
670  }
671  for(d = 0; d < *dcount; ++d) {
672  ierr = PetscMalloc(sizeof(char**)*(*dsizes)[d], (*dlists)+d); CHKERRQ(ierr);
673  }
674  /* Now copy strings and destroy tokens. */
675  tok = llfirst; d = 0; i = 0;
676  while(tok) {
677  if(tok->s) {
678  ierr = PetscStrallocpy(tok->s, (*dlists)[d]+i); CHKERRQ(ierr);
679  ++i;
680  }
681  else {
682  ++d;
683  i = 0;
684  }
685  llfirst = tok;
686  tok = tok->next;
687  ierr = PetscFree(llfirst); CHKERRQ(ierr);
688  }
689  /* Deallocate workspace. */
690  ierr = PetscFree(s0); CHKERRQ(ierr);
691  PetscFunctionReturn(0);
692 }
