96 void PointLocatorTree::init (Trees::BuildType build_type)
97 {
98  libmesh_assert (!this->_tree);
99 
100  if (this->_initialized)
101  {
102  // Warn that we are already initialized
103  libMesh::err << "Warning: PointLocatorTree already initialized! Will ignore this call..." << std::endl;
104 
105  // Further warn if we try to init() again with a different build_type
106  if (_build_type != build_type)
107  {
108  libMesh::err << "Warning: PointLocatorTree is using build_type = " << _build_type << ".\n"
109  << "Your requested build_type, " << build_type << " will not be used!" << std::endl;
110  }
111  }
112 
113  else
114  {
115  // Let the requested build_type override the _build_type we were
116  // constructed with. This is no big deal since we have not been
117  // initialized before.
118  _build_type = build_type;
119 
120  if (this->_master == NULL)
121  {
122  START_LOG("init(no master)", "PointLocatorTree");
123 
124  if (this->_mesh.mesh_dimension() == 3)
125  _tree = new Trees::OctTree (this->_mesh, get_target_bin_size(), _build_type);
126  else
127  {
128  // A 1D/2D mesh in 3D space needs special consideration.
129  // If the mesh is planar XY, we want to build a QuadTree
130  // to search efficiently. If the mesh is truly a manifold,
131  // then we need an octree
132 #if LIBMESH_DIM > 2
133  bool is_planar_xy = false;
134 
135  // Build the bounding box for the mesh. If the delta-z bound is
136  // negligibly small then we can use a quadtree.
137  {
138  MeshTools::BoundingBox bbox = MeshTools::bounding_box(this->_mesh);
139 
140  const Real
141  Dx = bbox.second(0) - bbox.first(0),
142  Dz = bbox.second(2) - bbox.first(2);
143 
144  if (std::abs(Dz/(Dx + 1.e-20)) < 1e-10)
145  is_planar_xy = true;
146  }
147 
148  if (!is_planar_xy)
149  _tree = new Trees::OctTree (this->_mesh, get_target_bin_size(), _build_type);
150  else
151 #endif
152 #if LIBMESH_DIM > 1
153  _tree = new Trees::QuadTree (this->_mesh, get_target_bin_size(), _build_type);
154 #else
155  _tree = new Trees::BinaryTree (this->_mesh, get_target_bin_size(), _build_type);
156 #endif
157  }
158 
159  STOP_LOG("init(no master)", "PointLocatorTree");
160  }
161 
162  else
163  {
164  // We are _not_ the master. Let our Tree point to
165  // the master's tree. But for this we first transform
166  // the master in a state for which we are friends.
167  // And make sure the master @e has a tree!
168  const PointLocatorTree* my_master =
169  cast_ptr<const PointLocatorTree*>(this->_master);
170 
171  if (my_master->initialized())
172  this->_tree = my_master->_tree;
173  else
174  libmesh_error_msg("ERROR: Initialize master first, then servants!");
175  }
176 
177  // Not all PointLocators may own a tree, but all of them
178  // use their own element pointer. Let the element pointer
179  // be unique for every interpolator.
180  // Suppose the interpolators are used concurrently
181  // at different locations in the mesh, then it makes quite
182  // sense to have unique start elements.
183  this->_element = NULL;
184  }
185 
186  // ready for take-off
187  this->_initialized = true;
188 }

192 const Elem* PointLocatorTree::operator() (const Point& p, const std::set<subdomain_id_type> *allowed_subdomains) const
193 {
194  libmesh_assert (this->_initialized);
195 
196  START_LOG("operator()", "PointLocatorTree");
197 
198  // If we're provided with an allowed_subdomains list and have a cached element, make sure it complies
199  if (allowed_subdomains && this->_element && !allowed_subdomains->count(this->_element->subdomain_id())) this->_element = NULL;
200 
201  // First check the element from last time before asking the tree
202  if (this->_element==NULL || !(this->_element->contains_point(p)))
203  {
204  // ask the tree
205  this->_element = this->_tree->find_element (p,allowed_subdomains);
206 
207  if (this->_element == NULL)
208  {
209  // No element seems to contain this point. Thus:
210  // 1.) If _out_of_mesh_mode == true, we can just return NULL
211  // without searching further.
212  // 2.) If _out_of_mesh_mode == false, we perform a linear
213  // search over all active (possibly local) elements.
214  // The idea here is that, in the case of curved elements,
215  // the bounding box computed in \p TreeNode::insert(const
216  // Elem*) might be slightly inaccurate and therefore we may
217  // have generated a false negative.
218  if (_out_of_mesh_mode == false)
219  {
220  this->_element = this->perform_linear_search(p, allowed_subdomains, /*use_close_to_point*/ false);
221 
222  STOP_LOG("operator()", "PointLocatorTree");
223  return this->_element;
224  }
225 
226  // If we haven't found the element, we may want to do a linear
227  // search using a tolerance. We only do this if _out_of_mesh_mode == true,
228  // since we're looking for a point that may be outside of the mesh (within the
229  // specified tolerance).
230  if( _use_close_to_point_tol )
231  {
232  if(_verbose)
233  {
234  libMesh::out << "Performing linear search using close-to-point tolerance "
235  << _close_to_point_tol
236  << std::endl;
237  }
238 
239  this->_element =
240  this->perform_linear_search(p,
241  allowed_subdomains,
242  /*use_close_to_point*/ true,
243  _close_to_point_tol);
244 
245  STOP_LOG("operator()", "PointLocatorTree");
246  return this->_element;
247  }
248  }
249  }
250 
251  // If we found an element, it should be active
252  libmesh_assert (!this->_element || this->_element->active());
253 
254  // If we found an element and have a restriction list, they better match
255  libmesh_assert (!this->_element || !allowed_subdomains || allowed_subdomains->count(this->_element->subdomain_id()));
256 
257  STOP_LOG("operator()", "PointLocatorTree");
258 
259  // return the element
260  return this->_element;
261 }

265 const Elem* PointLocatorTree::perform_linear_search(const Point& p,
266  const std::set<subdomain_id_type> *allowed_subdomains,
267  bool use_close_to_point,
268  Real close_to_point_tolerance) const
269 {
270  START_LOG("perform_linear_search", "PointLocatorTree");
271 
272  // The type of iterator depends on the Trees::BuildType
273  // used for this PointLocator. If it's
274  // TREE_LOCAL_ELEMENTS, we only want to double check
275  // local elements during this linear search.
276  MeshBase::const_element_iterator pos =
277  this->_build_type == Trees::LOCAL_ELEMENTS ?
278  this->_mesh.active_local_elements_begin() : this->_mesh.active_elements_begin();
279 
280  const MeshBase::const_element_iterator end_pos =
281  this->_build_type == Trees::LOCAL_ELEMENTS ?
282  this->_mesh.active_local_elements_end() : this->_mesh.active_elements_end();
283 
284  for ( ; pos != end_pos; ++pos)
285  {
286  if (!allowed_subdomains ||
287  allowed_subdomains->count((*pos)->subdomain_id()))
288  {
289  if(!use_close_to_point)
290  {
291  if ((*pos)->contains_point(p))
292  {
293  STOP_LOG("perform_linear_search", "PointLocatorTree");
294  return (*pos);
295  }
296  }
297  else
298  {
299  if ((*pos)->close_to_point(p, close_to_point_tolerance))
300  {
301  STOP_LOG("perform_linear_search", "PointLocatorTree");
302  return (*pos);
303  }
304  }
305  }
306  }
307 
308  STOP_LOG("perform_linear_search", "PointLocatorTree");
309  return NULL;
310 }

313 void PointLocatorTree::enable_out_of_mesh_mode ()
314 {
315  // Out-of-mesh mode is currently only supported if all of the
316  // elements have affine mappings. The reason is that for quadratic
317  // mappings, it is not easy to construct a reliable bounding box of
318  // the element, and thus, the fallback linear search in \p
319  // operator() is required. Hence, out-of-mesh mode would be
320  // extremely slow.
321  if (_out_of_mesh_mode == false)
322  {
323 #ifdef DEBUG
324  MeshBase::const_element_iterator pos = this->_mesh.active_elements_begin();
325  const MeshBase::const_element_iterator end_pos = this->_mesh.active_elements_end();
326  for ( ; pos != end_pos; ++pos)
327  if (!(*pos)->has_affine_map())
328  libmesh_error_msg("ERROR: Out-of-mesh mode is currently only supported if all elements have affine mappings.");
329 #endif
330 
331  _out_of_mesh_mode = true;
332  }
333 }
