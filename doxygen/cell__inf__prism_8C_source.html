63 dof_id_type InfPrism::key (const unsigned int s) const
64 {
65  libmesh_assert_less (s, this->n_sides());
66 
67  switch (s)
68  {
69  case 0: // the triangular face at z=-1, base face
70  return this->compute_key (this->node_id(InfPrism6::side_nodes_map[s][0]),
71  this->node_id(InfPrism6::side_nodes_map[s][1]),
72  this->node_id(InfPrism6::side_nodes_map[s][2]));
73 
74  case 1: // the quad face at y=0

106 {
107  libmesh_assert_less (i, this->n_sides());
108 
109  std::unique_ptr<Elem> face;
110 
111  switch (i)
112  {
113  case 0: // the triangular face at z=-1, base face
114  {
115  face = libmesh_make_unique<Tri3>();
116  break;
117  }
118 
119  case 1: // the quad face at y=0
120  case 2: // the other quad face
121  case 3: // the quad face at x=0
122  {
123  face = libmesh_make_unique<InfQuad4>();
124  break;
125  }
126 
127  default:
128  libmesh_error_msg("Invalid side i = " << i);
129  }
130 
131  // Set the nodes
132  for (unsigned n=0; n<face->n_nodes(); ++n)
133  face->set_node(n) = this->node_ptr(InfPrism6::side_nodes_map[i][n]);

161 bool InfPrism::contains_point (const Point & p, Real tol) const
162 {
163  // For infinite elements with linear base interpolation:
164  // make use of the fact that infinite elements do not
165  // live inside the envelope. Use a fast scheme to
166  // check whether point \p p is inside or outside
167  // our relevant part of the envelope. Note that
168  // this is not exclusive: only when the distance is less,
169  // we are safe. Otherwise, we cannot say anything. The
170  // envelope may be non-spherical, the physical point may lie
171  // inside the envelope, outside the envelope, or even inside
172  // this infinite element. Therefore if this fails,
173  // fall back to the FEInterface::inverse_map()
174  const Point my_origin (this->origin());
175 
176  // determine the minimal distance of the base from the origin
177  // use norm_sq(), it is faster than norm() and produces
178  // the same behavior. Shift my_origin to center first:
179  Point pt0_o(this->point(0) - my_origin);
180  Point pt1_o(this->point(1) - my_origin);
181  Point pt2_o(this->point(2) - my_origin);
182  const Real min_distance_sq = std::min(pt0_o.norm_sq(),
183  std::min(pt1_o.norm_sq(),
184  pt2_o.norm_sq()));
185 
186  // work with 1% allowable deviation. We can still fall
187  // back to the InfFE::inverse_map()
188  const Real conservative_p_dist_sq = 1.01 * (Point(p - my_origin).norm_sq());
189 
190  if (conservative_p_dist_sq < min_distance_sq)
191  {
192  // the physical point is definitely not contained in the element:
193  return false;
194  }
195 
196  // this captures the case that the point is not (almost) in the direction of the element.:
197  // first, project the problem onto the unit sphere:
198  Point p_o(p - my_origin);
199  pt0_o /= pt0_o.norm();
200  pt1_o /= pt1_o.norm();
201  pt2_o /= pt2_o.norm();
202  p_o /= p_o.norm();
203 
204  // now, check if it is in the projected face; by comparing the distance of
205  // any point in the element to \p p with the largest distance between this point
206  // to any other point in the element.
207  if ((p_o - pt0_o).norm_sq() > std::max((pt0_o - pt1_o).norm_sq(), (pt0_o - pt2_o).norm_sq()) ||
208  (p_o - pt1_o).norm_sq() > std::max((pt1_o - pt2_o).norm_sq(), (pt1_o - pt0_o).norm_sq()) ||
209  (p_o - pt2_o).norm_sq() > std::max((pt2_o - pt0_o).norm_sq(), (pt2_o - pt1_o).norm_sq()) )
210  {
211  // the physical point is definitely not contained in the element
212  return false;
213  }
214 
215 
216  // It seems that \p p is at least close to this element.
217  //
218  // Declare a basic FEType. Will use default in the base,
219  // and something else (not important) in radial direction.
220  FEType fe_type(default_order());
221 
222  const Point mapped_point = FEInterface::inverse_map(dim(),
223  fe_type,
224  this,
225  p,
226  tol,
227  false);
228 
229  return FEInterface::on_reference_element(mapped_point, this->type(), tol);
230 }
