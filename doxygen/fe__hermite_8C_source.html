35 void hermite_nodal_soln(const Elem* elem,
36  const Order order,
37  const std::vector<Number>& elem_soln,
38  std::vector<Number>& nodal_soln,
39  unsigned Dim)
40 {
41  const unsigned int n_nodes = elem->n_nodes();
42 
43  const ElemType elem_type = elem->type();
44 
45  nodal_soln.resize(n_nodes);
46 
47  const Order totalorder = static_cast<Order>(order + elem->p_level());
48 
49  // FEType object to be passed to various FEInterface functions below.
50  FEType fe_type(totalorder, HERMITE);
51 
52  const unsigned int n_sf =
53  // FE<Dim,T>::n_shape_functions(elem_type, totalorder);
54  FEInterface::n_shape_functions(Dim, fe_type, elem_type);
55 
56  std::vector<Point> refspace_nodes;
57  FEBase::get_refspace_nodes(elem_type,refspace_nodes);
58  libmesh_assert_equal_to (refspace_nodes.size(), n_nodes);
59 
60  for (unsigned int n=0; n<n_nodes; n++)
61  {
62  libmesh_assert_equal_to (elem_soln.size(), n_sf);
63 
64  // Zero before summation
65  nodal_soln[n] = 0;
66 
67  // u_i = Sum (alpha_i phi_i)
68  for (unsigned int i=0; i<n_sf; i++)
69  nodal_soln[n] += elem_soln[i] *
70  // FE<Dim,T>::shape(elem, order, i, mapped_point);
71  FEInterface::shape(Dim, fe_type, elem, i, refspace_nodes[n]);
72  }
73 } // hermite_nodal_soln()

77 unsigned int hermite_n_dofs(const ElemType t, const Order o)
78 {
79 #ifdef DEBUG
80  if (o < 3)
81  libmesh_error_msg("Error: Hermite elements require order>=3, but you asked for order=" << o);
82 #endif
83 
84  // Piecewise (bi/tri)cubic C1 Hermite splines
85  switch (t)
86  {
87  case NODEELEM:
88  return 1;
89  case EDGE2:
90  libmesh_assert_less (o, 4);
91  case EDGE3:
92  return (o+1);
93 
94  case QUAD4:
95  case QUAD8:
96  libmesh_assert_less (o, 4);
97  case QUAD9:
98  return ((o+1)*(o+1));
99 
100  case HEX8:
101  case HEX20:
102  libmesh_assert_less (o, 4);
103  case HEX27:
104  return ((o+1)*(o+1)*(o+1));
105 
106  case INVALID_ELEM:
107  return 0;
108 
109  default:
110  libmesh_error_msg("ERROR: Bad ElemType = " << Utility::enum_to_string(t) << " for " << Utility::enum_to_string(o) << " order approximation!");
111  }
112 
113  libmesh_error_msg("We'll never get here!");
114  return 0;
115 } // hermite_n_dofs()

120 unsigned int hermite_n_dofs_at_node(const ElemType t,
121  const Order o,
122  const unsigned int n)
123 {
124  libmesh_assert_greater (o, 2);
125  // Piecewise (bi/tri)cubic C1 Hermite splines
126  switch (t)
127  {
128  case NODEELEM:
129  return 1;
130  case EDGE2:
131  case EDGE3:
132  {
133  switch (n)
134  {
135  case 0:
136  case 1:
137  return 2;
138  case 2:
139  // Interior DoFs are carried on Elems
140  // return (o-3);
141  return 0;
142 
143  default:
144  libmesh_error_msg("ERROR: Invalid node ID " << n << " selected for EDGE2/3!");
145  }
146  }
147 
148  case QUAD4:
149  libmesh_assert_less (o, 4);
150  case QUAD8:
151  case QUAD9:
152  {
153  switch (n)
154  {
155  // Vertices
156  case 0:
157  case 1:
158  case 2:
159  case 3:
160  return 4;
161  // Edges

163  case 5:
164  case 6:
165  case 7:
166  return (2*(o-3));
167  case 8:
168  // Interior DoFs are carried on Elems
169  // return ((o-3)*(o-3));

173  libmesh_error_msg("ERROR: Invalid node ID " << n << " selected for QUAD4/8/9!");
174  }
175  }

177  case HEX8:
178  case HEX20:
179  libmesh_assert_less (o, 4);

184  // Vertices

186  case 1:
187  case 2:
188  case 3:
189  case 4:
190  case 5:

197  case 10:

208  // Faces

211  case 22:

223  }

226  case INVALID_ELEM:
227  return 0;
228 
229  default:
