213 SystemNorm::SystemNorm(const std::vector<FEMNormType> &norms,
214  std::vector<std::vector<Real> > &weights):
215  _norms(norms),
216  _weights(weights.size()),
217  _weights_sq(weights.size()),
218  _off_diagonal_weights(weights)
219 {
220  if(_norms.empty())
221  _norms.push_back(DISCRETE_L2);
222 
223  if (_weights.empty())
224  {
225  _weights.push_back(1.0);
226  _weights_sq.push_back(1.0);
227  }
228  else
229  {
230  // Loop over the entries of the user provided matrix and store its entries in
231  // the _off_diagonal_weights or _diagonal_weights
232  for(std::size_t i=0; i!=_off_diagonal_weights.size(); ++i)
233  {
234  if(_off_diagonal_weights[i].size() > i)
235  {
236  _weights[i] = _off_diagonal_weights[i][i];
237  _off_diagonal_weights[i][i] = 0;
238  }
239  else
240  _weights[i] = 1.0;
241  }
242  for (std::size_t i=0; i != _weights.size(); ++i)
243  _weights_sq[i] = _weights[i] * _weights[i];
244  }
245 }

345 inline
346 Real SystemNorm::calculate_norm(const std::vector<Real>& v1, const std::vector<Real>& v2)
347 {
348  // The vectors are assumed to both be vectors of the (same number
349  // of) components
350  std::size_t vsize = v1.size();
351  libmesh_assert_equal_to (vsize, v2.size());
352 
353  // We'll support implicitly defining weights, but if the user sets
354  // more weights than he uses then something's probably wrong
355  std::size_t diagsize = this->_weights.size();
356  libmesh_assert_greater_equal (vsize, diagsize);
357 
358  // Initialize the variable val
359  Real val = 0.;
360 
361  // Loop over all the components of the system with explicit
362  // weights
363  for(std::size_t i = 0; i != diagsize; i++)
364  {
365  val += this->_weights[i] * v1[i] * v2[i];
366  }
367  // Loop over all the components of the system with implicit
368  // weights
369  for(std::size_t i = diagsize; i < vsize; i++)
370  {
371  val += v1[i] * v2[i];
372  }
373 
374  // Loop over the components of the system
375  std::size_t nrows = this->_off_diagonal_weights.size();
376  libmesh_assert_less_equal (vsize, nrows);
377 
378  for(std::size_t i = 0; i != nrows; i++)
379  {
380  std::size_t ncols = this->_off_diagonal_weights[i].size();
381  for(std::size_t j=0; j != ncols; j++)
382  {
383  // Note that the diagonal weights here were set to zero
384  // in the constructor
385  val += this->_off_diagonal_weights[i][j] * v1[i] * v2[j];
386  }
387  }
388 
389  return(val);
390 }

398 inline
399 bool SystemNorm::is_identity()
400 {
401  std::size_t nrows = this->_off_diagonal_weights.size();
402 
403  // If any of the off-diagonal elements is not 0, then we are in the non-identity case
404  for(std::size_t i = 0; i != nrows; i++)
405  {
406  std::size_t ncols = this->_off_diagonal_weights[i].size();
407  for(std::size_t j = 0; j != ncols; j++)
408  {
409  if(_off_diagonal_weights[i][j] != 0)
410  {
411  return(false);
412  }
413  }
414  }
415 
416  // If any of the diagonal elements is not 1, then we are in the non-identity case
417  nrows = this->_weights.size();
418  for(std::size_t i = 0; i != nrows; i++)
419  if(_weights[i] != 1)
420  return(false);
421 
422  // If all the off-diagonals elements are 0, and diagonal elements 1, then we are in an identity case
423  return(true);
424 }
