106 bool Prism18::has_affine_map() const
107 {
108  // Make sure z edges are affine
109  Point v = this->point(3) - this->point(0);
110  if (!v.relative_fuzzy_equals(this->point(4) - this->point(1)) ||
111  !v.relative_fuzzy_equals(this->point(5) - this->point(2)))
112  return false;
113  // Make sure edges are straight
114  v /= 2;
115  if (!v.relative_fuzzy_equals(this->point(9) - this->point(0)) ||
116  !v.relative_fuzzy_equals(this->point(10) - this->point(1)) ||
117  !v.relative_fuzzy_equals(this->point(11) - this->point(2)) ||
118  !v.relative_fuzzy_equals(this->point(15) - this->point(6)) ||
119  !v.relative_fuzzy_equals(this->point(16) - this->point(7)) ||
120  !v.relative_fuzzy_equals(this->point(17) - this->point(8)))
121  return false;
122  v = (this->point(1) - this->point(0))/2;
123  if (!v.relative_fuzzy_equals(this->point(6) - this->point(0)) ||
124  !v.relative_fuzzy_equals(this->point(12) - this->point(3)))
125  return false;

693  // The quadrature rule for the Prism18 is a tensor product between a
694  // FOURTH-order TRI rule (in xi, eta) and a FIFTH-order EDGE rule
695  // in zeta.
696 
697  // Number of points in the 2D quadrature rule.
698  const int N2D = 6;
699 
700  // Parameters of the 2D rule
701  static const Real
702  w1 = 1.1169079483900573284750350421656140e-01L,
703  w2 = 5.4975871827660933819163162450105264e-02L,
704  a1 = 4.4594849091596488631832925388305199e-01L,
705  a2 = 9.1576213509770743459571463402201508e-02L;
706 
707  // Points and weights of the 2D rule
708  static const Real w2D[N2D] = {w1, w1, w1, w2, w2, w2};
709 
710  // Quadrature point locations raised to powers. xi[0][2] is
711  // quadrature point 0, squared, xi[1][1] is quadrature point 1 to the
712  // first power, etc. This lets us avoid calling std::pow inside the
713  // loops below.
714  static const Real xi[N2D][3] =
715  {
716  // ^0 ^1 ^2
717  { 1., a1, a1*a1},
718  { 1., 1-2*a1, (1-2*a1)*(1-2*a1)},
719  { 1., a1, a1*a1},
720  { 1., a2, a2*a2},
721  { 1., 1-2*a2, (1-2*a2)*(1-2*a2)},
722  { 1., a2, a2*a2}
723  };
724 
725  static const Real eta[N2D][3] =
726  {
727  // ^0 ^1 ^2
728  { 1., a1, a1*a1},
729  { 1., a1, a1*a1},
730  { 1., 1-2*a1, (1-2*a1)*(1-2*a1)},
731  { 1., a2, a2*a2},
732  { 1., a2, a2*a2},
733  { 1., 1-2*a2, (1-2*a2)*(1-2*a2)}
734  };
735 
736  // Number of points in the 1D quadrature rule.
737  const int N1D = 3;
738 
739  // Points and weights of the 1D quadrature rule.
740  static const Real w1D[N1D] = {5./9, 8./9, 5./9};
741 
742  const Real zeta[N1D][3] =
743  {
744  //^0 ^1 ^2
745  { 1., -std::sqrt(15)/5., 15./25},
746  { 1., 0., 0.},
747  { 1., std::sqrt(15)/5., 15./25}
748  };
749 
750  // The integer exponents for each term.
751  static const int exponents[n_components][3] =
752  {
753  {0, 0, 0},
754  {0, 0, 1},
755  {0, 0, 2},
756  {0, 1, 0},
757  {0, 1, 1},
758  {0, 1, 2},
759  {0, 2, 0},
760  {0, 2, 1},
761  {1, 0, 0},
762  {1, 0, 1},
763  {1, 0, 2},
764  {1, 1, 0},
765  {1, 1, 1},
766  {1, 2, 0},
767  {2, 0, 0},
768  {2, 0, 1}
769  };
770 
771  Real vol = 0.;
772  for (int i=0; i<N2D; ++i)
773  for (int j=0; j<N1D; ++j)
774  {
775  // Compute dx_dxi, dx_deta, dx_dzeta at the current quadrature point.
776  Point dx_dxi_q, dx_deta_q, dx_dzeta_q;
777  for (int c=0; c<n_components; ++c)
778  {
779  Real coeff =
780  xi[i][exponents[c][0]]*
781  eta[i][exponents[c][1]]*
782  zeta[j][exponents[c][2]];
783 
784  dx_dxi_q += coeff * dx_dxi[c];
785  dx_deta_q += coeff * dx_deta[c];
786  dx_dzeta_q += coeff * dx_dzeta[c];
787  }
788 
789  // Compute scalar triple product, multiply by weight, and accumulate volume.
790  vol += w2D[i] * w1D[j] * triple_product(dx_dxi_q, dx_deta_q, dx_dzeta_q);
791  }
792 
793  return vol;
794 }
