38 // A custom comparison function, based on Point::operator<,
39 // that tries to ignore floating point differences in components
40 // of the point
41 class FuzzyPointCompare
42 {
43 private:
44  Real _tol;
45 
46 public:
47  // Constructor takes the tolerance to be used in fuzzy comparisons
48  FuzzyPointCompare(Real tol) : _tol(tol) {}
49 
50  // This is inspired directly by Point::operator<
51  bool operator()(const Point & lhs, const Point & rhs)
52  {
53  for (unsigned i=0; i<LIBMESH_DIM; ++i)
54  {
55  // If the current components are within some tolerance
56  // of one another, then don't attempt the less-than comparison.
57  // Note that this may cause something strange to happen, as Roy
58  // believes he can prove it is not a total ordering...
59  Real rel_size = std::max(std::abs(lhs(i)), std::abs(rhs(i)));
60 
61  // Don't use relative tolerance if both numbers are already small.
62  // How small? Some possible options are:
63  // * std::numeric_limits<Real>::epsilon()
64  // * TOLERANCE
65  // * 1.0
66  // If we use std::numeric_limits<Real>::epsilon(), we'll
67  // do more relative comparisons for small numbers, but
68  // increase the chance for false positives? If we pick 1.0,
69  // we'll never "increase" the difference between small numbers
70  // in the test below.
71  if (rel_size < 1.)
72  rel_size = 1.;
73 
74  // Don't attempt the comparison if lhs(i) and rhs(i) are too close
75  // together.
76  if ( std::abs(lhs(i) - rhs(i)) / rel_size < _tol)
77  continue;
78 
79  if (lhs(i) < rhs(i))
80  return true;
81  if (lhs(i) > rhs(i))
82  return false;
83  }
84 
85  // We compared all the components without returning yet, so
86  // each component was neither greater than nor less than they other.
87  // They might be equal, so return false.
88  return false;
89  }
90 
91  // Needed by std::sort on vector< pair<Point,id> >
92  bool operator()(const std::pair<Point, dof_id_type> & lhs,
93  const std::pair<Point, dof_id_type> & rhs)
94  {
95  return (*this)(lhs.first, rhs.first);
96  }
97 
98  // comparison function where lhs is a Point and rhs is a pair<Point,dof_id_type>.
99  // This is used in routines like lower_bound, where a specific value is being
100  // searched for.
101  bool operator()(const Point & lhs, std::pair<Point, dof_id_type> & rhs)
102  {
103  return (*this)(lhs, rhs.first);
104  }
105 
106  // And the other way around...
107  bool operator()(std::pair<Point, dof_id_type> & lhs, const Point & rhs)
108  {
109  return (*this)(lhs.first, rhs);
110  }
111 };

347 {
348  libmesh_assert(e);
349 
350  // Initialize an iterator to eventually point to the element we want to delete
351  std::vector<Elem *>::iterator pos = _elements.end();
352 
353  // In many cases, e->id() gives us a clue as to where e
354  // is located in the _elements vector. Try that first
355  // before trying the O(n_elem) search.
356  libmesh_assert_less (e->id(), _elements.size());

360  // We found it!

404  const dof_id_type id,
405  const processor_id_type proc_id)
406 {
407  // // We only append points with ReplicatedMesh
408  // libmesh_assert(id == DofObject::invalid_id || id == _nodes.size());
409  // Node *n = Node::build(p, _nodes.size()).release();
410  // n->processor_id() = proc_id;
411  // _nodes.push_back (n);

414 
415  // If the user requests a valid id, either
416  // provide the existing node or resize the container
417  // to fit the new node.
418  if (id != DofObject::invalid_id)
419  if (id < _nodes.size())
420  n = _nodes[id];
421  else
422  _nodes.resize(id+1);
423  else

435  n->processor_id() = proc_id;

476 Node * ReplicatedMesh::insert_node(Node * n)
477 {
478  if (!n)
479  libmesh_error_msg("Error, attempting to insert NULL node.");
480 
481  if (n->id() == DofObject::invalid_id)
482  libmesh_error_msg("Error, cannot insert node with invalid id.");
483 
484  if (n->id() < _nodes.size())
485  {
486  // Don't allow inserting on top of an existing Node.
487 
488  // Doing so doesn't have to be *error*, in the case where a
489  // redundant insert is done, but when that happens we ought to
490  // always be able to make the code more efficient by avoiding
491  // the redundant insert, so let's keep screaming "Error" here.
492  if (_nodes[ n->id() ] != libmesh_nullptr)
493  libmesh_error_msg("Error, cannot insert node on top of existing node.");
494  }
495  else
496  {
497  // Allocate just enough space to store the new node. This will
498  // cause highly non-ideal memory allocation behavior if called
499  // repeatedly...
500  _nodes.resize(n->id() + 1);
501  }
502 
503 #ifdef LIBMESH_ENABLE_UNIQUE_ID
504  if (!n->valid_unique_id())
505  n->set_unique_id() = _next_unique_id++;
506 #endif
507 
508  // We have enough space and this spot isn't already occupied by
509  // another node, so go ahead and add it.
510  _nodes[ n->id() ] = n;
511 
512  // If we made it this far, we just inserted the node the user handed
513  // us, so we can give it right back.
514  return n;
515 }

633 void ReplicatedMesh::renumber_nodes_and_elements ()
634 {
635  LOG_SCOPE("renumber_nodes_and_elem()", "Mesh");
636 
637  // node and element id counters
638  dof_id_type next_free_elem = 0;
639  dof_id_type next_free_node = 0;
640 
641  // Will hold the set of nodes that are currently connected to elements
642  LIBMESH_BEST_UNORDERED_SET<Node *> connected_nodes;
643 
644  // Loop over the elements. Note that there may
645  // be NULLs in the _elements vector from the coarsening
646  // process. Pack the elements in to a contiguous array
647  // and then trim any excess.
648  {
649  std::vector<Elem *>::iterator in = _elements.begin();
650  std::vector<Elem *>::iterator out_iter = _elements.begin();
651  const std::vector<Elem *>::iterator end = _elements.end();
652 
653  for (; in != end; ++in)
654  if (*in != libmesh_nullptr)
655  {
656  Elem * el = *in;
657 
658  *out_iter = *in;
659  ++out_iter;
660 
661  // Increment the element counter
662  el->set_id (next_free_elem++);
663 
664  if (_skip_renumber_nodes_and_elements)
665  {
666  // Add this elements nodes to the connected list
667  for (unsigned int n=0; n<el->n_nodes(); n++)
668  connected_nodes.insert(el->node_ptr(n));
669  }
670  else // We DO want node renumbering
671  {
672  // Loop over this element's nodes. Number them,
673  // if they have not been numbered already. Also,
674  // position them in the _nodes vector so that they
675  // are packed contiguously from the beginning.
676  for (unsigned int n=0; n<el->n_nodes(); n++)
677  if (el->node_id(n) == next_free_node) // don't need to process
678  next_free_node++; // [(src == dst) below]
679 
680  else if (el->node_id(n) > next_free_node) // need to process
681  {
682  // The source and destination indices
683  // for this node
684  const dof_id_type src_idx = el->node_id(n);
685  const dof_id_type dst_idx = next_free_node++;
686 
687  // ensure we want to swap a valid nodes
688  libmesh_assert(_nodes[src_idx]);
689 
690  // Swap the source and destination nodes
691  std::swap(_nodes[src_idx],
692  _nodes[dst_idx] );
693 
694  // Set proper indices where that makes sense
695  if (_nodes[src_idx] != libmesh_nullptr)
696  _nodes[src_idx]->set_id (src_idx);
697  _nodes[dst_idx]->set_id (dst_idx);
698  }
699  }
700  }
701 
702  // Erase any additional storage. These elements have been
703  // copied into NULL voids by the procedure above, and are
704  // thus repeated and unnecessary.
705  _elements.erase (out_iter, end);
706  }
707 
708 
709  if (_skip_renumber_nodes_and_elements)
710  {
711  // Loop over the nodes. Note that there may
712  // be NULLs in the _nodes vector from the coarsening
713  // process. Pack the nodes in to a contiguous array
714  // and then trim any excess.
715 
716  std::vector<Node *>::iterator in = _nodes.begin();
717  std::vector<Node *>::iterator out_iter = _nodes.begin();
718  const std::vector<Node *>::iterator end = _nodes.end();
719 
720  for (; in != end; ++in)
721  if (*in != libmesh_nullptr)
722  {
723  // This is a reference so that if we change the pointer it will change in the vector
724  Node * & nd = *in;
725 
726  // If this node is still connected to an elem, put it in the list
727  if (connected_nodes.find(nd) != connected_nodes.end())
728  {
729  *out_iter = nd;
730  ++out_iter;
731 
732  // Increment the node counter
733  nd->set_id (next_free_node++);
734  }
735  else // This node is orphaned, delete it!
736  {
737  this->get_boundary_info().remove (nd);
738 
739  // delete the node
740  delete nd;
741  nd = libmesh_nullptr;
742  }
743  }
744 
745  // Erase any additional storage. Whatever was
746  _nodes.erase (out_iter, end);
747  }
748  else // We really DO want node renumbering
749  {
750  // Any nodes in the vector >= _nodes[next_free_node]
751  // are not connected to any elements and may be deleted
752  // if desired.
753

777  }
778 
779  _nodes.erase (nd, end);
780  }
781  }
782

784  libmesh_assert_equal_to (next_free_node, _nodes.size());
785

845 void ReplicatedMesh::stitching_helper (ReplicatedMesh * other_mesh,
846  boundary_id_type this_mesh_boundary_id,
847  boundary_id_type other_mesh_boundary_id,
848  Real tol,
849  bool clear_stitched_boundary_ids,
850  bool verbose,
851  bool use_binary_search,
852  bool enforce_all_nodes_match_on_boundaries,
853  bool skip_find_neighbors)
854 {
855  std::map<dof_id_type, dof_id_type> node_to_node_map, other_to_this_node_map; // The second is the inverse map of the first
856  std::map<dof_id_type, std::vector<dof_id_type> > node_to_elems_map;
857 
858  typedef dof_id_type key_type;
859  typedef std::pair<Elem *, unsigned char> val_type;
860  typedef std::pair<key_type, val_type> key_val_pair;
861  typedef LIBMESH_BEST_UNORDERED_MULTIMAP<key_type, val_type> map_type;
862  // Mapping between all side keys in this mesh and elements+side numbers relevant to the boundary in this mesh as well.
863  map_type side_to_elem_map;
864 
865  // If there is only one mesh (i.e. other_mesh==NULL), then loop over this mesh twice
866  if (!other_mesh)
867  {
868  other_mesh = this;
869  }
870 
871  if ((this_mesh_boundary_id != BoundaryInfo::invalid_id) &&
872  (other_mesh_boundary_id != BoundaryInfo::invalid_id))
873  {
874  LOG_SCOPE("stitch_meshes node merging", "ReplicatedMesh");
875 
876  // While finding nodes on the boundary, also find the minimum edge length
877  // of all faces on both boundaries. This will later be used in relative
878  // distance checks when stitching nodes.
879  Real h_min = std::numeric_limits<Real>::max();
880  bool h_min_updated = false;
881 
882  // Loop below fills in these sets for the two meshes.
883  std::set<dof_id_type> this_boundary_node_ids, other_boundary_node_ids;
884  {
885  // Make temporary fixed-size arrays for loop
886  boundary_id_type id_array[2] = {this_mesh_boundary_id, other_mesh_boundary_id};
887  std::set<dof_id_type> * set_array[2] = {&this_boundary_node_ids, &other_boundary_node_ids};
888  ReplicatedMesh * mesh_array[2] = {this, other_mesh};
889 
890  for (unsigned i=0; i<2; ++i)
891  {
892  // First we deal with node boundary IDs.
893  // We only enter this loop if we have at least one
894  // nodeset.
895  if (mesh_array[i]->get_boundary_info().n_nodeset_conds() > 0)
896  {
897  std::vector<numeric_index_type> node_id_list;
898  std::vector<boundary_id_type> bc_id_list;
899 
900  // Get the list of nodes with associated boundary IDs
901  mesh_array[i]->get_boundary_info().build_node_list(node_id_list, bc_id_list);
902 
903  for (std::size_t node_index=0; node_index<bc_id_list.size(); node_index++)
904  {
905  boundary_id_type node_bc_id = bc_id_list[node_index];
906  if (node_bc_id == id_array[i])
907  {
908  dof_id_type this_node_id = node_id_list[node_index];
909  set_array[i]->insert( this_node_id );
910 
911  // We need to set h_min to some value. It's too expensive to
912  // search for the element that actually contains this node,
913  // since that would require a PointLocator. As a result, we
914  // just use the first element in the mesh to give us hmin.
915  const Elem * first_active_elem = *mesh_array[i]->active_elements_begin();
916  h_min = first_active_elem->hmin();
917  h_min_updated = true;
918  }
919  }
920  }
921 
922  // Container to catch boundary IDs passed back from BoundaryInfo.
923  std::vector<boundary_id_type> bc_ids;
924 
925  MeshBase::element_iterator elem_it = mesh_array[i]->elements_begin();
926  MeshBase::element_iterator elem_end = mesh_array[i]->elements_end();
927  for ( ; elem_it != elem_end; ++elem_it)
928  {
929  Elem * el = *elem_it;
930 
931  // Now check whether elem has a face on the specified boundary
932  for (unsigned char side_id=0; side_id<el->n_sides(); ++side_id)
933  if (el->neighbor_ptr(side_id) == libmesh_nullptr)
934  {
935  // Get *all* boundary IDs on this side, not just the first one!
936  mesh_array[i]->get_boundary_info().boundary_ids (el, side_id, bc_ids);
937 
938  if (std::find(bc_ids.begin(), bc_ids.end(), id_array[i]) != bc_ids.end())
939  {
940  UniquePtr<Elem> side (el->build_side_ptr(side_id));
941  for (unsigned int node_i=0; node_i<side->n_nodes(); ++node_i)
942  set_array[i]->insert( side->node_id(node_i) );
943 
944  h_min = std::min(h_min, side->hmin());
945  h_min_updated = true;
946 
947  // This side is on the boundary, add its information to side_to_elem
948  if (skip_find_neighbors && (i==0))
949  {
950  key_type key = el->key(side_id);
951  val_type val;
952  val.first = el;
953  val.second = side_id;
954 
955  key_val_pair kvp;
956  kvp.first = key;
957  kvp.second = val;
958  // side_to_elem_map[key] = val;
959 #if defined(LIBMESH_HAVE_UNORDERED_MAP) || defined(LIBMESH_HAVE_TR1_UNORDERED_MAP) || defined(LIBMESH_HAVE_HASH_MAP) || defined(LIBMESH_HAVE_EXT_HASH_MAP)
960  side_to_elem_map.insert (kvp);
961 #else
962  side_to_elem_map.insert (side_to_elem_map.begin(),kvp);
963 #endif
964  }
965  }
966 
967  // Also, check the edges on this side. We don't have to worry about
968  // updating neighbor info in this case since elements don't store
969  // neighbor info on edges.
970  for (unsigned short edge_id=0; edge_id<el->n_edges(); ++edge_id)
971  {
972  if (el->is_edge_on_side(edge_id, side_id))
973  {
974  // Get *all* boundary IDs on this edge, not just the first one!
975  mesh_array[i]->get_boundary_info().edge_boundary_ids (el, edge_id, bc_ids);
976 
977  if (std::find(bc_ids.begin(), bc_ids.end(), id_array[i]) != bc_ids.end())
978  {
979  UniquePtr<Elem> edge (el->build_edge_ptr(edge_id));
980  for (unsigned int node_i=0; node_i<edge->n_nodes(); ++node_i)
981  set_array[i]->insert( edge->node_id(node_i) );
982 
983  h_min = std::min(h_min, edge->hmin());
984  h_min_updated = true;
985  }
986  }
987  }
988  }
989  }
990  }
991  }
992 
993  if (verbose)
994  {
995  libMesh::out << "In ReplicatedMesh::stitch_meshes:\n"
996  << "This mesh has " << this_boundary_node_ids.size()
997  << " nodes on boundary " << this_mesh_boundary_id << ".\n"
998  << "Other mesh has " << other_boundary_node_ids.size()
999  << " nodes on boundary " << other_mesh_boundary_id << ".\n";
1000 
1001  if (h_min_updated)
1002  {
1003  libMesh::out << "Minimum edge length on both surfaces is " << h_min << ".\n";
1004  }
1005  else
1006  {
1007  libMesh::out << "No elements on specified surfaces." << std::endl;
1008  }
1009  }
1010 
1011 
1012  if (use_binary_search)
1013  {
1014  // Store points from both stitched faces in sorted vectors for faster
1015  // searching later.
1016  typedef std::vector< std::pair<Point, dof_id_type> > PointVector;
1017  PointVector
1018  this_sorted_bndry_nodes(this_boundary_node_ids.size()),
1019  other_sorted_bndry_nodes(other_boundary_node_ids.size());
1020 
1021  // Comparison object that will be used later. So far, I've had reasonable success
1022  // with TOLERANCE...
1023  FuzzyPointCompare mein_comp(TOLERANCE);
1024 
1025  // Create and sort the vectors we will use to do the geometric searching
1026  {
1027  std::set<dof_id_type> * set_array[2] = {&this_boundary_node_ids, &other_boundary_node_ids};
1028  ReplicatedMesh * mesh_array[2] = {this, other_mesh};
1029  PointVector * vec_array[2] = {&this_sorted_bndry_nodes, &other_sorted_bndry_nodes};
1030 
1031  for (unsigned i=0; i<2; ++i)
1032  {
1033  std::set<dof_id_type>::iterator
1034  set_it = set_array[i]->begin(),
1035  set_it_end = set_array[i]->end();
1036 
1037  // Fill up the vector with the contents of the set...
1038  for (unsigned ctr=0; set_it != set_it_end; ++set_it, ++ctr)
1039  {
1040  (*vec_array[i])[ctr] = std::make_pair(mesh_array[i]->point(*set_it), // The geometric point
1041  *set_it); // Its ID
1042  }
1043 
1044  // Sort the vectors based on the FuzzyPointCompare struct op()
1045  std::sort(vec_array[i]->begin(), vec_array[i]->end(), mein_comp);
1046  }
1047  }
1048 
1049  // Build up the node_to_node_map and node_to_elems_map using the sorted vectors of Points.
1050  for (std::size_t i=0; i<this_sorted_bndry_nodes.size(); ++i)
1051  {
1052  // Current point we're working on
1053  Point this_point = this_sorted_bndry_nodes[i].first;
1054 
1055  // FuzzyPointCompare does a fuzzy equality comparison internally to handle
1056  // slight differences between the list of nodes on each mesh.
1057  PointVector::iterator other_iter = Utility::binary_find(other_sorted_bndry_nodes.begin(),
1058  other_sorted_bndry_nodes.end(),
1059  this_point,
1060  mein_comp);
1061 
1062  // Not every node on this_sorted_bndry_nodes will necessarily be stitched, so
1063  // if its pair is not found on other_mesh, just continue.
1064  if (other_iter != other_sorted_bndry_nodes.end())
1065  {
1066  // Check that the points do indeed match - should not be necessary unless something
1067  // is wrong with binary_find. To be on the safe side, we'll check.
1068  {
1069  // Grab the other point from the iterator
1070  Point other_point = other_iter->first;
1071 
1072  if (!this_point.absolute_fuzzy_equals(other_point, tol*h_min))
1073  libmesh_error_msg("Error: mismatched points: " << this_point << " and " << other_point);
1074  }
1075 
1076 
1077  // Associate these two nodes in both the node_to_node_map and the other_to_this_node_map
1078  dof_id_type
1079  this_node_id = this_sorted_bndry_nodes[i].second,
1080  other_node_id = other_iter->second;
1081  node_to_node_map[this_node_id] = other_node_id;
1082  other_to_this_node_map[other_node_id] = this_node_id;
1083  }
1084 
1085  }
1086  }
1087  else
1088  {
1089  // Otherwise, use a simple N^2 search to find the closest matching points. This can be helpful
1090  // in the case that we have tolerance issues which cause mismatch between the two surfaces
1091  // that are being stitched.
1092 
1093  std::set<dof_id_type>::iterator set_it = this_boundary_node_ids.begin();
1094  std::set<dof_id_type>::iterator set_it_end = this_boundary_node_ids.end();
1095  for ( ; set_it != set_it_end; ++set_it)
1096  {
1097  dof_id_type this_node_id = *set_it;
1098  Node & this_node = this->node_ref(this_node_id);
1099 
1100  bool found_matching_nodes = false;
1101 
1102  std::set<dof_id_type>::iterator other_set_it = other_boundary_node_ids.begin();
1103  std::set<dof_id_type>::iterator other_set_it_end = other_boundary_node_ids.end();
1104  for ( ; other_set_it != other_set_it_end; ++other_set_it)
1105  {
1106  dof_id_type other_node_id = *other_set_it;
1107  Node & other_node = other_mesh->node_ref(other_node_id);
1108 
1109  Real node_distance = (this_node - other_node).norm();
1110 
1111  if (node_distance < tol*h_min)
1112  {
1113  // Make sure we didn't already find a matching node!
1114  if (found_matching_nodes)
1115  libmesh_error_msg("Error: Found multiple matching nodes in stitch_meshes");
1116 
1117  node_to_node_map[this_node_id] = other_node_id;
1118  other_to_this_node_map[other_node_id] = this_node_id;
1119 
1120  found_matching_nodes = true;
1121  }
1122  }
1123  }
1124  }
1125 
1126  // Build up the node_to_elems_map, using only one loop over other_mesh
1127  {
1128  MeshBase::element_iterator other_elem_it = other_mesh->elements_begin();
1129  MeshBase::element_iterator other_elem_end = other_mesh->elements_end();
1130  for (; other_elem_it != other_elem_end; ++other_elem_it)
1131  {
1132  Elem * el = *other_elem_it;
1133 
1134  // For each node on the element, find the corresponding node
1135  // on "this" Mesh, 'this_node_id', if it exists, and push
1136  // the current element ID back onto node_to_elems_map[this_node_id].
1137  // For that we will use the reverse mapping we created at
1138  // the same time as the forward mapping.
1139  for (unsigned n=0; n<el->n_nodes(); ++n)
1140  {
1141  dof_id_type other_node_id = el->node_id(n);
1142  std::map<dof_id_type, dof_id_type>::iterator it =
1143  other_to_this_node_map.find(other_node_id);
1144 
1145  if (it != other_to_this_node_map.end())
1146  {
1147  dof_id_type this_node_id = it->second;
1148  node_to_elems_map[this_node_id].push_back( el->id() );
1149  }
1150  }
1151  }
1152  }
1153 
1154  if (verbose)
1155  {
1156  libMesh::out << "In ReplicatedMesh::stitch_meshes:\n"
1157  << "Found " << node_to_node_map.size()
1158  << " matching nodes.\n"
1159  << std::endl;
1160  }
1161 
1162  if (enforce_all_nodes_match_on_boundaries)
1163  {
1164  std::size_t n_matching_nodes = node_to_node_map.size();
1165  std::size_t this_mesh_n_nodes = this_boundary_node_ids.size();
1166  std::size_t other_mesh_n_nodes = other_boundary_node_ids.size();
1167  if ((n_matching_nodes != this_mesh_n_nodes) || (n_matching_nodes != other_mesh_n_nodes))
1168  libmesh_error_msg("Error: We expected the number of nodes to match.");
1169  }
1170  }
1171  else
1172  {
1173  if (verbose)
1174  {
1175  libMesh::out << "Skip node merging in ReplicatedMesh::stitch_meshes:" << std::endl;
1176  }
1177  }
1178 
1179 
1180 
1181  dof_id_type node_delta = this->max_node_id();
1182  dof_id_type elem_delta = this->max_elem_id();
1183 
1184  // If other_mesh!=NULL, then we have to do a bunch of work
1185  // in order to copy it to this mesh
1186  if (this!=other_mesh)
1187  {
1188  LOG_SCOPE("stitch_meshes copying", "ReplicatedMesh");
1189 
1190  // need to increment node and element IDs of other_mesh before copying to this mesh
1191  MeshBase::node_iterator node_it = other_mesh->nodes_begin();
1192  MeshBase::node_iterator node_end = other_mesh->nodes_end();
1193  for (; node_it != node_end; ++node_it)
1194  {
1195  Node * nd = *node_it;
1196  dof_id_type new_id = nd->id() + node_delta;
1197  nd->set_id(new_id);
1198  }
1199 
1200  MeshBase::element_iterator elem_it = other_mesh->elements_begin();
1201  MeshBase::element_iterator elem_end = other_mesh->elements_end();
1202  for (; elem_it != elem_end; ++elem_it)
1203  {
1204  Elem * el = *elem_it;
1205  dof_id_type new_id = el->id() + elem_delta;
1206  el->set_id(new_id);
1207  }
1208 
1209  // Also, increment the node_to_node_map and node_to_elems_map
1210  std::map<dof_id_type, dof_id_type>::iterator node_map_it = node_to_node_map.begin();
1211  std::map<dof_id_type, dof_id_type>::iterator node_map_it_end = node_to_node_map.end();
1212  for ( ; node_map_it != node_map_it_end; ++node_map_it)
1213  {
1214  node_map_it->second += node_delta;
1215  }
1216  std::map<dof_id_type, std::vector<dof_id_type> >::iterator elem_map_it = node_to_elems_map.begin();
1217  std::map<dof_id_type, std::vector<dof_id_type> >::iterator elem_map_it_end = node_to_elems_map.end();
1218  for ( ; elem_map_it != elem_map_it_end; ++elem_map_it)
1219  {
1220  std::size_t n_elems = elem_map_it->second.size();
1221  for (std::size_t i=0; i<n_elems; i++)
1222  {
1223  (elem_map_it->second)[i] += elem_delta;
1224  }
1225  }
1226 
1227  // Copy mesh data. If we skip the call to find_neighbors(), the lists
1228  // of neighbors will be copied verbatim from the other mesh
1229  this->copy_nodes_and_elements(*other_mesh, skip_find_neighbors);
1230 
1231  // Decrement node IDs of mesh to return to original state
1232  node_it = other_mesh->nodes_begin();
1233  node_end = other_mesh->nodes_end();
1234  for (; node_it != node_end; ++node_it)
1235  {
1236  Node * nd = *node_it;
1237  dof_id_type new_id = nd->id() - node_delta;
1238  nd->set_id(new_id);
1239  }
1240 
1241  // Container to catch boundary IDs passed back from BoundaryInfo.
1242  std::vector<boundary_id_type> bc_ids;
1243 
1244  elem_it = other_mesh->elements_begin();
1245  elem_end = other_mesh->elements_end();
1246  for (; elem_it != elem_end; ++elem_it)
1247  {
1248  Elem * other_elem = *elem_it;
1249 
1250  // Decrement elem IDs of other_mesh to return it to original state
1251  dof_id_type new_id = other_elem->id() - elem_delta;
1252  other_elem->set_id(new_id);
1253  }
1254 
1255  // Copy BoundaryInfo from other_mesh too. We do this via the
1256  // list APIs rather than element-by-element for speed.
1257  BoundaryInfo & boundary = this->get_boundary_info();
1258  const BoundaryInfo & other_boundary = other_mesh->get_boundary_info();
1259 
1260  {
1261  std::vector<dof_id_type> node_id_list;
1262  std::vector<boundary_id_type> bc_id_list;
1263 
1264  other_boundary.build_node_list(node_id_list, bc_id_list);
1265  for (std::size_t i=0; i != node_id_list.size(); ++i)
1266  {
1267  const dof_id_type our_id = node_id_list[i] + node_delta;
1268  boundary.add_node(our_id, bc_id_list[i]);
1269  }
1270  }
1271 
1272  {
1273  std::vector<dof_id_type> elem_id_list;
1274  std::vector<unsigned short int> side_list;
1275  std::vector<boundary_id_type> bc_id_list;
1276 
1277  other_boundary.build_side_list(elem_id_list, side_list, bc_id_list);
1278  for (std::size_t i=0; i != elem_id_list.size(); ++i)
1279  {
1280  const dof_id_type our_id = elem_id_list[i] + elem_delta;
1281  boundary.add_side(our_id, side_list[i], bc_id_list[i]);
1282  }
1283  }
1284 
1285  {
1286  std::vector<dof_id_type> elem_id_list;
1287  std::vector<unsigned short int> edge_list;
1288  std::vector<boundary_id_type> bc_id_list;
1289 
1290  other_boundary.build_edge_list(elem_id_list, edge_list, bc_id_list);
1291  for (std::size_t i=0; i != elem_id_list.size(); ++i)
1292  {
1293  const dof_id_type our_id = elem_id_list[i] + elem_delta;
1294  boundary.add_edge(our_id, edge_list[i], bc_id_list[i]);
1295  }
1296  }
1297 
1298  {
1299  std::vector<dof_id_type> elem_id_list;
1300  std::vector<unsigned short int> shellface_list;
