41 void SFCPartitioner::partition_range(MeshBase & mesh,
42  MeshBase::element_iterator beg,
43  MeshBase::element_iterator end,
44  unsigned int n)
45 {
46  libmesh_assert_greater (n, 0);
47 
48  // Check for an easy return
49  if (n == 1)
50  {
51  this->single_partition_range (beg, end);
52  return;
53  }
54 
55  // What to do if the sfcurves library IS NOT present
56 #ifndef LIBMESH_HAVE_SFCURVES
57 
58  libmesh_here();
59  libMesh::err << "ERROR: The library has been built without" << std::endl
60  << "Space Filling Curve support. Using a linear" << std::endl
61  << "partitioner instead!" << std::endl;
62 
63  LinearPartitioner lp;
64  lp.partition_range (mesh, beg, end, n);
65 
66  // What to do if the sfcurves library IS present
67 #else
68 
69  LOG_SCOPE("partition_range()", "SFCPartitioner");
70 
71  // We don't yet support distributed meshes with this Partitioner
72  if (!mesh.is_serial())
73  libmesh_not_implemented();
74 
75  const dof_id_type n_range_elem = std::distance(beg, end);
76  const dof_id_type n_elem = mesh.n_elem();
77 
78  // The forward_map maps the range's element ids into a contiguous
79  // block of indices.
80  std::vector<dof_id_type> forward_map (n_elem, DofObject::invalid_id);
81 
82  // the reverse_map maps the contiguous ids back
83  // to active elements
84  std::vector<Elem *> reverse_map (n_range_elem, libmesh_nullptr);
85 
86  std::vector<double> x (n_range_elem);
87  std::vector<double> y (n_range_elem);
88  std::vector<double> z (n_range_elem);
89  std::vector<int> table (n_range_elem);
90 
91  // Map the range's element ids into a contiguous range.
92  {
93  MeshBase::element_iterator it = beg;
94  dof_id_type el_num = 0;
95 
96  for (; it != end; ++it)
97  {
98  libmesh_assert_less ((*it)->id(), forward_map.size());
99  libmesh_assert_less (el_num, reverse_map.size());
100 
101  forward_map[(*it)->id()] = el_num;
102  reverse_map[el_num] = *it;
103  el_num++;
104  }
105  libmesh_assert_equal_to (el_num, n_range_elem);
106  }
107 
108  // Get the centroid for each range element.
109  {
110  MeshBase::element_iterator it = beg;
111 
112  for (; it != end; ++it)
113  {
114  const Elem * elem = *it;
115 
116  libmesh_assert_less (elem->id(), forward_map.size());
117 
118  const Point p = elem->centroid();
119 
120  x[forward_map[elem->id()]] = p(0);
121  y[forward_map[elem->id()]] = p(1);
122  z[forward_map[elem->id()]] = p(2);
123  }
124  }
125 
126  // We need an integer reference to pass to the Sfc interface.
127  int size = static_cast<int>(n_range_elem);
128 
129  // build the space-filling curve
130  if (_sfc_type == "Hilbert")
131  Sfc::hilbert (&x[0], &y[0], &z[0], &size, &table[0]);
132 
133  else if (_sfc_type == "Morton")
134  Sfc::morton (&x[0], &y[0], &z[0], &size, &table[0]);
135 
136  else
137  {
138  libmesh_here();
139  libMesh::err << "ERROR: Unknown type: " << _sfc_type << std::endl
140  << " Valid types are" << std::endl
141  << " \"Hilbert\"" << std::endl
142  << " \"Morton\"" << std::endl
143  << " " << std::endl
144  << "Proceeding with a Hilbert curve." << std::endl;
145 
146  Sfc::hilbert (&x[0], &y[0], &z[0], &size, &table[0]);
147  }
148 
149 
150  // Assign the partitioning to the range elements
151  {
152  // {
153  // std::ofstream out ("sfc.dat");
154  // out << "variables=x,y,z" << std::endl;
155  // out << "zone f=point" << std::endl;
156  // for (unsigned int i=0; i<n_range_elem; i++)
157  // out << x[i] << " " << y[i] << " " << z[i] << std::endl;
158  // }
159 
160  const dof_id_type blksize = (n_range_elem + n - 1) / n;
161 
162  for (dof_id_type i=0; i<n_range_elem; i++)
163  {
164  libmesh_assert_less (static_cast<unsigned int>(table[i] - 1), reverse_map.size());
165 
166  Elem * elem = reverse_map[table[i] - 1];
167 
168  elem->processor_id() = cast_int<processor_id_type>(i/blksize);
169  }
170  }
171 
172 #endif
173 }
