176 template <typename Output, typename OutputGradient>
177 inline
178 void
179 ParsedFunction<Output,OutputGradient>::reparse (const std::string & expression)
180 {
181  variables = "x";
182 #if LIBMESH_DIM > 1
183  variables += ",y";
184 #endif
185 #if LIBMESH_DIM > 2
186  variables += ",z";
187 #endif
188  variables += ",t";
189 
190  // If additional vars were passed, append them to the string
191  // that we send to the function parser. Also add them to the
192  // end of our spacetime vector
193  for (unsigned int i=0; i < _additional_vars.size(); ++i)
194  {
195  variables += "," + _additional_vars[i];
196  // Initialize extra variables to the vector passed in or zero
197  // Note: The initial_vals vector can be shorter than the additional_vars vector
198  _spacetime[LIBMESH_DIM+1 + i] =
199  (i < _initial_vals.size()) ? _initial_vals[i] : 0;
200  }
201 
202  this->partial_reparse(expression);
203 
204  this->_is_time_dependent = this->expression_is_time_dependent(expression);
205 }

312 template <typename Output, typename OutputGradient>
313 inline
314 Output
315 ParsedFunction<Output,OutputGradient>::get_inline_value (const std::string & inline_var_name) const
316 {
317  libmesh_assert_greater (_subexpressions.size(), 0);
318 
319 #ifndef NDEBUG
320  bool found_var_name = false;
321 #endif
322  Output old_var_value;
323 
324  for (unsigned int s=0; s != _subexpressions.size(); ++s)
325  {
326  const std::string & subexpression = _subexpressions[s];
327  const std::size_t varname_i =
328  find_name(inline_var_name, subexpression);
329  if (varname_i == std::string::npos)
330  continue;
331 
332  const std::size_t assignment_i =
333  subexpression.find(":", varname_i+1);
334 
335  libmesh_assert_not_equal_to(assignment_i, std::string::npos);
336 
337  libmesh_assert_equal_to(subexpression[assignment_i+1], '=');
338  for (unsigned int i = varname_i+1; i != assignment_i; ++i)
339  libmesh_assert_equal_to(subexpression[i], ' ');
340 
341  std::size_t end_assignment_i =
342  subexpression.find(";", assignment_i+1);
343 
344  libmesh_assert_not_equal_to(end_assignment_i, std::string::npos);
345 
346  std::string new_subexpression =
347  subexpression.substr(0, end_assignment_i+1) +
348  inline_var_name;
349 
350 #ifdef LIBMESH_HAVE_FPARSER
351  // Parse and evaluate the new subexpression.
352  // Add the same constants as we used originally.
353  FunctionParserADBase<Output> fp;
354  fp.AddConstant("NaN", std::numeric_limits<Real>::quiet_NaN());
355  fp.AddConstant("pi", std::acos(Real(-1)));
356  fp.AddConstant("e", std::exp(Real(1)));
357  if (fp.Parse(new_subexpression, variables) != -1) // -1 for success
358  libmesh_error_msg
359  ("ERROR: FunctionParser is unable to parse modified expression: "
360  << new_subexpression << '\n' << fp.ErrorMsg());
361 
362  Output new_var_value = this->eval(fp, new_subexpression, 0);
363 #ifdef NDEBUG
364  return new_var_value;
365 #else
366  if (found_var_name)
367  {
368  libmesh_assert_equal_to(old_var_value, new_var_value);
369  }
370  else
371  {
372  old_var_value = new_var_value;
373  found_var_name = true;
374  }
375 #endif
376 
377 #else
378  libmesh_error_msg("ERROR: This functionality requires fparser!");
379 #endif
380  }
381 
382  libmesh_assert(found_var_name);
383  return old_var_value;
384 }

388 inline
389 void
390 ParsedFunction<Output,OutputGradient>::set_inline_value (const std::string & inline_var_name,
391  Output newval)
392 {
393  libmesh_assert_greater (_subexpressions.size(), 0);
394 
395 #ifndef NDEBUG
396  bool found_var_name = false;
397 #endif
398  for (unsigned int s=0; s != _subexpressions.size(); ++s)
399  {
400  const std::string & subexpression = _subexpressions[s];
401  const std::size_t varname_i =
402  find_name(inline_var_name, subexpression);
403  if (varname_i == std::string::npos)
404  continue;
405 
406 #ifndef NDEBUG
407  found_var_name = true;
408 #endif
409  const std::size_t assignment_i =
410  subexpression.find(":", varname_i+1);
411 
412  libmesh_assert_not_equal_to(assignment_i, std::string::npos);
413 
414  libmesh_assert_equal_to(subexpression[assignment_i+1], '=');
415  for (unsigned int i = varname_i+1; i != assignment_i; ++i)
416  libmesh_assert_equal_to(subexpression[i], ' ');
417 
418  std::size_t end_assignment_i =
419  subexpression.find(";", assignment_i+1);
420 
421  libmesh_assert_not_equal_to(end_assignment_i, std::string::npos);
422 
423  std::ostringstream new_subexpression;
424  new_subexpression << subexpression.substr(0, assignment_i+2)
425  << std::setprecision(std::numeric_limits<Output>::digits10+2)
426 #ifdef LIBMESH_USE_COMPLEX_NUMBERS
427  << '(' << newval.real() << '+'
428  << newval.imag() << 'i' << ')'
429 #else
430  << newval
431 #endif
432  << subexpression.substr(end_assignment_i,
433  std::string::npos);
434  _subexpressions[s] = new_subexpression.str();
435  }
436 
437  libmesh_assert(found_var_name);
438 
439  std::string new_expression;
440 
441  for (unsigned int s=0; s != _subexpressions.size(); ++s)
442  {
443  new_expression += '{';
444  new_expression += _subexpressions[s];
445  new_expression += '}';
446  }
447 
448  this->partial_reparse(new_expression);
449 }

452 template <typename Output, typename OutputGradient>
453 inline
454 void
455 ParsedFunction<Output,OutputGradient>::partial_reparse (const std::string & expression)
456 {
457  _expression = expression;
458  _subexpressions.clear();
459  parsers.clear();
460 
461  size_t nextstart = 0, end = 0;
462 
463  while (end != std::string::npos)
464  {
465  // If we're past the end of the string, we can't make any more
466  // subparsers
467  if (nextstart >= expression.size())
468  break;
469 
470  // If we're at the start of a brace delimited section, then we
471  // parse just that section:
472  if (expression[nextstart] == '{')
473  {
474  nextstart++;
475  end = expression.find('}', nextstart);
476  }
477  // otherwise we parse the whole thing
478  else
479  end = std::string::npos;
480 
481  // We either want the whole end of the string (end == npos) or
482  // a substring in the middle.
483  _subexpressions.push_back
484  (expression.substr(nextstart, (end == std::string::npos) ?
485  std::string::npos : end - nextstart));
486 
487  // fparser can crash on empty expressions
488  if (_subexpressions.back().empty())
489  libmesh_error_msg("ERROR: FunctionParser is unable to parse empty expression.\n");
490 
491  // Parse (and optimize if possible) the subexpression.
492  // Add some basic constants, to Real precision.
493  FunctionParserADBase<Output> fp;
494  fp.AddConstant("NaN", std::numeric_limits<Real>::quiet_NaN());
495  fp.AddConstant("pi", std::acos(Real(-1)));
496  fp.AddConstant("e", std::exp(Real(1)));
497  if (fp.Parse(_subexpressions.back(), variables) != -1) // -1 for success
498  libmesh_error_msg
499  ("ERROR: FunctionParser is unable to parse expression: "
500  << _subexpressions.back() << '\n' << fp.ErrorMsg());
501 
502  // use of derivatives is optional. suppress error output on the console
503  // use the has_derivatives() method to check if AutoDiff was successful.
504  // also enable immediate optimization
505  fp.SetADFlags(FunctionParserADBase<Output>::ADSilenceErrors |
506  FunctionParserADBase<Output>::ADAutoOptimize);
507 
508  // optimize original function
509  fp.Optimize();
510  parsers.push_back(fp);
511 
512  // generate derivatives through automatic differentiation
513  FunctionParserADBase<Output> dx_fp(fp);
514  if (dx_fp.AutoDiff("x") != -1) // -1 for success
515  _valid_derivatives = false;
516  dx_parsers.push_back(dx_fp);
517 #if LIBMESH_DIM > 1
518  FunctionParserADBase<Output> dy_fp(fp);
519  if (dy_fp.AutoDiff("y") != -1) // -1 for success
520  _valid_derivatives = false;
521  dy_parsers.push_back(dy_fp);
522 #endif
523 #if LIBMESH_DIM > 2
524  FunctionParserADBase<Output> dz_fp(fp);
525  if (dz_fp.AutoDiff("z") != -1) // -1 for success
526  _valid_derivatives = false;
527  dz_parsers.push_back(dz_fp);
528 #endif
529  FunctionParserADBase<Output> dt_fp(fp);
530  if (dt_fp.AutoDiff("t") != -1) // -1 for success
531  _valid_derivatives = false;
532  dt_parsers.push_back(dt_fp);
533 
534  // If at end, use nextstart=maxSize. Else start at next
535  // character.
536  nextstart = (end == std::string::npos) ?
537  std::string::npos : end + 1;
538  }

542 template <typename Output, typename OutputGradient>
543 inline
544 std::size_t
545 ParsedFunction<Output,OutputGradient>::find_name (const std::string & varname,
546  const std::string & expr) const
547 {
548  const std::size_t namesize = varname.size();
549  std::size_t varname_i = expr.find(varname);
550 
551  while ((varname_i != std::string::npos) &&
552  (((varname_i > 0) &&
553  (std::isalnum(expr[varname_i-1]) ||
554  (expr[varname_i-1] == '_'))) ||
555  ((varname_i+namesize < expr.size()) &&
556  (std::isalnum(expr[varname_i+namesize]) ||
557  (expr[varname_i+namesize] == '_')))))
558  {
559  varname_i = expr.find(varname, varname_i+1);
560  }
561 
562  return varname_i;
563 }

602 Output
603 ParsedFunction<Output,OutputGradient>::eval (FunctionParserADBase<Output> & parser,
604  const std::string & libmesh_dbg_var(function_name),
605  unsigned int libmesh_dbg_var(component_idx)) const
606 {
607 #ifndef NDEBUG
608  Output result = parser.Eval(&_spacetime[0]);
609  int error_code = parser.EvalError();
610  if (error_code)
611  {
612  libMesh::err << "ERROR: FunctionParser is unable to evaluate component "
613  << component_idx
614  << " of expression '"
615  << function_name
616  << "' with arguments:\n";
617  for (unsigned int j=0; j<_spacetime.size(); ++j)
618  libMesh::err << '\t' << _spacetime[j] << '\n';
619  libMesh::err << '\n';
620 
621  // Currently no API to report error messages, we'll do it manually
622  std::string error_message = "Reason: ";
623 
624  switch (error_code)
625  {
626  case 1:
627  error_message += "Division by zero";
628  break;
629  case 2:
630  error_message += "Square Root error (negative value)";
631  break;
632  case 3:
633  error_message += "Log error (negative value)";
634  break;
635  case 4:
636  error_message += "Trigonometric error (asin or acos of illegal value)";
637  break;
638  case 5:
639  error_message += "Maximum recursion level reached";
640  break;
641  default:
642  error_message += "Unknown";
643  break;
644  }
645  libmesh_error_msg(error_message);
646  }
647 
648  return result;
649 #else
650  return parser.Eval(&_spacetime[0]);
651 #endif
652 }
