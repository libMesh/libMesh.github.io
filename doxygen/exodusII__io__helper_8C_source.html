330 void ExodusII_IO_Helper::open(const char * filename, bool read_only)
331 {
332  // Version of Exodus you are using
333  float ex_version = 0.;
334 
335  int comp_ws = 0;
336 
337  if (_single_precision)
338  comp_ws = cast_int<int>(sizeof(float));
339 
340  // Fall back on double precision when necessary since ExodusII
341  // doesn't seem to support long double
342  else
343  comp_ws = cast_int<int>(std::min(sizeof(Real), sizeof(double)));
344 
345  // Word size in bytes of the floating point data as they are stored
346  // in the ExodusII file. "If this argument is 0, the word size of the
347  // floating point data already stored in the file is returned"
348  int io_ws = 0;
349 
350  ex_id = exII::ex_open(filename,
351  read_only ? EX_READ : EX_WRITE,
352  &comp_ws,
353  &io_ws,
354  &ex_version);
355 
356  std::string err_msg = std::string("Error opening ExodusII mesh file: ") + std::string(filename);
357  EX_CHECK_ERR(ex_id, err_msg);
358  if (verbose) libMesh::out << "File opened successfully." << std::endl;
359 
360  if (read_only)
361  opened_for_reading = true;
362  else
363  opened_for_writing = true;
364 
365  current_filename = std::string(filename);
366 }

400 void ExodusII_IO_Helper::read_qa_records()
401 {
402  // The QA records are four MAX_STR_LENGTH-byte character strings.
403  int num_qa_rec =
404  this->inquire(exII::EX_INQ_QA, "Error retrieving number of QA records");
405 
406  if (verbose)
407  libMesh::out << "Found "
408  << num_qa_rec
409  << " QA record(s) in the Exodus file."
410  << std::endl;
411 
412  if (num_qa_rec > 0)
413  {
414  // How to dynamically allocate an array of fixed-size char * arrays in C++.
415  // http://stackoverflow.com/questions/8529359/creating-a-dynamic-sized-array-of-fixed-sized-int-arrays-in-c
416  typedef char * inner_array_t[4];
417  inner_array_t * qa_record = new inner_array_t[num_qa_rec];
418 
419  for (int i=0; i<num_qa_rec; i++)
420  for (int j=0; j<4; j++)
421  qa_record[i][j] = new char[MAX_STR_LENGTH+1];
422 
423  ex_err = exII::ex_get_qa (ex_id, qa_record);
424  EX_CHECK_ERR(ex_err, "Error reading the QA records.");
425 
426  // Print the QA records
427  if (verbose)
428  {
429  for (int i=0; i<num_qa_rec; i++)
430  {
431  libMesh::out << "QA Record: " << i << std::endl;
432  for (int j=0; j<4; j++)
433  libMesh::out << qa_record[i][j] << std::endl;
434  }
435  }
436 
437 
438  // Clean up dynamically-allocated memory
439  for (int i=0; i<num_qa_rec; i++)
440  for (int j=0; j<4; j++)
441  delete [] qa_record[i][j];
442 
443  delete [] qa_record;
444  }
445 }

1111 void ExodusII_IO_Helper::initialize(std::string str_title, const MeshBase & mesh, bool use_discontinuous)
1112 {

1114  unsigned int n_active_elem = mesh.n_active_elem();
1115 
1116  if ((_run_only_on_proc0) && (this->processor_id() != 0))
1117  return;
1118 
1119  // If _write_as_dimension is nonzero, use it to set num_dim in the Exodus file.
1120  if (_write_as_dimension)
1121  num_dim = _write_as_dimension;
1122  else if (_use_mesh_dimension_instead_of_spatial_dimension)
1123  num_dim = mesh.mesh_dimension();
1124  else
1125  num_dim = mesh.spatial_dimension();
1126 
1127  num_elem = mesh.n_elem();
1128 
1129  if (!use_discontinuous)
1130  {
1131  // Don't rely on mesh.n_nodes() here. If nodes have been
1132  // deleted recently, it will be incorrect.
1133  num_nodes = cast_int<int>(std::distance(mesh.nodes_begin(),
1134  mesh.nodes_end()));
1135  }
1136  else
1137  {
1138  MeshBase::const_element_iterator it = mesh.active_elements_begin();
1139  const MeshBase::const_element_iterator end = mesh.active_elements_end();
1140  for (; it!=end; ++it)
1141  num_nodes += (*it)->n_nodes();
1142  }
1143 
1144  std::vector<boundary_id_type> unique_side_boundaries;
1145  std::vector<boundary_id_type> unique_node_boundaries;
1146 
1147  mesh.get_boundary_info().build_side_boundary_ids(unique_side_boundaries);
1148  {
1149  // Add shell face boundaries to the list of side boundaries, since ExodusII
1150  // treats these the same way.
1151  std::vector<boundary_id_type> shellface_boundaries;
1152  mesh.get_boundary_info().build_shellface_boundary_ids(shellface_boundaries);
1153  for (std::size_t i=0; i<shellface_boundaries.size(); i++)
1154  unique_side_boundaries.push_back(shellface_boundaries[i]);
1155  }
1156  mesh.get_boundary_info().build_node_boundary_ids(unique_node_boundaries);
1157 
1158  num_side_sets = cast_int<int>(unique_side_boundaries.size());
1159  num_node_sets = cast_int<int>(unique_node_boundaries.size());
1160 
1161  //loop through element and map between block and element vector
1162  std::map<subdomain_id_type, std::vector<unsigned int> > subdomain_map;
1163 
1164  MeshBase::const_element_iterator it = mesh.active_elements_begin();
1165  const MeshBase::const_element_iterator end = mesh.active_elements_end();
1166  for (; it!=end; ++it)
1167  {
1168  const Elem * elem = *it;
1169  subdomain_id_type cur_subdomain = elem->subdomain_id();
1170 
1171  subdomain_map[cur_subdomain].push_back(elem->id());
1172  }
1173  num_elem_blk = cast_int<int>(subdomain_map.size());
1174 
1175  if (str_title.size() > MAX_LINE_LENGTH)
1176  {
1177  libMesh::err << "Warning, Exodus files cannot have titles longer than "
1178  << MAX_LINE_LENGTH
1179  << " characters. Your title will be truncated."
1180  << std::endl;
1181  str_title.resize(MAX_LINE_LENGTH);
1182  }
1183 
1184  ex_err = exII::ex_put_init(ex_id,

1872 void ExodusII_IO_Helper::write_element_values(const MeshBase & mesh, const std::vector<Real> & values, int timestep)
1873 {
1874  if ((_run_only_on_proc0) && (this->processor_id() != 0))
1875  return;
1876 
1877  // Loop over the element blocks and write the data one block at a time
1878  std::map<unsigned int, std::vector<unsigned int> > subdomain_map;
1879 
1880  // Ask the file how many element vars it has, store it in the num_elem_vars variable.
1881  ex_err = exII::ex_get_var_param(ex_id, "e", &num_elem_vars);
1882  EX_CHECK_ERR(ex_err, "Error reading number of elemental variables.");
1883 
1884  MeshBase::const_element_iterator mesh_it = mesh.active_elements_begin();
1885  const MeshBase::const_element_iterator end = mesh.active_elements_end();
1886 
1887  // loop through element and map between block and element vector
1888  for (; mesh_it!=end; ++mesh_it)
1889  {
1890  const Elem * elem = *mesh_it;
1891  subdomain_map[elem->subdomain_id()].push_back(elem->id());
1892  }
1893 
1894  // Use mesh.n_elem() to access into the values vector rather than
1895  // the number of elements the Exodus writer thinks the mesh has,
1896  // which may not include inactive elements.
1897  dof_id_type n_elem = mesh.n_elem();
1898 
1899  // For each variable, create a 'data' array which holds all the elemental variable
1900  // values *for a given block* on this processor, then write that data vector to file
1901  // before moving onto the next block.
1902  for (unsigned int i=0; i<static_cast<unsigned>(num_elem_vars); ++i)
1903  {
1904  // The size of the subdomain map is the number of blocks.
1905  std::map<unsigned int, std::vector<unsigned int> >::iterator it = subdomain_map.begin();
1906 
1907  for (unsigned int j=0; it!=subdomain_map.end(); ++it, ++j)
1908  {
1909  const std::vector<unsigned int> & elem_nums = (*it).second;
1910  const unsigned int num_elems_this_block =
1911  cast_int<unsigned int>(elem_nums.size());
1912  std::vector<Real> data(num_elems_this_block);
1913 
1914  for (unsigned int k=0; k<num_elems_this_block; ++k)
1915  data[k] = values[i*n_elem + elem_nums[k]];
1916 
1917  if (_single_precision)
1918  {
1919  std::vector<float> cast_data(data.begin(), data.end());
1920 
1921  ex_err = exII::ex_put_elem_var(ex_id,
1922  timestep,
1923  i+1,
1924  this->get_block_id(j),
1925  num_elems_this_block,
1926  &cast_data[0]);
1927  }
1928  else
1929  {
1930  ex_err = exII::ex_put_elem_var(ex_id,
1931  timestep,
1932  i+1,
1933  this->get_block_id(j),
1934  num_elems_this_block,
1935  &data[0]);
1936  }
1937  EX_CHECK_ERR(ex_err, "Error writing element values.");
1938  }
1939  }
1940 
1941  ex_err = exII::ex_update(ex_id);
1942  EX_CHECK_ERR(ex_err, "Error flushing buffers to file.");
1943 }
