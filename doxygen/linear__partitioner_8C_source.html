28 void LinearPartitioner::partition_range(MeshBase & mesh,
29  MeshBase::element_iterator it,
30  MeshBase::element_iterator end,
31  const unsigned int n)
32 {
33  libmesh_assert_greater (n, 0);
34 
35  // Check for an easy return
36  if (n == 1)
37  {
38  this->single_partition_range (it, end);
39  return;
40  }
41 
42  // Create a simple linear partitioning
43  LOG_SCOPE ("partition_range()", "LinearPartitioner");
44 
45  const bool mesh_is_serial = mesh.is_serial();
46 
47  // This has to be an ordered set
48  std::set<dof_id_type> element_ids;
49 
50  // If we're on a serialized mesh, we know our range is the same on
51  // every processor.
52  if (mesh_is_serial)
53  {
54  const dof_id_type blksize = std::distance(it, end) / n;
55 
56  dof_id_type e = 0;
57  for ( ; it != end; ++it)
58  {
59  Elem * elem = *it;
60  if ((e/blksize) < n)
61  elem->processor_id() = cast_int<processor_id_type>(e/blksize);
62  else
63  elem->processor_id() = 0;
64 
65  e++;
66  }
67  }
68  // If we're on a replicated mesh, we might have different ranges on
69  // different processors, and we'll need to gather the full range.
70  //
71  // This is not an efficient way to do this, but if you want to be
72  // efficient then you want to be using a different partitioner to
73  // begin with; LinearPartitioner is more for debugging than
74  // performance.
75  else
76  {
77  for (; it != end; ++it)
78  element_ids.insert((*it)->id());
79 
80  mesh.comm().set_union(element_ids);
81 
82  const dof_id_type blksize = element_ids.size();
83 
84  dof_id_type e = 0;
85  for (auto eid : element_ids)
86  {
87  Elem * elem = mesh.query_elem_ptr(eid);
88  if (elem)
89  {
90  if ((e/blksize) < n)
91  elem->processor_id() = cast_int<processor_id_type>(e/blksize);
92  else
93  elem->processor_id() = 0;
94  }
95 
96  e++;
97  }
98  }
99 }
