105 void ImplicitSystem::init_matrices ()
106 {
107  libmesh_assert(matrix);
108 
109  // Check for quick return in case the system matrix
110  // (and by extension all the matrices) has already
111  // been initialized
112  if (matrix->initialized())
113  return;
114 
115  // Get a reference to the DofMap
116  DofMap & dof_map = this->get_dof_map();
117 
118  // no chance to add other matrices
119  _can_add_matrices = false;
120 
121  // Tell the matrices about the dof map, and vice versa
122  for (matrices_iterator pos = _matrices.begin();
123  pos != _matrices.end(); ++pos)
124  {
125  SparseMatrix<Number> & m = *(pos->second);
126  libmesh_assert (!m.initialized());
127 
128  // We want to allow repeated init() on systems, but we don't
129  // want to attach the same matrix to the DofMap twice
130  if (!dof_map.is_attached(m))
131  dof_map.attach_matrix (m);
132  }
133 
134  // Compute the sparsity pattern for the current
135  // mesh and DOF distribution. This also updates
136  // additional matrices, \p DofMap now knows them
137  dof_map.compute_sparsity (this->get_mesh());
138 
139  // Initialize matrices
140  for (matrices_iterator pos = _matrices.begin();
141  pos != _matrices.end(); ++pos)
142  pos->second->init ();
143 
144  // Set the additional matrices to 0.
145  for (matrices_iterator pos = _matrices.begin();
146  pos != _matrices.end(); ++pos)
147  pos->second->zero ();
148 }

303 std::pair<unsigned int, Real>
304 ImplicitSystem::sensitivity_solve (const ParameterVector & parameters)
305 {
306  // Log how long the linear solve takes.
307  LOG_SCOPE("sensitivity_solve()", "ImplicitSystem");
308 
309  // The forward system should now already be solved.
310  // Now assemble the corresponding sensitivity system.
311 
312  if (this->assemble_before_solve)
313  {
314  // Build the Jacobian
315  this->assembly(false, true);
316  this->matrix->close();
317 
318  // Reset and build the RHS from the residual derivatives
319  this->assemble_residual_derivatives(parameters);
320  }
321 
322  // The sensitivity problem is linear
323  LinearSolver<Number> * linear_solver = this->get_linear_solver();
324 
325  // Our iteration counts and residuals will be sums of the individual
326  // results
327  std::pair<unsigned int, Real> solver_params =
328  this->get_linear_solve_parameters();
329  std::pair<unsigned int, Real> totalrval = std::make_pair(0,0.0);
330 
331  // Solve the linear system.
332  SparseMatrix<Number> * pc = this->request_matrix("Preconditioner");
333  for (unsigned int p=0; p != parameters.size(); ++p)
334  {
335  std::pair<unsigned int, Real> rval =
336  linear_solver->solve (*matrix, pc,
337  this->add_sensitivity_solution(p),
338  this->get_sensitivity_rhs(p),
339  solver_params.second,
340  solver_params.first);
341 
342  totalrval.first += rval.first;
343  totalrval.second += rval.second;
344  }
345 
346  // The linear solver may not have fit our constraints exactly
347 #ifdef LIBMESH_ENABLE_CONSTRAINTS
348  for (unsigned int p=0; p != parameters.size(); ++p)
349  this->get_dof_map().enforce_constraints_exactly
350  (*this, &this->get_sensitivity_solution(p),
351  /* homogeneous = */ true);
352 #endif
353 
354  this->release_linear_solver(linear_solver);
355 
356  return totalrval;
357 }

414 std::pair<unsigned int, Real>
415 ImplicitSystem::weighted_sensitivity_adjoint_solve (const ParameterVector & parameters_in,
416  const ParameterVector & weights,
417  const QoISet & qoi_indices)
418 {
419  // Log how long the linear solve takes.
420  LOG_SCOPE("weighted_sensitivity_adjoint_solve()", "ImplicitSystem");
421 
422  // We currently get partial derivatives via central differencing
423  const Real delta_p = TOLERANCE;
424 
425  ParameterVector & parameters =
426  const_cast<ParameterVector &>(parameters_in);
427 
428  // The forward system should now already be solved.
429  // The adjoint system should now already be solved.
430  // Now we're assembling a weighted sum of adjoint-adjoint systems:
431  //
432  // dR/du (u, sum_l(w_l*z^l)) = sum_l(w_l*(Q''_ul - R''_ul (u, z)))
433 
434  // FIXME: The derivation here does not yet take adjoint boundary
435  // conditions into account.
436  for (unsigned int i=0; i != this->qoi.size(); ++i)
437  if (qoi_indices.has_index(i))
438  libmesh_assert(!this->get_dof_map().has_adjoint_dirichlet_boundaries(i));
439 
440  // We'll assemble the rhs first, because the R'' term will require
441  // perturbing the jacobian
442 
443  // We'll use temporary rhs vectors, because we haven't (yet) found
444  // any good reasons why users might want to save these:
445 
446  std::vector<NumericVector<Number> *> temprhs(this->qoi.size());
447  for (unsigned int i=0; i != this->qoi.size(); ++i)
448  if (qoi_indices.has_index(i))
449  temprhs[i] = this->rhs->zero_clone().release();
450 
451  // We approximate the _l partial derivatives via a central
452  // differencing perturbation in the w_l direction:
453  //
454  // sum_l(w_l*v_l) ~= (v(p + dp*w_l*e_l) - v(p - dp*w_l*e_l))/(2*dp)
455 
456  // PETSc doesn't implement SGEMX, so neither does NumericVector,
457  // so we want to avoid calculating f -= R'*z. We'll thus evaluate
458  // the above equation by first adding -v(p+dp...), then multiplying
459  // the intermediate result vectors by -1, then adding -v(p-dp...),
460  // then finally dividing by 2*dp.
461 
462  ParameterVector oldparameters, parameterperturbation;
463  parameters.deep_copy(oldparameters);
464  weights.deep_copy(parameterperturbation);
465  parameterperturbation *= delta_p;
466  parameters += parameterperturbation;
467 
468  this->assembly(false, true);
469  this->matrix->close();
470 
471  // Take the discrete adjoint, so that we can calculate R_u(u,z) with
472  // a matrix-vector product of R_u and z.
473  matrix->get_transpose(*matrix);
474 
475  this->assemble_qoi_derivative(qoi_indices,
476  /* include_liftfunc = */ false,
477  /* apply_constraints = */ true);
478  for (unsigned int i=0; i != this->qoi.size(); ++i)
479  if (qoi_indices.has_index(i))
480  {
481  this->get_adjoint_rhs(i).close();
482  *(temprhs[i]) -= this->get_adjoint_rhs(i);
483  this->matrix->vector_mult_add(*(temprhs[i]), this->get_adjoint_solution(i));
484  *(temprhs[i]) *= -1.0;
485  }
486 
487  oldparameters.value_copy(parameters);
488  parameterperturbation *= -1.0;
489  parameters += parameterperturbation;
490 
491  this->assembly(false, true);
492  this->matrix->close();
493  matrix->get_transpose(*matrix);
494 
495  this->assemble_qoi_derivative(qoi_indices,
496  /* include_liftfunc = */ false,
497  /* apply_constraints = */ true);
498  for (unsigned int i=0; i != this->qoi.size(); ++i)
499  if (qoi_indices.has_index(i))
500  {
501  this->get_adjoint_rhs(i).close();
502  *(temprhs[i]) -= this->get_adjoint_rhs(i);
503  this->matrix->vector_mult_add(*(temprhs[i]), this->get_adjoint_solution(i));
504  *(temprhs[i]) /= (2.0*delta_p);
505  }
506 
507  // Finally, assemble the jacobian at the non-perturbed parameter
508  // values. Ignore assemble_before_solve; if we had a good
509  // non-perturbed matrix before we've already overwritten it.
510  oldparameters.value_copy(parameters);
511 
512  // if (this->assemble_before_solve)
513  {
514  // Build the Jacobian
515  this->assembly(false, true);
516  this->matrix->close();
517 
518  // Take the discrete adjoint
519  matrix->get_transpose(*matrix);
520  }
521 
522  // The weighted adjoint-adjoint problem is linear
523  LinearSolver<Number> * linear_solver = this->get_linear_solver();
524 
525  // Our iteration counts and residuals will be sums of the individual
526  // results
527  std::pair<unsigned int, Real> solver_params =
528  this->get_linear_solve_parameters();
529  std::pair<unsigned int, Real> totalrval = std::make_pair(0,0.0);
530 
531  for (unsigned int i=0; i != this->qoi.size(); ++i)
532  if (qoi_indices.has_index(i))
533  {
534  const std::pair<unsigned int, Real> rval =
535  linear_solver->solve (*matrix, this->add_weighted_sensitivity_adjoint_solution(i),
536  *(temprhs[i]),
537  solver_params.second,
538  solver_params.first);
539 
540  totalrval.first += rval.first;
541  totalrval.second += rval.second;
542  }
543 
544  this->release_linear_solver(linear_solver);
545 
546  for (unsigned int i=0; i != this->qoi.size(); ++i)
547  if (qoi_indices.has_index(i))
548  delete temprhs[i];
549 
550  // The linear solver may not have fit our constraints exactly
551 #ifdef LIBMESH_ENABLE_CONSTRAINTS
552  for (unsigned int i=0; i != this->qoi.size(); ++i)
553  if (qoi_indices.has_index(i))
554  this->get_dof_map().enforce_constraints_exactly
555  (*this, &this->get_weighted_sensitivity_adjoint_solution(i),
556  /* homogeneous = */ true);
557 #endif
558 
559  return totalrval;
560 }

564 std::pair<unsigned int, Real>
565 ImplicitSystem::weighted_sensitivity_solve (const ParameterVector & parameters_in,
566  const ParameterVector & weights)
567 {
568  // Log how long the linear solve takes.
569  LOG_SCOPE("weighted_sensitivity_solve()", "ImplicitSystem");
570 
571  // We currently get partial derivatives via central differencing
572  const Real delta_p = TOLERANCE;
573 
574  ParameterVector & parameters =
575  const_cast<ParameterVector &>(parameters_in);
576 
577  // The forward system should now already be solved.
578 
579  // Now we're assembling a weighted sum of sensitivity systems:
580  //
581  // dR/du (u, v)(sum(w_l*u'_l)) = -sum_l(w_l*R'_l (u, v)) forall v
582 
583  // We'll assemble the rhs first, because the R' term will require
584  // perturbing the system, and some applications may not be able to
585  // assemble a perturbed residual without simultaneously constructing
586  // a perturbed jacobian.
587 
588  // We approximate the _l partial derivatives via a central
589  // differencing perturbation in the w_l direction:
590  //
591  // sum_l(w_l*v_l) ~= (v(p + dp*w_l*e_l) - v(p - dp*w_l*e_l))/(2*dp)
592 
593  ParameterVector oldparameters, parameterperturbation;
594  parameters.deep_copy(oldparameters);
595  weights.deep_copy(parameterperturbation);
596  parameterperturbation *= delta_p;
597  parameters += parameterperturbation;
598 
599  this->assembly(true, false, true);
600  this->rhs->close();
601 
602  UniquePtr<NumericVector<Number> > temprhs = this->rhs->clone();
603 
604  oldparameters.value_copy(parameters);
605  parameterperturbation *= -1.0;
606  parameters += parameterperturbation;
607 
608  this->assembly(true, false, true);
609  this->rhs->close();
610 
611  *temprhs -= *(this->rhs);
612  *temprhs /= (2.0*delta_p);
613 
614  // Finally, assemble the jacobian at the non-perturbed parameter
615  // values
616  oldparameters.value_copy(parameters);
617 
618  // Build the Jacobian
619  this->assembly(false, true);
620  this->matrix->close();
621 
622  // The weighted sensitivity problem is linear
623  LinearSolver<Number> * linear_solver = this->get_linear_solver();
624 
625  std::pair<unsigned int, Real> solver_params =
626  this->get_linear_solve_parameters();
627 
628  const std::pair<unsigned int, Real> rval =
629  linear_solver->solve (*matrix, this->add_weighted_sensitivity_solution(),
630  *temprhs,
631  solver_params.second,
632  solver_params.first);
633 
634  this->release_linear_solver(linear_solver);
635 
636  // The linear solver may not have fit our constraints exactly
637 #ifdef LIBMESH_ENABLE_CONSTRAINTS
638  this->get_dof_map().enforce_constraints_exactly
639  (*this, &this->get_weighted_sensitivity_solution(),
640  /* homogeneous = */ true);
641 #endif
642 
643  return rval;
644 }

648 void ImplicitSystem::assemble_residual_derivatives(const ParameterVector & parameters_in)
649 {
650  ParameterVector & parameters =
651  const_cast<ParameterVector &>(parameters_in);
652 
653  const unsigned int Np = cast_int<unsigned int>
654  (parameters.size());
655 
656  for (unsigned int p=0; p != Np; ++p)
657  {
658  NumericVector<Number> & sensitivity_rhs = this->add_sensitivity_rhs(p);
659 
660  // Approximate -(partial R / partial p) by
661  // (R(p-dp) - R(p+dp)) / (2*dp)
662 
663  Number old_parameter = *parameters[p];
664 
665  const Real delta_p =
666  TOLERANCE * std::max(std::abs(old_parameter), 1e-3);
667 
668  *parameters[p] -= delta_p;
669 
670  // this->assembly(true, false, true);
671  this->assembly(true, false, false);
672  this->rhs->close();
673  sensitivity_rhs = *this->rhs;
674 
675  *parameters[p] = old_parameter + delta_p;
676 
677  // this->assembly(true, false, true);
678  this->assembly(true, false, false);
679  this->rhs->close();
680 
681  sensitivity_rhs -= *this->rhs;
682  sensitivity_rhs /= (2*delta_p);
683  sensitivity_rhs.close();
684 
685  *parameters[p] = old_parameter;
686  }
687 }

691 void ImplicitSystem::adjoint_qoi_parameter_sensitivity (const QoISet & qoi_indices,
692  const ParameterVector & parameters_in,
693  SensitivityData & sensitivities)
694 {
695  ParameterVector & parameters =
696  const_cast<ParameterVector &>(parameters_in);
697 
698  const unsigned int Np = cast_int<unsigned int>
699  (parameters.size());
700  const unsigned int Nq = cast_int<unsigned int>
701  (qoi.size());
702 
703  // An introduction to the problem:
704  //
705  // Residual R(u(p),p) = 0
706  // partial R / partial u = J = system matrix
707  //
708  // This implies that:
709  // d/dp(R) = 0
710  // (partial R / partial p) +
711  // (partial R / partial u) * (partial u / partial p) = 0
712 
713  // We first do an adjoint solve:
714  // J^T * z = (partial q / partial u)
715  // if we havent already or dont have an initial condition for the adjoint
716  if (!this->is_adjoint_already_solved())
717  {
718  this->adjoint_solve(qoi_indices);
719  }
720 
721  this->assemble_residual_derivatives(parameters_in);
722 
723  // Get ready to fill in senstivities:
724  sensitivities.allocate_data(qoi_indices, *this, parameters);
725 
726  // We use the identities:
727  // dq/dp = (partial q / partial p) + (partial q / partial u) *
728  // (partial u / partial p)
729  // dq/dp = (partial q / partial p) + (J^T * z) *
730  // (partial u / partial p)
731  // dq/dp = (partial q / partial p) + z * J *
732  // (partial u / partial p)
733 
734  // Leading to our final formula:
735  // dq/dp = (partial q / partial p) - z * (partial R / partial p)
736 
737  // In the case of adjoints with heterogenous Dirichlet boundary
738  // function phi, where
739  // q := S(u) - R(u,phi)
740  // the final formula works out to:
741  // dq/dp = (partial S / partial p) - z * (partial R / partial p)
742  // Because we currently have no direct access to
743  // (partial S / partial p), we use the identity
744  // (partial S / partial p) = (partial q / partial p) +
745  // phi * (partial R / partial p)
746  // to derive an equivalent equation:
747  // dq/dp = (partial q / partial p) - (z-phi) * (partial R / partial p)
748 
749  // Since z-phi degrees of freedom are zero for constrained indices,
750  // we can use the same constrained -(partial R / partial p) that we
751  // use for forward sensitivity solves, taking into account the
752  // differing sign convention.
753  //
754  // Since that vector is constrained, its constrained indices are
755  // zero, so its product with phi is zero, so we can neglect the
756  // evaluation of phi terms.
757 
758  for (unsigned int j=0; j != Np; ++j)
759  {
760  // We currently get partial derivatives via central differencing
761 
762  // (partial q / partial p) ~= (q(p+dp)-q(p-dp))/(2*dp)
763  // (partial R / partial p) ~= (rhs(p+dp) - rhs(p-dp))/(2*dp)
764 
765  Number old_parameter = *parameters[j];
766 
767  const Real delta_p =
768  TOLERANCE * std::max(std::abs(old_parameter), 1e-3);
769 
770  *parameters[j] = old_parameter - delta_p;
771  this->assemble_qoi(qoi_indices);
772  std::vector<Number> qoi_minus = this->qoi;
773 
774  NumericVector<Number> & neg_partialR_partialp = this->get_sensitivity_rhs(j);
775 
776  *parameters[j] = old_parameter + delta_p;
777  this->assemble_qoi(qoi_indices);
778  std::vector<Number> & qoi_plus = this->qoi;
779 
780  std::vector<Number> partialq_partialp(Nq, 0);
781  for (unsigned int i=0; i != Nq; ++i)
782  if (qoi_indices.has_index(i))
783  partialq_partialp[i] = (qoi_plus[i] - qoi_minus[i]) / (2.*delta_p);
784 
785  // Don't leave the parameter changed
786  *parameters[j] = old_parameter;
787 
788  for (unsigned int i=0; i != Nq; ++i)
789  if (qoi_indices.has_index(i))
790  sensitivities[i][j] = partialq_partialp[i] +
791  neg_partialR_partialp.dot(this->get_adjoint_solution(i));
792  }
793 
794  // All parameters have been reset.
795  // Reset the original qoi.
796 
797  this->assemble_qoi(qoi_indices);
798 }

802 void ImplicitSystem::forward_qoi_parameter_sensitivity (const QoISet & qoi_indices,
803  const ParameterVector & parameters_in,
804  SensitivityData & sensitivities)
805 {

807  const_cast<ParameterVector &>(parameters_in);
808 
809  const unsigned int Np = cast_int<unsigned int>
810  (parameters.size());
811  const unsigned int Nq = cast_int<unsigned int>
812  (qoi.size());
813 
814  // An introduction to the problem:
815  //
816  // Residual R(u(p),p) = 0
817  // partial R / partial u = J = system matrix
818  //
819  // This implies that:
820  // d/dp(R) = 0
821  // (partial R / partial p) +
822  // (partial R / partial u) * (partial u / partial p) = 0
823 
824  // We first solve for (partial u / partial p) for each parameter:
825  // J * (partial u / partial p) = - (partial R / partial p)
826 
827  this->sensitivity_solve(parameters);
828 
829  // Get ready to fill in senstivities:
830  sensitivities.allocate_data(qoi_indices, *this, parameters);
831 
832  // We use the identity:
833  // dq/dp = (partial q / partial p) + (partial q / partial u) *
834  // (partial u / partial p)
835 
836  // We get (partial q / partial u) from the user
837  this->assemble_qoi_derivative(qoi_indices,
838  /* include_liftfunc = */ true,
839  /* apply_constraints = */ false);
840 
841  // We don't need these to be closed() in this function, but libMesh
842  // standard practice is to have them closed() by the time the
843  // function exits
844  for (unsigned int i=0; i != this->qoi.size(); ++i)
845  if (qoi_indices.has_index(i))
846  this->get_adjoint_rhs(i).close();
847 
848  for (unsigned int j=0; j != Np; ++j)
849  {
850  // We currently get partial derivatives via central differencing
851 
852  // (partial q / partial p) ~= (q(p+dp)-q(p-dp))/(2*dp)
853 
854  Number old_parameter = *parameters[j];
855 
856  const Real delta_p =
857  TOLERANCE * std::max(std::abs(old_parameter), 1e-3);
858 
859  *parameters[j] = old_parameter - delta_p;
860  this->assemble_qoi(qoi_indices);
861  std::vector<Number> qoi_minus = this->qoi;
862 
863  *parameters[j] = old_parameter + delta_p;
864  this->assemble_qoi(qoi_indices);
865  std::vector<Number> & qoi_plus = this->qoi;
866 
867  std::vector<Number> partialq_partialp(Nq, 0);
868  for (unsigned int i=0; i != Nq; ++i)
869  if (qoi_indices.has_index(i))
870  partialq_partialp[i] = (qoi_plus[i] - qoi_minus[i]) / (2.*delta_p);
871 
872  // Don't leave the parameter changed
873  *parameters[j] = old_parameter;
874 
875  for (unsigned int i=0; i != Nq; ++i)
876  if (qoi_indices.has_index(i))
877  sensitivities[i][j] = partialq_partialp[i] +
878  this->get_adjoint_rhs(i).dot(this->get_sensitivity_solution(j));
879  }
880 
881  // All parameters have been reset.
882  // We didn't cache the original rhs or matrix for memory reasons,
883  // but we can restore them to a state consistent solution -
884  // principle of least surprise.
885  this->assembly(true, true);
886  this->rhs->close();
887  this->matrix->close();
888  this->assemble_qoi(qoi_indices);

893 void ImplicitSystem::qoi_parameter_hessian_vector_product (const QoISet & qoi_indices,
894  const ParameterVector & parameters_in,
895  const ParameterVector & vector,
896  SensitivityData & sensitivities)
897 {
898  // We currently get partial derivatives via finite differencing
899  const Real delta_p = TOLERANCE;
900 
901  ParameterVector & parameters =
902  const_cast<ParameterVector &>(parameters_in);
903 
904  // We'll use a single temporary vector for matrix-vector-vector products
905  UniquePtr<NumericVector<Number> > tempvec = this->solution->zero_clone();
906 
907  const unsigned int Np = cast_int<unsigned int>
908  (parameters.size());
909  const unsigned int Nq = cast_int<unsigned int>
910  (qoi.size());
911 
912  // For each quantity of interest q, the parameter sensitivity
913  // Hessian is defined as q''_{kl} = {d^2 q}/{d p_k d p_l}.
914  // Given a vector of parameter perturbation weights w_l, this
915  // function evaluates the hessian-vector product sum_l(q''_{kl}*w_l)
916  //
917  // We calculate it from values and partial derivatives of the
918  // quantity of interest function Q, solution u, adjoint solution z,
919  // parameter sensitivity adjoint solutions z^l, and residual R, as:
920  //
921  // sum_l(q''_{kl}*w_l) =
922  // sum_l(w_l * Q''_{kl}) + Q''_{uk}(u)*(sum_l(w_l u'_l)) -
923  // R'_k(u, sum_l(w_l*z^l)) - R'_{uk}(u,z)*(sum_l(w_l u'_l) -
924  // sum_l(w_l*R''_{kl}(u,z))
925  //
926  // See the adjoints model document for more details.
927 
928  // We first do an adjoint solve to get z for each quantity of
929  // interest
930  // if we havent already or dont have an initial condition for the adjoint
931  if (!this->is_adjoint_already_solved())
932  {
933  this->adjoint_solve(qoi_indices);
934  }
935 
936  // Get ready to fill in senstivities:
937  sensitivities.allocate_data(qoi_indices, *this, parameters);
938 
939  // We can't solve for all the solution sensitivities u'_l or for all
940  // of the parameter sensitivity adjoint solutions z^l without
941  // requiring O(Nq*Np) linear solves. So we'll solve directly for their
942  // weighted sum - this is just O(Nq) solves.
943 
944  // First solve for sum_l(w_l u'_l).
945  this->weighted_sensitivity_solve(parameters, vector);
946 
947  // Then solve for sum_l(w_l z^l).
948  this->weighted_sensitivity_adjoint_solve(parameters, vector, qoi_indices);
949 
950  for (unsigned int k=0; k != Np; ++k)
951  {
952  // We approximate sum_l(w_l * Q''_{kl}) with a central
953  // differencing pertubation:
954  // sum_l(w_l * Q''_{kl}) ~=
955  // (Q(p + dp*w_l*e_l + dp*e_k) - Q(p - dp*w_l*e_l + dp*e_k) -
956  // Q(p + dp*w_l*e_l - dp*e_k) + Q(p - dp*w_l*e_l - dp*e_k))/(4*dp^2)
957 
958  // The sum(w_l*R''_kl) term requires the same sort of pertubation,
959  // and so we subtract it in at the same time:
960  // sum_l(w_l * R''_{kl}) ~=
961  // (R(p + dp*w_l*e_l + dp*e_k) - R(p - dp*w_l*e_l + dp*e_k) -
962  // R(p + dp*w_l*e_l - dp*e_k) + R(p - dp*w_l*e_l - dp*e_k))/(4*dp^2)
963 
964  ParameterVector oldparameters, parameterperturbation;
965  parameters.deep_copy(oldparameters);
966  vector.deep_copy(parameterperturbation);
967  parameterperturbation *= delta_p;
968  parameters += parameterperturbation;
969 
970  Number old_parameter = *parameters[k];
971 
972  *parameters[k] = old_parameter + delta_p;
973  this->assemble_qoi(qoi_indices);
974  this->assembly(true, false, true);
975  this->rhs->close();
976  std::vector<Number> partial2q_term = this->qoi;
977  std::vector<Number> partial2R_term(this->qoi.size());
978  for (unsigned int i=0; i != Nq; ++i)
979  if (qoi_indices.has_index(i))
980  partial2R_term[i] = this->rhs->dot(this->get_adjoint_solution(i));
981 
982  *parameters[k] = old_parameter - delta_p;
983  this->assemble_qoi(qoi_indices);
984  this->assembly(true, false, true);
985  this->rhs->close();
986  for (unsigned int i=0; i != Nq; ++i)
987  if (qoi_indices.has_index(i))
988  {
989  partial2q_term[i] -= this->qoi[i];
990  partial2R_term[i] -= this->rhs->dot(this->get_adjoint_solution(i));
991  }
992 
993  oldparameters.value_copy(parameters);
994  parameterperturbation *= -1.0;
995  parameters += parameterperturbation;
996 
997  // Re-center old_parameter, which may be affected by vector
998  old_parameter = *parameters[k];
999 
1000  *parameters[k] = old_parameter + delta_p;
1001  this->assemble_qoi(qoi_indices);
1002  this->assembly(true, false, true);
1003  this->rhs->close();
1004  for (unsigned int i=0; i != Nq; ++i)
1005  if (qoi_indices.has_index(i))
1006  {
1007  partial2q_term[i] -= this->qoi[i];
1008  partial2R_term[i] -= this->rhs->dot(this->get_adjoint_solution(i));
1009  }
1010 
1011  *parameters[k] = old_parameter - delta_p;
1012  this->assemble_qoi(qoi_indices);
1013  this->assembly(true, false, true);
1014  this->rhs->close();
1015  for (unsigned int i=0; i != Nq; ++i)
1016  if (qoi_indices.has_index(i))
1017  {
1018  partial2q_term[i] += this->qoi[i];
1019  partial2R_term[i] += this->rhs->dot(this->get_adjoint_solution(i));
1020  }
1021 
1022  for (unsigned int i=0; i != Nq; ++i)
1023  if (qoi_indices.has_index(i))
1024  {
1025  partial2q_term[i] /= (4. * delta_p * delta_p);
1026  partial2R_term[i] /= (4. * delta_p * delta_p);
1027  }
1028 
1029  for (unsigned int i=0; i != Nq; ++i)
1030  if (qoi_indices.has_index(i))
1031  sensitivities[i][k] = partial2q_term[i] - partial2R_term[i];
1032 
1033  // We get (partial q / partial u), R, and
1034  // (partial R / partial u) from the user, but centrally
1035  // difference to get q_uk, R_k, and R_uk terms:
1036  // (partial R / partial k)
1037  // R_k*sum(w_l*z^l) = (R(p+dp*e_k)*sum(w_l*z^l) - R(p-dp*e_k)*sum(w_l*z^l))/(2*dp)
1038  // (partial^2 q / partial u partial k)
1039  // q_uk = (q_u(p+dp*e_k) - q_u(p-dp*e_k))/(2*dp)
1040  // (partial^2 R / partial u partial k)
1041  // R_uk*z*sum(w_l*u'_l) = (R_u(p+dp*e_k)*z*sum(w_l*u'_l) - R_u(p-dp*e_k)*z*sum(w_l*u'_l))/(2*dp)
1042 
1043  // To avoid creating Nq temporary vectors for q_uk or R_uk, we add
1044  // subterms to the sensitivities output one by one.
1045  //
1046  // FIXME: this is probably a bad order of operations for
1047  // controlling floating point error.
1048 
1049  *parameters[k] = old_parameter + delta_p;
1050  this->assembly(true, true);
1051  this->rhs->close();
1052  this->matrix->close();
1053  this->assemble_qoi_derivative(qoi_indices,
1054  /* include_liftfunc = */ true,
1055  /* apply_constraints = */ false);
1056 
1057  this->matrix->vector_mult(*tempvec, this->get_weighted_sensitivity_solution());
1058 
1059  for (unsigned int i=0; i != Nq; ++i)
1060  if (qoi_indices.has_index(i))
1061  {
1062  this->get_adjoint_rhs(i).close();
1063  sensitivities[i][k] += (this->get_adjoint_rhs(i).dot(this->get_weighted_sensitivity_solution()) -
1064  this->rhs->dot(this->get_weighted_sensitivity_adjoint_solution(i)) -
1065  this->get_adjoint_solution(i).dot(*tempvec)) / (2.*delta_p);
1066  }
1067 
1068  *parameters[k] = old_parameter - delta_p;
1069  this->assembly(true, true);
1070  this->rhs->close();
1071  this->matrix->close();
1072  this->assemble_qoi_derivative(qoi_indices,
1073  /* include_liftfunc = */ true,
1074  /* apply_constraints = */ false);
1075 
1076  this->matrix->vector_mult(*tempvec, this->get_weighted_sensitivity_solution());
1077 
1078  for (unsigned int i=0; i != Nq; ++i)
1079  if (qoi_indices.has_index(i))
1080  {
1081  this->get_adjoint_rhs(i).close();
1082  sensitivities[i][k] += (-this->get_adjoint_rhs(i).dot(this->get_weighted_sensitivity_solution()) +
1083  this->rhs->dot(this->get_weighted_sensitivity_adjoint_solution(i)) +
1084  this->get_adjoint_solution(i).dot(*tempvec)) / (2.*delta_p);
1085  }
1086  }
1087 
1088  // All parameters have been reset.
1089  // Don't leave the qoi or system changed - principle of least
1090  // surprise.
1091  this->assembly(true, true);
1092  this->rhs->close();
1093  this->matrix->close();
1094  this->assemble_qoi(qoi_indices);
1095 }

1099 void ImplicitSystem::qoi_parameter_hessian (const QoISet & qoi_indices,
1100  const ParameterVector & parameters_in,
1101  SensitivityData & sensitivities)
1102 {
1103  // We currently get partial derivatives via finite differencing
1104  const Real delta_p = TOLERANCE;
1105 
1106  ParameterVector & parameters =
1107  const_cast<ParameterVector &>(parameters_in);
1108 
1109  // We'll use one temporary vector for matrix-vector-vector products
1110  UniquePtr<NumericVector<Number> > tempvec = this->solution->zero_clone();
1111 
1112  // And another temporary vector to hold a copy of the true solution
1113  // so we can safely perturb this->solution.
1114  UniquePtr<NumericVector<Number> > oldsolution = this->solution->clone();
1115 
1116  const unsigned int Np = cast_int<unsigned int>
1117  (parameters.size());
1118  const unsigned int Nq = cast_int<unsigned int>
1119  (qoi.size());
1120 
1121  // For each quantity of interest q, the parameter sensitivity
1122  // Hessian is defined as q''_{kl} = {d^2 q}/{d p_k d p_l}.
1123  //
1124  // We calculate it from values and partial derivatives of the
1125  // quantity of interest function Q, solution u, adjoint solution z,
1126  // and residual R, as:
1127  //
1128  // q''_{kl} =
1129  // Q''_{kl} + Q''_{uk}(u)*u'_l + Q''_{ul}(u) * u'_k +
1130  // Q''_{uu}(u)*u'_k*u'_l -
1131  // R''_{kl}(u,z) -
1132  // R''_{uk}(u,z)*u'_l - R''_{ul}(u,z)*u'_k -
1133  // R''_{uu}(u,z)*u'_k*u'_l
1134  //
1135  // See the adjoints model document for more details.
1136 
1137  // We first do an adjoint solve to get z for each quantity of
1138  // interest
1139  // if we havent already or dont have an initial condition for the adjoint
1140  if (!this->is_adjoint_already_solved())
1141  {
1142  this->adjoint_solve(qoi_indices);
1143  }
1144 
1145  // And a sensitivity solve to get u_k for each parameter
1146  this->sensitivity_solve(parameters);
1147 
1148  // Get ready to fill in second derivatives:
1149  sensitivities.allocate_hessian_data(qoi_indices, *this, parameters);
1150 
1151  for (unsigned int k=0; k != Np; ++k)
1152  {
1153  Number old_parameterk = *parameters[k];
1154 
1155  // The Hessian is symmetric, so we just calculate the lower
1156  // triangle and the diagonal, and we get the upper triangle from
1157  // the transpose of the lower
1158 
1159  for (unsigned int l=0; l != k+1; ++l)
1160  {
1161  // The second partial derivatives with respect to parameters
1162  // are all calculated via a central finite difference
1163  // stencil:
1164  // F''_{kl} ~= (F(p+dp*e_k+dp*e_l) - F(p+dp*e_k-dp*e_l) -
1165  // F(p-dp*e_k+dp*e_l) + F(p-dp*e_k-dp*e_l))/(4*dp^2)
1166  // We will add Q''_{kl}(u) and subtract R''_{kl}(u,z) at the
1167  // same time.
1168  //
1169  // We have to be careful with the perturbations to handle
1170  // the k=l case
1171 
1172  Number old_parameterl = *parameters[l];
1173 
1174  *parameters[k] += delta_p;
1175  *parameters[l] += delta_p;
1176  this->assemble_qoi(qoi_indices);
1177  this->assembly(true, false, true);
1178  this->rhs->close();
1179  std::vector<Number> partial2q_term = this->qoi;
1180  std::vector<Number> partial2R_term(this->qoi.size());
1181  for (unsigned int i=0; i != Nq; ++i)
1182  if (qoi_indices.has_index(i))
1183  partial2R_term[i] = this->rhs->dot(this->get_adjoint_solution(i));
1184 
1185  *parameters[l] -= 2.*delta_p;
1186  this->assemble_qoi(qoi_indices);
1187  this->assembly(true, false, true);
1188  this->rhs->close();
1189  for (unsigned int i=0; i != Nq; ++i)
1190  if (qoi_indices.has_index(i))
1191  {
1192  partial2q_term[i] -= this->qoi[i];
1193  partial2R_term[i] -= this->rhs->dot(this->get_adjoint_solution(i));
1194  }
1195 
1196  *parameters[k] -= 2.*delta_p;
1197  this->assemble_qoi(qoi_indices);
1198  this->assembly(true, false, true);
1199  this->rhs->close();
1200  for (unsigned int i=0; i != Nq; ++i)
1201  if (qoi_indices.has_index(i))
1202  {
1203  partial2q_term[i] += this->qoi[i];
1204  partial2R_term[i] += this->rhs->dot(this->get_adjoint_solution(i));
1205  }
1206 
1207  *parameters[l] += 2.*delta_p;
1208  this->assemble_qoi(qoi_indices);
1209  this->assembly(true, false, true);
1210  this->rhs->close();
1211  for (unsigned int i=0; i != Nq; ++i)
1212  if (qoi_indices.has_index(i))
1213  {
1214  partial2q_term[i] -= this->qoi[i];
1215  partial2R_term[i] -= this->rhs->dot(this->get_adjoint_solution(i));
1216  partial2q_term[i] /= (4. * delta_p * delta_p);
1217  partial2R_term[i] /= (4. * delta_p * delta_p);
1218  }
1219 
1220  for (unsigned int i=0; i != Nq; ++i)
1221  if (qoi_indices.has_index(i))
1222  {
1223  Number current_terms = partial2q_term[i] - partial2R_term[i];
1224  sensitivities.second_derivative(i,k,l) += current_terms;
1225  if (k != l)
1226  sensitivities.second_derivative(i,l,k) += current_terms;
1227  }
1228 
1229  // Don't leave the parameters perturbed
1230  *parameters[l] = old_parameterl;
1231  *parameters[k] = old_parameterk;
1232  }
1233 
1234  // We get (partial q / partial u) and
1235  // (partial R / partial u) from the user, but centrally
1236  // difference to get q_uk and R_uk terms:
1237  // (partial^2 q / partial u partial k)
1238  // q_uk*u'_l = (q_u(p+dp*e_k)*u'_l - q_u(p-dp*e_k)*u'_l)/(2*dp)
1239  // R_uk*z*u'_l = (R_u(p+dp*e_k)*z*u'_l - R_u(p-dp*e_k)*z*u'_l)/(2*dp)
1240  //
1241  // To avoid creating Nq temporary vectors, we add these
1242  // subterms to the sensitivities output one by one.
1243  //
1244  // FIXME: this is probably a bad order of operations for
1245  // controlling floating point error.
1246 
1247  *parameters[k] = old_parameterk + delta_p;
