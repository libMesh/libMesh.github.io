38 template <typename T>
39 std::unique_ptr<LinearSolver<T>>
40 LinearSolver<T>::build(const libMesh::Parallel::Communicator & comm,
41  const SolverPackage solver_package)
42 {
43  // Avoid unused parameter warnings when no solver packages are enabled.
44  libmesh_ignore(comm);
45 
46  // Build the appropriate solver
47  switch (solver_package)
48  {
49 #ifdef LIBMESH_HAVE_LASPACK
50  case LASPACK_SOLVERS:
51  return libmesh_make_unique<LaspackLinearSolver<T>>(comm);
52 #endif
53 
54 
55 #ifdef LIBMESH_HAVE_PETSC
56  case PETSC_SOLVERS:
57  return libmesh_make_unique<PetscLinearSolver<T>>(comm);
58 #endif
59 
60 
61 #ifdef LIBMESH_TRILINOS_HAVE_AZTECOO
62  case TRILINOS_SOLVERS:
63  return libmesh_make_unique<AztecLinearSolver<T>>(comm);
64 #endif
65 
66 
67 #ifdef LIBMESH_HAVE_EIGEN
68  case EIGEN_SOLVERS:
69  return libmesh_make_unique<EigenSparseLinearSolver<T>>(comm);
70 #endif
71 
72  default:
73  libmesh_error_msg("ERROR: Unrecognized solver package: " << solver_package);
74  }
75 
76  return std::unique_ptr<LinearSolver<T>>();
77 }

127 template <typename T>
128 std::pair<unsigned int, Real> LinearSolver<T>::adjoint_solve (SparseMatrix<T> & mat,
129  NumericVector<T> & sol,
130  NumericVector<T> & rhs,
131  const double tol,
132  const unsigned int n_iter)
133 {
134  // Log how long the linear solve takes.
135  LOG_SCOPE("adjoint_solve()", "LinearSolver");
136 
137  // Take the discrete adjoint
138  mat.close();
139  mat.get_transpose(mat);
140 
141  // Call the solve function for the relevant linear algebra library and
142  // solve the transpose matrix
143  const std::pair<unsigned int, Real> totalrval = this->solve (mat, sol, rhs, tol, n_iter);
144 
145  // Now transpose back and restore the original matrix
146  // by taking the discrete adjoint
147  mat.get_transpose(mat);
148 
149  return totalrval;
150 }
