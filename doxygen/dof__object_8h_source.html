538 inline
539 void DofObject::invalidate_dofs (const unsigned int sys_num)
540 {
541  // If the user does not specify the system number...
542  if (sys_num >= this->n_systems())
543  {
544  for (unsigned int s=0; s<this->n_systems(); s++)
545  for (unsigned int vg=0; vg<this->n_var_groups(s); vg++)
546  if (this->n_comp_group(s,vg))
547  this->set_vg_dof_base(s,vg,invalid_id);
548  }
549  // ...otherwise invalidate the dofs for all systems
550  else
551  for (unsigned int vg=0; vg<this->n_var_groups(sys_num); vg++)
552  if (this->n_comp_group(sys_num,vg))
553  this->set_vg_dof_base(sys_num,vg,invalid_id);
554 }

597 unsigned int DofObject::n_dofs (const unsigned int s,
598  const unsigned int var) const
599 {
600  libmesh_assert_less (s, this->n_systems());
601 
602  unsigned int num = 0;
603 
604  // Count all variables
605  if (var == libMesh::invalid_uint)
606  for (unsigned int v=0; v<this->n_vars(s); v++)
607  num += this->n_comp(s,v);
608 
609  // Only count specified variable
610  else
611  num = this->n_comp(s,var);
612 
613  return num;

796 inline
797 dof_id_type DofObject::dof_number(const unsigned int s,
798  const unsigned int var,
799  const unsigned int comp) const
800 {
801  libmesh_assert_less (s, this->n_systems());
802  libmesh_assert_less (var, this->n_vars(s));
803  libmesh_assert_less (comp, this->n_comp(s,var));
804 
805  const unsigned int
806  vg = this->var_to_vg(s,var),
807  start_idx_sys = this->start_idx(s);
808 
809  libmesh_assert_less ((start_idx_sys + 2*vg + 1), _idx_buf.size());
810 
811  const dof_id_type
812  base_idx = _idx_buf[start_idx_sys + 2*vg + 1];
813 
814  // if the first component is invalid, they
815  // are all invalid
816  if (base_idx == invalid_id)
817  return invalid_id;
818 
819  // otherwise the index is the first component
820  // index augemented by the component number
821  else
822  {
823  const unsigned int
824  ncg = this->n_comp_group(s,vg),
825  vig = this->system_var_to_vg_var(s,vg,var);
826 
827  // std::cout << "base_idx, var, vg, vig, ncg, comp="
828  // << base_idx << " "
829  // << var << " "
830  // << vg << " "
831  // << vig << " "
832  // << ncg << " "
833  // << comp << '\n';
834 
835  return cast_int<dof_id_type>(base_idx + vig*ncg + comp);
836  }
837 }
