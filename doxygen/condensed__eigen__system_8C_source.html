44 void
45 CondensedEigenSystem::initialize_condensed_dofs(const std::set<dof_id_type> & global_dirichlet_dofs_set)
46 {
47  const DofMap & dof_map = this->get_dof_map();
48 
49  // First, put all unconstrained local dofs into non_dirichlet_dofs_set
50  std::set<dof_id_type> local_non_condensed_dofs_set;
51  for (dof_id_type i=this->get_dof_map().first_dof(); i<this->get_dof_map().end_dof(); i++)
52  if (!dof_map.is_constrained_dof(i))
53  local_non_condensed_dofs_set.insert(i);
54 
55  // Now erase the condensed dofs
56  std::set<dof_id_type>::iterator iter = global_dirichlet_dofs_set.begin();
57  std::set<dof_id_type>::iterator iter_end = global_dirichlet_dofs_set.end();
58 
59  for ( ; iter != iter_end ; ++iter)
60  {
61  dof_id_type condensed_dof_index = *iter;
62  if ( (this->get_dof_map().first_dof() <= condensed_dof_index) &&
63  (condensed_dof_index < this->get_dof_map().end_dof()) )
64  {
65  local_non_condensed_dofs_set.erase(condensed_dof_index);
66  }
67  }
68 
69  // Finally, move local_non_condensed_dofs_set over to a vector for convenience in solve()
70  iter = local_non_condensed_dofs_set.begin();
71  iter_end = local_non_condensed_dofs_set.end();
72 
73  this->local_non_condensed_dofs_vector.clear();
74 
75  for ( ; iter != iter_end; ++iter)
76  {
77  dof_id_type non_condensed_dof_index = *iter;
78 
79  this->local_non_condensed_dofs_vector.push_back(non_condensed_dof_index);
80  }
81 
82  condensed_dofs_initialized = true;
83 }

85 dof_id_type CondensedEigenSystem::n_global_non_condensed_dofs() const
86 {

101 void CondensedEigenSystem::solve()
102 {
103  LOG_SCOPE("solve()", "CondensedEigenSystem");
104 
105  // If we haven't initialized any condensed dofs,
106  // just use the default eigen_system
107  if (!condensed_dofs_initialized)
108  {
109  Parent::solve();
110  return;
111  }
112 
113  // A reference to the EquationSystems
114  EquationSystems & es = this->get_equation_systems();
115 
116  // check that necessary parameters have been set
117  libmesh_assert (es.parameters.have_parameter<unsigned int>("eigenpairs"));
118  libmesh_assert (es.parameters.have_parameter<unsigned int>("basis vectors"));
119 
120  if (this->assemble_before_solve)
121  // Assemble the linear system
122  this->assemble ();
123 
124  // If we reach here, then there should be some non-condensed dofs
125  libmesh_assert(!local_non_condensed_dofs_vector.empty());
126 
127  // Now condense the matrices
128  matrix_A->create_submatrix(*condensed_matrix_A,
129  local_non_condensed_dofs_vector,
130  local_non_condensed_dofs_vector);
131 
132  if (generalized())
133  {
134  matrix_B->create_submatrix(*condensed_matrix_B,
135  local_non_condensed_dofs_vector,
136  local_non_condensed_dofs_vector);
137  }
138 
139 
140  // Get the tolerance for the solver and the maximum
141  // number of iterations. Here, we simply adopt the linear solver
142  // specific parameters.
143  const Real tol =
144  es.parameters.get<Real>("linear solver tolerance");
145 
146  const unsigned int maxits =
147  es.parameters.get<unsigned int>("linear solver maximum iterations");
148 
149  const unsigned int nev =
150  es.parameters.get<unsigned int>("eigenpairs");
151 
152  const unsigned int ncv =
153  es.parameters.get<unsigned int>("basis vectors");
154 
155  std::pair<unsigned int, unsigned int> solve_data;
156 
157  // call the solver depending on the type of eigenproblem
158  if (generalized())
159  {
160  //in case of a generalized eigenproblem
161  solve_data = eigen_solver->solve_generalized
162  (*condensed_matrix_A,*condensed_matrix_B, nev, ncv, tol, maxits);
163  }
164 
165  else
166  {
167  libmesh_assert (!matrix_B);
168 
169  //in case of a standard eigenproblem
170  solve_data = eigen_solver->solve_standard (*condensed_matrix_A, nev, ncv, tol, maxits);
171  }
172 
173  set_n_converged(solve_data.first);
174  set_n_iterations(solve_data.second);
175 }

179 std::pair<Real, Real> CondensedEigenSystem::get_eigenpair(dof_id_type i)
180 {
181  LOG_SCOPE("get_eigenpair()", "CondensedEigenSystem");
182 
183  // If we haven't initialized any condensed dofs,
184  // just use the default eigen_system
185  if (!condensed_dofs_initialized)
186  return Parent::get_eigenpair(i);
187 
188  // If we reach here, then there should be some non-condensed dofs
189  libmesh_assert(!local_non_condensed_dofs_vector.empty());
190 
191  // This function assumes that condensed_solve has just been called.
192  // If this is not the case, then we will trip an asset in get_eigenpair
193  UniquePtr< NumericVector<Number> > temp = NumericVector<Number>::build(this->comm());
194  dof_id_type n_local = local_non_condensed_dofs_vector.size();
195  dof_id_type n = n_local;
196  this->comm().sum(n);
197 
198  temp->init (n, n_local, false, PARALLEL);
199 
200  std::pair<Real, Real> eval = eigen_solver->get_eigenpair (i, *temp);
201 
202  // Now map temp to solution. Loop over local entries of local_non_condensed_dofs_vector
203  this->solution->zero();
204  for (std::size_t j=0; j<local_non_condensed_dofs_vector.size(); j++)
205  {
206  dof_id_type index = local_non_condensed_dofs_vector[j];
207  solution->set(index,(*temp)(temp->first_local_index()+j));
208  }
209 
210  solution->close();
211  this->update();
212 
213  return eval;
214 }
