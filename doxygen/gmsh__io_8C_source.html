335 void GmshIO::read_mesh(std::istream& in)
336 {
337  // This is a serial-only process for now;
338  // the Mesh should be read on processor 0 and
339  // broadcast later
340  libmesh_assert_equal_to (MeshOutput<MeshBase>::mesh().processor_id(), 0);
341 
342  libmesh_assert(in.good());
343 
344  // initialize the map with element types
345  init_eletypes();
346 
347  // clear any data in the mesh
348  MeshBase& mesh = MeshInput<MeshBase>::mesh();
349  mesh.clear();
350 
351  // some variables
352  int format=0, size=0;
353  Real version = 1.0;
354 
355  // map to hold the node numbers for translation
356  // note the the nodes can be non-consecutive
357  std::map<unsigned int, unsigned int> nodetrans;
358 
359  // For reading the file line by line
360  std::string s;
361 
362  while (true)
363  {
364  // Try to read something. This may set EOF!
365  std::getline(in, s);
366 
367  if (in)
368  {
369  // Process s...
370 
371  if (s.find("$MeshFormat") == static_cast<std::string::size_type>(0))
372  {
373  in >> version >> format >> size;
374  if ((version != 2.0) && (version != 2.1) && (version != 2.2))
375  {
376  // Some notes on gmsh mesh versions:
377  //
378  // Mesh version 2.0 goes back as far as I know. It's not explicitly
379  // mentioned here: http://www.geuz.org/gmsh/doc/VERSIONS.txt
380  //
381  // As of gmsh-2.4.0:
382  // bumped mesh version format to 2.1 (small change in the $PhysicalNames
383  // section, where the group dimension is now required);
384  // [Since we don't even parse the PhysicalNames section at the time
385  // of this writing, I don't think this change affects us.]
386  //
387  // Mesh version 2.2 tested by Manav Bhatia; no other
388  // libMesh code changes were required for support
389  libmesh_error_msg("Error: Unknown msh file version " << version);
390  }
391 
392  if (format)
393  libmesh_error_msg("Error: Unknown data format for mesh in Gmsh reader.");
394  }
395 
396  // read the node block
397  else if (s.find("$NOD") == static_cast<std::string::size_type>(0) ||
398  s.find("$NOE") == static_cast<std::string::size_type>(0) ||
399  s.find("$Nodes") == static_cast<std::string::size_type>(0))
400  {
401  unsigned int num_nodes = 0;
402  in >> num_nodes;
403  mesh.reserve_nodes (num_nodes);
404 
405  // read in the nodal coordinates and form points.
406  Real x, y, z;
407  unsigned int id;
408 
409  // add the nodal coordinates to the mesh
410  for (unsigned int i=0; i<num_nodes; ++i)
411  {
412  in >> id >> x >> y >> z;
413  mesh.add_point (Point(x, y, z), i);
414  nodetrans[id] = i;
415  }
416 
417  // read the $ENDNOD delimiter
418  std::getline(in, s);
419  }
420 
421 
422  // Read the element block
423  else if (s.find("$ELM") == static_cast<std::string::size_type>(0) ||
424  s.find("$Elements") == static_cast<std::string::size_type>(0))
425  {
426  // For reading the number of elements and the node ids from the stream
427  unsigned int
428  num_elem = 0,
429  node_id = 0;
430 
431  // read how many elements are there, and reserve space in the mesh
432  in >> num_elem;
433  mesh.reserve_elem (num_elem);
434 
435  // As of version 2.2, the format for each element line is:
436  // elm-number elm-type number-of-tags < tag > ... node-number-list
437  // From the Gmsh docs:
438  // * the first tag is the number of the
439  // physical entity to which the element belongs
440  // * the second is the number of the elementary geometrical
441  // entity to which the element belongs
442  // * the third is the number of mesh partitions to which the element
443  // belongs
444  // * The rest of the tags are the partition ids (negative
445  // partition ids indicate ghost cells). A zero tag is
446  // equivalent to no tag. Gmsh and most codes using the
447  // MSH 2 format require at least the first two tags
448  // (physical and elementary tags).
449 
450  // Keep track of all the element dimensions seen
451  std::vector<unsigned> elem_dimensions_seen(3);
452 
453  // read the elements
454  for (unsigned int iel=0; iel<num_elem; ++iel)
455  {
456  unsigned int
457  id, type,
458  physical=1, elementary=1,
459  nnodes=0, ntags;
460 
461  // Note: tag has to be an int because it could be negative,
462  // see above.
463  int tag;
464 
465  if (version <= 1.0)
466  in >> id >> type >> physical >> elementary >> nnodes;
467 
468  else
469  {
470  in >> id >> type >> ntags;
471 
472  if (ntags > 2)
473  libmesh_do_once(libMesh::err << "Warning, ntags=" << ntags << ", but we currently only support reading 2 flags." << std::endl;);
474 
475  for (unsigned int j = 0; j < ntags; j++)
476  {
477  in >> tag;
478  if (j == 0)
479  physical = tag;
480  else if (j == 1)
481  elementary = tag;
482  }
483  }
484 
485  // Consult the import element table to determine which element to build
486  std::map<unsigned int, elementDefinition>::iterator eletypes_it = eletypes_imp.find(type);
487 
488  // Make sure we actually found something
489  if (eletypes_it == eletypes_imp.end())
490  libmesh_error_msg("Element type " << type << " not found!");
491 
492  // Get a reference to the elementDefinition
493  const elementDefinition& eletype = eletypes_it->second;
494 
495  // If we read nnodes, make sure it matches the number in eletype.nnodes
496  if (nnodes != 0 && nnodes != eletype.nnodes)
497  libmesh_error_msg("nnodes = " << nnodes << " and eletype.nnodes = " << eletype.nnodes << " do not match.");
498 
499  // Assign the value from the eletype object.
500  nnodes = eletype.nnodes;
501 
502  // Don't add 0-dimensional "point" elements to the
503  // Mesh. They should *always* be treated as boundary
504  // "nodeset" data.
505  if (eletype.dim > 0)
506  {
507  // Record this element dimension as being "seen".
508  // We will treat all elements with dimension <
509  // max(dimension) as specifying boundary conditions,
510  // but we won't know what max_elem_dimension_seen is
511  // until we read the entire file.
512  elem_dimensions_seen[eletype.dim-1] = 1;
513 
514  // Add the element to the mesh
515  {
516  Elem* elem = Elem::build(eletype.type).release();
517  elem->set_id(iel);
518  elem = mesh.add_elem(elem);
519 
520  // Make sure that the libmesh element we added has nnodes nodes.
521  if (elem->n_nodes() != nnodes)
522  libmesh_error_msg("Number of nodes for element " \
523  << id \
524  << " of type " << eletypes_imp[type].type \
525  << " (Gmsh type " << type \
526  << ") does not match Libmesh definition. " \
527  << "I expected " << elem->n_nodes() \
528  << " nodes, but got " << nnodes);
529 
530  // Add node pointers to the elements.
531  // If there is a node translation table, use it.
532  if (eletype.nodes.size() > 0)
533  for (unsigned int i=0; i<nnodes; i++)
534  {
535  in >> node_id;
536  elem->set_node(eletype.nodes[i]) = mesh.node_ptr(nodetrans[node_id]);
537  }
538  else
539  {
540  for (unsigned int i=0; i<nnodes; i++)
541  {
542  in >> node_id;
543  elem->set_node(i) = mesh.node_ptr(nodetrans[node_id]);
544  }
545  }
546 
547  // Finally, set the subdomain ID to physical. If this is a lower-dimension element, this ID will
548  // eventually go into the Mesh's BoundaryInfo object.
549  elem->subdomain_id() = static_cast<subdomain_id_type>(physical);
550  }
551  }
552 
553  // Handle 0-dimensional elements (points) by adding
554  // them to the BoundaryInfo object with
555  // boundary_id == physical.
556  else
557  {
558  // This seems like it should always be the same
559  // number as the 'id' we already read in on this
560  // line. At least it was in the example gmsh
561  // file I had...
562  in >> node_id;
563  mesh.get_boundary_info().add_node
564  (nodetrans[node_id],
565  static_cast<boundary_id_type>(physical));
566  }
567  } // element loop
568 
569  // read the $ENDELM delimiter
570  std::getline(in, s);
571 
572  // Record the max and min element dimension seen while reading the file.
573  unsigned char
574  max_elem_dimension_seen=1,
575  min_elem_dimension_seen=3;
576 
577  for (unsigned char i=0; i<elem_dimensions_seen.size(); ++i)
578  if (elem_dimensions_seen[i])
579  {
580  // Debugging
581  // libMesh::out << "Seen elements of dimension " << i+1 << std::endl;
582  max_elem_dimension_seen =
583  std::max(max_elem_dimension_seen, cast_int<unsigned char>(i+1));
584  min_elem_dimension_seen =
585  std::min(min_elem_dimension_seen, cast_int<unsigned char>(i+1));
586  }
587 
588  // Debugging:
589  // libMesh::out << "max_elem_dimension_seen=" << max_elem_dimension_seen << std::endl;
590  // libMesh::out << "min_elem_dimension_seen=" << min_elem_dimension_seen << std::endl;
591 
592  // If the difference between the max and min element dimension seen is larger than
593  // 1, (e.g. the file has 1D and 3D elements only) we don't handle this case.
594  if (max_elem_dimension_seen - min_elem_dimension_seen > 1)
595  libmesh_error_msg("Cannot handle meshes with dimension mismatch greater than 1.");
596 
597  // How many different element dimensions did we see while reading from file?
598  unsigned n_dims_seen = std::accumulate(elem_dimensions_seen.begin(),
599  elem_dimensions_seen.end(),
600  static_cast<unsigned>(0),
601  std::plus<unsigned>());
602 
603  // Have not yet tested a case where 1, 2, and 3D elements are all in the same Mesh,
604  // though it should theoretically be possible to handle.
605  if (n_dims_seen == 3)
606  libmesh_error_msg("Reading meshes with 1, 2, and 3D elements not currently supported.");
607 
608  // Set mesh_dimension based on the largest element dimension seen.
609  mesh.set_mesh_dimension(max_elem_dimension_seen);
610 
611  if (n_dims_seen > 1)
612  {
613  // map from (node ids) -> elem of lower dimensional elements that can provide boundary conditions
614  typedef std::map<std::vector<dof_id_type>, Elem*> provide_container_t;
615  provide_container_t provide_bcs;
616 
617  // 1st loop over active elements - get info about lower-dimensional elements.
618  {
619  MeshBase::element_iterator it = mesh.active_elements_begin();
620  const MeshBase::element_iterator end = mesh.active_elements_end();
621  for ( ; it != end; ++it)
622  {
623  Elem* elem = *it;
624 
625  if (elem->dim() < max_elem_dimension_seen)
626  {
627  // Debugging status
628  // libMesh::out << "Processing Elem " << elem->id() << " as a boundary element." << std::endl;
629 
630  // To be pushed into the provide_bcs data structure
631  std::vector<dof_id_type> node_ids(elem->n_nodes());
632 
633  // To be consistent with the previous GmshIO behavior, add all the lower-dimensional elements' nodes to
634  // the Mesh's BoundaryInfo object with the lower-dimensional element's subdomain ID.
635  for (unsigned n=0; n<elem->n_nodes(); n++)
636  {
637  mesh.get_boundary_info().add_node
638  (elem->node(n), elem->subdomain_id());
639 
640  // And save for our local data structure
641  node_ids[n] = elem->node(n);
642  }
643 
644  // Sort before putting into the map
645  std::sort(node_ids.begin(), node_ids.end());
646  provide_bcs[node_ids] = elem;
647  }
648  }
649  } // end 1st loop over active elements
650 
651  // Debugging: What did we put in the provide_bcs data structure?
652  // {
653  // provide_container_t::iterator provide_it = provide_bcs.begin();
654  // provide_container_t::iterator provide_end = provide_bcs.end();
655  // for ( ; provide_it != provide_end; ++provide_it)
656  // {
657  // std::vector<dof_id_type> node_list = (*provide_it).first;
658  // Elem* elem = (*provide_it).second;
659  //
660  // libMesh::out << "Elem " << elem->id() << " provides BCs for the face: ";
661  // for (unsigned i=0; i<node_list.size(); ++i)
662  // libMesh::out << node_list[i] << " ";
663  // libMesh::out << std::endl;
664  // }
665  // }
666 
667  // 2nd loop over active elements - use lower dimensional element data to set BCs for higher dimensional elements
668  {
669  MeshBase::element_iterator it = mesh.active_elements_begin();
670  const MeshBase::element_iterator end = mesh.active_elements_end();
671  for ( ; it != end; ++it)
672  {
673  Elem* elem = *it;
674 
675  if (elem->dim() == max_elem_dimension_seen)
676  {
677  // This is a max-dimension element that
678  // may require BCs. For each of its
679  // sides, including internal sides, we'll
680  // see if a lower-dimensional element
681  // provides boundary information for it.
682  // Note that we have not yet called
683  // find_neighbors(), so we can't use
684  // elem->neighbor(sn) in this algorithm...
685 
686  for (unsigned short sn=0;
687  sn<elem->n_sides(); sn++)
688  {
689  AutoPtr<Elem> side (elem->build_side(sn));
690 
691  // Build up a node_ids vector, which is the key
692  std::vector<dof_id_type> node_ids(side->n_nodes());

789 void GmshIO::write_mesh (std::ostream& out_stream)
790 {
791  // Be sure that the stream is valid.
792  libmesh_assert (out_stream.good());
793 
794  // initialize the map with element types
795  init_eletypes();
796 
797  // Get a const reference to the mesh
798  const MeshBase& mesh = MeshOutput<MeshBase>::mesh();
799 
800  // Note: we are using version 2.0 of the gmsh output format.
801 
802  // Write the file header.
803  out_stream << "$MeshFormat\n";
804  out_stream << "2.0 0 " << sizeof(Real) << '\n';
805  out_stream << "$EndMeshFormat\n";
806 
807  // write the nodes in (n x y z) format
808  out_stream << "$Nodes\n";
809  out_stream << mesh.n_nodes() << '\n';
810 
811  for (unsigned int v=0; v<mesh.n_nodes(); v++)
812  out_stream << mesh.node(v).id()+1 << " "
813  << mesh.node(v)(0) << " "
814  << mesh.node(v)(1) << " "
815  << mesh.node(v)(2) << '\n';
816  out_stream << "$EndNodes\n";
817 
818  {
819  // write the connectivity
820  out_stream << "$Elements\n";
821  out_stream << mesh.n_active_elem() << '\n';
822 
823  MeshBase::const_element_iterator it = mesh.active_elements_begin();
824  const MeshBase::const_element_iterator end = mesh.active_elements_end();
825 
826  // loop over the elements
827  for ( ; it != end; ++it)
828  {
829  const Elem* elem = *it;
830 
831  // Make sure we have a valid entry for
832  // the current element type.
833  libmesh_assert (eletypes_exp.count(elem->type()));
834 
835  // consult the export element table
836  const elementDefinition& eletype = eletypes_exp[elem->type()];
837 
838  // The element mapper better not require any more nodes
839  // than are present in the current element!
840  libmesh_assert_less_equal (eletype.nodes.size(), elem->n_nodes());
841 
842  // elements ids are 1 based in Gmsh
843  out_stream << elem->id()+1 << " ";
844 
845  // element type
846  out_stream << eletype.exptype;
847 
848  // write the number of tags and
849  // tag1 (physical entity), tag2 (geometric entity), and tag3 (partition entity)
850  out_stream << " 3 "
851  << static_cast<unsigned int>(elem->subdomain_id())
852  << " 1 "
853  << (elem->processor_id()+1) << " ";
854 
855  // if there is a node translation table, use it
856  if (eletype.nodes.size() > 0)
857  for (unsigned int i=0; i < elem->n_nodes(); i++)
858  out_stream << elem->node(eletype.nodes[i])+1 << " "; // gmsh is 1-based
859  // otherwise keep the same node order
860  else
861  for (unsigned int i=0; i < elem->n_nodes(); i++)
862  out_stream << elem->node(i)+1 << " "; // gmsh is 1-based
863  out_stream << "\n";
864  } // element loop
865  out_stream << "$EndElements\n";
866  }
867 }
