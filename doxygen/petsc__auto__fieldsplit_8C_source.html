60 void petsc_auto_fieldsplit (PC my_pc,
61  const System &sys)
62 {
63  std::string sys_prefix = "--solver_group_";
64 
65  if (libMesh::on_command_line("--solver_system_names"))
66  {
67  sys_prefix = sys_prefix + sys.name() + "_";
68  }
69 
70  std::map<std::string, std::vector<dof_id_type> > group_indices;
71 
72  if (libMesh::on_command_line("--solver_variable_names"))
73  {
74  for (unsigned int v = 0; v != sys.n_vars(); ++v)
75  {
76  const std::string& var_name = sys.variable_name(v);
77 
78  std::vector<dof_id_type> var_idx;
79  sys.get_dof_map().local_variable_indices
80  (var_idx, sys.get_mesh(), v);
81 
82  std::string group_command = sys_prefix + var_name;
83 
84  const std::string empty_string;
85 
86  std::string group_name = libMesh::command_line_value
87  (group_command, empty_string);
88 
89  if (group_name != empty_string)
90  {
91  std::vector<dof_id_type> &indices =
92  group_indices[group_name];
93  const bool prior_indices = !indices.empty();
94  indices.insert(indices.end(), var_idx.begin(),
95  var_idx.end());
96  if (prior_indices)
97  std::sort(indices.begin(), indices.end());
98  }
99  else
100  {
101  indices_to_fieldsplit (sys.comm(), var_idx, my_pc, var_name);
102  }
103  }
104  }
105 
106  for (std::map<std::string, std::vector<dof_id_type> >::const_iterator
107  i = group_indices.begin(); i != group_indices.end(); ++i)
108  {
109  indices_to_fieldsplit(sys.comm(), i->second, my_pc, i->first);
110  }
111 }
