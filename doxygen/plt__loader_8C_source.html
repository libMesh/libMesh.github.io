60 void PltLoader::set_n_vars (const unsigned int nv)
61 {
62  _n_vars = nv;
63 
64  _var_types.resize (this->n_vars());
65  _var_names.resize (this->n_vars());
66 
67  // Default to float data

69

73  _data.resize (this->n_zones());
