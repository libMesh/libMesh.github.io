102 bool Prism15::has_affine_map() const
103 {
104  // Make sure z edges are affine
105  Point v = this->point(3) - this->point(0);
106  if (!v.relative_fuzzy_equals(this->point(4) - this->point(1)) ||
107  !v.relative_fuzzy_equals(this->point(5) - this->point(2)))
108  return false;
109  // Make sure edges are straight
110  v /= 2;
111  if (!v.relative_fuzzy_equals(this->point(9) - this->point(0)) ||
112  !v.relative_fuzzy_equals(this->point(10) - this->point(1)) ||
113  !v.relative_fuzzy_equals(this->point(11) - this->point(2)))
114  return false;
115  v = (this->point(1) - this->point(0))/2;
116  if (!v.relative_fuzzy_equals(this->point(6) - this->point(0)) ||
117  !v.relative_fuzzy_equals(this->point(12) - this->point(3)))
118  return false;
119  v = (this->point(2) - this->point(0))/2;
120  if (!v.relative_fuzzy_equals(this->point(8) - this->point(0)) ||
121  !v.relative_fuzzy_equals(this->point(14) - this->point(3)))
122  return false;
123  v = (this->point(2) - this->point(1))/2;
124  if (!v.relative_fuzzy_equals(this->point(7) - this->point(1)) ||
125  !v.relative_fuzzy_equals(this->point(13) - this->point(4)))
126  return false;
127  return true;
128 }

132 UniquePtr<Elem> Prism15::build_side_ptr (const unsigned int i,
133  bool proxy)
134 {
135  libmesh_assert_less (i, this->n_sides());
136 
137  if (proxy)
138  {
139  switch (i)
140  {
141  case 0: // the triangular face at z=-1
142  case 4:
143  return UniquePtr<Elem>(new Side<Tri6,Prism15>(this,i));
144 
145  case 1:
146  case 2:
147  case 3:
148  return UniquePtr<Elem>(new Side<Quad8,Prism15>(this,i));
149 
150  default:
151  libmesh_error_msg("Invalid side i = " << i);
152  }
153  }
154 
155  else
156  {
157  // Create NULL pointer to be initialized, returned later.
158  Elem * face = libmesh_nullptr;
159 
160  switch (i)
161  {
162  case 0: // the triangular face at z=-1
163  case 4: // the triangular face at z=1
164  {
165  face = new Tri6;
166  break;
167  }
168  case 1: // the quad face at y=0
169  case 2: // the other quad face
170  case 3: // the quad face at x=0
171  {
172  face = new Quad8;
173  break;
174  }
175  default:
176  libmesh_error_msg("Invalid side i = " << i);
177  }
178 
179  face->subdomain_id() = this->subdomain_id();
180 
181  // Set the nodes
182  for (unsigned n=0; n<face->n_nodes(); ++n)
183  face->set_node(n) = this->node_ptr(Prism15::side_nodes_map[i][n]);
184 
185  return UniquePtr<Elem>(face);
186  }
187 
188  libmesh_error_msg("We'll never get here!");
189  return UniquePtr<Elem>();
190 }

290 Real Prism15::volume () const
291 {
292  // Make copies of our points. It makes the subsequent calculations a bit
293  // shorter and avoids dereferencing the same pointer multiple times.
294  Point
295  x0 = point(0), x1 = point(1), x2 = point(2), x3 = point(3), x4 = point(4),
296  x5 = point(5), x6 = point(6), x7 = point(7), x8 = point(8), x9 = point(9),
297  x10 = point(10), x11 = point(11), x12 = point(12), x13 = point(13), x14 = point(14);
298 
299  // Terms are copied directly from a Python script.
300  Point dx_dxi[10] =
301  {
302  -x0 - x1 + x10 + 2*x12 - x3 - x4 + 2*x6 - x9,
303  3*x0/2 + x1/2 + 2*x12 - 3*x3/2 - x4/2 - 2*x6,
304  -x0/2 + x1/2 - x10 - x3/2 + x4/2 + x9,
305  2*x0 - 2*x12 + 2*x13 - 2*x14 + 2*x3 - 2*x6 + 2*x7 - 2*x8,
306  -2*x0 - 2*x12 + 2*x13 - 2*x14 + 2*x3 + 2*x6 - 2*x7 + 2*x8,
307  Point(0,0,0),
308  2*x0 + 2*x1 - 4*x12 + 2*x3 + 2*x4 - 4*x6,
309  -2*x0 - 2*x1 - 4*x12 + 2*x3 + 2*x4 + 4*x6,
310  Point(0,0,0),
311  Point(0,0,0)
312  };
313 
314  Point dx_deta[10] =
315  {
316  -x0 + x11 + 2*x14 - x2 - x3 - x5 + 2*x8 - x9,
317  3*x0/2 + 2*x14 + x2/2 - 3*x3/2 - x5/2 - 2*x8,
318  -x0/2 - x11 + x2/2 - x3/2 + x5/2 + x9,
319  2*x0 - 4*x14 + 2*x2 + 2*x3 + 2*x5 - 4*x8,
320  -2*x0 - 4*x14 - 2*x2 + 2*x3 + 2*x5 + 4*x8,
321  Point(0,0,0),
322  2*x0 - 2*x12 + 2*x13 - 2*x14 + 2*x3 - 2*x6 + 2*x7 - 2*x8,
323  -2*x0 - 2*x12 + 2*x13 - 2*x14 + 2*x3 + 2*x6 - 2*x7 + 2*x8,
324  Point(0,0,0),
325  Point(0,0,0)
326  };
327 
328  Point dx_dzeta[10] =
329  {
330  -x0/2 + x3/2,
331  x0 + x3 - 2*x9,
332  Point(0,0,0),
333  3*x0/2 + 2*x14 + x2/2 - 3*x3/2 - x5/2 - 2*x8,
334  -x0 - 2*x11 + x2 - x3 + x5 + 2*x9,
335  -x0 - 2*x14 - x2 + x3 + x5 + 2*x8,
336  3*x0/2 + x1/2 + 2*x12 - 3*x3/2 - x4/2 - 2*x6,
337  -x0 + x1 - 2*x10 - x3 + x4 + 2*x9,
338  -2*x0 - 2*x12 + 2*x13 - 2*x14 + 2*x3 + 2*x6 - 2*x7 + 2*x8,
339  -x0 - x1 - 2*x12 + x3 + x4 + 2*x6
340  };
341 
342  // The quadrature rule for the Prism15 is a tensor product between a
343  // FOURTH-order TRI3 rule (in xi, eta) and a FIFTH-order EDGE2 rule
344  // in zeta.
345 
346  // Number of points in the 2D quadrature rule.
347  const int N2D = 6;
348 
349  // Parameters of the 2D rule
350  static const Real
351  w1 = 1.1169079483900573284750350421656140e-01L,
352  w2 = 5.4975871827660933819163162450105264e-02L,
353  a1 = 4.4594849091596488631832925388305199e-01L,
354  a2 = 9.1576213509770743459571463402201508e-02L;
355 
356  // Points and weights of the 2D rule
357  static const Real w2D[N2D] = {w1, w1, w1, w2, w2, w2};
358 
359  // Quadrature point locations raised to powers. xi[0][2] is
360  // quadrature point 0, squared, xi[1][1] is quadrature point 1 to the
361  // first power, etc. This lets us avoid calling std::pow inside the
362  // loops below.
363  static const Real xi[N2D][3] =
364  {
365  // ^0 ^1 ^2
366  { 1., a1, a1*a1},
367  { 1., 1-2*a1, (1-2*a1)*(1-2*a1)},
368  { 1., a1, a1*a1},
369  { 1., a2, a2*a2},
370  { 1., 1-2*a2, (1-2*a2)*(1-2*a2)},
371  { 1., a2, a2*a2}
372  };
373 
374  static const Real eta[N2D][3] =
375  {
376  // ^0 ^1 ^2
377  { 1., a1, a1*a1},
378  { 1., a1, a1*a1},
379  { 1., 1-2*a1, (1-2*a1)*(1-2*a1)},
380  { 1., a2, a2*a2},
381  { 1., a2, a2*a2},
382  { 1., 1-2*a2, (1-2*a2)*(1-2*a2)}
383  };
384 
385  // Number of points in the 1D quadrature rule.
386  const int N1D = 3;
387 
388  // Points and weights of the 1D quadrature rule.
389  static const Real w1D[N1D] = {5./9, 8./9, 5./9};
390 
391  const Real zeta[N1D][3] =
392  {
393  //^0 ^1 ^2
394  { 1., -std::sqrt(15)/5., 15./25},
395  { 1., 0., 0.},
396  { 1., std::sqrt(15)/5., 15./25}
397  };
398 
399  // The integer exponents for each term.
400  static const int exponents[10][3] =
401  {
402  {0, 0, 0},
403  {0, 0, 1},
404  {0, 0, 2},
405  {0, 1, 0},
406  {0, 1, 1},
407  {0, 2, 0},
408  {1, 0, 0},
409  {1, 0, 1},
410  {1, 1, 0},
411  {2, 0, 0}
412  };
413 
414  Real vol = 0.;
415  for (int i=0; i<N2D; ++i)
416  for (int j=0; j<N1D; ++j)
417  {
418  // Compute dx_dxi, dx_deta, dx_dzeta at the current quadrature point.
419  Point dx_dxi_q, dx_deta_q, dx_dzeta_q;
420  for (int c=0; c<10; ++c)
421  {
422  Real coeff =
423  xi[i][exponents[c][0]]*
424  eta[i][exponents[c][1]]*
425  zeta[j][exponents[c][2]];
426 
427  dx_dxi_q += coeff * dx_dxi[c];
428  dx_deta_q += coeff * dx_deta[c];
429  dx_dzeta_q += coeff * dx_dzeta[c];
430  }
431 
432  // Compute scalar triple product, multiply by weight, and accumulate volume.
433  vol += w2D[i] * w1D[j] * triple_product(dx_dxi_q, dx_deta_q, dx_dzeta_q);
434  }
435 
436  return vol;
437 }
