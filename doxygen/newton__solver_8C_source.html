38 Real NewtonSolver::line_search(Real tol,
39  Real last_residual,
40  Real &current_residual,
41  NumericVector<Number> &newton_iterate,
42  const NumericVector<Number> &linear_solution)
43 {
44  // Take a full step if we got a residual reduction or if we
45  // aren't substepping
46  if ((current_residual < last_residual) ||
47  (!require_residual_reduction &&
48  (!require_finite_residual || !libmesh_isnan(current_residual))))
49  return 1.;
50 
51  // The residual vector
52  NumericVector<Number> &rhs = *(_system.rhs);
53 
54  Real ax = 0.; // First abscissa, don't take negative steps
55  Real cx = 1.; // Second abscissa, don't extrapolate steps
56 
57  // Find bx, a step length that gives lower residual than ax or cx
58  Real bx = 1.;
59 
60  while (libmesh_isnan(current_residual) ||
61  (current_residual > last_residual &&
62  require_residual_reduction))
63  {
64  // Reduce step size to 1/2, 1/4, etc.
65  Real substepdivision;
66  if (brent_line_search && !libmesh_isnan(current_residual))
67  {
68  substepdivision = std::min(0.5, last_residual/current_residual);
69  substepdivision = std::max(substepdivision, tol*2.);
70  }
71  else
72  substepdivision = 0.5;
73 
74  newton_iterate.add (bx * (1.-substepdivision),
75  linear_solution);
76  newton_iterate.close();
77  bx *= substepdivision;
78  if (verbose)
79  libMesh::out << " Shrinking Newton step to "
80  << bx << std::endl;
81 
82  // Check residual with fractional Newton step
83  _system.assembly (true, false);
84 
85  rhs.close();
86  current_residual = rhs.l2_norm();
87  if (verbose)
88  libMesh::out << " Current Residual: "
89  << current_residual << std::endl;
90 
91  if (bx/2. < minsteplength &&
92  (libmesh_isnan(current_residual) ||
93  (current_residual > last_residual)))
94  {
95  libMesh::out << "Inexact Newton step FAILED at step "
96  << _outer_iterations << std::endl;
97 
98  if (!continue_after_backtrack_failure)
99  {
100  libmesh_convergence_failure();
101  }
102  else
103  {
104  libMesh::out << "Continuing anyway ..." << std::endl;
105  _solve_result = DiffSolver::DIVERGED_BACKTRACKING_FAILURE;
106  return bx;
107  }
108  }
109  } // end while (current_residual > last_residual)
110 
111  // Now return that reduced-residual step, or use Brent's method to
112  // find a more optimal step.
113 
114  if (!brent_line_search)
115  return bx;
116 
117  // Brent's method adapted from Numerical Recipes in C, ch. 10.2
118  Real e = 0.;
119 
120  Real x = bx, w = bx, v = bx;
121 
122  // Residuals at bx
123  Real fx = current_residual,
124  fw = current_residual,
125  fv = current_residual;
126 
127  // Max iterations for Brent's method loop
128  const unsigned int max_i = 20;
129 
130  // for golden ratio steps
131  const Real golden_ratio = 1.-(std::sqrt(5.)-1.)/2.;
132 
133  for (unsigned int i=1; i <= max_i; i++)
134  {
135  Real xm = (ax+cx)*0.5;
136  Real tol1 = tol * std::abs(x) + tol*tol;
137  Real tol2 = 2.0 * tol1;
138 
139  // Test if we're done
140  if (std::abs(x-xm) <= (tol2 - 0.5 * (cx - ax)))
141  return x;
142 
143  Real d;
144 
145  // Construct a parabolic fit
146  if (std::abs(e) > tol1)
147  {
148  Real r = (x-w)*(fx-fv);
149  Real q = (x-v)*(fx-fw);
150  Real p = (x-v)*q-(x-w)*r;
151  q = 2. * (q-r);
152  if (q > 0.)
153  p = -p;
154  else
155  q = std::abs(q);
156  if (std::abs(p) >= std::abs(0.5*q*e) ||
157  p <= q * (ax-x) ||
158  p >= q * (cx-x))
159  {
160  // Take a golden section step
161  e = x >= xm ? ax-x : cx-x;
162  d = golden_ratio * e;
163  }
164  else
165  {
166  // Take a parabolic fit step
167  d = p/q;
168  if (x+d-ax < tol2 || cx-(x+d) < tol2)
169  d = SIGN(tol1, xm - x);
170  }
171  }
172  else
173  {
174  // Take a golden section step
175  e = x >= xm ? ax-x : cx-x;
176  d = golden_ratio * e;
177  }
178 
179  Real u = std::abs(d) >= tol1 ? x+d : x + SIGN(tol1,d);
180 
181  // Assemble the residual at the new steplength u
182  newton_iterate.add (bx - u, linear_solution);
183  newton_iterate.close();
184  bx = u;
185  if (verbose)
186  libMesh::out << " Shrinking Newton step to "
187  << bx << std::endl;
188 
189  _system.assembly (true, false);
190 
191  rhs.close();
192  Real fu = current_residual = rhs.l2_norm();
193  if (verbose)
194  libMesh::out << " Current Residual: "
195  << fu << std::endl;
196 
197  if (fu <= fx)
198  {
199  if (u >= x)
200  ax = x;
201  else
202  cx = x;
203  v = w; w = x; x = u;
204  fv = fw; fw = fx; fx = fu;
205  }
206  else
207  {
208  if (u < x)
209  ax = u;
210  else
211  cx = u;
212  if (fu <= fw || w == x)
213  {
214  v = w; w = u;
215  fv = fw; fw = fu;
216  }
217  else if (fu <= fv || v == x || v == w)
218  {
219  v = u;
220  fv = fu;
221  }
222  }
223  }
224 
225  if (!quiet)
226  libMesh::out << "Warning! Too many iterations used in Brent line search!"
227  << std::endl;
228  return bx;
229 }

277 unsigned int NewtonSolver::solve()
278 {
279  START_LOG("solve()", "NewtonSolver");
280 
281  // Reset any prior solve result
282  _solve_result = INVALID_SOLVE_RESULT;
283 
284  NumericVector<Number> &newton_iterate = *(_system.solution);
285 
286  AutoPtr<NumericVector<Number> > linear_solution_ptr = newton_iterate.zero_clone();
287  NumericVector<Number> &linear_solution = *linear_solution_ptr;
288  NumericVector<Number> &rhs = *(_system.rhs);
289 
290  newton_iterate.close();
291  linear_solution.close();
292  rhs.close();
293 
294 #ifdef LIBMESH_ENABLE_CONSTRAINTS
295  _system.get_dof_map().enforce_constraints_exactly(_system);
296 #endif
297 
298  SparseMatrix<Number> &matrix = *(_system.matrix);
299 
300  // Set starting linear tolerance
301  Real current_linear_tolerance = initial_linear_tolerance;
302 
303  // Start counting our linear solver steps
304  _inner_iterations = 0;
305 
306  // Now we begin the nonlinear loop
307  for (_outer_iterations=0; _outer_iterations<max_nonlinear_iterations;
308  ++_outer_iterations)
309  {
310  if (verbose)
311  libMesh::out << "Assembling the System" << std::endl;
312 
313  _system.assembly(true, true);
314  rhs.close();
315  Real current_residual = rhs.l2_norm();
316 
317  if (libmesh_isnan(current_residual))
318  {
319  libMesh::out << " Nonlinear solver DIVERGED at step "
320  << _outer_iterations
321  << " with residual Not-a-Number"
322  << std::endl;
323  libmesh_convergence_failure();
324  continue;
325  }
326 
327  if (current_residual == 0)
328  {
329  if (verbose)
330  libMesh::out << "Linear solve unnecessary; residual = 0"
331  << std::endl;
332 
333  // We're not doing a solve, but other code may reuse this
334  // matrix.
335  matrix.close();
336 
337  if (absolute_residual_tolerance > 0)
338  _solve_result |= CONVERGED_ABSOLUTE_RESIDUAL;
339  if (relative_residual_tolerance > 0)
340  _solve_result |= CONVERGED_RELATIVE_RESIDUAL;
341  if (absolute_step_tolerance > 0)
342  _solve_result |= CONVERGED_ABSOLUTE_STEP;
343  if (relative_step_tolerance > 0)
344  _solve_result |= CONVERGED_RELATIVE_STEP;
345 
346  break;
347  }
348 
349  // Prepare to take incomplete steps
350  Real last_residual = current_residual;
351 
352  max_residual_norm = std::max (current_residual,
353  max_residual_norm);
354 
355  // Compute the l2 norm of the whole solution
356  Real norm_total = newton_iterate.l2_norm();
357 
358  max_solution_norm = std::max(max_solution_norm, norm_total);
359 
360  if (verbose)
361  libMesh::out << "Nonlinear Residual: "
362  << current_residual << std::endl;
363 
364  // Make sure our linear tolerance is low enough
365  current_linear_tolerance = std::min (current_linear_tolerance,
366  current_residual * linear_tolerance_multiplier);
367 
368  // But don't let it be too small
369  if (current_linear_tolerance < minimum_linear_tolerance)
370  {
371  current_linear_tolerance = minimum_linear_tolerance;
372  }
373 
374  // At this point newton_iterate is the current guess, and
375  // linear_solution is now about to become the NEGATIVE of the next
376  // Newton step.
377 
378  // Our best initial guess for the linear_solution is zero!
379  linear_solution.zero();
380 
381  if (verbose)
382  libMesh::out << "Linear solve starting, tolerance "
383  << current_linear_tolerance << std::endl;
384 
385  // Solve the linear system.
386  const std::pair<unsigned int, Real> rval =
387  linear_solver->solve (matrix, _system.request_matrix("Preconditioner"),
388  linear_solution, rhs, current_linear_tolerance,
389  max_linear_iterations);
390 
391  if (track_linear_convergence)
392  {
393  LinearConvergenceReason linear_c_reason = linear_solver->get_converged_reason();
394 
395  // Check if something went wrong during the linear solve
396  if (linear_c_reason < 0)
397  {
398  // The linear solver failed somehow
399  _solve_result |= DiffSolver::DIVERGED_LINEAR_SOLVER_FAILURE;
400  // Print a message
401  libMesh::out << "Linear solver failed during Newton step, dropping out."
402  << std::endl;
403  break;
404  }
405  }
406 
407  // We may need to localize a parallel solution
408  _system.update ();
409  // The linear solver may not have fit our constraints exactly
410 #ifdef LIBMESH_ENABLE_CONSTRAINTS
411  _system.get_dof_map().enforce_constraints_exactly
412  (_system, &linear_solution, /* homogeneous = */ true);
413 #endif
414 
415  const unsigned int linear_steps = rval.first;
416  libmesh_assert_less_equal (linear_steps, max_linear_iterations);
417  _inner_iterations += linear_steps;
418 
419  const bool linear_solve_finished =
420  !(linear_steps == max_linear_iterations);
421 
422  if (verbose)
423  libMesh::out << "Linear solve finished, step " << linear_steps
424  << ", residual " << rval.second
425  << std::endl;
426 
427  // Compute the l2 norm of the nonlinear update
428  Real norm_delta = linear_solution.l2_norm();
429 
430  if (verbose)
431  libMesh::out << "Trying full Newton step" << std::endl;
432  // Take a full Newton step
433  newton_iterate.add (-1., linear_solution);
434  newton_iterate.close();
435 
436  if (this->linear_solution_monitor.get())
437  {
438  // Compute the l2 norm of the whole solution
439  norm_total = newton_iterate.l2_norm();
440  rhs.close();
441  (*this->linear_solution_monitor)(linear_solution, norm_delta,
442  newton_iterate, norm_total,
443  rhs, rhs.l2_norm(), _outer_iterations);
444  }
445 
446  // Check residual with full Newton step, if that's useful for determining
447  // whether to line search, whether to quit early, or whether to die after
448  // hitting our max iteration count
449  if (this->require_residual_reduction ||
450  this->require_finite_residual ||
451  _outer_iterations+1 < max_nonlinear_iterations ||
452  !continue_after_max_iterations)
453  {
454  _system.assembly(true, false);
455 
456  rhs.close();
457  current_residual = rhs.l2_norm();
458  if (verbose)
459  libMesh::out << " Current Residual: "
460  << current_residual << std::endl;
461 
462  // don't fiddle around if we've already converged
463  if (test_convergence(current_residual, norm_delta,
464  linear_solve_finished &&
465  current_residual <= last_residual))
466  {
467  if (!quiet)
468  print_convergence(_outer_iterations, current_residual,
469  norm_delta, linear_solve_finished &&
470  current_residual <= last_residual);
471  _outer_iterations++;
472  break; // out of _outer_iterations for loop
473  }
474  }
475 
476  // since we're not converged, backtrack if necessary
477  Real steplength =
478  this->line_search(std::sqrt(TOLERANCE),
479  last_residual, current_residual,
480  newton_iterate, linear_solution);
481  norm_delta *= steplength;
482 
483  // Check to see if backtracking failed,
484  // and break out of the nonlinear loop if so...
485  if (_solve_result == DiffSolver::DIVERGED_BACKTRACKING_FAILURE)
486  {
487  _outer_iterations++;
488  break; // out of _outer_iterations for loop
489  }
490 
491  if (_outer_iterations + 1 >= max_nonlinear_iterations)
492  {
493  libMesh::out << " Nonlinear solver reached maximum step "
494  << max_nonlinear_iterations << ", latest evaluated residual "
495  << current_residual << std::endl;
496  if (continue_after_max_iterations)
497  {
498  _solve_result = DiffSolver::DIVERGED_MAX_NONLINEAR_ITERATIONS;
499  libMesh::out << " Continuing..." << std::endl;
500  }
501  else
502  {
503  libmesh_convergence_failure();
504  }
505  continue;
506  }
507 
508  // Compute the l2 norm of the whole solution
509  norm_total = newton_iterate.l2_norm();
510 
511  max_solution_norm = std::max(max_solution_norm, norm_total);
512 
513  // Print out information for the
514  // nonlinear iterations.
515  if (verbose)
516  libMesh::out << " Nonlinear step: |du|/|u| = "
517  << norm_delta / norm_total
518  << ", |du| = " << norm_delta
519  << std::endl;
520 
521  // Terminate the solution iteration if the difference between
522  // this iteration and the last is sufficiently small.
523  if (test_convergence(current_residual, norm_delta / steplength,
524  linear_solve_finished))
525  {
526  if (!quiet)
527  print_convergence(_outer_iterations, current_residual,
528  norm_delta / steplength,
529  linear_solve_finished);
530  _outer_iterations++;
531  break; // out of _outer_iterations for loop
532  }
533  } // end nonlinear loop
534 
535  // The linear solver may not have fit our constraints exactly
536 #ifdef LIBMESH_ENABLE_CONSTRAINTS
537  _system.get_dof_map().enforce_constraints_exactly(_system);
538 #endif
539 
540  // We may need to localize a parallel solution
541  _system.update ();
542 
543  STOP_LOG("solve()", "NewtonSolver");
544 
545  // Make sure we are returning something sensible as the
546  // _solve_result, except in the edge case where we weren't really asked to
547  // solve.
548  libmesh_assert (_solve_result != DiffSolver::INVALID_SOLVE_RESULT ||
549  !max_nonlinear_iterations);
550 
551  return _solve_result;
552 }

556 bool NewtonSolver::test_convergence(Real current_residual,
557  Real step_norm,
558  bool linear_solve_finished)
559 {
560  // We haven't converged unless we pass a convergence test
561  bool has_converged = false;
562 
563  // Is our absolute residual low enough?
564  if (current_residual < absolute_residual_tolerance)
565  {
566  _solve_result |= CONVERGED_ABSOLUTE_RESIDUAL;
567  has_converged = true;
568  }
569 
570  // Is our relative residual low enough?
571  if ((current_residual / max_residual_norm) <
572  relative_residual_tolerance)
573  {
574  _solve_result |= CONVERGED_RELATIVE_RESIDUAL;
575  has_converged = true;
576  }
577 
578  // For incomplete linear solves, it's not safe to test step sizes
579  if (!linear_solve_finished)
580  {
581  return has_converged;
582  }
583 
584  // Is our absolute Newton step size small enough?
585  if (step_norm < absolute_step_tolerance)
586  {
587  _solve_result |= CONVERGED_ABSOLUTE_STEP;
588  has_converged = true;
589  }
590 
591  // Is our relative Newton step size small enough?
592  if (step_norm / max_solution_norm <
593  relative_step_tolerance)
594  {
595  _solve_result |= CONVERGED_RELATIVE_STEP;
596  has_converged = true;
597  }
598 
599  return has_converged;
600 }

603 void NewtonSolver::print_convergence(unsigned int step_num,
604  Real current_residual,
605  Real step_norm,
606  bool linear_solve_finished)
607 {
608  // Is our absolute residual low enough?
609  if (current_residual < absolute_residual_tolerance)
610  {
611  libMesh::out << " Nonlinear solver converged, step " << step_num
612  << ", residual " << current_residual
613  << std::endl;
614  }
615  else if (absolute_residual_tolerance)
616  {
617  if (verbose)
618  libMesh::out << " Nonlinear solver current_residual "
619  << current_residual << " > "
620  << (absolute_residual_tolerance) << std::endl;
621  }
622 
623  // Is our relative residual low enough?
624  if ((current_residual / max_residual_norm) <
625  relative_residual_tolerance)
626  {
627  libMesh::out << " Nonlinear solver converged, step " << step_num
628  << ", residual reduction "
629  << current_residual / max_residual_norm
630  << " < " << relative_residual_tolerance
631  << std::endl;
632  }
633  else if (relative_residual_tolerance)
634  {
635  if (verbose)
636  libMesh::out << " Nonlinear solver relative residual "
637  << (current_residual / max_residual_norm)
638  << " > " << relative_residual_tolerance
639  << std::endl;
640  }
641 
642  // For incomplete linear solves, it's not safe to test step sizes
643  if (!linear_solve_finished)
644  return;
645 
646  // Is our absolute Newton step size small enough?
647  if (step_norm < absolute_step_tolerance)
648  {
649  libMesh::out << " Nonlinear solver converged, step " << step_num
650  << ", absolute step size "
651  << step_norm
652  << " < " << absolute_step_tolerance
653  << std::endl;
654  }
655  else if (absolute_step_tolerance)
656  {
657  if (verbose)
658  libMesh::out << " Nonlinear solver absolute step size "
659  << step_norm
660  << " > " << absolute_step_tolerance
661  << std::endl;
662  }
663 
664  // Is our relative Newton step size small enough?
665  if (step_norm / max_solution_norm <
666  relative_step_tolerance)
667  {
668  libMesh::out << " Nonlinear solver converged, step " << step_num
669  << ", relative step size "
670  << (step_norm / max_solution_norm)
671  << " < " << relative_step_tolerance
672  << std::endl;
673  }
674  else if (relative_step_tolerance)
675  {
676  if (verbose)
677  libMesh::out << " Nonlinear solver relative step size "
678  << (step_norm / max_solution_norm)
679  << " > " << relative_step_tolerance
680  << std::endl;
681  }
682 }
