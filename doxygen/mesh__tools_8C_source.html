341 MeshTools::create_bounding_box (const MeshBase & mesh)
342 {
343  // This function must be run on all processors at once
344  libmesh_parallel_only(mesh.comm());
345 
346  FindBBox find_bbox;
347 
348  // Start with any unpartitioned elements we know about locally
349  Threads::parallel_reduce (ConstElemRange (mesh.pid_elements_begin(DofObject::invalid_processor_id),
350  mesh.pid_elements_end(DofObject::invalid_processor_id)),
351  find_bbox);
352 
353  // And combine with our local elements
354  find_bbox.bbox().union_with(MeshTools::create_local_bounding_box(mesh));
355 
356  // Compare the bounding boxes across processors
357  mesh.comm().min(find_bbox.min());
358  mesh.comm().max(find_bbox.max());
359 
360  return find_bbox.bbox();

734  el = nodes_to_elem_map[global_id].begin(),
735  end_el = nodes_to_elem_map[global_id].end();
736 
737  // Look through the elements that contain this node
738  // find the local node id... then find the side that
739  // node lives on in the element
740  // next, look for the _other_ node on that side
741  // That other node is a "nodal_neighbor"... save it
742  for (; el != end_el; ++el)
743  {
744  // Grab an Elem pointer to use in the subsequent loop
745  const Elem * elem = *el;
746 
747  // We only care about active elements...
748  if (elem->active())
749  {
750  // Which local node number is global_id?
751  unsigned local_node_number = elem->local_node(global_id);
752 
753  // Make sure it was found
754  libmesh_assert_not_equal_to(local_node_number, libMesh::invalid_uint);
755 
756  // If this element has no edges, the edge-based algorithm below doesn't make sense.
757  if (elem->n_edges() == 0)
758  {
759  switch (elem->type())
760  {
761  case EDGE2:
762  {
763  switch (local_node_number)
764  {
765  case 0:
766  // The other node is a nodal neighbor
767  neighbor_set.insert(elem->node_ptr(1));
768  break;
769 
770  case 1:
771  // The other node is a nodal neighbor
772  neighbor_set.insert(elem->node_ptr(0));
773  break;
774 
775  default:
776  libmesh_error_msg("Invalid local node number: " << local_node_number << " found." << std::endl);
777  }
778  break;
779  }
780 
781  case EDGE3:
782  {
783  switch (local_node_number)
784  {
785  // The outside nodes have node 2 as a neighbor
786  case 0:
787  case 1:
788  neighbor_set.insert(elem->node_ptr(2));
789  break;
790 
791  // The middle node has the outer nodes as neighbors
792  case 2:
793  neighbor_set.insert(elem->node_ptr(0));
794  neighbor_set.insert(elem->node_ptr(1));
795  break;
796 
797  default:
798  libmesh_error_msg("Invalid local node number: " << local_node_number << " found." << std::endl);
799  }
800  break;
801  }
802 
803  case EDGE4:
804  {
805  switch (local_node_number)
806  {
807  case 0:
808  // The left-middle node is a nodal neighbor
809  neighbor_set.insert(elem->node_ptr(2));
810  break;
811 
812  case 1:
813  // The right-middle node is a nodal neighbor
814  neighbor_set.insert(elem->node_ptr(3));
815  break;
816 
817  // The left-middle node
818  case 2:
819  neighbor_set.insert(elem->node_ptr(0));
820  neighbor_set.insert(elem->node_ptr(3));
821  break;
822 
823  // The right-middle node
824  case 3:
825  neighbor_set.insert(elem->node_ptr(1));
826  neighbor_set.insert(elem->node_ptr(2));
827  break;
828 
829  default:
830  libmesh_error_msg("Invalid local node number: " << local_node_number << " found." << std::endl);
831  }
832  break;
833  }
834 
835  default:
836  libmesh_error_msg("Unrecognized ElemType: " << Utility::enum_to_string(elem->type()) << std::endl);
837  }
838  }
839 
840  // Index of the current edge
841  unsigned current_edge = 0;
842 
843  while (current_edge < elem->n_edges())
844  {
845  // Find the edge the node is on
846  bool found_edge = false;
847  for (; current_edge<elem->n_edges(); ++current_edge)
848  if (elem->is_node_on_edge(local_node_number, current_edge))
849  {
850  found_edge = true;
851  break;
852  }
853 
854  // Did we find one?
855  if (found_edge)
856  {
857  const Node * node_to_save = libmesh_nullptr;
858 
859  // Find another node in this element on this edge
860  for (unsigned other_node_this_edge = 0; other_node_this_edge<elem->n_nodes(); other_node_this_edge++)
861  if ( (elem->is_node_on_edge(other_node_this_edge, current_edge)) && // On the current edge
862  (elem->node_id(other_node_this_edge) != global_id)) // But not the original node
863  {
864  // We've found a nodal neighbor! Save a pointer to it..
865  node_to_save = elem->node_ptr(other_node_this_edge);
866  break;
867  }
868 
869  // Make sure we found something
870  libmesh_assert(node_to_save != libmesh_nullptr);
871 
872  neighbor_set.insert(node_to_save);
873  }
874 
875  // Keep looking for edges, node may be on more than one edge
876  current_edge++;
877  }
878  } // if (elem->active())
879  } // for
880 
881  // Assign the entries from the set to the vector. Note: this
882  // replaces any existing contents in neighbors and modifies its size
883  // accordingly.
884  neighbors.assign(neighbor_set.begin(), neighbor_set.end());
885 }

890  std::map<dof_id_type, std::vector<dof_id_type> > & hanging_nodes)

893  const MeshBase::const_element_iterator end = mesh.active_local_elements_end();

902  //libmesh_assert_equal_to (elem->type(), QUAD4);
903  if (elem->type() == QUAD4)
904  {
905  //Loop over the sides looking for sides that have hanging nodes
906  //This code is inspired by compute_proj_constraints()
907  for (unsigned int s=0; s<elem->n_sides(); s++)
908  {
909  //If not a boundary node
910  if (elem->neighbor_ptr(s) != libmesh_nullptr)
911  {
912  // Get pointers to the element's neighbor.
913  const Elem * neigh = elem->neighbor_ptr(s);
914 
915  //Is there a coarser element next to this one?
916  if (neigh->level() < elem->level())
917  {
918  const Elem * ancestor = elem;
919  while (neigh->level() < ancestor->level())
920  ancestor = ancestor->parent();

922  libmesh_assert_less (s_neigh, neigh->n_neighbors());
923 
924  //Couple of helper uints...
925  unsigned int local_node1=0;
926  unsigned int local_node2=0;
927 
928  bool found_in_neighbor = false;
929 
930  // Find the two vertices that make up this side
931  while (!elem->is_node_on_side(local_node1++,s)) { }
932  local_node1--;
933 
934  // Start looking for the second one with the next node
935  local_node2=local_node1+1;
936 
937  // Find the other one
938  while (!elem->is_node_on_side(local_node2++,s)) { }
939  local_node2--;
940 
941  //Pull out their global ids:
942  dof_id_type node1 = elem->node_id(local_node1);
943  dof_id_type node2 = elem->node_id(local_node2);
944 
945  //Now find which node is present in the neighbor
946  //FIXME This assumes a level one rule!
947  //The _other_ one is the hanging node
948 
949  //First look for the first one
950  //FIXME could be streamlined a bit
951  for (unsigned int n=0;n<neigh->n_sides();n++)
952  {
953  if (neigh->node_id(n) == node1)
954  found_in_neighbor=true;
955  }
956 
957  dof_id_type hanging_node=0;
958 
959  if (!found_in_neighbor)
960  hanging_node=node1;
961  else // If it wasn't node1 then it must be node2!
962  hanging_node=node2;
963 
964  // Reset these for reuse
965  local_node1=0;
966  local_node2=0;
967 
968  // Find the first node that makes up the side in the neighbor (these should be the parent nodes)
969  while (!neigh->is_node_on_side(local_node1++,s_neigh)) { }
970  local_node1--;
971 
972  local_node2=local_node1+1;
973 
974  //Find the second node...
975  while (!neigh->is_node_on_side(local_node2++,s_neigh)) { }
976  local_node2--;
977 
978  //Save them if we haven't already found the parents for this one
979  if (hanging_nodes[hanging_node].size()<2)
980  {
981  hanging_nodes[hanging_node].push_back(neigh->node_id(local_node1));
982  hanging_nodes[hanging_node].push_back(neigh->node_id(local_node2));
983  }
984  }
985  }

1854 void MeshTools::correct_node_proc_ids (MeshBase & mesh)
1855 {
1856  // This function must be run on all processors at once
1857  libmesh_parallel_only(mesh.comm());
1858 
1859  // Fix all nodes' processor ids. Coarsening may have left us with
1860  // nodes which are no longer touched by any elements of the same
1861  // processor id, and for DofMap to work we need to fix that.
1862 
1863  // This is harder now that libMesh no longer requires a distributed
1864  // mesh to ghost all nodal neighbors: it is possible for two active
1865  // elements on two different processors to share the same node in
1866  // such a way that neither processor knows the others' element
1867  // exists!
1868 
1869  // We require all processors to agree on nodal processor ids before
1870  // going into this algorithm.
1871 #ifdef DEBUG
1872  MeshTools::libmesh_assert_parallel_consistent_procids<Node>(mesh);
1873 #endif
1874 
1875  // We build up a set of compatible processor ids for each node
1876  proc_id_map_type new_proc_ids;
1877 
1878  MeshBase::element_iterator e_it = mesh.active_elements_begin();
1879  const MeshBase::element_iterator e_end = mesh.active_elements_end();
1880  for (; e_it != e_end; ++e_it)
1881  {
1882  Elem * elem = *e_it;
1883  processor_id_type pid = elem->processor_id();
1884 
1885  for (unsigned int n=0; n != elem->n_nodes(); ++n)
1886  {
1887  const Node & node = elem->node_ref(n);
1888  const dof_id_type id = node.id();
1889  const proc_id_map_type::iterator it = new_proc_ids.find(id);
1890  if (it == new_proc_ids.end())
1891  new_proc_ids.insert(std::make_pair(id,pid));
1892  else
1893  it->second = std::min(it->second, pid);
1894  }
1895  }
1896 
1897  // Sort the new pids to push to each processor
1898  std::vector<std::vector<std::pair<dof_id_type, processor_id_type> > >
1899  ids_to_push(mesh.n_processors());
1900 
1901  for (MeshBase::const_node_iterator
1902  n_it = mesh.nodes_begin(),
1903  n_end = mesh.nodes_end();
1904  n_it != n_end; ++n_it)
1905  {
1906  const Node * node = *n_it;
1907  const dof_id_type id = node->id();
1908  const proc_id_map_type::iterator it = new_proc_ids.find(id);
1909  if (it == new_proc_ids.end())
1910  continue;
1911  const processor_id_type pid = it->second;
1912  ids_to_push[node->processor_id()].push_back(std::make_pair(id, pid));
1913  }
1914 
1915  // Push using non-blocking I/O
1916  std::vector<Parallel::Request> push_requests(mesh.n_processors());
1917 
1918  for (processor_id_type p=1; p != mesh.n_processors(); ++p)
1919  {
1920  const processor_id_type procup =
1921  cast_int<processor_id_type>
1922  ((mesh.comm().rank() + p) % mesh.comm().size());
1923 
1924  mesh.comm().send(procup, ids_to_push[procup], push_requests[procup]);
1925  }
1926 
1927  for (processor_id_type p=0; p != mesh.n_processors(); ++p)
1928  {
1929  const processor_id_type procdown =
1930  cast_int<processor_id_type>
1931  ((mesh.comm().size() + mesh.comm().rank() - p) %
1932  mesh.comm().size());
1933 
1934  std::vector<std::pair<dof_id_type, processor_id_type> >
1935  ids_to_pull;
1936 
1937  if (p)
1938  mesh.comm().receive(procdown, ids_to_pull);
1939  else
1940  ids_to_pull.swap(ids_to_push[procdown]);
1941 
1942  std::vector<std::pair<dof_id_type, processor_id_type> >::iterator
1943  pulled_ids_it = ids_to_pull.begin(),
1944  pulled_ids_end = ids_to_pull.end();
1945  for (; pulled_ids_it != pulled_ids_end; ++pulled_ids_it)
1946  {
1947  const dof_id_type id = pulled_ids_it->first;
1948  const processor_id_type pid = pulled_ids_it->second;
1949  const proc_id_map_type::iterator it = new_proc_ids.find(id);
1950  if (it == new_proc_ids.end())
1951  new_proc_ids.insert(std::make_pair(id,pid));
1952  else
1953  it->second = std::min(it->second, pid);
1954  }
1955  }
1956 
1957  // Now new_proc_ids is correct for every node we used to own. Let's
1958  // ask every other processor about the nodes they used to own. But
1959  // first we'll need to keep track of which nodes we used to own,
1960  // lest we get them confused with nodes we newly own.
1961  LIBMESH_BEST_UNORDERED_SET<Node *> ex_local_nodes;
1962  for (MeshBase::node_iterator
1963  n_it = mesh.local_nodes_begin(),
1964  n_end = mesh.local_nodes_end();
1965  n_it != n_end; ++n_it)
1966  {
1967  Node * node = *n_it;
1968  const proc_id_map_type::iterator it = new_proc_ids.find(node->id());
1969  if (it != new_proc_ids.end() && it->second != mesh.processor_id())
1970  ex_local_nodes.insert(node);
1971  }
1972 
1973  // Let's finish with previous I/O before we start more.
1974  Parallel::wait(push_requests);
1975 
1976  SyncProcIdsFromMap sync(new_proc_ids, mesh);
1977  Parallel::sync_dofobject_data_by_id
1978  (mesh.comm(), mesh.nodes_begin(), mesh.nodes_end(), sync);
1979 
1980  // And finally let's update the nodes we used to own.
1981  for (LIBMESH_BEST_UNORDERED_SET<Node *>::iterator
1982  n_it = ex_local_nodes.begin(),
1983  n_end = ex_local_nodes.end();
1984  n_it != n_end; ++n_it)
1985  {
1986  Node * node = *n_it;
1987  const dof_id_type id = node->id();
1988  const proc_id_map_type::iterator it = new_proc_ids.find(id);
1989  libmesh_assert(it != new_proc_ids.end());
1990  node->processor_id() = it->second;
1991  }
1992 
1993  // We should still have consistent nodal processor ids coming out of
1994  // this algorithm.
1995 #ifdef DEBUG
1996  MeshTools::libmesh_assert_valid_procids<Node>(mesh);
1997 #endif
