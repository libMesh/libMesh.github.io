91 void ParmetisPartitioner::_do_repartition (MeshBase & mesh,
92  const unsigned int n_sbdmns)
93 {
94  libmesh_assert_greater (n_sbdmns, 0);
95 
96  // Check for an easy return
97  if (n_sbdmns == 1)
98  {
99  this->single_partition(mesh);
100  return;
101  }
102 
103  // This function must be run on all processors at once
104  libmesh_parallel_only(mesh.comm());
105 
106  // What to do if the Parmetis library IS NOT present
107 #ifndef LIBMESH_HAVE_PARMETIS
108 
109  libmesh_here();
110  libMesh::err << "ERROR: The library has been built without" << std::endl
111  << "Parmetis support. Using a Metis" << std::endl
112  << "partitioner instead!" << std::endl;
113 
114  MetisPartitioner mp;
115 
116  mp.partition (mesh, n_sbdmns);
117 
118  // What to do if the Parmetis library IS present
119 #else
120 
121  // Revert to METIS on one processor.
122  if (mesh.n_processors() == 1)
123  {
124  MetisPartitioner mp;
125  mp.partition (mesh, n_sbdmns);
126  return;
127  }
128 
129  START_LOG("repartition()", "ParmetisPartitioner");
130 
131  // Initialize the data structures required by ParMETIS
132  this->initialize (mesh, n_sbdmns);
133 
134  // Make sure all processors have enough active local elements.
135  // Parmetis tends to crash when it's given only a couple elements
136  // per partition.
137  {
138  bool all_have_enough_elements = true;
139  for (processor_id_type pid=0; pid<_n_active_elem_on_proc.size(); pid++)
140  if (_n_active_elem_on_proc[pid] < MIN_ELEM_PER_PROC)
141  all_have_enough_elements = false;
142 
143  // Parmetis will not work unless each processor has some
144  // elements. Specifically, it will abort when passed a NULL
145  // partition array on *any* of the processors.
146  if (!all_have_enough_elements)
147  {
148  // FIXME: revert to METIS, although this requires a serial mesh
149  MeshSerializer serialize(mesh);
150 
151  STOP_LOG ("repartition()", "ParmetisPartitioner");
152 
153  MetisPartitioner mp;
154  mp.partition (mesh, n_sbdmns);
155 
156  return;
157  }
158  }
159 
160  // build the graph corresponding to the mesh
161  this->build_graph (mesh);
162 
163 
164  // Partition the graph
165  std::vector<Parmetis::idx_t> vsize(_pmetis->vwgt.size(), 1);
166  Parmetis::real_t itr = 1000000.0;
167  MPI_Comm mpi_comm = mesh.comm().get();
168 
169  // Call the ParMETIS adaptive repartitioning method. This respects the
170  // original partitioning when computing the new partitioning so as to
171  // minimize the required data redistribution.
172  Parmetis::ParMETIS_V3_AdaptiveRepart(_pmetis->vtxdist.empty() ? libmesh_nullptr : &_pmetis->vtxdist[0],
173  _pmetis->xadj.empty() ? libmesh_nullptr : &_pmetis->xadj[0],
174  _pmetis->adjncy.empty() ? libmesh_nullptr : &_pmetis->adjncy[0],
175  _pmetis->vwgt.empty() ? libmesh_nullptr : &_pmetis->vwgt[0],
176  vsize.empty() ? libmesh_nullptr : &vsize[0],
177  libmesh_nullptr,
178  &_pmetis->wgtflag,
179  &_pmetis->numflag,
180  &_pmetis->ncon,
181  &_pmetis->nparts,
182  _pmetis->tpwgts.empty() ? libmesh_nullptr : &_pmetis->tpwgts[0],
183  _pmetis->ubvec.empty() ? libmesh_nullptr : &_pmetis->ubvec[0],
184  &itr,
185  &_pmetis->options[0],
186  &_pmetis->edgecut,
187  _pmetis->part.empty() ? libmesh_nullptr : &_pmetis->part[0],
188  &mpi_comm);
189 
190  // Assign the returned processor ids
191  this->assign_partitioning (mesh);
192 
193 
194  STOP_LOG ("repartition()", "ParmetisPartitioner");
195 
196 #endif // #ifndef LIBMESH_HAVE_PARMETIS ... else ...
197 
198 }

205 void ParmetisPartitioner::initialize (const MeshBase & mesh,
206  const unsigned int n_sbdmns)
207 {
208  const dof_id_type n_active_local_elem = mesh.n_active_local_elem();
209 
210  // Set parameters.
211  _pmetis->wgtflag = 2; // weights on vertices only
212  _pmetis->ncon = 1; // one weight per vertex
213  _pmetis->numflag = 0; // C-style 0-based numbering
214  _pmetis->nparts = static_cast<Parmetis::idx_t>(n_sbdmns); // number of subdomains to create
215  _pmetis->edgecut = 0; // the numbers of edges cut by the
216  // partition
217 
218  // Initialize data structures for ParMETIS
219  _pmetis->vtxdist.resize (mesh.n_processors()+1); std::fill (_pmetis->vtxdist.begin(), _pmetis->vtxdist.end(), 0);
220  _pmetis->tpwgts.resize (_pmetis->nparts); std::fill (_pmetis->tpwgts.begin(), _pmetis->tpwgts.end(), 1./_pmetis->nparts);
221  _pmetis->ubvec.resize (_pmetis->ncon); std::fill (_pmetis->ubvec.begin(), _pmetis->ubvec.end(), 1.05);
222  _pmetis->part.resize (n_active_local_elem); std::fill (_pmetis->part.begin(), _pmetis->part.end(), 0);
223  _pmetis->options.resize (5);
224  _pmetis->vwgt.resize (n_active_local_elem);
225 
226  // Set the options
227  _pmetis->options[0] = 1; // don't use default options
228  _pmetis->options[1] = 0; // default (level of timing)
229  _pmetis->options[2] = 15; // random seed (default)
230  _pmetis->options[3] = 2; // processor distribution and subdomain distribution are decoupled
231 
232  // Find the number of active elements on each processor. We cannot use
233  // mesh.n_active_elem_on_proc(pid) since that only returns the number of
234  // elements assigned to pid which are currently stored on the calling
235  // processor. This will not in general be correct for parallel meshes
236  // when (pid!=mesh.processor_id()).
237  _n_active_elem_on_proc.resize(mesh.n_processors());
238  mesh.comm().allgather(n_active_local_elem, _n_active_elem_on_proc);
239 
240  // count the total number of active elements in the mesh. Note we cannot
241  // use mesh.n_active_elem() in general since this only returns the number
242  // of active elements which are stored on the calling processor.
243  // We should not use n_active_elem for any allocation because that will
244  // be inheritly unscalable, but it can be useful for libmesh_assertions.
245  dof_id_type n_active_elem=0;
246 
247  // Set up the vtxdist array. This will be the same on each processor.
248  // ***** Consult the Parmetis documentation. *****
249  libmesh_assert_equal_to (_pmetis->vtxdist.size(),
250  cast_int<std::size_t>(mesh.n_processors()+1));
251  libmesh_assert_equal_to (_pmetis->vtxdist[0], 0);
252 
253  for (processor_id_type pid=0; pid<mesh.n_processors(); pid++)
254  {
255  _pmetis->vtxdist[pid+1] = _pmetis->vtxdist[pid] + _n_active_elem_on_proc[pid];
256  n_active_elem += _n_active_elem_on_proc[pid];
257  }
258  libmesh_assert_equal_to (_pmetis->vtxdist.back(), static_cast<Parmetis::idx_t>(n_active_elem));
259 
260  // ParMetis expects the elements to be numbered in contiguous blocks
261  // by processor, i.e. [0, ne0), [ne0, ne0+ne1), ...
262  // Since we only partition active elements we should have no expectation
263  // that we currently have such a distribution. So we need to create it.
264  // Also, at the same time we are going to map all the active elements into a globally
265  // unique range [0,n_active_elem) which is *independent* of the current partitioning.
266  // This can be fed to ParMetis as the initial partitioning of the subdomains (decoupled
267  // from the partitioning of the objects themselves). This allows us to get the same
268  // resultant partitioning independed of the input partitioning.
269  MeshTools::BoundingBox bbox =
270  MeshTools::bounding_box(mesh);
271 
272  _global_index_by_pid_map.clear();
273 
274  // Maps active element ids into a contiguous range independent of partitioning.
275  // (only needs local scope)
276  vectormap<dof_id_type, dof_id_type> global_index_map;
277 
278  {
279  std::vector<dof_id_type> global_index;
280 
281  // create the mapping which is contiguous by processor
282  dof_id_type pid_offset=0;
283  for (processor_id_type pid=0; pid<mesh.n_processors(); pid++)
284  {
285  MeshBase::const_element_iterator it = mesh.active_pid_elements_begin(pid);
286  const MeshBase::const_element_iterator end = mesh.active_pid_elements_end(pid);
287 
288  // note that we may not have all (or any!) the active elements which belong on this processor,
289  // but by calling this on all processors a unique range in [0,_n_active_elem_on_proc[pid])
290  // is constructed. Only the indices for the elements we pass in are returned in the array.
291  MeshCommunication().find_global_indices (mesh.comm(),
292  bbox, it, end,
293  global_index);
294 
295  for (dof_id_type cnt=0; it != end; ++it)
296  {
297  const Elem * elem = *it;
298  libmesh_assert (!_global_index_by_pid_map.count(elem->id()));
299  libmesh_assert_less (cnt, global_index.size());
300  libmesh_assert_less (global_index[cnt], _n_active_elem_on_proc[pid]);
301 
302  _global_index_by_pid_map.insert(std::make_pair(elem->id(), global_index[cnt++] + pid_offset));
303  }
304 
305  pid_offset += _n_active_elem_on_proc[pid];
306  }
307 
308  // create the unique mapping for all active elements independent of partitioning
309  {
310  MeshBase::const_element_iterator it = mesh.active_elements_begin();
311  const MeshBase::const_element_iterator end = mesh.active_elements_end();
312 
313  // Calling this on all processors a unique range in [0,n_active_elem) is constructed.
314  // Only the indices for the elements we pass in are returned in the array.
315  MeshCommunication().find_global_indices (mesh.comm(),
316  bbox, it, end,
317  global_index);
318 
319  for (dof_id_type cnt=0; it != end; ++it)
320  {
321  const Elem * elem = *it;
322  libmesh_assert (!global_index_map.count(elem->id()));
323  libmesh_assert_less (cnt, global_index.size());
324  libmesh_assert_less (global_index[cnt], n_active_elem);
325 
326  global_index_map.insert(std::make_pair(elem->id(), global_index[cnt++]));
327  }
328  }
329  // really, shouldn't be close!
330  libmesh_assert_less_equal (global_index_map.size(), n_active_elem);
331  libmesh_assert_less_equal (_global_index_by_pid_map.size(), n_active_elem);
332 
333  // At this point the two maps should be the same size. If they are not
334  // then the number of active elements is not the same as the sum over all
335  // processors of the number of active elements per processor, which means
336  // there must be some unpartitioned objects out there.
337  if (global_index_map.size() != _global_index_by_pid_map.size())
338  libmesh_error_msg("ERROR: ParmetisPartitioner cannot handle unpartitioned objects!");
339  }
340 
341  // Finally, we need to initialize the vertex (partition) weights and the initial subdomain
342  // mapping. The subdomain mapping will be independent of the processor mapping, and is
343  // defined by a simple mapping of the global indices we just found.
344  {
345  std::vector<dof_id_type> subdomain_bounds(mesh.n_processors());
346 
347  const dof_id_type first_local_elem = _pmetis->vtxdist[mesh.processor_id()];
348 
349  for (processor_id_type pid=0; pid<mesh.n_processors(); pid++)
350  {
351  dof_id_type tgt_subdomain_size = 0;
352 
353  // watch out for the case that n_subdomains < n_processors
354  if (pid < static_cast<unsigned int>(_pmetis->nparts))
355  {
356  tgt_subdomain_size = n_active_elem/std::min
357  (cast_int<Parmetis::idx_t>(mesh.n_processors()), _pmetis->nparts);
358 
359  if (pid < n_active_elem%_pmetis->nparts)
360  tgt_subdomain_size++;
361  }
362  if (pid == 0)
363  subdomain_bounds[0] = tgt_subdomain_size;
364  else
365  subdomain_bounds[pid] = subdomain_bounds[pid-1] + tgt_subdomain_size;
366  }
367 
368  libmesh_assert_equal_to (subdomain_bounds.back(), n_active_elem);
369 
370  MeshBase::const_element_iterator elem_it = mesh.active_local_elements_begin();
371  const MeshBase::const_element_iterator elem_end = mesh.active_local_elements_end();
372 
373  for (; elem_it != elem_end; ++elem_it)
374  {
375  const Elem * elem = *elem_it;
376 
377  libmesh_assert (_global_index_by_pid_map.count(elem->id()));
378  const dof_id_type global_index_by_pid =
379  _global_index_by_pid_map[elem->id()];
380  libmesh_assert_less (global_index_by_pid, n_active_elem);
381 
382  const dof_id_type local_index =
383  global_index_by_pid - first_local_elem;
384 
385  libmesh_assert_less (local_index, n_active_local_elem);
386  libmesh_assert_less (local_index, _pmetis->vwgt.size());
387 
388  // TODO:[BSK] maybe there is a better weight?
389  _pmetis->vwgt[local_index] = elem->n_nodes();
390 
391  // find the subdomain this element belongs in
392  libmesh_assert (global_index_map.count(elem->id()));
393  const dof_id_type global_index =
394  global_index_map[elem->id()];
395 
396  libmesh_assert_less (global_index, subdomain_bounds.back());
397 
398  const unsigned int subdomain_id =
399  std::distance(subdomain_bounds.begin(),
400  std::lower_bound(subdomain_bounds.begin(),
401  subdomain_bounds.end(),
402  global_index));
403  libmesh_assert_less (subdomain_id, static_cast<unsigned int>(_pmetis->nparts));
404  libmesh_assert_less (local_index, _pmetis->part.size());
405 
406  _pmetis->part[local_index] = subdomain_id;
407  }
408  }

413 void ParmetisPartitioner::build_graph (const MeshBase & mesh)
414 {
415  // build the graph in distributed CSR format. Note that
416  // the edges in the graph will correspond to
417  // face neighbors
418  const dof_id_type n_active_local_elem = mesh.n_active_local_elem();
419 
420  // If we have boundary elements in this mesh, we want to account for
421  // the connectivity between them and interior elements. We can find
422  // interior elements from boundary elements, but we need to build up
423  // a lookup map to do the reverse.
424 
425  typedef LIBMESH_BEST_UNORDERED_MULTIMAP<const Elem *, const Elem *>
426  map_type;
427  map_type interior_to_boundary_map;
428 
429  {
430  MeshBase::const_element_iterator elem_it = mesh.active_elements_begin();
431  const MeshBase::const_element_iterator elem_end = mesh.active_elements_end();
432 
433  for (; elem_it != elem_end; ++elem_it)
434  {
435  const Elem * elem = *elem_it;
436 
437  // If we don't have an interior_parent then there's nothing to look us
438  // up.
439  if ((elem->dim() >= LIBMESH_DIM) ||
440  !elem->interior_parent())
441  continue;
442 
443  // get all relevant interior elements
444  std::set<const Elem *> neighbor_set;
445  elem->find_interior_neighbors(neighbor_set);
446 
447  std::set<const Elem *>::iterator n_it = neighbor_set.begin();
448  for (; n_it != neighbor_set.end(); ++n_it)
449  {
450  // FIXME - non-const versions of the Elem set methods
451  // would be nice
452  Elem * neighbor = const_cast<Elem *>(*n_it);
453 
454 #if defined(LIBMESH_HAVE_UNORDERED_MULTIMAP) || \
455  defined(LIBMESH_HAVE_TR1_UNORDERED_MAP) || \
456  defined(LIBMESH_HAVE_HASH_MAP) || \
457  defined(LIBMESH_HAVE_EXT_HASH_MAP)
458  interior_to_boundary_map.insert
459  (std::make_pair(neighbor, elem));
460 #else
461  interior_to_boundary_map.insert
462  (interior_to_boundary_map.begin(),
463  std::make_pair(neighbor, elem));
464 #endif
465  }
466  }
467  }
468 
469 #ifdef LIBMESH_ENABLE_AMR
470  std::vector<const Elem *> neighbors_offspring;
471 #endif
472 
473  std::vector<std::vector<dof_id_type> > graph(n_active_local_elem);
474  dof_id_type graph_size=0;
475 
476  const dof_id_type first_local_elem = _pmetis->vtxdist[mesh.processor_id()];
477 
478  MeshBase::const_element_iterator elem_it = mesh.active_local_elements_begin();
479  const MeshBase::const_element_iterator elem_end = mesh.active_local_elements_end();
480 
481  for (; elem_it != elem_end; ++elem_it)
482  {
483  const Elem * elem = *elem_it;
484 
485  libmesh_assert (_global_index_by_pid_map.count(elem->id()));
486  const dof_id_type global_index_by_pid =
487  _global_index_by_pid_map[elem->id()];
488 
489  const dof_id_type local_index =
490  global_index_by_pid - first_local_elem;
491  libmesh_assert_less (local_index, n_active_local_elem);
492 
493  std::vector<dof_id_type> & graph_row = graph[local_index];
494 
495  // Loop over the element's neighbors. An element
496  // adjacency corresponds to a face neighbor
497  for (unsigned int ms=0; ms<elem->n_neighbors(); ms++)
498  {
499  const Elem * neighbor = elem->neighbor(ms);
500 
501  if (neighbor != libmesh_nullptr)
502  {
503  // If the neighbor is active treat it
504  // as a connection
505  if (neighbor->active())
506  {
507  libmesh_assert(_global_index_by_pid_map.count(neighbor->id()));
508  const dof_id_type neighbor_global_index_by_pid =
509  _global_index_by_pid_map[neighbor->id()];
510 
511  graph_row.push_back(neighbor_global_index_by_pid);
512  graph_size++;
513  }
514 
515 #ifdef LIBMESH_ENABLE_AMR
516 
517  // Otherwise we need to find all of the
518  // neighbor's children that are connected to
519  // us and add them
520  else
521  {
522  // The side of the neighbor to which
523  // we are connected
524  const unsigned int ns =
525  neighbor->which_neighbor_am_i (elem);
526  libmesh_assert_less (ns, neighbor->n_neighbors());
527 
528  // Get all the active children (& grandchildren, etc...)
529  // of the neighbor
530 
531  // FIXME - this is the wrong thing, since we
532  // should be getting the active family tree on
533  // our side only. But adding too many graph
534  // links may cause hanging nodes to tend to be
535  // on partition interiors, which would reduce
536  // communication overhead for constraint
537  // equations, so we'll leave it.
538 
539  neighbor->active_family_tree (neighbors_offspring);
540 
541  // Get all the neighbor's children that
542  // live on that side and are thus connected
543  // to us
544  for (unsigned int nc=0; nc<neighbors_offspring.size(); nc++)
545  {
546  const Elem * child =
547  neighbors_offspring[nc];
548 
549  // This does not assume a level-1 mesh.
550  // Note that since children have sides numbered
551  // coincident with the parent then this is a sufficient test.
552  if (child->neighbor(ns) == elem)
553  {
554  libmesh_assert (child->active());
555  libmesh_assert (_global_index_by_pid_map.count(child->id()));
556  const dof_id_type child_global_index_by_pid =
557  _global_index_by_pid_map[child->id()];
558 
559  graph_row.push_back(child_global_index_by_pid);
560  graph_size++;
561  }
562  }
563  }
564 
565 #endif /* ifdef LIBMESH_ENABLE_AMR */
566 
567 
568  }
569  }
570 
571  if ((elem->dim() < LIBMESH_DIM) &&
572  elem->interior_parent())
573  {
574  // get all relevant interior elements
575  std::set<const Elem *> neighbor_set;
576  elem->find_interior_neighbors(neighbor_set);
577 
578  std::set<const Elem *>::iterator n_it = neighbor_set.begin();
579  for (; n_it != neighbor_set.end(); ++n_it)
580  {
581  // FIXME - non-const versions of the Elem set methods
582  // would be nice
583  Elem * neighbor = const_cast<Elem *>(*n_it);
584 
585  const dof_id_type neighbor_global_index_by_pid =
586  _global_index_by_pid_map[neighbor->id()];
587 
588  graph_row.push_back(neighbor_global_index_by_pid);
589  graph_size++;
590  }
591  }
592 
593  // Check for any boundary neighbors
594  typedef map_type::iterator map_it_type;
595  std::pair<map_it_type, map_it_type>
596  bounds = interior_to_boundary_map.equal_range(elem);
597

599  {
600  const Elem * neighbor = it->second;
601 
602  const dof_id_type neighbor_global_index_by_pid =
603  _global_index_by_pid_map[neighbor->id()];
604 
605  graph_row.push_back(neighbor_global_index_by_pid);
606  graph_size++;
607  }
608  }
609 
610  // Reserve space in the adjacency array
611  _pmetis->xadj.clear();
612  _pmetis->xadj.reserve (n_active_local_elem + 1);
613  _pmetis->adjncy.clear();
614  _pmetis->adjncy.reserve (graph_size);
615 
616  for (std::size_t r=0; r<graph.size(); r++)
617  {
618  _pmetis->xadj.push_back(_pmetis->adjncy.size());
619  std::vector<dof_id_type> graph_row; // build this emtpy
620  graph_row.swap(graph[r]); // this will deallocate at the end of scope
621  _pmetis->adjncy.insert(_pmetis->adjncy.end(),
622  graph_row.begin(),
623  graph_row.end());
624  }
625 
626  // The end of the adjacency array for the last elem
627  _pmetis->xadj.push_back(_pmetis->adjncy.size());
628 
629  libmesh_assert_equal_to (_pmetis->xadj.size(), n_active_local_elem+1);
630  libmesh_assert_equal_to (_pmetis->adjncy.size(), graph_size);
631 }

635 void ParmetisPartitioner::assign_partitioning (MeshBase & mesh)
636 {
637  // This function must be run on all processors at once
638  libmesh_parallel_only(mesh.comm());
639 
640  const dof_id_type
641  first_local_elem = _pmetis->vtxdist[mesh.processor_id()];
642 
643  std::vector<std::vector<dof_id_type> >
644  requested_ids(mesh.n_processors()),
645  requests_to_fill(mesh.n_processors());
646 
647  MeshBase::element_iterator elem_it = mesh.active_elements_begin();
648  MeshBase::element_iterator elem_end = mesh.active_elements_end();
649 
650  for (; elem_it != elem_end; ++elem_it)
651  {
652  Elem * elem = *elem_it;
653 
654  // we need to get the index from the owning processor
655  // (note we cannot assign it now -- we are iterating
656  // over elements again and this will be bad!)
657  libmesh_assert_less (elem->processor_id(), requested_ids.size());
658  requested_ids[elem->processor_id()].push_back(elem->id());
659  }
660 
661  // Trade with all processors (including self) to get their indices
662  for (processor_id_type pid=0; pid<mesh.n_processors(); pid++)
663  {
664  // Trade my requests with processor procup and procdown
665  const processor_id_type procup = (mesh.processor_id() + pid) % mesh.n_processors();
666  const processor_id_type procdown = (mesh.n_processors() +
667  mesh.processor_id() - pid) % mesh.n_processors();
668 
669  mesh.comm().send_receive (procup, requested_ids[procup],
670  procdown, requests_to_fill[procdown]);
671 
672  // we can overwrite these requested ids in-place.
673  for (std::size_t i=0; i<requests_to_fill[procdown].size(); i++)
674  {
675  const dof_id_type requested_elem_index =
676  requests_to_fill[procdown][i];
677 
678  libmesh_assert(_global_index_by_pid_map.count(requested_elem_index));
679 
680  const dof_id_type global_index_by_pid =
681  _global_index_by_pid_map[requested_elem_index];
682 
683  const dof_id_type local_index =
684  global_index_by_pid - first_local_elem;
685 
686  libmesh_assert_less (local_index, _pmetis->part.size());
687  libmesh_assert_less (local_index, mesh.n_active_local_elem());
688 
689  const unsigned int elem_procid =
690  static_cast<unsigned int>(_pmetis->part[local_index]);
691 
692  libmesh_assert_less (elem_procid, static_cast<unsigned int>(_pmetis->nparts));
693 
694  requests_to_fill[procdown][i] = elem_procid;
695  }
696 
697  // Trade back
698  mesh.comm().send_receive (procdown, requests_to_fill[procdown],
699  procup, requested_ids[procup]);
700  }
701 
702  // and finally assign the partitioning.
703  // note we are iterating in exactly the same order
704  // used to build up the request, so we can expect the
705  // required entries to be in the proper sequence.
706  elem_it = mesh.active_elements_begin();
707  elem_end = mesh.active_elements_end();
708 
709  for (std::vector<unsigned int> counters(mesh.n_processors(), 0);
710  elem_it != elem_end; ++elem_it)
711  {
712  Elem * elem = *elem_it;
713 
714  const processor_id_type current_pid = elem->processor_id();
715 
716  libmesh_assert_less (counters[current_pid], requested_ids[current_pid].size());
717 
718  const processor_id_type elem_procid =
719  requested_ids[current_pid][counters[current_pid]++];
720 
721  libmesh_assert_less (elem_procid, static_cast<unsigned int>(_pmetis->nparts));
722  elem->processor_id() = elem_procid;
723  }
724 }
