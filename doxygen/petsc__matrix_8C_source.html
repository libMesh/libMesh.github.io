62 // historic libMesh n_nz & n_oz arrays are set up for PETSc's AIJ format.
63 // however, when the blocksize is >1, we need to transform these into
64 // their BAIJ counterparts.
65 inline
66 void transform_preallocation_arrays (const PetscInt blocksize,
67  const std::vector<numeric_index_type> & n_nz,
68  const std::vector<numeric_index_type> & n_oz,
69  std::vector<numeric_index_type> & b_n_nz,
70  std::vector<numeric_index_type> & b_n_oz)
71 {
72  libmesh_assert_equal_to (n_nz.size(), n_oz.size());
73  libmesh_assert_equal_to (n_nz.size()%blocksize, 0);
74 
75  b_n_nz.clear(); b_n_nz.reserve(n_nz.size()/blocksize);
76  b_n_oz.clear(); b_n_oz.reserve(n_oz.size()/blocksize);
77 
78  for (unsigned int nn=0; nn<n_nz.size(); nn += blocksize)
79  {
80  b_n_nz.push_back (n_nz[nn]/blocksize);
81  b_n_oz.push_back (n_oz[nn]/blocksize);
82  }
83 }
84 }

129 template <typename T>
130 void PetscMatrix<T>::init (const numeric_index_type m_in,
131  const numeric_index_type n_in,
132  const numeric_index_type m_l,
133  const numeric_index_type n_l,
134  const numeric_index_type nnz,
135  const numeric_index_type noz,
136  const numeric_index_type blocksize_in)
137 {
138  // So compilers don't warn when !LIBMESH_ENABLE_BLOCKED_STORAGE
139  libmesh_ignore(blocksize_in);
140 
141  // Clear initialized matrices
142  if (this->initialized())
143  this->clear();
144 
145  this->_is_initialized = true;
146 
147 
148  PetscErrorCode ierr = 0;
149  PetscInt m_global = static_cast<PetscInt>(m_in);
150  PetscInt n_global = static_cast<PetscInt>(n_in);
151  PetscInt m_local = static_cast<PetscInt>(m_l);
152  PetscInt n_local = static_cast<PetscInt>(n_l);
153  PetscInt n_nz = static_cast<PetscInt>(nnz);
154  PetscInt n_oz = static_cast<PetscInt>(noz);
155 
156  ierr = MatCreate(this->comm().get(), &_mat);
157  LIBMESH_CHKERR(ierr);
158  ierr = MatSetSizes(_mat, m_local, n_local, m_global, n_global);
159  LIBMESH_CHKERR(ierr);
160 
161 #ifdef LIBMESH_ENABLE_BLOCKED_STORAGE
162  PetscInt blocksize = static_cast<PetscInt>(blocksize_in);
163  if (blocksize > 1)
164  {
165  // specified blocksize, bs>1.
166  // double check sizes.
167  libmesh_assert_equal_to (m_local % blocksize, 0);
168  libmesh_assert_equal_to (n_local % blocksize, 0);
169  libmesh_assert_equal_to (m_global % blocksize, 0);
170  libmesh_assert_equal_to (n_global % blocksize, 0);
171  libmesh_assert_equal_to (n_nz % blocksize, 0);
172  libmesh_assert_equal_to (n_oz % blocksize, 0);
173 
174  ierr = MatSetType(_mat, MATBAIJ); // Automatically chooses seqbaij or mpibaij
175  LIBMESH_CHKERR(ierr);
176  ierr = MatSetBlockSize(_mat, blocksize);
177  LIBMESH_CHKERR(ierr);
178  ierr = MatSeqBAIJSetPreallocation(_mat, blocksize, n_nz/blocksize, PETSC_NULL);
179  LIBMESH_CHKERR(ierr);
180  ierr = MatMPIBAIJSetPreallocation(_mat, blocksize,
181  n_nz/blocksize, PETSC_NULL,
182  n_oz/blocksize, PETSC_NULL);
183  LIBMESH_CHKERR(ierr);
184  }
185  else
186 #endif
187  {
188  ierr = MatSetType(_mat, MATAIJ); // Automatically chooses seqaij or mpiaij
189  LIBMESH_CHKERR(ierr);
190  ierr = MatSeqAIJSetPreallocation(_mat, n_nz, PETSC_NULL);
191  LIBMESH_CHKERR(ierr);
192  ierr = MatMPIAIJSetPreallocation(_mat, n_nz, PETSC_NULL, n_oz, PETSC_NULL);
193  LIBMESH_CHKERR(ierr);
194  }
195 
196  // Make it an error for PETSc to allocate new nonzero entries during assembly
197 #if PETSC_VERSION_LESS_THAN(3,0,0)
198  ierr = MatSetOption(_mat, MAT_NEW_NONZERO_ALLOCATION_ERR);
199 #else
200  ierr = MatSetOption(_mat, MAT_NEW_NONZERO_ALLOCATION_ERR, PETSC_TRUE);
201 #endif
202  LIBMESH_CHKERR(ierr);
203 
204  // Is prefix information available somewhere? Perhaps pass in the system name?
205  ierr = MatSetOptionsPrefix(_mat, "");
206  LIBMESH_CHKERR(ierr);
207  ierr = MatSetFromOptions(_mat);
208  LIBMESH_CHKERR(ierr);
209 
210  this->zero ();
211 }

501  if(!rows.empty())
502  ierr = MatZeroRows(_mat, cast_int<PetscInt>(rows.size()),

504  PETSC_NULL, PETSC_NULL);
505  else
506  ierr = MatZeroRows(_mat, 0, PETSC_NULL, diag_value, PETSC_NULL,
507  PETSC_NULL);
508 #endif

510  LIBMESH_CHKERR(ierr);

651 template <typename T>
652 void PetscMatrix<T>::print_personal(std::ostream & os) const
653 {
654  libmesh_assert (this->initialized());
655 
656  // Routine must be called in parallel on parallel matrices
657  // and serial on serial matrices.
658  semiparallel_only();
659 
660  // #ifndef NDEBUG
661  // if (os != std::cout)
662  // libMesh::err << "Warning! PETSc can only print to std::cout!" << std::endl;
663  // #endif
664 
665  // Matrix must be in an assembled state to be printed
666  this->close();
667 
668  PetscErrorCode ierr=0;
669 
670  // Print to screen if ostream is stdout
671  if (os.rdbuf() == std::cout.rdbuf())
672  {
673  ierr = MatView(_mat, PETSC_VIEWER_STDOUT_SELF);
674  LIBMESH_CHKERR(ierr);
675  }
676 
677  // Otherwise, print to the requested file, in a roundabout way...
678  else
679  {
680  // We will create a temporary filename, and file, for PETSc to
681  // write to.
682  std::string temp_filename;
683 
684  {
685  // Template for temporary filename
686  char c[] = "temp_petsc_matrix.XXXXXX";
687 
688  // Generate temporary, unique filename only on processor 0. We will
689  // use this filename for PetscViewerASCIIOpen, before copying it into
690  // the user's stream
691  if (this->processor_id() == 0)
692  {
693  int fd = mkstemp(c);
694 
695  // Check to see that mkstemp did not fail.
696  if (fd == -1)
697  libmesh_error_msg("mkstemp failed in PetscMatrix::print_personal()");
698 
699  // mkstemp returns a file descriptor for an open file,
700  // so let's close it before we hand it to PETSc!
701  ::close (fd);
702  }
703 
704  // Store temporary filename as string, makes it easier to broadcast
705  temp_filename = c;
706  }
707 
708  // Now broadcast the filename from processor 0 to all processors.
709  this->comm().broadcast(temp_filename);
710 
711  // PetscViewer object for passing to MatView
712  PetscViewer petsc_viewer;
713 
714  // This PETSc function only takes a string and handles the opening/closing
715  // of the file internally. Since print_personal() takes a reference to
716  // an ostream, we have to do an extra step... print_personal() should probably
717  // have a version that takes a string to get rid of this problem.
718  ierr = PetscViewerASCIIOpen( this->comm().get(),
719  temp_filename.c_str(),
720  &petsc_viewer);
721  LIBMESH_CHKERR(ierr);
722 
723  // Probably don't need to set the format if it's default...
724  // ierr = PetscViewerSetFormat (petsc_viewer,
725  // PETSC_VIEWER_DEFAULT);
726  // LIBMESH_CHKERR(ierr);
727 
728  // Finally print the matrix using the viewer
729  ierr = MatView (_mat, petsc_viewer);
730  LIBMESH_CHKERR(ierr);
731 
732  if (this->processor_id() == 0)
733  {
734  // Now the inefficient bit: open temp_filename as an ostream and copy the contents
735  // into the user's desired ostream. We can't just do a direct file copy, we don't even have the filename!
736  std::ifstream input_stream(temp_filename.c_str());
737  os << input_stream.rdbuf(); // The "most elegant" way to copy one stream into another.
738  // os.close(); // close not defined in ostream
739 
740  // Now remove the temporary file
741  input_stream.close();
742  std::remove(temp_filename.c_str());
743  }
744  }

1092 template <typename T>
1093 T PetscMatrix<T>::operator () (const numeric_index_type i_in,
1094  const numeric_index_type j_in) const
1095 {
1096  libmesh_assert (this->initialized());
1097 
1098  // PETSc 2.2.1 & newer
1099  const PetscScalar * petsc_row;
1100  const PetscInt * petsc_cols;
1101 
1102  // If the entry is not in the sparse matrix, it is 0.
1103  T value=0.;
1104 
1105  PetscErrorCode
1106  ierr=0;
1107  PetscInt
1108  ncols=0,
1109  i_val=static_cast<PetscInt>(i_in),
1110  j_val=static_cast<PetscInt>(j_in);
1111 
1112 
1113  // the matrix needs to be closed for this to work
1114  // this->close();
1115  // but closing it is a semiparallel operation; we want operator()
1116  // to run on one processor.
1117  libmesh_assert(this->closed());
1118 
1119  ierr = MatGetRow(_mat, i_val, &ncols, &petsc_cols, &petsc_row);
1120  LIBMESH_CHKERR(ierr);
1121 
1122  // Perform a binary search to find the contiguous index in
1123  // petsc_cols (resp. petsc_row) corresponding to global index j_val
1124  std::pair<const PetscInt *, const PetscInt *> p =
1125  std::equal_range (&petsc_cols[0], &petsc_cols[0] + ncols, j_val);
1126 
1127  // Found an entry for j_val
1128  if (p.first != p.second)
1129  {
1130  // The entry in the contiguous row corresponding
1131  // to the j_val column of interest
1132  const std::size_t j =
1133  std::distance (const_cast<PetscInt *>(&petsc_cols[0]),
1134  const_cast<PetscInt *>(p.first));
1135 
1136  libmesh_assert_less (static_cast<PetscInt>(j), ncols);
1137  libmesh_assert_equal_to (petsc_cols[j], j_val);
1138 
1139  value = static_cast<T> (petsc_row[j]);
1140  }
1141 
1142  ierr = MatRestoreRow(_mat, i_val,
1143  &ncols, &petsc_cols, &petsc_row);
1144  LIBMESH_CHKERR(ierr);
1145 
1146  return value;
1147 }
