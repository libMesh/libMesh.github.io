47 void QGrundmann_Moller::gm_rule(unsigned int s, unsigned int dim)
48 {
49  // Only dim==2 or dim==3 is allowed
50  libmesh_assert(dim==2 || dim==3);
51 
52  // A GM rule of index s can integrate polynomials of degree 2*s+1 exactly
53  const unsigned int degree = 2*s+1;
54 
55  // The number of points for rule of index s is
56  // (dim+1+s)! / (dim+1)! / s!
57  // In 3D, this is = 1/24 * Pi_{i=1}^4 (s+i)
58  // In 2D, this is = 1/6 * Pi_{i=1}^3 (s+i)
59  const unsigned int n_pts = dim==2 ? (s+3)*(s+2)*(s+1) / 6 : (s+4)*(s+3)*(s+2)*(s+1) / 24;
60  //libMesh::out << "n_pts=" << n_pts << std::endl;
61 
62  // Allocate space for points and weights
63  _points.resize(n_pts);
64  _weights.resize(n_pts);
65 
66  // (-1)^i -> This one flips sign at each iteration of the i-loop below.
67  int one_pm=1;
68 
69  // Where we store all the integer point compositions/permutations
70  std::vector<std::vector<unsigned int> > permutations;
71 
72  // Index into the vector where we should start adding the next round of points/weights
73  std::size_t offset=0;
74 
75  // Implement the GM formula 4.1 on page 286 of the paper
76  for (unsigned int i=0; i<=s; ++i)
77  {
78  // Get all the ordered compositions (and their permutations)
79  // of |beta| = s-i into dim+1 parts
80  compose_all(s-i, dim+1, permutations);
81  //libMesh::out << "n. permutations=" << permutations.size() << std::endl;
82 
83  for (unsigned int p=0; p<permutations.size(); ++p)
84  {
85  // We use the first dim entries of each permutation to
86  // construct an integration point.
87  for (unsigned int j=0; j<dim; ++j)
88  _points[offset+p](j) =
89  static_cast<Real>(2.*permutations[p][j] + 1.) /
90  static_cast<Real>( degree + dim - 2.*i );
91  }
92 
93  // Compute the weight for this i, being careful to avoid overflow.
94  // This technique is borrowed from Burkardt's code as well.
95  // Use once for each of the points obtained from the permutations array.
96  Real weight = one_pm;
97 
98  // This for loop needs to run for dim, degree, or dim+degree-i iterations,
99  // whichever is largest.
100  const unsigned int weight_loop_index =
101  std::max(dim, std::max(degree, degree+dim-i))+1;
102 
103  for (unsigned int j=1; j<weight_loop_index; ++j)
104  {
105  if (j <= degree) // Accumulate (d+n-2i)^d term
106  weight *= static_cast<Real>(degree+dim-2*i);
107 
108  if (j <= 2*s) // Accumulate 2^{-2s}
109  weight *= 0.5;
110 
111  if (j <= i) // Accumulate (i!)^{-1}
112  weight /= static_cast<Real>(j);
113 
114  if (j <= degree+dim-i) // Accumulate ( (d+n-i)! )^{-1}
115  weight /= static_cast<Real>(j);
116  }
117 
118  // This is the weight for each of the points computed previously
119  for (unsigned int j=0; j<permutations.size(); ++j)
120  _weights[offset+j] = weight;
121 
122  // Change sign for next iteration
123  one_pm = -one_pm;
124 
125  // Update offset for the next set of points
126  offset += permutations.size();
127  }
128 }

133 // This routine for computing compositions and their permutations is
134 // originall due to:
135 //
136 // Albert Nijenhuis, Herbert Wilf,
137 // Combinatorial Algorithms for Computers and Calculators,
138 // Second Edition,
139 // Academic Press, 1978,
140 // ISBN: 0-12-519260-6,
141 // LC: QA164.N54.
142 //
143 // The routine is deceptively simple: I still don't understand exactly
144 // why it works, but it does.
145 void QGrundmann_Moller::compose_all(unsigned int s, // number to be compositioned
146  unsigned int p, // # of partitions
147  std::vector<std::vector<unsigned int> > & result)
148 {
149  // Clear out results remaining from previous calls
150  result.clear();
151 
152  // Allocate storage for a workspace. The workspace will periodically
153  // be copied into the result container.
154  std::vector<unsigned int> workspace(p);
155 
156  // The first result is always (s,0,...,0)
157  workspace[0] = s;
158  result.push_back(workspace);
159 
160  // the value of the first non-zero entry
161  unsigned int head_value=s;
162 
163  // When head_index=-1, it refers to "off the front" of the array. Therefore,
164  // this needs to be a regular int rather than unsigned. I initially tried to
165  // do this with head_index unsigned and an else statement below, but then there
166  // is the special case: (1,0,...,0) which does not work correctly.
167  int head_index = -1;
168 
169  // At the end, all the entries will be in the final slot of workspace
170  while (workspace.back() != s)
171  {
172  // Uncomment for debugging
173  //libMesh::out << "previous head_value=" << head_value << " -> ";
174 
175  // If the previous head value is still larger than 1, reset the index
176  // to "off the front" of the array
177  if (head_value > 1)
178  head_index = -1;
179 
180  // Either move the index onto the front of the array or on to
181  // the next value.
182  head_index++;
183 
184  // Get current value of the head entry
185  head_value = workspace[head_index];
186 
187  // Uncomment for debugging
188  //std::copy(workspace.begin(), workspace.end(), std::ostream_iterator<int>(libMesh::out, " "));
189  //libMesh::out << ", head_index=" << head_index;
190  //libMesh::out << ", head_value=" << head_value << " -> ";
191 
192  // Put a zero into the head_index of the array. If head_index==0,
193  // this will be overwritten in the next line with head_value-1.
194  workspace[head_index] = 0;
195 
196  // The initial entry gets the current head value, minus 1.
197  // If head_value > 1, the next loop iteration will start back
198  // at workspace[0] again.
199  libmesh_assert_greater (head_value, 0);
200  workspace[0] = head_value - 1;
201 
202  // Increment the head+1 value
203  workspace[head_index+1] += 1;
204 
205  // Save this composition in the results
206  result.push_back(workspace);
207 
208  // Uncomment for debugging
209  //std::copy(workspace.begin(), workspace.end(), std::ostream_iterator<int>(libMesh::out, " "));
210  //libMesh::out<<"\n";
211  }
212 }
