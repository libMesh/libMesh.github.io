108 dof_id_type InfHex18::key (const unsigned int s) const
109 {
110  libmesh_assert_less (s, this->n_sides());
111 
112  // Think of a unit cube: (-1,1) x (-1,1) x (1,1)
113  switch (s)
114  {
115  case 0: // the base face
116  return this->compute_key (this->node(16));
117 
118  case 1: // the face at y = -1
119  case 2: // the face at x = 1
120  case 3: // the face at y = 1
121  case 4: // the face at x = -1
122  return InfHex::key(s);
123 
124  default:
125  libmesh_error_msg("Invalid side s = " << s);
126  }
127 
128  libmesh_error_msg("We'll never get here!");
129  return 0;
130 }

134 UniquePtr<Elem> InfHex18::build_side (const unsigned int i,
135  bool proxy) const
136 {
137  libmesh_assert_less (i, this->n_sides());
138 
139  if (proxy)
140  {
141  switch (i)
142  {
143  // base
144  case 0:
145  return UniquePtr<Elem>(new Side<Quad9,InfHex18>(this,i));
146 
147  // ifem sides
148  case 1:
149  case 2:
150  case 3:
151  case 4:
152  return UniquePtr<Elem>(new Side<InfQuad6,InfHex18>(this,i));
153 
154  default:
155  libmesh_error_msg("Invalid side i = " << i);
156  }
157  }
158 
159  else
160  {
161  // Create NULL pointer to be initialized, returned later.
162  Elem* face = NULL;
163 
164  // Think of a unit cube: (-1,1) x (-1,1) x (1,1)
165  switch (i)
166  {
167  // the base face
168  case 0:
169  {
170  face = new Quad9;
171  break;
172  }
173 
174  // connecting to another infinite element
175  case 1:
176  case 2:
177  case 3:
178  case 4:
179  {
180  face = new InfQuad6;
181  break;
182  }
183 
184  default:
185  libmesh_error_msg("Invalid side i = " << i);
186  }
187 
188  face->subdomain_id() = this->subdomain_id();
189 
190  // Set the nodes
191  for (unsigned n=0; n<face->n_nodes(); ++n)
192  face->set_node(n) = this->get_node(InfHex18::side_nodes_map[i][n]);
193 
194  return UniquePtr<Elem>(face);
195  }
196 
197  libmesh_error_msg("We'll never get here!");
198  return UniquePtr<Elem>();
199 }

322 unsigned short int InfHex18::second_order_adjacent_vertex (const unsigned int n,
323  const unsigned int v) const
324 {
325  libmesh_assert_greater_equal (n, this->n_vertices());
326  libmesh_assert_less (n, this->n_nodes());
327  libmesh_assert_less (v, this->n_second_order_adjacent_vertices(n));
328 
329  if (n == 16)
330  /*
331  * for the bubble node in the base the return value is
332  * simply v. Why? -- the user asks for the v-th
333  * adjacent vertex, from \p n_second_order_adjacent_vertices()
334  * there are 4 adjacent vertices, and these happen to be
335  * 0..3
336  */
337  return static_cast<unsigned short int>(v);
338  else if (n == 17)
339  /*
340  * for the bubble node further out similar reasoning works,
341  * but v must be shifted to the further-out nodes:
342  * simply add 4
343  */
344  return static_cast<unsigned short int>(v+4);
345 
346  else
347  /*
348  * all others are stored in the vertices matrix -- note
349  * that this matrix is kept in \p InfHex to foster
350  * code-reuse
351  */
352  return _second_order_adjacent_vertices[n-this->n_vertices()][v];
353 }
