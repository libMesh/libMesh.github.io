48  // Attach a new subfunction, along with a map from the indices of
49  // that subfunction to the indices of the global function.
50  // (*this)(index_map[i]) will return f(i).
51  void attach_subfunction (const FEMFunctionBase<Output> & f,
52  const std::vector<unsigned int> & index_map)
53  {
54  const unsigned int subfunction_index = subfunctions.size();
55  libmesh_assert_equal_to(subfunctions.size(), index_maps.size());
56 
57  subfunctions.push_back(f.clone().release());
58  index_maps.push_back(index_map);
59 
60  unsigned int max_index =
61  *std::max_element(index_map.begin(), index_map.end());
62 
63  if (max_index >= reverse_index_map.size())
64  reverse_index_map.resize
65  (max_index+1, std::make_pair(libMesh::invalid_uint,
66  libMesh::invalid_uint));
67 
68  for (unsigned int j=0; j != index_map.size(); ++j)
69  {
70  libmesh_assert_less(index_map[j], reverse_index_map.size());
71  libmesh_assert_equal_to(reverse_index_map[index_map[j]].first,
72  libMesh::invalid_uint);
73  libmesh_assert_equal_to(reverse_index_map[index_map[j]].second,
74  libMesh::invalid_uint);
75  reverse_index_map[index_map[j]] =
76  std::make_pair(subfunction_index, j);
77  }
78  }
