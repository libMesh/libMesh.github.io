57 template <>
58 Real FE<1,XYZ>::shape(const Elem* elem,
59  const Order libmesh_dbg_var(order),
60  const unsigned int i,
61  const Point& point_in)
62 {
63  libmesh_assert(elem);
64  libmesh_assert_less_equal (i, order + elem->p_level());
65 
66  // Only recompute the centroid if the element
67  // has changed from the last one we computed.
68  // This avoids repeated centroid calculations
69  // when called in succession with the same element.
70  if (elem->id() != old_elem_id)
71  {
72  centroid = elem->centroid();
73  old_elem_id = elem->id();
74  max_distance = 0.;
75  for (unsigned int p = 0; p < elem->n_nodes(); p++)
76  {
77  const Real distance = std::abs(centroid(0) - elem->point(p)(0));
78  max_distance = std::max(distance, max_distance);
79  }
80  }
81 
82  // Using static globals for old_elem_id, etc. will fail
83  // horribly with more than one thread.
84  libmesh_assert_equal_to (libMesh::n_threads(), 1);
85 
86  const Real x = point_in(0);
87  const Real xc = centroid(0);
88  const Real dx = (x - xc)/max_distance;
89 
90  // monomials. since they are hierarchic we only need one case block.
91  switch (i)
92  {
93  case 0:
94  return 1.;
95 
96  case 1:
97  return dx;
98 
99  case 2:
100  return dx*dx;
101 
102  case 3:
103  return dx*dx*dx;
104 
105  case 4:
106  return dx*dx*dx*dx;
107 
108  default:
109  Real val = 1.;
110  for (unsigned int index = 0; index != i; ++index)
111  val *= dx;
112  return val;
113  }
114 
115  libmesh_error_msg("We'll never get here!");
116  return 0.;
117
