41 ExactSolution::ExactSolution(const EquationSystems& es) :
42  _equation_systems(es),
43  _equation_systems_fine(NULL),
44  _extra_order(0)
45 {
46  // Initialize the _errors data structure which holds all
47  // the eventual values of the error.
48  for (unsigned int sys=0; sys<_equation_systems.n_systems(); ++sys)
49  {
50  // Reference to the system
51  const System& system = _equation_systems.get_system(sys);
52 
53  // The name of the system
54  const std::string& sys_name = system.name();
55 
56  // The SystemErrorMap to be inserted
57  ExactSolution::SystemErrorMap sem;
58 
59  for (unsigned int var=0; var<system.n_vars(); ++var)
60  {
61  // The name of this variable
62  const std::string& var_name = system.variable_name(var);
63  sem[var_name] = std::vector<Real>(5, 0.);
64  }
65 
66  _errors[sys_name] = sem;
67  }
68 }

261 std::vector<Real>& ExactSolution::_check_inputs(const std::string& sys_name,
262  const std::string& unknown_name)
263 {
264  // If no exact solution function or fine grid solution has been
265  // attached, we now just compute the solution norm (i.e. the
266  // difference from an "exact solution" of zero
267 
268  // Make sure the requested sys_name exists.
269  std::map<std::string, SystemErrorMap>::iterator sys_iter =
270  _errors.find(sys_name);
271 
272  if (sys_iter == _errors.end())
273  libmesh_error_msg("Sorry, couldn't find the requested system '" << sys_name << "'.");
274 
275  // Make sure the requested unknown_name exists.
276  SystemErrorMap::iterator var_iter = (*sys_iter).second.find(unknown_name);
277 
278  if (var_iter == (*sys_iter).second.end())
279  libmesh_error_msg("Sorry, couldn't find the requested variable '" << unknown_name << "'.");
280 
281  // Return a reference to the proper error entry
282  return (*var_iter).second;
283 }

287 void ExactSolution::compute_error(const std::string& sys_name,
288  const std::string& unknown_name)
289 {
290  // Check the inputs for validity, and get a reference
291  // to the proper location to store the error
292  std::vector<Real>& error_vals = this->_check_inputs(sys_name,
293  unknown_name);
294 
295  libmesh_assert( _equation_systems.has_system(sys_name) );
296  const System& sys = _equation_systems.get_system<System>( sys_name );
297 
298  libmesh_assert( sys.has_variable( unknown_name ) );
299  switch( FEInterface::field_type(sys.variable_type( unknown_name )) )
300  {
301  case TYPE_SCALAR:
302  {
303  this->_compute_error<Real>(sys_name,
304  unknown_name,
305  error_vals);
306  break;
307  }
308  case TYPE_VECTOR:
309  {
310  this->_compute_error<RealGradient>(sys_name,
311  unknown_name,
312  error_vals);
313  break;
314  }
315  default:
316  libmesh_error_msg("Invalid variable type!");
317  }
318 }

324 Real ExactSolution::error_norm(const std::string& sys_name,
325  const std::string& unknown_name,
326  const FEMNormType& norm)
327 {
328  // Check the inputs for validity, and get a reference
329  // to the proper location to store the error
330  std::vector<Real>& error_vals = this->_check_inputs(sys_name,
331  unknown_name);
332 
333  libmesh_assert(_equation_systems.has_system(sys_name));
334  libmesh_assert(_equation_systems.get_system(sys_name).has_variable( unknown_name ));
335  const FEType& fe_type = _equation_systems.get_system(sys_name).variable_type(unknown_name);
336 
337  switch (norm)
338  {
339  case L2:
340  return std::sqrt(error_vals[0]);
341  case H1:
342  return std::sqrt(error_vals[0] + error_vals[1]);
343  case H2:
344  return std::sqrt(error_vals[0] + error_vals[1] + error_vals[2]);
345  case HCURL:
346  {
347  if(FEInterface::field_type(fe_type) == TYPE_SCALAR)
348  libmesh_error_msg("Cannot compute HCurl error norm of scalar-valued variables!");
349  else
350  return std::sqrt(error_vals[0] + error_vals[5]);
351  }
352  case HDIV:
353  {
354  if(FEInterface::field_type(fe_type) == TYPE_SCALAR)
355  libmesh_error_msg("Cannot compute HDiv error norm of scalar-valued variables!");
356  else
357  return std::sqrt(error_vals[0] + error_vals[6]);
358  }
359  case H1_SEMINORM:
360  return std::sqrt(error_vals[1]);
361  case H2_SEMINORM:
362  return std::sqrt(error_vals[2]);
363  case HCURL_SEMINORM:
364  {
365  if(FEInterface::field_type(fe_type) == TYPE_SCALAR)
366  libmesh_error_msg("Cannot compute HCurl error seminorm of scalar-valued variables!");
367  else
368  return std::sqrt(error_vals[5]);
369  }
370  case HDIV_SEMINORM:
371  {
372  if(FEInterface::field_type(fe_type) == TYPE_SCALAR)
373  libmesh_error_msg("Cannot compute HDiv error seminorm of scalar-valued variables!");
374  else
375  return std::sqrt(error_vals[6]);
376  }
377  case L1:
378  return error_vals[3];
379  case L_INF:
380  return error_vals[4];
381 
382  default:
383  libmesh_error_msg("Currently only Sobolev norms/seminorms are supported!");
384  }
385 }

507 template< typename OutputShape>
508 void ExactSolution::_compute_error(const std::string& sys_name,
509  const std::string& unknown_name,
510  std::vector<Real>& error_vals)
511 {
512  // Make sure we aren't "overconfigured"
513  libmesh_assert (!(_exact_values.size() && _equation_systems_fine));
514 
515  // We need a commmunicator.
516  const Parallel::Communicator &communicator(_equation_systems.comm());
517 
518  // This function must be run on all processors at once
519  libmesh_parallel_only(communicator);
520 
521  // Get a reference to the system whose error is being computed.
522  // If we have a fine grid, however, we'll integrate on that instead
523  // for more accuracy.
524  const System& computed_system = _equation_systems_fine ?
525  _equation_systems_fine->get_system(sys_name) :
526  _equation_systems.get_system (sys_name);
527 
528  const Real time = _equation_systems.get_system(sys_name).time;
529 
530  const unsigned int sys_num = computed_system.number();
531  const unsigned int var = computed_system.variable_number(unknown_name);
532  const unsigned int var_component =
533  computed_system.variable_scalar_number(var, 0);
534 
535  // Prepare a global solution and a MeshFunction of the coarse system if we need one
536  AutoPtr<MeshFunction> coarse_values;
537  AutoPtr<NumericVector<Number> > comparison_soln = NumericVector<Number>::build(_equation_systems.comm());
538  if (_equation_systems_fine)
539  {
540  const System& comparison_system
541  = _equation_systems.get_system(sys_name);
542 
543  std::vector<Number> global_soln;
544  comparison_system.update_global_solution(global_soln);
545  comparison_soln->init(comparison_system.solution->size(), true, SERIAL);
546  (*comparison_soln) = global_soln;
547 
548  coarse_values = AutoPtr<MeshFunction>
549  (new MeshFunction(_equation_systems,
550  *comparison_soln,
551  comparison_system.get_dof_map(),
552  comparison_system.variable_number(unknown_name)));
553  coarse_values->init();
554  }
555 
556  // Initialize any functors we're going to use
557  for (unsigned int i=0; i != _exact_values.size(); ++i)
558  if (_exact_values[i])
559  _exact_values[i]->init();
560 
561  for (unsigned int i=0; i != _exact_derivs.size(); ++i)
562  if (_exact_derivs[i])
563  _exact_derivs[i]->init();
564 
565  for (unsigned int i=0; i != _exact_hessians.size(); ++i)
566  if (_exact_hessians[i])
567  _exact_hessians[i]->init();
568 
569  // Get a reference to the dofmap and mesh for that system
570  const DofMap& computed_dof_map = computed_system.get_dof_map();
571 
572  const MeshBase& _mesh = computed_system.get_mesh();
573 
574  const unsigned int dim = _mesh.mesh_dimension();
575 
576  // Zero the error before summation
577  // 0 - sum of square of function error (L2)
578  // 1 - sum of square of gradient error (H1 semi)
579  // 2 - sum of square of Hessian error (H2 semi)
580  // 3 - sum of sqrt(square of function error) (L1)
581  // 4 - max of sqrt(square of function error) (Linfty)
582  // 5 - sum of square of curl error (HCurl semi)
583  // 6 - sum of square of div error (HDiv semi)
584  error_vals = std::vector<Real>(7, 0.);
585 
586  // Construct Quadrature rule based on default quadrature order
587  const FEType& fe_type = computed_dof_map.variable_type(var);
588 
589  unsigned int n_vec_dim = FEInterface::n_vec_dim( _mesh, fe_type );
590 
591  // FIXME: MeshFunction needs to be updated to support vector-valued
592  // elements before we can use a reference solution.
593  if( (n_vec_dim > 1) && _equation_systems_fine )
594  {
595  libMesh::err << "Error calculation using reference solution not yet\n"
596  << "supported for vector-valued elements."
597  << std::endl;
598  libmesh_not_implemented();
599  }
600 
601  AutoPtr<QBase> qrule =
602  fe_type.default_quadrature_rule (dim,
603  _extra_order);
604 
605  // Construct finite element object
606 
607  AutoPtr<FEGenericBase<OutputShape> > fe(FEGenericBase<OutputShape>::build(dim, fe_type));
608 
609  // Attach quadrature rule to FE object
610  fe->attach_quadrature_rule (qrule.get());
611 
612  // The Jacobian*weight at the quadrature points.
613  const std::vector<Real>& JxW = fe->get_JxW();
614 
615  // The value of the shape functions at the quadrature points
616  // i.e. phi(i) = phi_values[i][qp]
617  const std::vector<std::vector<OutputShape> >& phi_values = fe->get_phi();
618 
619  // The value of the shape function gradients at the quadrature points
620  const std::vector<std::vector<typename FEGenericBase<OutputShape>::OutputGradient> >&
621  dphi_values = fe->get_dphi();
622 
623  // The value of the shape function curls at the quadrature points
624  // Only computed for vector-valued elements
625  const std::vector<std::vector<typename FEGenericBase<OutputShape>::OutputShape> >* curl_values = NULL;
626 
627  // The value of the shape function divergences at the quadrature points
628  // Only computed for vector-valued elements
629  const std::vector<std::vector<typename FEGenericBase<OutputShape>::OutputDivergence> >* div_values = NULL;
630 
631  if( FEInterface::field_type(fe_type) == TYPE_VECTOR )
632  {
633  curl_values = &fe->get_curl_phi();
634  div_values = &fe->get_div_phi();
635  }
636 
637 #ifdef LIBMESH_ENABLE_SECOND_DERIVATIVES
638  // The value of the shape function second derivatives at the quadrature points
639  const std::vector<std::vector<typename FEGenericBase<OutputShape>::OutputTensor> >&
640  d2phi_values = fe->get_d2phi();
641 #endif
642 
643  // The XYZ locations (in physical space) of the quadrature points
644  const std::vector<Point>& q_point = fe->get_xyz();
645 
646  // The global degree of freedom indices associated
647  // with the local degrees of freedom.
648  std::vector<dof_id_type> dof_indices;
649 
650 
651  //
652  // Begin the loop over the elements
653  //
654  // TODO: this ought to be threaded (and using subordinate
655  // MeshFunction objects in each thread rather than a single
656  // master)
657  MeshBase::const_element_iterator el = _mesh.active_local_elements_begin();
658  const MeshBase::const_element_iterator end_el = _mesh.active_local_elements_end();
659 
660  for ( ; el != end_el; ++el)
661  {
662  // Store a pointer to the element we are currently
663  // working on. This allows for nicer syntax later.
664  const Elem* elem = *el;
665 
666  // reinitialize the element-specific data
667  // for the current element
668  fe->reinit (elem);
669 
670  // Get the local to global degree of freedom maps
671  computed_dof_map.dof_indices (elem, dof_indices, var);
672 
673  // The number of quadrature points
674  const unsigned int n_qp = qrule->n_points();
675 
676  // The number of shape functions
677  const unsigned int n_sf =
678  cast_int<unsigned int>(dof_indices.size());
679 
680  //
681  // Begin the loop over the Quadrature points.
682  //
683  for (unsigned int qp=0; qp<n_qp; qp++)
684  {
685  // Real u_h = 0.;
686  // RealGradient grad_u_h;
687 
688  typename FEGenericBase<OutputShape>::OutputNumber u_h = 0.;
689 
690  typename FEGenericBase<OutputShape>::OutputNumberGradient grad_u_h;
691 #ifdef LIBMESH_ENABLE_SECOND_DERIVATIVES
692  typename FEGenericBase<OutputShape>::OutputNumberTensor grad2_u_h;
693 #endif
694  typename FEGenericBase<OutputShape>::OutputNumber curl_u_h = 0.0;
695  typename FEGenericBase<OutputShape>::OutputNumberDivergence div_u_h = 0.0;
696 
697  // Compute solution values at the current
698  // quadrature point. This reqiures a sum
699  // over all the shape functions evaluated
700  // at the quadrature point.
701  for (unsigned int i=0; i<n_sf; i++)
702  {
703  // Values from current solution.
704  u_h += phi_values[i][qp]*computed_system.current_solution (dof_indices[i]);
705  grad_u_h += dphi_values[i][qp]*computed_system.current_solution (dof_indices[i]);
706 #ifdef LIBMESH_ENABLE_SECOND_DERIVATIVES
707  grad2_u_h += d2phi_values[i][qp]*computed_system.current_solution (dof_indices[i]);
708 #endif
709  if( FEInterface::field_type(fe_type) == TYPE_VECTOR )
710  {
711  curl_u_h += (*curl_values)[i][qp]*computed_system.current_solution (dof_indices[i]);
712  div_u_h += (*div_values)[i][qp]*computed_system.current_solution (dof_indices[i]);
713  }
714  }
715 
716  // Compute the value of the error at this quadrature point
717  typename FEGenericBase<OutputShape>::OutputNumber exact_val = 0;
718  RawAccessor<typename FEGenericBase<OutputShape>::OutputNumber> exact_val_accessor( exact_val, dim );
719  if (_exact_values.size() > sys_num && _exact_values[sys_num])
720  {
721  for( unsigned int c = 0; c < n_vec_dim; c++)
722  exact_val_accessor(c) =
723  _exact_values[sys_num]->
724  component(var_component+c, q_point[qp], time);
725  }
726  else if (_equation_systems_fine)
727  {
728  // FIXME: Needs to be updated for vector-valued elements
729  exact_val = (*coarse_values)(q_point[qp]);
730  }
731  const typename FEGenericBase<OutputShape>::OutputNumber val_error = u_h - exact_val;
732 
733  // Add the squares of the error to each contribution
734  Real error_sq = TensorTools::norm_sq(val_error);
735  error_vals[0] += JxW[qp]*error_sq;
736 
737  Real norm = sqrt(error_sq);
738  error_vals[3] += JxW[qp]*norm;
739 
740  if(error_vals[4]<norm) { error_vals[4] = norm; }
741 
742  // Compute the value of the error in the gradient at this
743  // quadrature point
744  typename FEGenericBase<OutputShape>::OutputNumberGradient exact_grad;
745  RawAccessor<typename FEGenericBase<OutputShape>::OutputNumberGradient> exact_grad_accessor( exact_grad, _mesh.spatial_dimension() );
746  if (_exact_derivs.size() > sys_num && _exact_derivs[sys_num])
747  {
748  for( unsigned int c = 0; c < n_vec_dim; c++)
749  for( unsigned int d = 0; d < _mesh.spatial_dimension(); d++ )
750  exact_grad_accessor(d + c*_mesh.spatial_dimension() ) =
751  _exact_derivs[sys_num]->
752  component(var_component+c, q_point[qp], time)(d);
753  }
754  else if (_equation_systems_fine)
755  {
756  // FIXME: Needs to be updated for vector-valued elements
757  exact_grad = coarse_values->gradient(q_point[qp]);
758  }
759 
760  const typename FEGenericBase<OutputShape>::OutputNumberGradient grad_error = grad_u_h - exact_grad;
761 
762  error_vals[1] += JxW[qp]*grad_error.size_sq();
763 
764 
765  if( FEInterface::field_type(fe_type) == TYPE_VECTOR )
766  {
767  // Compute the value of the error in the curl at this
768  // quadrature point
769  typename FEGenericBase<OutputShape>::OutputNumber exact_curl = 0.0;
770  if (_exact_derivs.size() > sys_num && _exact_derivs[sys_num])
771  {
772  exact_curl = TensorTools::curl_from_grad( exact_grad );
773  }
774  else if (_equation_systems_fine)
775  {
776  // FIXME: Need to implement curl for MeshFunction and support reference
777  // solution for vector-valued elements
778  }
779 
780  const typename FEGenericBase<OutputShape>::OutputNumber curl_error = curl_u_h - exact_curl;
781 
782  error_vals[5] += JxW[qp]*TensorTools::norm_sq(curl_error);
783 
784  // Compute the value of the error in the divergence at this
785  // quadrature point
786  typename FEGenericBase<OutputShape>::OutputNumberDivergence exact_div = 0.0;
787  if (_exact_derivs.size() > sys_num && _exact_derivs[sys_num])
788  {
789  exact_div = TensorTools::div_from_grad( exact_grad );
790  }
791  else if (_equation_systems_fine)
792  {
793  // FIXME: Need to implement div for MeshFunction and support reference
794  // solution for vector-valued elements
795  }
796 
797  const typename FEGenericBase<OutputShape>::OutputNumberDivergence div_error = div_u_h - exact_div;
798 
799  error_vals[6] += JxW[qp]*TensorTools::norm_sq(div_error);
800  }
801 
802 #ifdef LIBMESH_ENABLE_SECOND_DERIVATIVES
803  // Compute the value of the error in the hessian at this
804  // quadrature point
805  typename FEGenericBase<OutputShape>::OutputNumberTensor exact_hess;
806  RawAccessor<typename FEGenericBase<OutputShape>::OutputNumberTensor> exact_hess_accessor( exact_hess, dim );
807  if (_exact_hessians.size() > sys_num && _exact_hessians[sys_num])
808  {
809  //FIXME: This needs to be implemented to support rank 3 tensors
810  // which can't happen until type_n_tensor is fully implemented
811  // and a RawAccessor<TypeNTensor> is fully implemented
812  if( FEInterface::field_type(fe_type) == TYPE_VECTOR )
813  libmesh_not_implemented();
814 
815  for( unsigned int c = 0; c < n_vec_dim; c++)
816  for( unsigned int d = 0; d < dim; d++ )
817  for( unsigned int e =0; e < dim; e++ )
818  exact_hess_accessor(d + e*dim + c*dim*dim) =
819  _exact_hessians[sys_num]->
820  component(var_component+c, q_point[qp], time)(d,e);
821  }
822  else if (_equation_systems_fine)
823  {
824  // FIXME: Needs to be updated for vector-valued elements
825  exact_hess = coarse_values->hessian(q_point[qp]);
826  }
827 
828  const typename FEGenericBase<OutputShape>::OutputNumberTensor grad2_error = grad2_u_h - exact_hess;
829 
830  // FIXME: PB: Is this what we want for rank 3 tensors?
831  error_vals[2] += JxW[qp]*grad2_error.size_sq();
832 #endif
833 
834  } // end qp loop
835  } // end element loop
836 
837  // Add up the error values on all processors, except for the L-infty
838  // norm, for which the maximum is computed.
839  Real l_infty_norm = error_vals[4];
840  communicator.max(l_infty_norm);
841  communicator.sum(error_vals);
842  error_vals[4] = l_infty_norm;
843 }
