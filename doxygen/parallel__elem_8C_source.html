57 template <>
58 unsigned int
59 Packing<const Elem *>::packed_size (std::vector<largest_id_type>::const_iterator in)
60 {
61 #ifndef NDEBUG
62  const largest_id_type packed_header = *in++;
63  libmesh_assert_equal_to (packed_header, elem_magic_header);
64 #endif
65 
66  // int 0: level
67  const unsigned int level =
68  cast_int<unsigned int>(*in);
69 
70  // int 4: element type
71  const int typeint = cast_int<int>(*(in+4));
72  libmesh_assert_greater_equal (typeint, 0);
73  libmesh_assert_less (typeint, INVALID_ELEM);
74  const ElemType type =
75  cast_int<ElemType>(typeint);
76 
77  const unsigned int n_nodes =
78  Elem::type_to_n_nodes_map[type];
79 
80  const unsigned int n_sides =
81  Elem::type_to_n_sides_map[type];
82 
83  const unsigned int n_edges =
84  Elem::type_to_n_edges_map[type];
85 
86  const unsigned int pre_indexing_size =
87  header_size + n_nodes + n_sides;
88 
89  const unsigned int indexing_size =
90  DofObject::unpackable_indexing_size(in+pre_indexing_size);
91 
92  unsigned int total_packed_bc_data = 0;
93  if (level == 0)
94  {
95  for (unsigned int s = 0; s != n_sides; ++s)
96  {
97  const int n_bcs = cast_int<int>
98  (*(in + pre_indexing_size + indexing_size +
99  total_packed_bc_data++));
100  libmesh_assert_greater_equal (n_bcs, 0);
101  total_packed_bc_data += n_bcs;
102  }
103 
104  for (unsigned int e = 0; e != n_edges; ++e)
105  {
106  const int n_bcs = cast_int<int>
107  (*(in + pre_indexing_size + indexing_size +
108  total_packed_bc_data++));
109  libmesh_assert_greater_equal (n_bcs, 0);
110  total_packed_bc_data += n_bcs;
111  }
112 
113  for (unsigned short sf=0; sf != 2; ++sf)
114  {
115  const int n_bcs = cast_int<int>
116  (*(in + pre_indexing_size + indexing_size +
117  total_packed_bc_data++));
118  libmesh_assert_greater_equal (n_bcs, 0);
119  total_packed_bc_data += n_bcs;
120  }
121  }
122 
123  return
124 #ifndef NDEBUG
125  1 + // Account for magic header
126 #endif
127  pre_indexing_size + indexing_size + total_packed_bc_data;
128 }

143 template <>
144 unsigned int
145 Packing<const Elem *>::packable_size (const Elem * const & elem,
146  const MeshBase * mesh)
147 {
148  unsigned int total_packed_bcs = 0;
149  if (elem->level() == 0)
150  {
151  total_packed_bcs += elem->n_sides();
152  for (unsigned short s = 0; s != elem->n_sides(); ++s)
153  total_packed_bcs +=
154  mesh->get_boundary_info().n_boundary_ids(elem,s);
155 
156  total_packed_bcs += elem->n_edges();
157  for (unsigned short e = 0; e != elem->n_edges(); ++e)
158  total_packed_bcs +=
159  mesh->get_boundary_info().n_edge_boundary_ids(elem,e);
160 
161  total_packed_bcs += 2; // shellfaces
162  for (unsigned short sf=0; sf != 2; ++sf)
163  total_packed_bcs +=
164  mesh->get_boundary_info().n_shellface_boundary_ids(elem,sf);
165  }
166 
167  return
168 #ifndef NDEBUG
169  1 + // add an int for the magic header when testing
170 #endif
171  header_size + elem->n_nodes() +
172  elem->n_neighbors() +
173  elem->packed_indexing_size() + total_packed_bcs;
174 }

200 template <>
201 template <>
202 void
203 Packing<const Elem *>::pack (const Elem * const & elem,
204  std::back_insert_iterator<std::vector<largest_id_type> > data_out,
205  const MeshBase * mesh)
206 {
207  libmesh_assert(elem);
208 
209 #ifndef NDEBUG
210  *data_out++ = elem_magic_header;
211 #endif
212 
213 #ifdef LIBMESH_ENABLE_AMR
214  *data_out++ = (static_cast<largest_id_type>(elem->level()));
215  *data_out++ = (static_cast<largest_id_type>(elem->p_level()));
216 
217  // Encode both the refinement flag and whether the element has
218  // children together. This coding is unambiguous because our
219  // refinement state encoding starts at 0 and ends at
220  // INVALID_REFINEMENTSTATE
221  largest_id_type refinement_info =
222  static_cast<largest_id_type>(elem->refinement_flag());
223  if (elem->has_children())
224  refinement_info +=
225  static_cast<largest_id_type>(Elem::INVALID_REFINEMENTSTATE) + 1;
226  *data_out++ = (refinement_info);
227 
228  *data_out++ = (static_cast<largest_id_type>(elem->p_refinement_flag()));
229 #else
230  *data_out++ = (0);
231  *data_out++ = (0);
232  *data_out++ = (0);
233  *data_out++ = (0);
234 #endif
235  *data_out++ = (static_cast<largest_id_type>(elem->type()));
236  *data_out++ = (elem->processor_id());
237  *data_out++ = (elem->subdomain_id());
238  *data_out++ = (elem->id());
239 
240 #ifdef LIBMESH_ENABLE_UNIQUE_ID
241  if (elem->valid_unique_id())
242  *data_out++ = (static_cast<largest_id_type>(elem->unique_id()));
243  else
244  // OK to send invalid unique id, we must not own this DOF
245  *data_out++ = (static_cast<largest_id_type>(DofObject::invalid_unique_id));
246 #endif
247 
248 #ifdef LIBMESH_ENABLE_AMR
249  // use parent_ID of invalid_id to indicate a level 0 element
250  if (elem->level() == 0)
251  {
252  *data_out++ =(DofObject::invalid_id);
253  *data_out++ =(DofObject::invalid_id);
254  }
255  else
256  {
257  *data_out++ =(elem->parent()->id());
258  *data_out++ =(elem->parent()->which_child_am_i(elem));
259  }
260 #else
261  *data_out++ = (DofObject::invalid_id);
262  *data_out++ = (DofObject::invalid_id);
263 #endif
264 
265  if ((elem->dim() < LIBMESH_DIM) &&
266  elem->interior_parent())
267  *data_out++ =(elem->interior_parent()->id());
268  else
269  *data_out++ =(DofObject::invalid_id);
270 
271  for (unsigned int n=0; n<elem->n_nodes(); n++)
272  *data_out++ = (elem->node_id(n));
273 
274  for (unsigned int n=0; n<elem->n_neighbors(); n++)
275  {
276  const Elem * neigh = elem->neighbor_ptr(n);
277  if (neigh)
278  *data_out++ = (neigh->id());
279  else
280  *data_out++ = (DofObject::invalid_id);
281  }
282 
283  // Add any DofObject indices
284  elem->pack_indexing(data_out);
285 
286  // If this is a coarse element,
287  // Add any element side boundary condition ids
288  if (elem->level() == 0)
289  {
290  std::vector<boundary_id_type> bcs;
291  for (unsigned short s = 0; s != elem->n_sides(); ++s)
292  {
293  mesh->get_boundary_info().boundary_ids(elem, s, bcs);
294 
295  *data_out++ =(bcs.size());
296 
297  for (std::vector<boundary_id_type>::iterator bc_it=bcs.begin();
298  bc_it != bcs.end(); ++bc_it)
299  *data_out++ =(*bc_it);
300  }
301 
302  for (unsigned short e = 0; e != elem->n_edges(); ++e)
303  {
304  mesh->get_boundary_info().edge_boundary_ids(elem, e, bcs);
305 
306  *data_out++ =(bcs.size());
307 
308  for (std::vector<boundary_id_type>::iterator bc_it=bcs.begin();
309  bc_it != bcs.end(); ++bc_it)
310  *data_out++ =(*bc_it);
311  }
312 
313  for (unsigned short sf=0; sf != 2; ++sf)
314  {
315  mesh->get_boundary_info().shellface_boundary_ids(elem, sf, bcs);
316 
317  *data_out++ =(bcs.size());
318 
319  for (std::vector<boundary_id_type>::iterator bc_it=bcs.begin();
320  bc_it != bcs.end(); ++bc_it)
321  *data_out++ =(*bc_it);
322  }
323  }
324 }

352 // FIXME - this needs serious work to be 64-bit compatible
353 template <>
354 template <>
355 Elem *
356 Packing<Elem *>::unpack (std::vector<largest_id_type>::const_iterator in,
357  MeshBase * mesh)
358 {
359 #ifndef NDEBUG
360  const std::vector<largest_id_type>::const_iterator original_in = in;
361 
362  const largest_id_type incoming_header = *in++;
363  libmesh_assert_equal_to (incoming_header, elem_magic_header);
364 #endif
365 
366  // int 0: level
367  const unsigned int level =
368  cast_int<unsigned int>(*in++);
369 
370 #ifdef LIBMESH_ENABLE_AMR
371  // int 1: p level
372  const unsigned int p_level =
373  cast_int<unsigned int>(*in++);
374 
375  // int 2: refinement flag and encoded has_children
376  const int rflag = cast_int<int>(*in++);
377  const int invalid_rflag =
378  cast_int<int>(Elem::INVALID_REFINEMENTSTATE);
379  libmesh_assert_greater_equal (rflag, 0);
380 
381  libmesh_assert_less (rflag, invalid_rflag*2+1);
382 
383  const bool has_children = (rflag > invalid_rflag);
384 
385  const Elem::RefinementState refinement_flag = has_children ?
386  cast_int<Elem::RefinementState>(rflag - invalid_rflag - 1) :
387  cast_int<Elem::RefinementState>(rflag);
388 
389  // int 3: p refinement flag
390  const int pflag = cast_int<int>(*in++);
391  libmesh_assert_greater_equal (pflag, 0);
392  libmesh_assert_less (pflag, Elem::INVALID_REFINEMENTSTATE);
393  const Elem::RefinementState p_refinement_flag =
394  cast_int<Elem::RefinementState>(pflag);
395 #else
396  in += 3;
397 #endif // LIBMESH_ENABLE_AMR
398 
399  // int 4: element type
400  const int typeint = cast_int<int>(*in++);
401  libmesh_assert_greater_equal (typeint, 0);
402  libmesh_assert_less (typeint, INVALID_ELEM);
403  const ElemType type =
404  cast_int<ElemType>(typeint);
405 
406  const unsigned int n_nodes =
407  Elem::type_to_n_nodes_map[type];
408 
409  // int 5: processor id
410  const processor_id_type processor_id =
411  cast_int<processor_id_type>(*in++);
412  libmesh_assert (processor_id < mesh->n_processors() ||
413  processor_id == DofObject::invalid_processor_id);
414 
415  // int 6: subdomain id
416  const subdomain_id_type subdomain_id =
417  cast_int<subdomain_id_type>(*in++);
418 
419  // int 7: dof object id
420  const dof_id_type id =
421  cast_int<dof_id_type>(*in++);
422  libmesh_assert_not_equal_to (id, DofObject::invalid_id);
423 
424 #ifdef LIBMESH_ENABLE_UNIQUE_ID
425  // int 8: dof object unique id
426  const unique_id_type unique_id =
427  cast_int<unique_id_type>(*in++);
428 #endif
429 
430 #ifdef LIBMESH_ENABLE_AMR
431  // int 9: parent dof object id.
432  // Note: If level==0, then (*in) == invalid_id. In
433  // this case, the equality check in cast_int<unsigned>(*in) will
434  // never succeed. Therefore, we should only attempt the more
435  // rigorous cast verification in cases where level != 0.
436  const dof_id_type parent_id =
437  (level == 0)
438  ? static_cast<dof_id_type>(*in++)
439  : cast_int<dof_id_type>(*in++);
440  libmesh_assert (level == 0 || parent_id != DofObject::invalid_id);
441  libmesh_assert (level != 0 || parent_id == DofObject::invalid_id);
442 
443  // int 10: local child id
444  // Note: If level==0, then which_child_am_i is not valid, so don't
445  // do the more rigorous cast verification.
446  const unsigned int which_child_am_i =
447  (level == 0)
448  ? static_cast<unsigned int>(*in++)
449  : cast_int<unsigned int>(*in++);
450 #else
451  in += 2;
452 #endif // LIBMESH_ENABLE_AMR
453 
454  const dof_id_type interior_parent_id =
455  static_cast<dof_id_type>(*in++);
456 
457  // Make sure we don't miscount above when adding the "magic" header
458  // plus the real data header
459  libmesh_assert_equal_to (in - original_in, header_size + 1);
460 
461  Elem * elem = mesh->query_elem_ptr(id);
462 
463  // if we already have this element, make sure its
464  // properties match, and update any missing neighbor
465  // links, but then go on
466  if (elem)
467  {
468  libmesh_assert_equal_to (elem->level(), level);
469  libmesh_assert_equal_to (elem->id(), id);
470  //#ifdef LIBMESH_ENABLE_UNIQUE_ID
471  // No check for unique id sanity
472  //#endif
473  libmesh_assert_equal_to (elem->processor_id(), processor_id);
474  libmesh_assert_equal_to (elem->subdomain_id(), subdomain_id);
475  libmesh_assert_equal_to (elem->type(), type);
476  libmesh_assert_equal_to (elem->n_nodes(), n_nodes);
477 
478 #ifndef NDEBUG
479  // All our nodes should be correct
480  for (unsigned int i=0; i != n_nodes; ++i)
481  libmesh_assert(elem->node_id(i) ==
482  cast_int<dof_id_type>(*in++));
483 #else
484  in += n_nodes;
485 #endif
486 
487 #ifdef LIBMESH_ENABLE_AMR
488  libmesh_assert_equal_to (elem->refinement_flag(), refinement_flag);
489  libmesh_assert_equal_to (elem->has_children(), has_children);
490 
491 #ifdef DEBUG
492  if (elem->active())
493  {
494  libmesh_assert_equal_to (elem->p_level(), p_level);
495  libmesh_assert_equal_to (elem->p_refinement_flag(), p_refinement_flag);
496  }
497 #endif
498 
499  libmesh_assert (!level || elem->parent() != libmesh_nullptr);
500  libmesh_assert (!level || elem->parent()->id() == parent_id);
501  libmesh_assert (!level || elem->parent()->child_ptr(which_child_am_i) == elem);
502 #endif
503  // Our interior_parent link should be "close to" correct - we
504  // may have to update it, but we can check for some
505  // inconsistencies.
506  {
507  // If the sending processor sees no interior_parent here, we'd
508  // better agree.
509  if (interior_parent_id == DofObject::invalid_id)
510  {
511  if (elem->dim() < LIBMESH_DIM)
512  libmesh_assert (!(elem->interior_parent()));
513  }
514 
515  // If the sending processor has a remote_elem interior_parent,
516  // then all we know is that we'd better have *some*
517  // interior_parent
518  else if (interior_parent_id == remote_elem->id())
519  {
520  libmesh_assert(elem->interior_parent());
521  }
522  else
523  {
524  Elem * ip = mesh->query_elem_ptr(interior_parent_id);
525 
526  // The sending processor sees an interior parent here, so
527  // if we don't have that interior element, then we'd
528  // better have a remote_elem signifying that fact.
529  if (!ip)
530  libmesh_assert_equal_to (elem->interior_parent(), remote_elem);
531  else
532  {
533  // The sending processor has an interior_parent here,
534  // and we have that element, but that does *NOT* mean
535  // we're already linking to it. Perhaps we initially
536  // received elem from a processor on which the
537  // interior_parent link was remote?
538  libmesh_assert(elem->interior_parent() == ip ||
539  elem->interior_parent() == remote_elem);
540 
541  // If the link was originally remote, update it
542  if (elem->interior_parent() == remote_elem)
543  {
544  elem->set_interior_parent(ip);
545  }
546  }
547  }
548  }
549 
550  // Our neighbor links should be "close to" correct - we may have
551  // to update a remote_elem link, and we can check for possible
552  // inconsistencies along the way.
553  //
554  // For subactive elements, we don't bother keeping neighbor
555  // links in good shape, so there's nothing we need to set or can
556  // safely assert here.
557  if (!elem->subactive())
558  for (unsigned int n=0; n != elem->n_neighbors(); ++n)
559  {
560  const dof_id_type neighbor_id =
561  cast_int<dof_id_type>(*in++);
562 
563  // If the sending processor sees a domain boundary here,
564  // we'd better agree.
565  if (neighbor_id == DofObject::invalid_id)
566  {
567  libmesh_assert (!(elem->neighbor_ptr(n)));
568  continue;
569  }
570 
571  // If the sending processor has a remote_elem neighbor here,
572  // then all we know is that we'd better *not* have a domain
573  // boundary.
574  if (neighbor_id == remote_elem->id())
575  {
576  libmesh_assert(elem->neighbor_ptr(n));
577  continue;
578  }
579 
580  Elem * neigh = mesh->query_elem_ptr(neighbor_id);
581 
582  // The sending processor sees a neighbor here, so if we
583  // don't have that neighboring element, then we'd better
584  // have a remote_elem signifying that fact.
585  if (!neigh)
586  {
587  libmesh_assert_equal_to (elem->neighbor_ptr(n), remote_elem);
588  continue;
589  }
590 
591  // The sending processor has a neighbor here, and we have
592  // that element, but that does *NOT* mean we're already
593  // linking to it. Perhaps we initially received both elem
594  // and neigh from processors on which their mutual link was
595  // remote?
596  libmesh_assert(elem->neighbor_ptr(n) == neigh ||
597  elem->neighbor_ptr(n) == remote_elem);
598 
599  // If the link was originally remote, we should update it,
600  // and make sure the appropriate parts of its family link
601  // back to us.
602  if (elem->neighbor_ptr(n) == remote_elem)
603  {
604  elem->set_neighbor(n, neigh);
605 
606  elem->make_links_to_me_local(n);
607  }
608  }
609 
610  // Our p level and refinement flags should be "close to" correct
611  // if we're not an active element - we might have a p level
612  // increased or decreased by changes in remote_elem children.
613  //
614  // But if we have remote_elem children, then we shouldn't be
615  // doing a projection on this inactive element on this
616  // processor, so we won't need correct p settings. Couldn't
617  // hurt to update, though.
618 #ifdef LIBMESH_ENABLE_AMR
619  if (elem->processor_id() != mesh->processor_id())
620  {
621  elem->hack_p_level(p_level);
622  elem->set_p_refinement_flag(p_refinement_flag);
623  }
624 #endif // LIBMESH_ENABLE_AMR
625 
626  // FIXME: We should add some debug mode tests to ensure that the
627  // encoded indexing and boundary conditions are consistent.
628  }
629  else
630  {
631  // We don't already have the element, so we need to create it.
632 
633  // Find the parent if necessary
634  Elem * parent = libmesh_nullptr;
635 #ifdef LIBMESH_ENABLE_AMR
636  // Find a child element's parent
637  if (level > 0)
638  {
639  // Note that we must be very careful to construct the send
640  // connectivity so that parents are encountered before
641  // children. If we get here and can't find the parent that
642  // is a fatal error.
643  parent = mesh->elem_ptr(parent_id);
644  }
645  // Or assert that the sending processor sees no parent
646  else
647  libmesh_assert_equal_to (parent_id, DofObject::invalid_id);
648 #else
649  // No non-level-0 elements without AMR
650  libmesh_assert_equal_to (level, 0);
651 #endif
652 
653  elem = Elem::build(type,parent).release();
654  libmesh_assert (elem);
655 
656 #ifdef LIBMESH_ENABLE_AMR
657  if (level != 0)
658  {
659  // Since this is a newly created element, the parent must
660  // have previously thought of this child as a remote element.
661  libmesh_assert_equal_to (parent->child_ptr(which_child_am_i), remote_elem);
662 
663  parent->add_child(elem, which_child_am_i);
664  }
665 
666  // Assign the refinement flags and levels
667  elem->set_p_level(p_level);
668  elem->set_refinement_flag(refinement_flag);

733  // neighbors to remote_elem ourselves and wait for a later
734  // packed element to give us better information.
735  if (neighbor_id == remote_elem->id())

738  continue;

742  // remote_elem until we get it.

744  if (!neigh)

747  continue;
748  }

751  // make sure the appropriate parts of its family link back
752  // to us.
753  elem->set_neighbor(n, neigh);
754 
755  elem->make_links_to_me_local(n);
756  }

759  }

764  // add any element side or edge boundary condition ids
765  if (level == 0)

768  {

775  }

781

783  mesh->get_boundary_info().add_edge
784  (elem, e, cast_int<boundary_id_type>(*in++));
785  }

788  {

791 
792  for (boundary_id_type bc_it=0; bc_it < num_bcs; bc_it++)
793  mesh->get_boundary_info().add_shellface
794  (elem, sf, cast_int<boundary_id_type>(*in++));
795  }
796  }

799  return elem;
