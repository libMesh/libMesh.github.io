125 std::unique_ptr<Elem> InfHex16::build_side_ptr (const unsigned int i,
126  bool proxy)
127 {
128  libmesh_assert_less (i, this->n_sides());
129 
130  if (proxy)
131  {
132  switch (i)
133  {
134  // base
135  case 0:
136  return libmesh_make_unique<Side<Quad8,InfHex16>>(this,i);
137 
138  // ifem sides
139  case 1:
140  case 2:
141  case 3:
142  case 4:
143  return libmesh_make_unique<Side<InfQuad6,InfHex16>>(this,i);
144 
145  default:
146  libmesh_error_msg("Invalid side i = " << i);
147  }
148  }
149 
150  else
151  {
152  // Return value
153  std::unique_ptr<Elem> face;
154 
155  // Think of a unit cube: (-1,1) x (-1,1) x (1,1)
156  switch (i)
157  {
158  // the base face
159  case 0:
160  {
161  face = libmesh_make_unique<Quad8>();
162  break;
163  }
164 
165  // connecting to another infinite element
166  case 1:
167  case 2:
168  case 3:
169  case 4:
170  {
171  face = libmesh_make_unique<InfQuad6>();
172  break;
173  }
174 
175  default:
176  libmesh_error_msg("Invalid side i = " << i);
177  }
178 
179  face->subdomain_id() = this->subdomain_id();
180 
181  // Set the nodes
182  for (unsigned n=0; n<face->n_nodes(); ++n)
183  face->set_node(n) = this->node_ptr(InfHex16::side_nodes_map[i][n]);
184 
185  return face;
186  }
187 }

201 void InfHex16::connectivity(const unsigned int sc,
202  const IOPackage iop,
203  std::vector<dof_id_type> & conn) const
204 {
205  libmesh_assert(_nodes);
206  libmesh_assert_less (sc, this->n_sub_elem());
207  libmesh_assert_not_equal_to (iop, INVALID_IO_PACKAGE);
208 
209  switch (iop)
210  {
211  case TECPLOT:
212  {
213  switch (sc)
214  {
215  case 0:
216 
217  conn[0] = this->node_id(0)+1;
218  conn[1] = this->node_id(1)+1;
219  conn[2] = this->node_id(2)+1;
220  conn[3] = this->node_id(3)+1;
221  conn[4] = this->node_id(4)+1;
222  conn[5] = this->node_id(5)+1;
223  conn[6] = this->node_id(6)+1;
224  conn[7] = this->node_id(7)+1;
225  return;
226 
227  default:
228  libmesh_error_msg("Invalid sc = " << sc);
229  }
230  }
231 
232  default:
233  libmesh_error_msg("Unsupported IO package " << iop);
234  }
235 }
