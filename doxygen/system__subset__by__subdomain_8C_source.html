113 void
114 SystemSubsetBySubdomain::
115 init (const SubdomainSelection & subdomain_selection)
116 {
117  _dof_ids.clear();
118 
119  std::vector<std::vector<dof_id_type>> dof_ids_per_processor(this->n_processors());
120 
121  const DofMap & dof_map = _system.get_dof_map();
122  std::vector<dof_id_type> dof_indices;
123 
124  const MeshBase & mesh = _system.get_mesh();
125  for (const auto & elem : mesh.active_local_element_ptr_range())
126  if (subdomain_selection(elem->subdomain_id()))
127  {
128  std::set<unsigned int>::const_iterator it = _var_nums.begin();
129  const std::set<unsigned int>::const_iterator itEnd = _var_nums.end();
130  for (; it!=itEnd; ++it)
131  {
132  dof_map.dof_indices (elem, dof_indices, *it);
133  for (std::size_t i=0; i<dof_indices.size(); i++)
134  {
135  const dof_id_type dof = dof_indices[i];
136  for (processor_id_type proc=0; proc<this->n_processors(); proc++)
137  if ((dof>=dof_map.first_dof(proc)) && (dof<dof_map.end_dof(proc)))
138  dof_ids_per_processor[proc].push_back(dof);
139  }
140  }
141  }
142 
143  /* Distribute information among processors. */
144  std::vector<Parallel::Request> request_per_processor(this->n_processors());
145  for (unsigned int proc=0; proc<this->n_processors(); proc++)
146  {
147  if (proc!=this->processor_id())
148  {
149  this->comm().send(proc,dof_ids_per_processor[proc],request_per_processor[proc]);
150  }
151  }
152  for (unsigned int proc=0; proc<this->n_processors(); proc++)
153  {
154  std::vector<dof_id_type> received_dofs;
155  if (proc==this->processor_id())
156  {
157  received_dofs = dof_ids_per_processor[proc];
158  }
159  else
160  {
161  this->comm().receive(proc,received_dofs);
162  }
163  for (std::size_t i=0; i<received_dofs.size(); i++)
164  _dof_ids.push_back(received_dofs[i]);
165  }
166 
167  /* Sort and unique the vector (using the same mechanism as in \p
168  DofMap::prepare_send_list()). */
169  std::sort(_dof_ids.begin(), _dof_ids.end());
170  std::vector<unsigned int>::iterator new_end = std::unique (_dof_ids.begin(), _dof_ids.end());
171  std::vector<unsigned int> (_dof_ids.begin(), new_end).swap (_dof_ids);
172 
173  /* Wait for sends to be complete. */
174  for (unsigned int proc=0; proc<this->n_processors(); proc++)
175  {
176  if (proc!=this->processor_id())
177  {
178  request_per_processor[proc].wait();
179  }
180  }
181 }
