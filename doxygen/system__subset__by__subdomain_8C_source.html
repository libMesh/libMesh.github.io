115 void
116 SystemSubsetBySubdomain::
117 init (const SubdomainSelection& subdomain_selection)
118 {
119  _dof_ids.clear();
120 
121  std::vector<std::vector<dof_id_type> > dof_ids_per_processor(this->n_processors());
122 
123  const DofMap & dof_map = _system.get_dof_map();
124  std::vector<dof_id_type> dof_indices;
125 
126  const MeshBase& mesh = _system.get_mesh();
127  MeshBase::const_element_iterator el = mesh.active_local_elements_begin();
128  const MeshBase::const_element_iterator end_el = mesh.active_local_elements_end();
129  for ( ; el != end_el; ++el)
130  {
131  const Elem* elem = *el;
132  if(subdomain_selection(elem->subdomain_id()))
133  {
134  std::set<unsigned int>::const_iterator it = _var_nums.begin();
135  const std::set<unsigned int>::const_iterator itEnd = _var_nums.end();
136  for (; it!=itEnd; ++it)
137  {
138  dof_map.dof_indices (elem, dof_indices, *it);
139  for(size_t i=0; i<dof_indices.size(); i++)
140  {
141  const dof_id_type dof = dof_indices[i];
142  for(processor_id_type proc=0; proc<this->n_processors(); proc++)
143  {
144  if((dof>=dof_map.first_dof(proc)) && (dof<dof_map.end_dof(proc)))
145  {
146  dof_ids_per_processor[proc].push_back(dof);
147  }
148  }
149  }
150  }
151  }
152  }
153 
154  /* Distribute information among processors. */
155  std::vector<Parallel::Request> request_per_processor(this->n_processors());
156  for(unsigned int proc=0; proc<this->n_processors(); proc++)
157  {
158  if(proc!=this->processor_id())
159  {
160  this->comm().send(proc,dof_ids_per_processor[proc],request_per_processor[proc]);
161  }
162  }
163  for(unsigned int proc=0; proc<this->n_processors(); proc++)
164  {
165  std::vector<dof_id_type> received_dofs;
166  if(proc==this->processor_id())
167  {
168  received_dofs = dof_ids_per_processor[proc];
169  }
170  else
171  {
172  this->comm().receive(proc,received_dofs);
173  }
174  for(unsigned int i=0; i<received_dofs.size(); i++)
175  {
176  _dof_ids.push_back(received_dofs[i]);
177  }
178  }
179 
180  /* Sort and unique the vector (using the same mechanism as in \p
181  DofMap::prepare_send_list()). */
182  std::sort(_dof_ids.begin(), _dof_ids.end());
183  std::vector<unsigned int>::iterator new_end = std::unique (_dof_ids.begin(), _dof_ids.end());
184  std::vector<unsigned int> (_dof_ids.begin(), new_end).swap (_dof_ids);
185 
186  /* Wait for sends to be complete. */
187  for(unsigned int proc=0; proc<this->n_processors(); proc++)
188  {
189  if(proc!=this->processor_id())
190  {
191  request_per_processor[proc].wait();
192  }
193  }
194 }
