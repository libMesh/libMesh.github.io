157 void XdrIO::write (const std::string & name)
158 {
159  if (this->legacy())
160  {
161  libmesh_deprecated();
162 
163  // We don't support writing parallel files in the legacy format
164  libmesh_assert(!this->_write_parallel);
165 
166  LegacyXdrIO(MeshOutput<MeshBase>::mesh(), this->binary()).write(name);
167  return;
168  }
169 
170  Xdr io ((this->processor_id() == 0) ? name : "", this->binary() ? ENCODE : WRITE);
171 
172  START_LOG("write()","XdrIO");
173 
174  // convenient reference to our mesh
175  const MeshBase & mesh = MeshOutput<MeshBase>::mesh();
176 
177  header_id_type n_elem = mesh.n_elem();
178  header_id_type n_nodes = mesh.n_nodes();
179 
180  libmesh_assert(n_elem == mesh.n_elem());
181  libmesh_assert(n_nodes == mesh.n_nodes());
182 
183  header_id_type n_bcs = cast_int<header_id_type>(mesh.get_boundary_info().n_boundary_conds());
184  header_id_type n_nodesets = cast_int<header_id_type>(mesh.get_boundary_info().n_nodeset_conds());
185  unsigned int n_p_levels = MeshTools::n_p_levels (mesh);
186 
187  bool write_parallel_files = this->write_parallel();
188 
189  //-------------------------------------------------------------
190  // For all the optional files -- the default file name is "n/a".
191  // However, the user may specify an optional external file.
192 
193  // If there are BCs and the user has not already provided a
194  // file name then write to "."
195  if ((n_bcs || n_nodesets) &&
196  this->boundary_condition_file_name() == "n/a")
197  this->boundary_condition_file_name() = ".";
198 
199  // If there are more than one subdomains and the user has not specified an
200  // external file then write the subdomain mapping to the default file "."
201  if ((mesh.n_subdomains() > 0) &&
202  (this->subdomain_map_file_name() == "n/a"))
203  this->subdomain_map_file_name() = ".";
204 
205  // In general we don't write the partition information.
206 
207  // If we have p levels and the user has not already provided
208  // a file name then write to "."
209  if ((n_p_levels > 1) &&
210  (this->polynomial_level_file_name() == "n/a"))
211  this->polynomial_level_file_name() = ".";
212 
213  // write the header
214  if (this->processor_id() == 0)
215  {
216  std::string full_ver = this->version() + (write_parallel_files ? " parallel" : "");
217  io.data (full_ver);
218 
219  io.data (n_elem, "# number of elements");
220  io.data (n_nodes, "# number of nodes");
221 
222  io.data (this->boundary_condition_file_name(), "# boundary condition specification file");
223  io.data (this->subdomain_map_file_name(), "# subdomain id specification file");
224  io.data (this->partition_map_file_name(), "# processor id specification file");
225  io.data (this->polynomial_level_file_name(), "# p-level specification file");
226 
227  // Version 0.9.2+ introduces sizes for each type
228  header_id_type write_size = sizeof(xdr_id_type), zero_size = 0;
229 
230  const bool
231  write_p_level = ("." == this->polynomial_level_file_name()),
232  write_partitioning = ("." == this->partition_map_file_name()),
233  write_subdomain_id = ("." == this->subdomain_map_file_name()),
234  write_bcs = ("." == this->boundary_condition_file_name());
235 
236  io.data (write_size, "# type size");
237  io.data (_write_unique_id ? write_size : zero_size, "# uid size");
238  io.data (write_partitioning ? write_size : zero_size, "# pid size");
239  io.data (write_subdomain_id ? write_size : zero_size, "# sid size");
240  io.data (write_p_level ? write_size : zero_size, "# p-level size");
241  // Boundary Condition sizes
242  io.data (write_bcs ? write_size : zero_size, "# eid size"); // elem id
243  io.data (write_bcs ? write_size : zero_size, "# side size "); // side number
244  io.data (write_bcs ? write_size : zero_size, "# bid size"); // boundary id
245  }
246 
247  if (write_parallel_files)
248  {
249  // Parallel xdr mesh files aren't implemented yet; until they
250  // are we'll just warn the user and write a serial file.
251  libMesh::out << "Warning! Parallel xda/xdr is not yet implemented.\n";
252  libMesh::out << "Writing a serialized file instead." << std::endl;
253 
254  // write subdomain names
255  this->write_serialized_subdomain_names(io);
256 
257  // write connectivity
258  this->write_serialized_connectivity (io, n_elem);
259 
260  // write the nodal locations
261  this->write_serialized_nodes (io, n_nodes);
262 
263  // write the boundary condition information
264  this->write_serialized_bcs (io, n_bcs);
265 
266  // write the nodeset information
267  this->write_serialized_nodesets (io, n_nodesets);
268  }
269  else
270  {
271  // write subdomain names
272  this->write_serialized_subdomain_names(io);
273 
274  // write connectivity
275  this->write_serialized_connectivity (io, n_elem);
276 
277  // write the nodal locations
278  this->write_serialized_nodes (io, n_nodes);
279 
280  // write the boundary condition information
281  this->write_serialized_bcs (io, n_bcs);
282 
283  // write the nodeset information
284  this->write_serialized_nodesets (io, n_nodesets);
285  }
286 
287  if(mesh.get_boundary_info().n_edge_conds() > 0)
288  {
289  libMesh::out << "Warning: Mesh contains edge boundary IDs, but these "
290  << "are not supported by the XDR format."
291  << std::endl;
292  }
293 
294  STOP_LOG("write()","XdrIO");
295 
296  // pause all processes until the writing ends -- this will
297  // protect for the pathological case where a write is
298  // followed immediately by a read. The write must be
299  // guaranteed to complete first.
300  io.close();
301  this->comm().barrier();
302 }

306 void XdrIO::write_serialized_subdomain_names(Xdr & io) const
307 {
308  if (this->processor_id() == 0)
309  {
310  const MeshBase & mesh = MeshOutput<MeshBase>::mesh();
311 
312  const std::map<subdomain_id_type, std::string> & subdomain_map = mesh.get_subdomain_name_map();
313 
314  std::vector<header_id_type> subdomain_ids; subdomain_ids.reserve(subdomain_map.size());
315  std::vector<std::string> subdomain_names; subdomain_names.reserve(subdomain_map.size());
316 
317  // We need to loop over the map and make sure that there aren't any invalid entries. Since we
318  // return writable references in mesh_base, it's possible for the user to leave some entity names
319  // blank. We can't write those to the XDA file.
320  header_id_type n_subdomain_names = 0;
321  std::map<subdomain_id_type, std::string>::const_iterator it_end = subdomain_map.end();
322  for (std::map<subdomain_id_type, std::string>::const_iterator it = subdomain_map.begin(); it != it_end; ++it)
323  {
324  if (!it->second.empty())
325  {
326  n_subdomain_names++;
327  subdomain_ids.push_back(it->first);
328  subdomain_names.push_back(it->second);
329  }
330  }
331 
332  io.data(n_subdomain_names, "# subdomain id to name map");
333  // Write out the ids and names in two vectors
334  if (n_subdomain_names)
335  {
336  io.data(subdomain_ids);
337  io.data(subdomain_names);
338  }
339  }
340 }

344 void XdrIO::write_serialized_connectivity (Xdr & io, const dof_id_type libmesh_dbg_var(n_elem)) const
345 {
346  libmesh_assert (io.writing());
347 
348  const bool
349  write_p_level = ("." == this->polynomial_level_file_name()),
350  write_partitioning = ("." == this->partition_map_file_name()),
351  write_subdomain_id = ("." == this->subdomain_map_file_name());
352 
353  // convenient reference to our mesh
354  const MeshBase & mesh = MeshOutput<MeshBase>::mesh();
355  libmesh_assert_equal_to (n_elem, mesh.n_elem());
356 
357  // We will only write active elements and their parents.
358  const unsigned int n_active_levels = MeshTools::n_active_levels (mesh);
359  std::vector<xdr_id_type> n_global_elem_at_level(n_active_levels);
360 
361  MeshBase::const_element_iterator it = mesh.local_elements_end(), end=it;
362 
363  // Find the number of local and global elements at each level
364 #ifndef NDEBUG
365  xdr_id_type tot_n_elem = 0;
366 #endif
367  for (unsigned int level=0; level<n_active_levels; level++)
368  {
369  it = mesh.local_level_elements_begin(level);
370  end = mesh.local_level_elements_end(level);
371 
372  n_global_elem_at_level[level] = MeshTools::n_elem(it, end);
373 
374  this->comm().sum(n_global_elem_at_level[level]);
375 #ifndef NDEBUG
376  tot_n_elem += n_global_elem_at_level[level];
377 #endif
378  libmesh_assert_less_equal (n_global_elem_at_level[level], n_elem);
379  libmesh_assert_less_equal (tot_n_elem, n_elem);
380  }
381 
382  std::vector<xdr_id_type>
383  xfer_conn, recv_conn;
384  std::vector<dof_id_type>
385  n_elem_on_proc(this->n_processors()), processor_offsets(this->n_processors());
386  std::vector<xdr_id_type> output_buffer;
387  std::vector<std::size_t>
388  xfer_buf_sizes(this->n_processors());
389 
390 #ifdef LIBMESH_ENABLE_AMR
391  typedef std::map<dof_id_type, std::pair<processor_id_type, dof_id_type> > id_map_type;
392  id_map_type parent_id_map, child_id_map;
393 #endif
394 
395  dof_id_type my_next_elem=0, next_global_elem=0;
396 
397  //-------------------------------------------
398  // First write the level-0 elements directly.
399  it = mesh.local_level_elements_begin(0);
400  end = mesh.local_level_elements_end(0);
401  for (; it != end; ++it, ++my_next_elem)
402  {
403  pack_element (xfer_conn, *it);
404 #ifdef LIBMESH_ENABLE_AMR
405  parent_id_map[(*it)->id()] = std::make_pair(this->processor_id(),
406  my_next_elem);
407 #endif
408  }
409  xfer_conn.push_back(my_next_elem); // toss in the number of elements transferred.
410 
411  std::size_t my_size = xfer_conn.size();
412  this->comm().gather (0, my_next_elem, n_elem_on_proc);
413  this->comm().gather (0, my_size, xfer_buf_sizes);
414 
415  processor_offsets[0] = 0;
416  for (unsigned int pid=1; pid<this->n_processors(); pid++)
417  processor_offsets[pid] = processor_offsets[pid-1] + n_elem_on_proc[pid-1];
418 
419  // All processors send their xfer buffers to processor 0.
420  // Processor 0 will receive the data and write out the elements.
421  if (this->processor_id() == 0)
422  {
423  // Write the number of elements at this level.
424  {
425  std::string comment = "# n_elem at level 0", legend = ", [ type ";
426  if (_write_unique_id)
427  legend += "uid ";
428  if (write_partitioning)
429  legend += "pid ";
430  if (write_subdomain_id)
431  legend += "sid ";
432  if (write_p_level)
433  legend += "p_level ";
434  legend += "(n0 ... nN-1) ]";
435  comment += legend;
436  io.data (n_global_elem_at_level[0], comment.c_str());
437  }
438 
439  for (unsigned int pid=0; pid<this->n_processors(); pid++)
440  {
441  recv_conn.resize(xfer_buf_sizes[pid]);
442  if (pid == 0)
443  recv_conn = xfer_conn;
444  else
445  this->comm().receive (pid, recv_conn);
446 
447  // at a minimum, the buffer should contain the number of elements,
448  // which could be 0.
449  libmesh_assert (!recv_conn.empty());
450 
451  {
452  const xdr_id_type n_elem_received = recv_conn.back();
453  std::vector<xdr_id_type>::const_iterator recv_conn_iter = recv_conn.begin();
454 
455  for (xdr_id_type elem=0; elem<n_elem_received; elem++, next_global_elem++)
456  {
457  output_buffer.clear();
458  const xdr_id_type n_nodes = *recv_conn_iter; ++recv_conn_iter;
459  output_buffer.push_back(*recv_conn_iter); /* type */ ++recv_conn_iter;
460 
461  if (_write_unique_id)
462  output_buffer.push_back(*recv_conn_iter); /* unique_id */ ++recv_conn_iter;
463 
464  if (write_partitioning)
465  output_buffer.push_back(*recv_conn_iter); /* processor id */ ++recv_conn_iter;
466 
467  if (write_subdomain_id)
468  output_buffer.push_back(*recv_conn_iter); /* subdomain id */ ++recv_conn_iter;
469 
470 #ifdef LIBMESH_ENABLE_AMR
471  if (write_p_level)
472  output_buffer.push_back(*recv_conn_iter); /* p level */ ++recv_conn_iter;
473 #endif
474  for (dof_id_type node=0; node<n_nodes; node++, ++recv_conn_iter)
475  output_buffer.push_back(*recv_conn_iter);
476 
477  io.data_stream
478  (&output_buffer[0],
479  cast_int<unsigned int>(output_buffer.size()),
480  cast_int<unsigned int>(output_buffer.size()));
481  }
482  }
483  }
484  }
485  else
486  this->comm().send (0, xfer_conn);
487 
488 #ifdef LIBMESH_ENABLE_AMR
489  //--------------------------------------------------------------------
490  // Next write the remaining elements indirectly through their parents.
491  // This will insure that the children are written in the proper order
492  // so they can be reconstructed properly.
493  for (unsigned int level=1; level<n_active_levels; level++)
494  {
495  xfer_conn.clear();
496 
497  it = mesh.local_level_elements_begin(level-1);
498  end = mesh.local_level_elements_end (level-1);
499 
500  dof_id_type my_n_elem_written_at_level = 0;
501  for (; it != end; ++it)
502  if (!(*it)->active()) // we only want the parents elements at this level, and
503  { // there is no direct iterator for this obscure use
504  const Elem * parent = *it;
505  id_map_type::iterator pos = parent_id_map.find(parent->id());
506  libmesh_assert (pos != parent_id_map.end());
507  const processor_id_type parent_pid = pos->second.first;
508  const dof_id_type parent_id = pos->second.second;
509  parent_id_map.erase(pos);
510 
511  for (unsigned int c=0; c<parent->n_children(); c++, my_next_elem++)
512  {
513  const Elem * child = parent->child(c);
514  pack_element (xfer_conn, child, parent_id, parent_pid);
515 
516  // this aproach introduces the possibility that we write
517  // non-local elements. These elements may well be parents
518  // at the next step
519  child_id_map[child->id()] = std::make_pair (child->processor_id(),
520  my_n_elem_written_at_level++);
521  }
522  }
523  xfer_conn.push_back(my_n_elem_written_at_level);
524  my_size = xfer_conn.size();
525  this->comm().gather (0, my_size, xfer_buf_sizes);
526 
527  // Processor 0 will receive the data and write the elements.
528  if (this->processor_id() == 0)
529  {
530  // Write the number of elements at this level.
531  {
532  char buf[80];
533  std::sprintf(buf, "# n_elem at level %u", level);
534  std::string comment(buf), legend = ", [ type ";
535 
536  if (_write_unique_id)
537  legend += "uid ";
538  legend += "parent ";
539  if (write_partitioning)
540  legend += "pid ";
541  if (write_subdomain_id)
542  legend += "sid ";
543  if (write_p_level)
544  legend += "p_level ";
545  legend += "(n0 ... nN-1) ]";
546  comment += legend;
547  io.data (n_global_elem_at_level[level], comment.c_str());
548  }
549 
550  for (unsigned int pid=0; pid<this->n_processors(); pid++)
551  {
552  recv_conn.resize(xfer_buf_sizes[pid]);
553  if (pid == 0)
554  recv_conn = xfer_conn;
555  else
556  this->comm().receive (pid, recv_conn);
557 
558  // at a minimum, the buffer should contain the number of elements,
559  // which could be 0.
560  libmesh_assert (!recv_conn.empty());
561 
562  {
563  const xdr_id_type n_elem_received = recv_conn.back();
564  std::vector<xdr_id_type>::const_iterator recv_conn_iter = recv_conn.begin();
565 
566  for (xdr_id_type elem=0; elem<n_elem_received; elem++, next_global_elem++)
567  {
568  output_buffer.clear();
569  const xdr_id_type n_nodes = *recv_conn_iter; ++recv_conn_iter;
570  output_buffer.push_back(*recv_conn_iter); /* type */ ++recv_conn_iter;
571  if (_write_unique_id)
572  output_buffer.push_back(*recv_conn_iter); /* unique_id */ ++recv_conn_iter;
573 
574  const xdr_id_type parent_local_id = *recv_conn_iter; ++recv_conn_iter;
575  const xdr_id_type parent_pid = *recv_conn_iter; ++recv_conn_iter;
576 
577  output_buffer.push_back (parent_local_id+processor_offsets[parent_pid]);
578 
579  if (write_partitioning)
580  output_buffer.push_back(*recv_conn_iter); /* processor id */ ++recv_conn_iter;
581 
582  if (write_subdomain_id)
583  output_buffer.push_back(*recv_conn_iter); /* subdomain id */ ++recv_conn_iter;
584 
585  if (write_p_level)
586  output_buffer.push_back(*recv_conn_iter); /* p level */ ++recv_conn_iter;
587 
588  for (xdr_id_type node=0; node<n_nodes; node++, ++recv_conn_iter)
589  output_buffer.push_back(*recv_conn_iter);
590 
591  io.data_stream
592  (&output_buffer[0],
593  cast_int<unsigned int>(output_buffer.size()),
594  cast_int<unsigned int>(output_buffer.size()));
595  }
596  }
597  }
598  }
599  else
600  this->comm().send (0, xfer_conn);
601 
602  // update the processor_offsets
603  processor_offsets[0] = processor_offsets.back() + n_elem_on_proc.back();
604  this->comm().gather (0, my_n_elem_written_at_level, n_elem_on_proc);
605  for (unsigned int pid=1; pid<this->n_processors(); pid++)
606  processor_offsets[pid] = processor_offsets[pid-1] + n_elem_on_proc[pid-1];
607 
608  // Now, at the next level we will again iterate over local parents. However,
609  // those parents may have been written by other processors (at this step),
610  // so we need to gather them into our *_id_maps.
611  {
612  std::vector<std::vector<dof_id_type> > requested_ids(this->n_processors());
613  std::vector<dof_id_type> request_to_fill;
614 
615  it = mesh.local_level_elements_begin(level);
616  end = mesh.local_level_elements_end(level);
617 
618  for (; it!=end; ++it)
619  if (!child_id_map.count((*it)->id()))
620  {
621  libmesh_assert_not_equal_to ((*it)->parent()->processor_id(), this->processor_id());
622  requested_ids[(*it)->parent()->processor_id()].push_back((*it)->id());
623  }
624 
625  // Next set the child_ids
626  for (unsigned int p=1; p != this->n_processors(); ++p)
627  {
628  // Trade my requests with processor procup and procdown
629  unsigned int procup = (this->processor_id() + p) %
630  this->n_processors();
631  unsigned int procdown = (this->n_processors() +
632  this->processor_id() - p) %
633  this->n_processors();
634 
635  this->comm().send_receive(procup, requested_ids[procup],
636  procdown, request_to_fill);
637 
638  // Fill those requests by overwriting the requested ids
639  for (std::size_t i=0; i<request_to_fill.size(); i++)
640  {
641  libmesh_assert (child_id_map.count(request_to_fill[i]));
642  libmesh_assert_equal_to (child_id_map[request_to_fill[i]].first, procdown);
643 
644  request_to_fill[i] = child_id_map[request_to_fill[i]].second;
645  }
646 
647  // Trade back the results
648  std::vector<dof_id_type> filled_request;
649  this->comm().send_receive(procdown, request_to_fill,
650  procup, filled_request);
651 
652  libmesh_assert_equal_to (filled_request.size(), requested_ids[procup].size());
653 
654  for (std::size_t i=0; i<filled_request.size(); i++)
655  child_id_map[requested_ids[procup][i]] =
656  std::make_pair (procup,
657  filled_request[i]);
658  }
659  // overwrite the parent_id_map with the child_id_map, but
660  // use std::map::swap() for efficiency.
661  parent_id_map.swap(child_id_map); child_id_map.clear();
662  }
663  }
664 #endif // LIBMESH_ENABLE_AMR
665  if (this->processor_id() == 0)
666  libmesh_assert_equal_to (next_global_elem, n_elem);
667
