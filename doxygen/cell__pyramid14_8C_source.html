160 std::unique_ptr<Elem> Pyramid14::build_side_ptr (const unsigned int i, bool proxy)
161 {
162  libmesh_assert_less (i, this->n_sides());
163 
164  if (proxy)
165  {
166  switch (i)
167  {
168  case 0:
169  case 1:
170  case 2:
171  case 3:
172  return libmesh_make_unique<Side<Tri6,Pyramid14>>(this,i);
173 
174  case 4:
175  return libmesh_make_unique<Side<Quad9,Pyramid14>>(this,i);
176 
177  default:
178  libmesh_error_msg("Invalid side i = " << i);
179  }
180  }
181 
182  else
183  {
184  // Return value
185  std::unique_ptr<Elem> face;
186 
187  switch (i)
188  {
189  case 0: // triangular face 1
190  case 1: // triangular face 2
191  case 2: // triangular face 3
192  case 3: // triangular face 4
193  {
194  face = libmesh_make_unique<Tri6>();
195  break;
196  }
197  case 4: // the quad face at z=0
198  {
199  face = libmesh_make_unique<Quad9>();
200  break;
201  }
202  default:
203  libmesh_error_msg("Invalid side i = " << i);
204  }
205 
206  face->subdomain_id() = this->subdomain_id();
207 
208  // Set the nodes
209  for (unsigned n=0; n<face->n_nodes(); ++n)
210  face->set_node(n) = this->node_ptr(Pyramid14::side_nodes_map[i][n]);
211 
212  return face;
213  }
214 }
