142 class GaussLobattoRedistributionFunction : public FunctionBase<Real>
143 {
144 public:
148  GaussLobattoRedistributionFunction(unsigned int nx,
149  Real xmin,
150  Real xmax,
151  unsigned int ny=0,
152  Real ymin=0,
153  Real ymax=0,
154  unsigned int nz=0,
155  Real zmin=0,
156  Real zmax=0) :
157  FunctionBase<Real>(libmesh_nullptr)
158  {
159  _nelem.resize(3);
160  _nelem[0] = nx;
161  _nelem[1] = ny;
162  _nelem[2] = nz;
163 
164  _mins.resize(3);
165  _mins[0] = xmin;
166  _mins[1] = ymin;
167  _mins[2] = zmin;
168 
169  _widths.resize(3);
170  _widths[0] = xmax - xmin;
171  _widths[1] = ymax - ymin;
172  _widths[2] = zmax - zmin;
173 
174  // Precompute the cosine values.
175  _cosines.resize(3);
176  for (unsigned dir=0; dir<3; ++dir)
177  if (_nelem[dir] != 0)
178  {
179  _cosines[dir].resize(_nelem[dir]+1);
180  for (unsigned i=0; i<_cosines[dir].size(); ++i)
181  _cosines[dir][i] = std::cos(libMesh::pi * i / _nelem[dir]);
182  }
183  }
184 
189  virtual UniquePtr<FunctionBase<Real> > clone () const libmesh_override
190  {
191  return UniquePtr<FunctionBase<Real> > (new GaussLobattoRedistributionFunction(*this));
192  }

199  virtual void operator() (const Point & p,
200  const Real /*time*/,
201  DenseVector<Real> & output) libmesh_override
202  {
203  output.resize(3);
204 
205  for (unsigned dir=0; dir<3; ++dir)
206  if (_nelem[dir] != 0)
207  {
208  // Figure out the index of the current point.
209  Real float_index = (p(dir) - _mins[dir]) * _nelem[dir] / _widths[dir];
210 
211  // std::modf separates the fractional and integer parts of the index.
212  Real integer_part = 0;
213  Real fractional_part = std::modf(float_index, &integer_part);
214 
215  // Vertex node?
216  if (std::abs(fractional_part) < TOLERANCE || std::abs(fractional_part - 1.0) < TOLERANCE)
217  {
218  int index = round(float_index);
219 
220  // Move node to the Gauss-Lobatto position.
221  output(dir) = _mins[dir] + _widths[dir] * 0.5 * (1.0 - _cosines[dir][index]);
222  }
223 
224  // Mid-edge (quadratic) node?
225  else if (std::abs(fractional_part - 0.5) < TOLERANCE)
226  {
227  // Move node to the Gauss-Lobatto position, which is the average of
228  // the node to the left and the node to the right.
229  output(dir) = _mins[dir] + _widths[dir] * 0.5 *
230  (1.0 - 0.5*(_cosines[dir][integer_part] + _cosines[dir][integer_part+1]));
231  }
232 
233  // 1D only: Left interior (cubic) node?
234  else if (std::abs(fractional_part - 1./3.) < TOLERANCE)
235  {
236  // Move node to the Gauss-Lobatto position, which is
237  // 2/3*left_vertex + 1/3*right_vertex.
238  output(dir) = _mins[dir] + _widths[dir] * 0.5 *
239  (1.0 - 2./3.*_cosines[dir][integer_part] - 1./3.*_cosines[dir][integer_part+1]);
240  }
241 
242  // 1D only: Right interior (cubic) node?
243  else if (std::abs(fractional_part - 2./3.) < TOLERANCE)
244  {
245  // Move node to the Gauss-Lobatto position, which is
246  // 1/3*left_vertex + 2/3*right_vertex.
247  output(dir) = _mins[dir] + _widths[dir] * 0.5 *
248  (1.0 - 1./3.*_cosines[dir][integer_part] - 2./3.*_cosines[dir][integer_part+1]);
249  }
250 
251  else
252  libmesh_error_msg("Cannot redistribute node: " << p);
253  }
254  }

263  libmesh_not_implemented();
264  }

1258  MeshTools::Modification::redistribute(mesh, func);
1259  }
1260  else // !gauss_lobatto_grid
1261  {
1262  for (unsigned int p=0; p<mesh.n_nodes(); p++)
1263  {
1264  mesh.node_ref(p)(0) = (mesh.node_ref(p)(0))*(xmax-xmin) + xmin;
1265  mesh.node_ref(p)(1) = (mesh.node_ref(p)(1))*(ymax-ymin) + ymin;
1266  mesh.node_ref(p)(2) = (mesh.node_ref(p)(2))*(zmax-zmin) + zmin;
1267  }
1268  }
1269 
1270 
1271 
1272  // Additional work for tets and pyramids: we take the existing
1273  // HEX27 discretization and split each element into 24
1274  // sub-tets or 6 sub-pyramids.
1275  //
1276  // 24 isn't the minimum-possible number of tets, but it
1277  // obviates any concerns about the edge orientations between
1278  // the various elements.
1279  if ((type == TET4) ||
1280  (type == TET10) ||
1281  (type == PYRAMID5) ||
1282  (type == PYRAMID13) ||
1283  (type == PYRAMID14))
1284  {
1285  // Temporary storage for new elements. (24 tets per hex, 6 pyramids)
1286  std::vector<Elem *> new_elements;
1287 
1288  if ((type == TET4) || (type == TET10))
1289  new_elements.reserve(24*mesh.n_elem());
1290  else
1291  new_elements.reserve(6*mesh.n_elem());
1292 
1293  // Create tetrahedra or pyramids
1294  {
1295  MeshBase::element_iterator el = mesh.elements_begin();
1296  const MeshBase::element_iterator end_el = mesh.elements_end();
1297 
1298  for ( ; el != end_el; ++el)
1299  {
1300  // Get a pointer to the HEX27 element.
1301  Elem * base_hex = *el;
1302 
1303  // Get a pointer to the node located at the HEX27 centroid
1304  Node * apex_node = base_hex->node_ptr(26);
1305 
1306  // Container to catch ids handed back from BoundaryInfo
1307  std::vector<boundary_id_type> ids;
1308 
1309  for (unsigned short s=0; s<base_hex->n_sides(); ++s)
1310  {
1311  // Get the boundary ID(s) for this side
1312  boundary_info.boundary_ids(*el, s, ids);
1313 
1314  // We're creating this Mesh, so there should be 0 or 1 boundary IDs.
1315  libmesh_assert(ids.size() <= 1);
1316 
1317  // A convenient name for the side's ID.
1318  boundary_id_type b_id = ids.empty() ? BoundaryInfo::invalid_id : ids[0];
1319 
1320  // Need to build the full-ordered side!
1321  UniquePtr<Elem> side = base_hex->build_side_ptr(s);
1322 
1323  if ((type == TET4) || (type == TET10))
1324  {
1325  // Build 4 sub-tets per side
1326  for (unsigned int sub_tet=0; sub_tet<4; ++sub_tet)
1327  {
1328  new_elements.push_back( new Tet4 );
1329  Elem * sub_elem = new_elements.back();
1330  sub_elem->set_node(0) = side->node_ptr(sub_tet);
1331  sub_elem->set_node(1) = side->node_ptr(8); // centroid of the face
1332  sub_elem->set_node(2) = side->node_ptr(sub_tet==3 ? 0 : sub_tet+1 ); // wrap-around
1333  sub_elem->set_node(3) = apex_node; // apex node always used!
1334 
1335  // If the original hex was a boundary hex, add the new sub_tet's side
1336  // 0 with the same b_id. Note: the tets are all aligned so that their
1337  // side 0 is on the boundary.
1338  if (b_id != BoundaryInfo::invalid_id)
1339  boundary_info.add_side(sub_elem, 0, b_id);
1340  }
1341  } // end if ((type == TET4) || (type == TET10))
1342 
1343  else // type==PYRAMID5 || type==PYRAMID13 || type==PYRAMID14
1344  {
1345  // Build 1 sub-pyramid per side.
1346  new_elements.push_back(new Pyramid5);
1347  Elem * sub_elem = new_elements.back();
1348 
1349  // Set the base. Note that since the apex is *inside* the base_hex,
1350  // and the pyramid uses a counter-clockwise base numbering, we need to
1351  // reverse the [1] and [3] node indices.
1352  sub_elem->set_node(0) = side->node_ptr(0);
1353  sub_elem->set_node(1) = side->node_ptr(3);
1354  sub_elem->set_node(2) = side->node_ptr(2);
1355  sub_elem->set_node(3) = side->node_ptr(1);
1356 
1357  // Set the apex
1358  sub_elem->set_node(4) = apex_node;
1359 
1360  // If the original hex was a boundary hex, add the new sub_pyr's side
1361  // 4 (the square base) with the same b_id.
1362  if (b_id != BoundaryInfo::invalid_id)
1363  boundary_info.add_side(sub_elem, 4, b_id);
1364  } // end else type==PYRAMID5 || type==PYRAMID13 || type==PYRAMID14
1365  }
1366  }
1367  }
1368 
1369 
1370  // Delete the original HEX27 elements from the mesh, and the boundary info structure.
1371  {
1372  MeshBase::element_iterator el = mesh.elements_begin();
1373  const MeshBase::element_iterator end_el = mesh.elements_end();
1374 
1375  for ( ; el != end_el; ++el)
1376  {
1377  boundary_info.remove(*el); // Safe even if *el has no boundary info.
1378  mesh.delete_elem(*el);
1379  }
1380  }
1381 
1382  // Add the new elements
1383  for (unsigned int i=0; i<new_elements.size(); ++i)
1384  mesh.add_elem(new_elements[i]);
1385 
1386  } // end if (type == TET4,TET10,PYRAMID5,PYRAMID13,PYRAMID14
1387 
1388 
1389  // Use all_second_order to convert the TET4's to TET10's or PYRAMID5's to PYRAMID14's
1390  if ((type == TET10) || (type == PYRAMID14))
1391  mesh.all_second_order();
1392 
1393  else if (type == PYRAMID13)
1394  mesh.all_second_order(/*full_ordered=*/false);
1395 
1396 
1397  // Add sideset names to boundary info (Z axis out of the screen)
1398  boundary_info.sideset_name(0) = "back";
1399  boundary_info.sideset_name(1) = "bottom";
1400  boundary_info.sideset_name(2) = "right";
1401  boundary_info.sideset_name(3) = "top";
1402  boundary_info.sideset_name(4) = "left";
1403  boundary_info.sideset_name(5) = "front";
1404 
1405  // Add nodeset names to boundary info
1406  boundary_info.nodeset_name(0) = "back";
1407  boundary_info.nodeset_name(1) = "bottom";
1408  boundary_info.nodeset_name(2) = "right";
1409  boundary_info.nodeset_name(3) = "top";
1410  boundary_info.nodeset_name(4) = "left";
1411  boundary_info.nodeset_name(5) = "front";
1412 
1413  break;
1414  } // end case dim==3
1415 
1416  default:
1417  libmesh_error_msg("Unknown dimension " << mesh.mesh_dimension());
1418  }
1419 
1420  STOP_LOG("build_cube()", "MeshTools::Generation");
1421 
1422 
1423 
1424  // Done building the mesh. Now prepare it for use.
1425  mesh.prepare_for_use (/*skip_renumber =*/ false);
1426 }
