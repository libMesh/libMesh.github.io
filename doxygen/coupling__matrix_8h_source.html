136  }
137 
138  operator bool() const {
139  const std::size_t max_size = std::numeric_limits<std::size_t>::max();
140 
141  // Find the range that might contain i,j
142  // lower_bound isn't *quite* what we want
143  CouplingMatrix::rc_type::const_iterator lb = std::upper_bound
144  (_mat._ranges.begin(), _mat._ranges.end(),
145  std::make_pair(_location, max_size));
146  if (lb!=_mat._ranges.begin())
147  --lb;
148  else
149  lb=_mat._ranges.end();
150 
151  // If no range might contain i,j then it's 0
152  if (lb == _mat._ranges.end())
153  return false;
154 
155  const std::size_t lastloc = lb->second;
156 
157 #ifdef DEBUG

163  next++;

165  {
166  // Ranges should be sorted and should not touch

169 #endif

224  }
225  else

234  next++;

289  libmesh_assert_greater(nextloc, lastloc+1);

302  }
303  else

329 class ConstCouplingRow
330 {
331 public:
332  ConstCouplingRow(unsigned int row_in,
333  const CouplingMatrix& mat_in) :
334  _row_i(row_in), _mat(mat_in)
335  {
336  libmesh_assert_less(_row_i, _mat.size());
337 
338  // Location for i,N
339  _begin_location = _row_i*_mat.size()+_mat.size()-1;
340 
341  const std::size_t max_size = std::numeric_limits<std::size_t>::max();
342 
343  // Find the range that might contain i,N
344  // lower_bound isn't *quite* what we want
345  _begin_it = std::upper_bound
346  (_mat._ranges.begin(), _mat._ranges.end(),
347  std::make_pair(_begin_location, max_size));
348  if (_begin_it !=_mat._ranges.begin())
349  --_begin_it;
350  else
351  _begin_it=_mat._ranges.end();
352 
353  // If that range doesn't exist then we're an empty row
354  if (_begin_it == _mat._ranges.end())
355  _begin_location = max_size;
356  else
357  {
358  const std::size_t lastloc = _begin_it->second;
359 #ifdef DEBUG
360  const std::size_t firstloc = _begin_it->first;
361  libmesh_assert_less_equal(firstloc, lastloc);
362 #endif
363 
364  // If that range ends before i,0 then we're an empty row
365  std::size_t zero_location = _row_i*_mat.size();
366  if (zero_location > lastloc)
367  {
368  _begin_location = max_size;
369  _begin_it = _mat._ranges.end();
370  }
371  else
372  // We have *some* entry(s) in this row, we just need to find
373  // the earliest
374  {
375  while (_begin_it != _mat._ranges.begin())
376  {
377  CouplingMatrix::rc_type::const_iterator prev =
378  _begin_it;
379  --prev;
380 
381  if (prev->second < zero_location)
382  break;
383 
384  _begin_it = prev;
385  }
386  if (_begin_it->first < zero_location)
387  _begin_location = zero_location;
388  else
389  _begin_location = _begin_it->first;
390  }
391  }
392  }
