71 void PostscriptIO::write (const std::string & fname)
72 {
73  // We may need to gather a DistributedMesh to output it, making that
74  // const qualifier in our constructor a dirty lie
75  MeshSerializer serialize(const_cast<MeshBase &>(this->mesh()), !_is_parallel_format);
76 
77  if (this->mesh().processor_id() == 0)
78  {
79  // Get a constant reference to the mesh.
80  const MeshBase & the_mesh = MeshOutput<MeshBase>::mesh();
81 
82  // Only works in 2D
83  libmesh_assert_equal_to (the_mesh.mesh_dimension(), 2);
84 
85  // Create output file stream.
86  // _out is now a private member of the class.
87  _out.open(fname.c_str());
88 
89  // Make sure it opened correctly
90  if (!_out.good())
91  libmesh_file_error(fname.c_str());
92 
93  // The mesh bounding box gives us info about what the
94  // Postscript bounding box should be.
95  MeshTools::BoundingBox bbox = MeshTools::bounding_box(the_mesh);
96 
97  // Add a little extra padding to the "true" bounding box so
98  // that we can still see the boundary
99  const Real percent_padding = 0.01;
100  const Real dx=bbox.second(0)-bbox.first(0); libmesh_assert_greater (dx, 0.0);
101  const Real dy=bbox.second(1)-bbox.first(1); libmesh_assert_greater (dy, 0.0);
102 
103  const Real x_min = bbox.first(0) - percent_padding*dx;
104  const Real y_min = bbox.first(1) - percent_padding*dy;
105  const Real x_max = bbox.second(0) + percent_padding*dx;
106  const Real y_max = bbox.second(1) + percent_padding*dy;
107 
108  // Width of the output as given in postscript units.
109  // This usually is given by the strange unit 1/72 inch.
110  // A width of 300 represents a size of roughly 10 cm.
111  const Real width = 300;
112  _scale = width / (x_max-x_min);
113  _offset(0) = x_min;
114  _offset(1) = y_min;
115 
116  // Header writing stuff stolen from Deal.II
117  std::time_t time1= std::time (0);
118  std::tm * time = std::localtime(&time1);
119  _out << "%!PS-Adobe-2.0 EPSF-1.2" << '\n'
120  //<< "%!PS-Adobe-1.0" << '\n' // Lars' PS version
121  << "%%Filename: " << fname << '\n'
122  << "%%Title: LibMesh Output" << '\n'
123  << "%%Creator: LibMesh: A C++ finite element library" << '\n'
124  << "%%Creation Date: "
125  << time->tm_year+1900 << "/"
126  << time->tm_mon+1 << "/"
127  << time->tm_mday << " - "
128  << time->tm_hour << ":"
129  << std::setw(2) << time->tm_min << ":"
130  << std::setw(2) << time->tm_sec << '\n'
131  << "%%BoundingBox: "
132  // lower left corner
133  << "0 0 "
134  // upper right corner
135  << static_cast<unsigned int>( rint((x_max-x_min) * _scale ))
136  << ' '
137  << static_cast<unsigned int>( rint((y_max-y_min) * _scale ))
138  << '\n';
139 
140  // define some abbreviations to keep
141  // the output small:
142  // m=move turtle to
143  // l=define a line
144  // s=set rgb color
145  // sg=set gray value
146  // lx=close the line and plot the line
147  // lf=close the line and fill the interior
148  _out << "/m {moveto} bind def" << '\n'
149  << "/l {lineto} bind def" << '\n'
150  << "/s {setrgbcolor} bind def" << '\n'
151  << "/sg {setgray} bind def" << '\n'
152  << "/cs {curveto stroke} bind def" << '\n'
153  << "/lx {lineto closepath stroke} bind def" << '\n'
154  << "/lf {lineto closepath fill} bind def" << '\n';
155 
156  _out << "%%EndProlog" << '\n';
157  // << '\n';
158 
159  // Set line width in the postscript file.
160  _out << line_width << " setlinewidth" << '\n';
161 
162  // Set line cap and join options
163  _out << "1 setlinecap" << '\n';
164  _out << "1 setlinejoin" << '\n';
165 
166  // allow only five digits for output (instead of the default
167  // six); this should suffice even for fine grids, but reduces
168  // the file size significantly
169  _out << std::setprecision (5);
170 
171  // Loop over the active elements, draw lines for the edges. We
172  // draw even quadratic elements with straight sides, i.e. a straight
173  // line sits between each pair of vertices. Also we draw every edge
174  // for an element regardless of the fact that it may overlap with
175  // another. This would probably be a useful optimization...
176  MeshBase::const_element_iterator el = the_mesh.active_elements_begin();
177  const MeshBase::const_element_iterator end_el = the_mesh.active_elements_end();
178  for ( ; el != end_el; ++el)
179  {
180  this->plot_linear_elem(*el);
181  //this->plot_quadratic_elem(*el); // Experimental
182  }
183 
184  // Issue the showpage command, and we're done.
185  _out << "showpage" << std::endl;
186 
187  } // end if (this->mesh().processor_id() == 0)
188 }

195 void PostscriptIO::plot_linear_elem(const Elem * elem)
196 {
197  // Clear the string contents. Yes, this really is how you do that...
198  _cell_string.str("");
199 
200  // The general strategy is:
201  // 1.) Use m := {moveto} to go to vertex 0.
202  // 2.) Use l := {lineto} commands to draw lines to vertex 1, 2, ... N-1.
203  // 3a.) Use lx := {lineto closepath stroke} command at vertex N to draw the last line.
204  // 3b.) lf := {lineto closepath fill} command to shade the cell just drawn
205  // All of our 2D elements' vertices are numbered in counterclockwise order,
206  // so we can just draw them in the same order.
207 
208  // 1.)
209  _current_point = (elem->point(0) - _offset) * _scale;
210  _cell_string << _current_point(0) << " " << _current_point(1) << " "; // write x y
211  _cell_string << "m ";
212 
213  // 2.)
214  const unsigned int nv=elem->n_vertices();
215  for (unsigned int v=1; v<nv-1; ++v)
216  {
217  _current_point = (elem->point(v) - _offset) * _scale;
218  _cell_string << _current_point(0) << " " << _current_point(1) << " "; // write x y
219  _cell_string << "l ";
220  }
221 
222  // 3.)
223  _current_point = (elem->point(nv-1) - _offset) * _scale;
224  _cell_string << _current_point(0) << " " << _current_point(1) << " "; // write x y
225 
226  // We draw the shaded (interior) parts first, if applicable.
227  if (shade_value > 0.0)
228  _out << shade_value << " sg " << _cell_string.str() << "lf\n";
229 
230  // Draw the black lines (I guess we will always do this)
231  _out << "0 sg " << _cell_string.str() << "lx\n";
232 }

237 void PostscriptIO::plot_quadratic_elem(const Elem * elem)
238 {
239  for (unsigned int ns=0; ns<elem->n_sides(); ++ns)
240  {
241  // Build the quadratic side
242  UniquePtr<const Elem> side = elem->build_side_ptr(ns);
243 
244  // Be sure it's quadratic (Edge2). Eventually we could
245  // handle cubic elements as well...
246  libmesh_assert_equal_to ( side->type(), EDGE3 );
247 
248  _out << "0 sg ";
249 
250  // Move to the first point on this side.
251  _current_point = (side->point(0) - _offset) * _scale;
252  _out << _current_point(0) << " " << _current_point(1) << " "; // write x y
253  _out << "m ";
254 
255  // Compute _bezier_coeffs for this edge. This fills up
256  // the _bezier_coeffs vector.
257  this->_compute_edge_bezier_coeffs(side.get());
258 
259  // Print curveto path to file
260  for (unsigned int i=0; i<_bezier_coeffs.size(); ++i)
261  _out << _bezier_coeffs[i](0) << " " << _bezier_coeffs[i](1) << " ";
262  _out << " cs\n";
263  }
264 }

269 void PostscriptIO::_compute_edge_bezier_coeffs(const Elem * elem)
270 {
271  // I only know how to do this for an Edge3!
272  libmesh_assert_equal_to (elem->type(), EDGE3);
273 
274  // Get x-coordinates into an array, transform them,
275  // and repeat for y.
276  float phys_coords[3] = {0., 0., 0.};
277  float bez_coords[3] = {0., 0., 0.};
278 
279  for (unsigned int i=0; i<2; ++i)
280  {
281  // Initialize vectors. Physical coordinates are initialized
282  // by their postscript-scaled values.
283  for (unsigned int j=0; j<3; ++j)
284  {
285  phys_coords[j] = static_cast<float>
286  ((elem->point(j)(i) - _offset(i)) * _scale);
287  bez_coords[j] = 0.; // zero out result vector
288  }
289 
290  // Multiply matrix times vector
291  for (unsigned int j=0; j<3; ++j)
292  for (unsigned int k=0; k<3; ++k)
293  bez_coords[j] += _bezier_transform[j][k]*phys_coords[k];
294 
295  // Store result in _bezier_coeffs
296  for (unsigned int j=0; j<3; ++j)
297  _bezier_coeffs[j](i) = phys_coords[j];
298  }
299 }
