153 void MeshBase::prepare_for_use (const bool skip_renumber_nodes_and_elements, const bool skip_find_neighbors)
154 {
155  parallel_object_only();
156 
157  libmesh_assert(this->comm().verify(this->is_serial()));
158 
159  // A distributed mesh may have processors with no elements (or
160  // processors with no elements of higher dimension, if we ever
161  // support mixed-dimension meshes), but we want consistent
162  // mesh_dimension anyways.
163  //
164  // cache_elem_dims() should get the elem_dimensions() and
165  // mesh_dimension() correct later, and we don't need it earlier.
166 
167 
168  // Renumber the nodes and elements so that they in contiguous
169  // blocks. By default, _skip_renumber_nodes_and_elements is false.
170  //
171  // We may currently change that by passing
172  // skip_renumber_nodes_and_elements==true to this function, but we
173  // should use the allow_renumbering() accessor instead.
174  //
175  // Instances where you if prepare_for_use() should not renumber the nodes
176  // and elements include reading in e.g. an xda/r or gmv file. In
177  // this case, the ordering of the nodes may depend on an accompanying
178  // solution, and the node ordering cannot be changed.
179 
180  if (skip_renumber_nodes_and_elements)
181  {
182  libmesh_deprecated();
183  this->allow_renumbering(false);
184  }
185 
186  // Mesh modification operations might not leave us with consistent
187  // id counts, but our partitioner might need that consistency.
188  if(!_skip_renumber_nodes_and_elements)
189  this->renumber_nodes_and_elements();
190  else
191  this->update_parallel_id_counts();
192 
193  // Let all the elements find their neighbors
194  if(!skip_find_neighbors)
195  this->find_neighbors();
196 
197  // Search the mesh for all the dimensions of the elements
198  // and cache them.
199  this->cache_elem_dims();
200 
201  // Search the mesh for elements that have a neighboring element
202  // of dim+1 and set that element as the interior parent
203  this->detect_interior_parents();
204 
205  // Fix up node unique ids in case mesh generation code didn't take
206  // exceptional care to do so.
207 // MeshCommunication().make_node_unique_ids_parallel_consistent(*this);
208 
209  // We're going to still require that mesh generation code gets
210  // element unique ids consistent.
211 #if defined(DEBUG) && defined(LIBMESH_ENABLE_UNIQUE_ID)
212  MeshTools::libmesh_assert_valid_unique_ids(*this);
213 #endif
214 
215  // Partition the mesh.
216  this->partition();
217 
218  // If we're using ParallelMesh, we'll want it parallelized.
219  this->delete_remote_elements();
220 
221  if(!_skip_renumber_nodes_and_elements)
222  this->renumber_nodes_and_elements();
223 
224  // Reset our PointLocator. This needs to happen any time the elements
225  // in the underlying elements in the mesh have changed, so we do it here.
226  this->clear_point_locator();
227 
228  // The mesh is now prepared for use.
229  _is_prepared = true;
230 
231 #if defined(DEBUG) && defined(LIBMESH_ENABLE_UNIQUE_ID)
232  MeshTools::libmesh_assert_valid_unique_ids(*this);
233 #endif
234 }

405 void MeshBase::partition (const unsigned int n_parts)
406 {
407  // If we get here and we have unpartitioned elements, we need that
408  // fixed.
409  if (this->n_unpartitioned_elem() > 0)
410  {
411  libmesh_assert (partitioner().get());
412  libmesh_assert (this->is_serial());
413  partitioner()->partition (*this, n_parts);
414  }
415  // NULL partitioner means don't repartition
416  // Non-serial meshes may not be ready for repartitioning here.
417  else if(!skip_partitioning() &&
418  partitioner().get())
419  {
420  partitioner()->partition (*this, n_parts);
421  }
422  else
423  {
424  // Adaptive coarsening may have "orphaned" nodes on processors
425  // whose elements no longer share them. We need to check for
426  // and possibly fix that.
427  Partitioner::set_node_processor_ids(*this);
428 
429  // Make sure locally cached partition count
430  this->recalculate_n_partitions();
431 
432  // Make sure any other locally cached data is correct
433  this->update_post_partitioning();
434  }
435 }

437 unsigned int MeshBase::recalculate_n_partitions()
438 {
439  // This requires an inspection on every processor
440  parallel_object_only();
441 
442  const_element_iterator el = this->active_local_elements_begin();
443  const_element_iterator end = this->active_local_elements_end();
444 
445  unsigned int max_proc_id=0;
446 
447  for (; el!=end; ++el)
448  max_proc_id = std::max(max_proc_id, static_cast<unsigned int>((*el)->processor_id()));
449 
450  // The number of partitions is one more than the max processor ID.
451  _n_parts = max_proc_id+1;
452 
453  this->comm().max(_n_parts);
454 
455  return _n_parts;
456 }

540 void MeshBase::cache_elem_dims()
541 {
542  // This requires an inspection on every processor
543  parallel_object_only();
544 
545  // Need to clear _elem_dims first in case all elements of a
546  // particular dimension have been deleted.
547  _elem_dims.clear();
548 
549  const_element_iterator el = this->active_elements_begin();
550  const_element_iterator end = this->active_elements_end();
551 
552  for (; el!=end; ++el)
553  _elem_dims.insert((*el)->dim());
554 
555  // Some different dimension elements may only live on other processors
556  this->comm().set_union(_elem_dims);
557 
558  // If the largest element dimension found is larger than the current
559  // _spatial_dimension, increase _spatial_dimension.
560  unsigned int max_dim = this->mesh_dimension();
561  if (max_dim > _spatial_dimension)
562  _spatial_dimension = cast_int<unsigned char>(max_dim);
563 
564  // _spatial_dimension may need to increase from 1->2 or 2->3 if the
565  // mesh is full of 1D elements but they are not x-aligned, or the
566  // mesh is full of 2D elements but they are not in the x-y plane.
567  // If the mesh is x-aligned or x-y planar, we will end up checking
568  // every node's coordinates and not breaking out of the loop
569  // early...
570  if (_spatial_dimension < 3)
571  {
572  const_node_iterator node_it = this->nodes_begin();
573  const_node_iterator node_end = this->nodes_end();
574  for (; node_it != node_end; ++node_it)
575  {
576  Node & node = **node_it;
577 
578  // Note: the exact floating point comparison is intentional,
579  // we don't want to get tripped up by tolerances.
580  if (node(1) != 0.)
581  {
582  _spatial_dimension = 2;
583 #if LIBMESH_DIM == 2
584  // If libmesh is compiled in 2D mode, this is the
585  // largest spatial dimension possible so we can break
586  // out.
587  break;
588 #endif
589  }
590 
591 #if LIBMESH_DIM > 2
592  if (node(2) != 0.)
593  {
594  // Spatial dimension can't get any higher than this, so
595  // we can break out.
596  _spatial_dimension = 3;
597  break;
598  }
599 #endif
600  }
601  }

604 void MeshBase::detect_interior_parents()
605 {
606  // This requires an inspection on every processor
607  parallel_object_only();
608 
609  // Check if the mesh contains mixed dimensions. If so, then set interior parents, otherwise return.
610  if (this->elem_dimensions().size() == 1)
611  return;
612 
613  //This map will be used to set interior parents
614  LIBMESH_BEST_UNORDERED_MAP<dof_id_type, std::vector<dof_id_type> > node_to_elem;
615 
616  const_element_iterator el = this->active_elements_begin();
617  const_element_iterator end = this->active_elements_end();
618 
619  for (; el!=end; ++el)
620  {
621  const Elem * elem = *el;
622 
623  // Populating the node_to_elem map, same as MeshTools::build_nodes_to_elem_map
624  for (unsigned int n=0; n<elem->n_vertices(); n++)
625  {
626  libmesh_assert_less (elem->id(), this->max_elem_id());
627 
628  node_to_elem[elem->node(n)].push_back(elem->id());
629  }
630  }
631 
632  // Automatically set interior parents
633  el = this->elements_begin();
634  for (; el!=end; ++el)
635  {
636  Elem * element = *el;
637 
638  // Ignore an 3D element or an element that already has an interior parent
639  if (element->dim()>=LIBMESH_DIM || element->interior_parent())
640  continue;
641 
642  // Start by generating a SET of elements that are dim+1 to the current
643  // element at each vertex of the current element, thus ignoring interior nodes.
644  // If one of the SET of elements is empty, then we will not have an interior parent
645  // since an interior parent must be connected to all vertices of the current element
646  std::vector< std::set<dof_id_type> > neighbors( element->n_vertices() );
647 
648  bool found_interior_parents = false;
649 
650  for (dof_id_type n=0; n < element->n_vertices(); n++)
651  {
652  std::vector<dof_id_type> & element_ids = node_to_elem[element->node(n)];
653  for (std::vector<dof_id_type>::iterator e_it = element_ids.begin();
654  e_it != element_ids.end(); e_it++)
655  {
656  dof_id_type eid = *e_it;
657  if (this->elem(eid)->dim() == element->dim()+1)
658  neighbors[n].insert(eid);
659  }
660  if (neighbors[n].size()>0)
661  {
662  found_interior_parents = true;
663  }
664  else
665  {
666  // We have found an empty set, no reason to continue
667  // Ensure we set this flag to false before the break since it could have
668  // been set to true for previous vertex
669  found_interior_parents = false;
670  break;
671  }
672  }
673 
674  // If we have successfully generated a set of elements for each vertex, we will compare
675  // the set for vertex 0 will the sets for the vertices until we find a id that exists in
676  // all sets. If found, this is our an interior parent id. The interior parent id found
677  // will be the lowest element id if there is potential for multiple interior parents.
678  if (found_interior_parents)
679  {
680  std::set<dof_id_type> & neighbors_0 = neighbors[0];
681  for (std::set<dof_id_type>::iterator e_it = neighbors_0.begin();
682  e_it != neighbors_0.end(); e_it++)
683  {
684  found_interior_parents=false;
685  dof_id_type interior_parent_id = *e_it;
686  for (dof_id_type n=1; n < element->n_vertices(); n++)
687  {
688  if (neighbors[n].find(interior_parent_id)!=neighbors[n].end())
689  {
690  found_interior_parents=true;
691  }
