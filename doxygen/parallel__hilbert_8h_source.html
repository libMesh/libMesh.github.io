103 void dofobjectkey_max_op (libMesh::Parallel::DofObjectKey * in,
104  libMesh::Parallel::DofObjectKey * inout,
105  int * len, void *)
106 {
107  // When (*in <= *inout), then inout already contains max(*in,*inout)
108  // Otherwise we need to copy from in.
109  for (int i=0; i<*len; i++, in++, inout++)
110  if (*inout < *in)
111  *inout = *in;
112 }
