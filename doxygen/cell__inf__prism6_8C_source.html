102 std::unique_ptr<Elem> InfPrism6::build_side_ptr (const unsigned int i,
103  bool proxy)
104 {
105  libmesh_assert_less (i, this->n_sides());
106 
107  if (proxy)
108  {
109  switch (i)
110  {
111  // base
112  case 0:
113  return libmesh_make_unique<Side<Tri3,InfPrism6>>(this,i);
114 
115  // ifem sides
116  case 1:
117  case 2:
118  case 3:
119  return libmesh_make_unique<Side<InfQuad4,InfPrism6>>(this,i);
120 
121  default:
122  libmesh_error_msg("Invalid side i = " << i);
123  }
124  }
125 
126  else
127  {
128  // Return value
129  std::unique_ptr<Elem> face;
130 
131  switch (i)
132  {
133  case 0: // the triangular face at z=-1, base face
134  {
135  face = libmesh_make_unique<Tri3>();
136  break;
137  }
138 
139  case 1: // the quad face at y=0
140  case 2: // the other quad face
141  case 3: // the quad face at x=0
142  {
143  face = libmesh_make_unique<InfQuad4>();
144  break;
145  }
146 
147  default:
148  libmesh_error_msg("Invalid side i = " << i);
149  }
150 
151  face->subdomain_id() = this->subdomain_id();
152 
153  // Set the nodes
154  for (unsigned n=0; n<face->n_nodes(); ++n)
155  face->set_node(n) = this->node_ptr(InfPrism6::side_nodes_map[i][n]);
156 
157  return face;
158  }
159 }

173 void InfPrism6::connectivity(const unsigned int libmesh_dbg_var(sc),
174  const IOPackage iop,
175  std::vector<dof_id_type> & conn) const
176 {
177  libmesh_assert(_nodes);
178  libmesh_assert_less (sc, this->n_sub_elem());
179  libmesh_assert_not_equal_to (iop, INVALID_IO_PACKAGE);
180 
181  switch (iop)
182  {
183  case TECPLOT:
184  {
185  conn.resize(8);
186  conn[0] = this->node_id(0)+1;
187  conn[1] = this->node_id(1)+1;
188  conn[2] = this->node_id(2)+1;
189  conn[3] = this->node_id(2)+1;
190  conn[4] = this->node_id(3)+1;
191  conn[5] = this->node_id(4)+1;
192  conn[6] = this->node_id(5)+1;
193  conn[7] = this->node_id(5)+1;
194  return;
195  }
196 
197  default:
198  libmesh_error_msg("Unsupported IO package " << iop);
199  }
