64 void NameBasedIO::read (const std::string& name)
65 {
66  MeshBase& mymesh = MeshInput<MeshBase>::mesh();
67 
68  // See if the file exists. Perform this check on all processors
69  // so that the code is terminated properly in the case that the
70  // file does not exist.
71 
72  // For Nemesis files, the name we try to read will have suffixes
73  // identifying processor rank
74  if (name.rfind(".nem") + 4 == name.size() ||
75  name.rfind(".n") + 2 == name.size())
76  {
77  std::ostringstream full_name;
78 
79  // Find the length of a string which represents the highest processor ID
80  full_name << (mymesh.n_processors());
81  int field_width = cast_int<int>(full_name.str().size());
82 
83  // reset the string stream
84  full_name.str("");
85 
86  // And build up the full filename
87  full_name << name
88  << '.' << mymesh.n_processors()
89  << '.' << std::setfill('0') << std::setw(field_width) << mymesh.processor_id();
90 
91  std::ifstream in (full_name.str().c_str());
92 
93  if (!in.good())
94  libmesh_error_msg("ERROR: cannot locate specified file:\n\t" << full_name.str());
95  }
96  else if(name.rfind(".cp")) {} // Do error checking in the reader
97  else
98  {
99  std::ifstream in (name.c_str());
100 
101  if (!in.good())
102  libmesh_error_msg("ERROR: cannot locate specified file:\n\t" << name);
103  }
104 
105  // Look for parallel formats first
106  if (is_parallel_file_format(name))
107  {
108  // no need to handle bz2 files here -- the Xdr class does that.
109  if ((name.rfind(".xda") < name.size()) ||
110  (name.rfind(".xdr") < name.size()))
111  {
112  XdrIO xdr_io(mymesh);
113 
114  // .xda* ==> bzip2/gzip/ASCII flavors
115  if (name.rfind(".xda") < name.size())
116  {
117  xdr_io.binary() = false;
118  xdr_io.read (name);
119  }
120  else // .xdr* ==> true binary XDR file
121  {
122  xdr_io.binary() = true;
123  xdr_io.read (name);
124  }
125 
126  // The xdr_io object gets constructed with legacy() == false.
127  // if legacy() == true then it means that a legacy file was detected and
128  // thus processor 0 performed the read. We therefore need to broadcast the
129  // mesh. Further, for this flavor of mesh solution data ordering is tied
130  // to the node ordering, so we better not reorder the nodes!
131  if (xdr_io.legacy())
132  {
133  mymesh.allow_renumbering(false);
134  MeshCommunication().broadcast(mymesh);
135  }
136 
137  // libHilbert-enabled libMesh builds should construct files
138  // with a canonical node ordering, which libHilbert-enabled
139  // builds will be able to read in again regardless of any
140  // renumbering. So in that case we're free to renumber.
141  // However, if either the writer or the reader of this file
142  // don't have libHilbert, then we'll have to skip
143  // renumbering because we need the numbering to remain
144  // consistent with any solution file we read in next.
145 #ifdef LIBMESH_HAVE_LIBHILBERT
146  // if (!xdr_io.libhilbert_ordering())
147  // skip_renumber_nodes_and_elements = true;
148 #else
149  mymesh.allow_renumbering(false);
150 #endif
151  }
152  else if (name.rfind(".nem") < name.size() ||
153  name.rfind(".n") < name.size())
154  Nemesis_IO(mymesh).read (name);
155  else if (name.rfind(".cp") < name.size())
156  {
157  if(name.rfind(".cpa") < name.size())
158  CheckpointIO(mymesh, false).read(name);
159  else
160  CheckpointIO(mymesh, true).read(name);
161  }
162  }
163 
164  // Serial mesh formats
165  else
166  {
167  START_LOG("read()", "NameBasedIO");
168 
169  // Read the file based on extension. Only processor 0
170  // needs to read the mesh. It will then broadcast it and
171  // the other processors will pick it up
172  if (mymesh.processor_id() == 0)
173  {
174  std::ostringstream pid_suffix;
175  pid_suffix << '_' << getpid();
176  // Nasty hack for reading/writing zipped files
177  std::string new_name = name;
178  if (name.size() - name.rfind(".bz2") == 4)
179  {
180 #ifdef LIBMESH_HAVE_BZIP
181  new_name.erase(new_name.end() - 4, new_name.end());
182  new_name += pid_suffix.str();
183  std::string system_string = "bunzip2 -f -k -c ";
184  system_string += name + " > " + new_name;
185  START_LOG("system(bunzip2)", "NameBasedIO");
186  if (std::system(system_string.c_str()))
187  libmesh_file_error(system_string);
188  STOP_LOG("system(bunzip2)", "NameBasedIO");
189 #else
190  libmesh_error_msg("ERROR: need bzip2/bunzip2 to open .bz2 file " << name);
191 #endif
192  }
193  else if (name.size() - name.rfind(".xz") == 3)
194  {
195 #ifdef LIBMESH_HAVE_XZ
196  new_name.erase(new_name.end() - 3, new_name.end());
197  new_name += pid_suffix.str();
198  std::string system_string = "xz -f -d -k -c ";
199  system_string += name + " > " + new_name;
200  START_LOG("system(xz -d)", "XdrIO");
201  if (std::system(system_string.c_str()))
202  libmesh_file_error(system_string);
203  STOP_LOG("system(xz -d)", "XdrIO");
204 #else
205  libmesh_error_msg("ERROR: need xz to open .xz file " << name);
206 #endif
207  }
208 
209  if (new_name.rfind(".mat") < new_name.size())
210  MatlabIO(mymesh).read(new_name);
211 
212  else if (new_name.rfind(".ucd") < new_name.size())
213  UCDIO(mymesh).read (new_name);
214 
215  else if ((new_name.rfind(".off") < new_name.size()) ||
216  (new_name.rfind(".ogl") < new_name.size()) ||
217  (new_name.rfind(".oogl") < new_name.size()))
218  OFFIO(mymesh).read (new_name);
219 
220  else if (new_name.rfind(".mgf") < new_name.size())
221  LegacyXdrIO(mymesh,true).read_mgf (new_name);
222 
223  else if (new_name.rfind(".unv") < new_name.size())
224  UNVIO(mymesh).read (new_name);
225 
226  else if ((new_name.rfind(".node") < new_name.size()) ||
227  (new_name.rfind(".ele") < new_name.size()))
228  TetGenIO(mymesh).read (new_name);
229 
230  else if (new_name.rfind(".exd") < new_name.size() ||
231  new_name.rfind(".e") < new_name.size())
232  ExodusII_IO(mymesh).read (new_name);
233 
234  else if (new_name.rfind(".msh") < new_name.size())
235  GmshIO(mymesh).read (new_name);
236 
237  else if (new_name.rfind(".gmv") < new_name.size())
238  GMVIO(mymesh).read (new_name);
239 
240  else if (new_name.rfind(".vtu") < new_name.size())
241  VTKIO(mymesh).read(new_name);
242 
243  else if (new_name.rfind(".inp") < new_name.size())
244  AbaqusIO(mymesh).read(new_name);
245 
246  else
247  {
248  libmesh_error_msg(" ERROR: Unrecognized file extension: " \
249  << name \
250  << "\n I understand the following:\n\n" \
251  << " *.e -- Sandia's ExodusII format\n" \
252  << " *.exd -- Sandia's ExodusII format\n" \
253  << " *.gmv -- LANL's General Mesh Viewer format\n" \
254  << " *.mat -- Matlab triangular ASCII file\n" \
255  << " *.n -- Sandia's Nemesis format\n" \
256  << " *.nem -- Sandia's Nemesis format\n" \
257  << " *.off -- OOGL OFF surface format\n" \
258  << " *.ucd -- AVS's ASCII UCD format\n" \
259  << " *.unv -- I-deas Universal format\n" \
260  << " *.vtu -- Paraview VTK format\n" \
261  << " *.inp -- Abaqus .inp format\n" \
262  << " *.xda -- libMesh ASCII format\n" \
263  << " *.xdr -- libMesh binary format\n" \
264  << " *.gz -- any above format gzipped\n" \
265  << " *.bz2 -- any above format bzip2'ed\n" \
266  << " *.xz -- any above format xzipped\n" \
267  << " *.cpa -- libMesh Checkpoint ASCII format\n" \
268  << " *.cpr -- libMesh Checkpoint binary format\n");
269  }
270 
271  // If we temporarily decompressed a file, remove the
272  // uncompressed version
273  if (name.size() - name.rfind(".bz2") == 4)
274  std::remove(new_name.c_str());
275  if (name.size() - name.rfind(".xz") == 3)
276  std::remove(new_name.c_str());
277  }
278 
279 
280  STOP_LOG("read()", "NameBasedIO");
281 
282  // Send the mesh & bcs (which are now only on processor 0) to the other
283  // processors
284  MeshCommunication().broadcast (mymesh);
285  }

289 void NameBasedIO::write (const std::string& name)
290 {
291  MeshBase& mymesh = MeshInput<MeshBase>::mesh();
292 
293  // parallel formats are special -- they may choose to write
294  // separate files, let's not try to handle the zipping here.
295  if (is_parallel_file_format(name))
296  {
297  // no need to handle bz2 files here -- the Xdr class does that.
298  if (name.rfind(".xda") < name.size())
299  XdrIO(mymesh).write(name);
300 
301  else if (name.rfind(".xdr") < name.size())
302  XdrIO(mymesh,true).write(name);
303 
304  else if (name.rfind(".nem") < name.size() ||
305  name.rfind(".n") < name.size())
306  Nemesis_IO(mymesh).write(name);
307  }
308 
309  // serial file formats
310  else
311  {
312  // Nasty hack for reading/writing zipped files
313  std::string new_name = name;
314  pid_t pid_0 = 0;
315  if (mymesh.processor_id() == 0)
316  pid_0 = getpid();
317  mymesh.comm().broadcast(pid_0);
318  std::ostringstream pid_suffix;
319  pid_suffix << '_' << pid_0;
320 
321  if (name.size() - name.rfind(".bz2") == 4)
322  {
323  new_name.erase(new_name.end() - 4, new_name.end());
324  new_name += pid_suffix.str();
325  }
326  else if (name.size() - name.rfind(".xz") == 3)
327  {
328  new_name.erase(new_name.end() - 3, new_name.end());
329  new_name += pid_suffix.str();
330  }
331 
332  // New scope so that io will close before we try to zip the file
333  {
334  // Write the file based on extension
335  if (new_name.rfind(".dat") < new_name.size())
336  TecplotIO(mymesh).write (new_name);
337 
338  else if (new_name.rfind(".plt") < new_name.size())
339  TecplotIO(mymesh,true).write (new_name);
340 
341  else if (new_name.rfind(".ucd") < new_name.size())
342  UCDIO (mymesh).write (new_name);
343 
344  else if (new_name.rfind(".gmv") < new_name.size())
345  if (mymesh.n_partitions() > 1)
346  GMVIO(mymesh).write (new_name);
347  else
348  {
349  GMVIO io(mymesh);
350  io.partitioning() = false;
351  io.write (new_name);
352  }
353 
354  else if (new_name.rfind(".ugrid") < new_name.size())
355  DivaIO(mymesh).write(new_name);
356  else if (new_name.rfind(".e") < new_name.size())
357  ExodusII_IO(mymesh).write(new_name);
358  else if (new_name.rfind(".mgf") < new_name.size())
359  LegacyXdrIO(mymesh,true).write_mgf(new_name);
360 
361  else if (new_name.rfind(".unv") < new_name.size())
362  UNVIO(mymesh).write (new_name);
363 
364  else if (new_name.rfind(".mesh") < new_name.size())
365  MEDITIO(mymesh).write (new_name);
366 
367  else if (new_name.rfind(".poly") < new_name.size())
368  TetGenIO(mymesh).write (new_name);
369 
370  else if (new_name.rfind(".msh") < new_name.size())
371  GmshIO(mymesh).write (new_name);
372 
373  else if (new_name.rfind(".fro") < new_name.size())
374  FroIO(mymesh).write (new_name);
375 
376  else if (new_name.rfind(".vtu") < new_name.size())
377  VTKIO(mymesh).write (new_name);
378 
379  else
380  {
381  libMesh::err
382  << " ERROR: Unrecognized file extension: " << name
383  << "\n I understand the following:\n\n"
384  << " *.dat -- Tecplot ASCII file\n"
385  << " *.e -- Sandia's ExodusII format\n"
386  << " *.exd -- Sandia's ExodusII format\n"
387  << " *.fro -- ACDL's surface triangulation file\n"
388  << " *.gmv -- LANL's GMV (General Mesh Viewer) format\n"
389  << " *.mesh -- MEdit mesh format\n"
390  << " *.mgf -- MGF binary mesh format\n"
391  << " *.msh -- GMSH ASCII file\n"
392  << " *.n -- Sandia's Nemesis format\n"
393  << " *.nem -- Sandia's Nemesis format\n"
394  << " *.plt -- Tecplot binary file\n"
395  << " *.poly -- TetGen ASCII file\n"
396  << " *.ucd -- AVS's ASCII UCD format\n"
397  << " *.ugrid -- Kelly's DIVA ASCII format\n"
398  << " *.unv -- I-deas Universal format\n"
399  << " *.vtu -- VTK (paraview-readable) format\n"
400  << " *.xda -- libMesh ASCII format\n"
401  << " *.xdr -- libMesh binary format,\n"
402  << std::endl
403  << "\n Exiting without writing output\n";
404  }
405  }
406 
407  // Nasty hack for reading/writing zipped files
408  if (name.size() - name.rfind(".bz2") == 4)
409  {
410  START_LOG("system(bzip2)", "NameBasedIO");
411  if (mymesh.processor_id() == 0)
412  {
413  std::string system_string = "bzip2 -f -c ";
414  system_string += new_name + " > " + name;
415  if (std::system(system_string.c_str()))
416  libmesh_file_error(system_string);
417  std::remove(new_name.c_str());
418  }
419  mymesh.comm().barrier();
420  STOP_LOG("system(bzip2)", "NameBasedIO");
421  }
422  if (name.size() - name.rfind(".xz") == 3)
423  {
424  START_LOG("system(xz)", "NameBasedIO");
425  if (mymesh.processor_id() == 0)
426  {
427  std::string system_string = "xz -f -c ";
428  system_string += new_name + " > " + name;
429  if (std::system(system_string.c_str()))
430  libmesh_file_error(system_string);
431  std::remove(new_name.c_str());
432  }
433  mymesh.comm().barrier();
434  STOP_LOG("system(xz)", "NameBasedIO");
435  }
436  }

441 void NameBasedIO::write_nodal_data (const std::string& name,
442  const std::vector<Number>& v,
443  const std::vector<std::string>& vn)
444 {
445  MeshBase& mymesh = MeshInput<MeshBase>::mesh();
446 
447  // Write the file based on extension

449  TecplotIO(mymesh).write_nodal_data (name, v, vn);
450 
451  else if (name.rfind(".e") < name.size())
452  ExodusII_IO(mymesh).write_nodal_data(name, v, vn);
453 
454  else if (name.rfind(".gmv") < name.size())
455  {
456  if (mymesh.n_subdomains() > 1)
457  GMVIO(mymesh).write_nodal_data (name, v, vn);

459  {
460  GMVIO io(mymesh);
461  io.partitioning() = false;
462  io.write_nodal_data (name, v, vn);
463  }
464  }
465 
466  else if (name.rfind(".mesh") < name.size())
467  MEDITIO(mymesh).write_nodal_data (name, v, vn);
468 
469  else if (name.rfind(".msh") < name.size())
470  GmshIO(mymesh).write_nodal_data (name, v, vn);
471 
472  else if (name.rfind(".nem") < name.size() ||
473  name.rfind(".n") < name.size())
474  Nemesis_IO(mymesh).write_nodal_data(name, v, vn);
475 
476  else if (name.rfind(".plt") < name.size())

495  << " *.msh -- GMSH ASCII file\n"
496  << " *.n -- Sandia's Nemesis format\n"
497  << " *.nem -- Sandia's Nemesis format\n"
498  << " *.plt -- Tecplot binary file\n"
499  << " *.pvtu -- Paraview VTK file\n"
500  << " *.ucd -- AVS's ASCII UCD format\n"

502  }

506 void NameBasedIO::write_equation_systems (const std::string& filename,
507  const EquationSystems& es,
508  const std::set<std::string>* system_names)
509 {
510  // XDA/XDR require a separate code path, and currently only support
511  // writing complete restarts
512  if (!system_names)
513  {
514  if (filename.rfind(".xda") < filename.size())
515  {
516  es.write(filename,WRITE,
517  EquationSystems::WRITE_DATA |
518  EquationSystems::WRITE_ADDITIONAL_DATA);
519  return;
520  }
521  else if (filename.rfind(".xdr") < filename.size())
522  {
523  es.write(filename,ENCODE,
524  EquationSystems::WRITE_DATA |
525  EquationSystems::WRITE_ADDITIONAL_DATA);
526  return;
527  }
528  }
529 
530  // Other formats just use the default "write nodal values" path
531  MeshOutput<MeshBase>::write_equation_systems
532  (filename, es, system_names);
533 }
