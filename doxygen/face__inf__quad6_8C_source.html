172  case 2:
173  {
174  edge = new InfEdge2;

176  }
177 
178  default:
179  libmesh_error_msg("Invalid side i = " << i);
180  }
181 
182  edge->subdomain_id() = this->subdomain_id();
183 
184  // Set the nodes
185  for (unsigned n=0; n<edge->n_nodes(); ++n)
186  edge->set_node(n) = this->get_node(InfQuad6::side_nodes_map[i][n]);
187 
188  return UniquePtr<Elem>(edge);
189  }
190 
191  libmesh_error_msg("We'll never get here!");
192  return UniquePtr<Elem>();

198 void InfQuad6::connectivity(const unsigned int sf,
199  const IOPackage iop,
200  std::vector<dof_id_type>& conn) const
201 {
202  libmesh_assert_less (sf, this->n_sub_elem());
203  libmesh_assert_not_equal_to (iop, INVALID_IO_PACKAGE);
204 
205  conn.resize(4);
206 
207  switch (iop)
208  {
209  case TECPLOT:
210  {
211  switch(sf)
212  {
213  case 0:
214  // linear sub-quad 0
215  conn[0] = this->node(0)+1;
216  conn[1] = this->node(4)+1;
217  conn[2] = this->node(5)+1;
218  conn[3] = this->node(2)+1;
219 
220  return;
221 
222  case 1:
223  // linear sub-quad 1
224  conn[0] = this->node(4)+1;
225  conn[1] = this->node(1)+1;
226  conn[2] = this->node(3)+1;
227  conn[3] = this->node(5)+1;
228 
229  return;
230 
231  default:
232  libmesh_error_msg("Invalid sf = " << sf);
233  }
234  }
235 
236  default:
237  libmesh_error_msg("Unsupported IO package " << iop);
238  }
239 }
