144 std::unique_ptr<Elem> InfQuad6::build_side_ptr (const unsigned int i,
145  bool proxy)
146 {
147  // libmesh_assert_less (i, this->n_sides());
148 
149  if (proxy)
150  {
151  switch (i)
152  {
153  case 0:
154  return libmesh_make_unique<Side<Edge3,InfQuad6>>(this,i);
155 
156  case 1:
157  case 2:
158  return libmesh_make_unique<Side<InfEdge2,InfQuad6>>(this,i);
159 
160  default:
161  libmesh_error_msg("Invalid side i = " << i);
162  }
163  }
164 
165  else
166  {
167  // Return value
168  std::unique_ptr<Elem> edge;
169 
170  switch (i)
171  {
172  case 0:
173  {
174  edge = libmesh_make_unique<Edge3>();
175  break;
176  }
177 
178  // adjacent to another infinite element
179  case 1:
180  case 2:
181  {
182  edge = libmesh_make_unique<InfEdge2>();
183  break;
184  }
185 
186  default:
187  libmesh_error_msg("Invalid side i = " << i);
188  }
189 
190  edge->subdomain_id() = this->subdomain_id();
191 
192  // Set the nodes
193  for (unsigned n=0; n<edge->n_nodes(); ++n)
194  edge->set_node(n) = this->node_ptr(InfQuad6::side_nodes_map[i][n]);
195 
196  return edge;
197  }

203 void InfQuad6::connectivity(const unsigned int sf,
204  const IOPackage iop,
205  std::vector<dof_id_type> & conn) const
206 {
207  libmesh_assert_less (sf, this->n_sub_elem());
208  libmesh_assert_not_equal_to (iop, INVALID_IO_PACKAGE);
209 
210  conn.resize(4);
211 
212  switch (iop)
213  {
214  case TECPLOT:
215  {
216  switch(sf)
217  {
218  case 0:
219  // linear sub-quad 0
220  conn[0] = this->node_id(0)+1;
221  conn[1] = this->node_id(4)+1;
222  conn[2] = this->node_id(5)+1;
223  conn[3] = this->node_id(2)+1;
224 
225  return;
226 
227  case 1:
228  // linear sub-quad 1
229  conn[0] = this->node_id(4)+1;
230  conn[1] = this->node_id(1)+1;
231  conn[2] = this->node_id(3)+1;
232  conn[3] = this->node_id(5)+1;
233 
234  return;
235 
236  default:
237  libmesh_error_msg("Invalid sf = " << sf);
238  }
239  }
240 
241  default:
242  libmesh_error_msg("Unsupported IO package " << iop);
243  }
244 }
