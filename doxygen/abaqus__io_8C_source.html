59 void add_eletype_entry(ElemType libmesh_elem_type,
60  const unsigned* node_map,
61  unsigned node_map_size,
62  const unsigned short* side_map,
63  unsigned side_map_size)
64 {
65  // If map entry does not exist, this will create it
66  ElementDefinition& map_entry = eletypes[libmesh_elem_type];
67 
68 
69  // Use the "swap trick" from Scott Meyer's "Effective STL" to swap
70  // an unnamed temporary vector into the map_entry's vector. Note:
71  // the vector(iter, iter) constructor is used.
72  std::vector<unsigned>
73  (node_map, node_map+node_map_size).swap
74  (map_entry.abaqus_zero_based_node_id_to_libmesh_node_id);
75 
76  std::vector<unsigned short>
77  (side_map, side_map+side_map_size).swap
78  (map_entry.abaqus_zero_based_side_id_to_libmesh_side_id);
79 }

205 void AbaqusIO::read (const std::string& fname)
206 {
207  // Get a reference to the mesh we are reading
208  MeshBase& the_mesh = MeshInput<MeshBase>::mesh();
209 
210  // Clear any existing mesh data
211  the_mesh.clear();
212 
213  // Open stream for reading
214  _in.open(fname.c_str());
215  libmesh_assert(_in.good());
216 
217  // Initialize the elems_of_dimension array. We will use this in a
218  // "1-based" manner so that elems_of_dimension[d]==true means
219  // elements of dimension d have been seen.
220  elems_of_dimension.resize(4, false);
221 
222  // Read file line-by-line... this is based on a set of different
223  // test input files. I have not looked at the full input file
224  // specs for Abaqus.
225  std::string s;
226  while (true)
227  {
228  // Try to read something. This may set EOF!
229  std::getline(_in, s);
230 
231  if (_in)
232  {
233  // Process s...
234  //
235  // There are many sections in Abaqus files, we read some
236  // but others are just ignored... Some sections may occur
237  // more than once. For example for a hybrid grid, you
238  // will have multiple *Element sections...
239 
240  // Some Abaqus files use all upper-case for section names,
241  // so we will just convert s to uppercase
242  std::string upper(s);
243  std::transform(upper.begin(), upper.end(), upper.begin(), ::toupper);
244 
245  // 0.) Look for the "*Part" section
246  if (upper.find("*PART") == static_cast<std::string::size_type>(0))
247  {
248  if (_already_seen_part)
249  libmesh_error_msg("We currently don't support reading Abaqus files with multiple PART sections");
250 
251  _already_seen_part = true;
252  }
253 
254  // 1.) Look for the "*Nodes" section
255  if (upper.find("*NODE") == static_cast<std::string::size_type>(0))
256  {
257  // Some sections that begin with *NODE are actually
258  // "*NODE OUTPUT" sections which we want to skip. I
259  // have only seen this with a single space, but it would
260  // probably be more robust to remove whitespace before
261  // making this check.
262  if (upper.find("*NODE OUTPUT") == static_cast<std::string::size_type>(0))
263  continue;
264 
265  // Some *Node sections also specify an Nset name on the same line.
266  // Look for one here.
267  std::string nset_name = this->parse_label(s, "nset");
268 
269  // Process any lines of comments that may be present
270  this->process_and_discard_comments();
271 
272  // Read a block of nodes
273  this->read_nodes(nset_name);
274  }
275 
276 
277 
278  // 2.) Look for the "*Element" section
279  else if (upper.find("*ELEMENT,") == static_cast<std::string::size_type>(0))
280  {
281  // Some sections that begin with *ELEMENT are actually
282  // "*ELEMENT OUTPUT" sections which we want to skip. I
283  // have only seen this with a single space, but it would
284  // probably be more robust to remove whitespace before
285  // making this check.
286  if (upper.find("*ELEMENT OUTPUT") == static_cast<std::string::size_type>(0))
287  continue;
288 
289  // Some *Element sections also specify an Elset name on the same line.
290  // Look for one here.
291  std::string elset_name = this->parse_label(s, "elset");
292 
293  // Process any lines of comments that may be present
294  this->process_and_discard_comments();
295 
296  // Read a block of elements
297  this->read_elements(upper, elset_name);
298  }
299 
300 
301 
302  // 3.) Look for a Nodeset section
303  else if (upper.find("*NSET") == static_cast<std::string::size_type>(0))
304  {
305  std::string nset_name = this->parse_label(s, "nset");
306 
307  // I haven't seen an unnamed nset yet, but let's detect it
308  // just in case...
309  if (nset_name == "")
310  libmesh_error_msg("Unnamed nset encountered!");
311 
312  // Process any lines of comments that may be present
313  this->process_and_discard_comments();
314 
315  // Read the IDs, storing them in _nodeset_ids
316  this->read_ids(nset_name, _nodeset_ids);
317  } // *Nodeset
318 
319 
320 
321  // 4.) Look for an Elset section
322  else if (upper.find("*ELSET") == static_cast<std::string::size_type>(0))
323  {
324  std::string elset_name = this->parse_label(s, "elset");
325 
326  // I haven't seen an unnamed elset yet, but let's detect it
327  // just in case...
328  if (elset_name == "")
329  libmesh_error_msg("Unnamed elset encountered!");
330 
331  // Process any lines of comments that may be present
332  this->process_and_discard_comments();
333 
334  // Read the IDs, storing them in _elemset_ids
335  this->read_ids(elset_name, _elemset_ids);
336  } // *Elset
337 
338 
339 
340  // 5.) Look for a Surface section. Need to be a little
341  // careful, since there are also "surface interaction"
342  // sections we don't want to read here.
343  else if (upper.find("*SURFACE,") == static_cast<std::string::size_type>(0))
344  {
345  // Get the name from the Name=Foo label. This will be the map key.
346  std::string sideset_name = this->parse_label(s, "name");
347 
348  // Process any lines of comments that may be present
349  this->process_and_discard_comments();
350 
351  // Read the sideset IDs
352  this->read_sideset(sideset_name, _sideset_ids);
353  }
354 
355  continue;
356  } // if (_in)
357 
358  // If !file, check to see if EOF was set. If so, break out
359  // of while loop.
360  if (_in.eof())
361  break;
362 
363  // If !in and !in.eof(), stream is in a bad state!
364  libmesh_error_msg("Stream is bad! Perhaps the file: " << fname << " does not exist?");
365  } // while
366 
367  // Set the Mesh dimension based on the highest dimension element seen.
368  the_mesh.set_mesh_dimension(this->max_elem_dimension_seen());
369 
370  // Set element IDs based on the element sets.
371  this->assign_subdomain_ids();
372 
373  // Assign nodeset values to the BoundaryInfo object
374  this->assign_boundary_node_ids();
375 
376  // Assign sideset values in the BoundaryInfo object
377  this->assign_sideset_ids();
378 
379  // Abaqus files only contain nodesets by default. To be useful in
380  // applying most types of BCs in libmesh, we will definitely need
381  // sidesets. So we can call the new BoundaryInfo function which
382  // generates sidesets from nodesets.
383  if (build_sidesets_from_nodesets)
384  the_mesh.get_boundary_info().build_side_list_from_node_list();
385 
386  // Delete lower-dimensional elements from the Mesh. We assume these
387  // were only used for setting BCs, and aren't part of the actual
388  // Mesh.
389  {
390  unsigned char max_dim = this->max_elem_dimension_seen();
391 
392  MeshBase::element_iterator el = the_mesh.elements_begin();
393  const MeshBase::element_iterator end_el = the_mesh.elements_end();
394 
395  for (; el != end_el; ++el)
396  {
397  Elem* elem = *el;
398 
399  if (elem->dim() < max_dim)
400  the_mesh.delete_elem(elem);
401  }
402  }
403 }

411 void AbaqusIO::read_nodes(std::string nset_name)
412 {
413  // Get a reference to the mesh we are reading
414  MeshBase& the_mesh = MeshInput<MeshBase>::mesh();
415 
416  // In the input files I have, Abaqus neither tells what
417  // the mesh dimension is nor how many nodes it has...
418  //
419  // The node line format is:
420  // id, x, y, z
421  // and you do have to parse out the commas.
422  // The z-coordinate will only be present for 3D meshes
423 
424  // Temporary variables for parsing lines of text
425  char c;
426  std::string dummy;
427 
428  // Defines the sequential node numbering used by libmesh. Since
429  // there can be multiple *NODE sections in an Abaqus file, we always
430  // start our numbering with the number of nodes currently in the
431  // Mesh.
432  dof_id_type libmesh_node_id = the_mesh.n_nodes();
433 
434  // We need to duplicate some of the read_ids code if this *NODE
435  // section also defines an NSET. We'll set up the id_storage
436  // pointer and push back IDs into this vector in the loop below...
437  std::vector<dof_id_type>* id_storage = NULL;
438  if (nset_name != "")
439  id_storage = &(_nodeset_ids[nset_name]);
440 
441  // We will read nodes until the next line begins with *, since that will be the
442  // next section.
443  // TODO: Is Abaqus guaranteed to start the line with '*' or can there be leading white space?
444  while (_in.peek() != '*' && _in.peek() != EOF)
445  {
446  // Re-Initialize variables to be read in from file
447  dof_id_type abaqus_node_id=0;
448  Real x=0, y=0, z=0;
449 
450  // Note: we assume *at least* 2D points here, should we worry about
451  // trying to read 1D Abaqus meshes?
452  _in >> abaqus_node_id >> c >> x >> c >> y;
453 
454  // Peek at the next character. If it is a comma, then there is another
455  // value to read!
456  if (_in.peek() == ',')
457  _in >> c >> z;
458 
459  // Read (and discard) the rest of the line, including the newline.
460  // This is required so that our 'peek()' at the beginning of this
461  // loop doesn't read the newline character, for example.
462  std::getline(_in, dummy);
463 
464  // If this *NODE section defines an NSET, also store the abaqus ID in id_storage
465  if (id_storage)
466  id_storage->push_back(abaqus_node_id);
467 
468  // Set up the abaqus -> libmesh node mapping. This is usually just the
469  // "off-by-one" map.
470  _abaqus_to_libmesh_node_mapping[abaqus_node_id] = libmesh_node_id;
471 
472  // Add the point to the mesh using libmesh's numbering,
473  // and post-increment the libmesh node counter.
474  the_mesh.add_point(Point(x,y,z), libmesh_node_id++);
475  } // while
476 }

482 void AbaqusIO::read_elements(std::string upper, std::string elset_name)
483 {
484  // Get a reference to the mesh we are reading
485  MeshBase& the_mesh = MeshInput<MeshBase>::mesh();
486 
487  // initialize the eletypes map (eletypes is a file-global variable)
488  init_eletypes();
489 
490  ElemType elem_type = INVALID_ELEM;
491  unsigned n_nodes_per_elem = 0;
492 
493  // Within s, we should have "type=XXXX"
494  if (upper.find("T3D2") != std::string::npos)
495  {
496  elem_type = EDGE2;
497  n_nodes_per_elem = 2;
498  elems_of_dimension[1] = true;
499  }
500  else if (upper.find("CPE4") != std::string::npos ||
501  upper.find("CPS4") != std::string::npos)
502  {
503  elem_type = QUAD4;
504  n_nodes_per_elem = 4;
505  elems_of_dimension[2] = true;
506  }
507  else if (upper.find("CPS3") != std::string::npos ||
508  upper.find("S3R") != std::string::npos)
509  {
510  elem_type = TRI3;
511  n_nodes_per_elem = 3;
512  elems_of_dimension[2] = true;
513  }
514  else if (upper.find("C3D8") != std::string::npos)
515  {
516  elem_type = HEX8;
517  n_nodes_per_elem = 8;
518  elems_of_dimension[3] = true;
519  }
520  else if (upper.find("C3D4") != std::string::npos)
521  {
522  elem_type = TET4;
523  n_nodes_per_elem = 4;
524  elems_of_dimension[3] = true;
525  }
526  else if (upper.find("C3D20") != std::string::npos)
527  {
528  elem_type = HEX20;
529  n_nodes_per_elem = 20;
530  elems_of_dimension[3] = true;
531  }
532  else if (upper.find("C3D6") != std::string::npos)
533  {
534  elem_type = PRISM6;
535  n_nodes_per_elem = 6;
536  elems_of_dimension[3] = true;
537  }
538  else if (upper.find("C3D15") != std::string::npos)
539  {
540  elem_type = PRISM15;
541  n_nodes_per_elem = 15;
542  elems_of_dimension[3] = true;
543  }
544  else if (upper.find("C3D10") != std::string::npos)
545  {
546  elem_type = TET10;
547  n_nodes_per_elem = 10;
548  elems_of_dimension[3] = true;
549  }
550  else
551  libmesh_error_msg("Unrecognized element type: " << upper);
552 
553  // Insert the elem type we detected into the set of all elem types for this mesh
554  _elem_types.insert(elem_type);
555 
556  // Grab a reference to the element definition for this element type
557  const ElementDefinition& eledef = eletypes[elem_type];
558 
559  // If the element definition was not found, the call above would have
560  // created one with an uninitialized struct. Check for that here...
561  if (eledef.abaqus_zero_based_node_id_to_libmesh_node_id.size() == 0)
562  libmesh_error_msg("No Abaqus->LibMesh mapping information for ElemType " \
563  << Utility::enum_to_string(elem_type) \
564  << "!");
565 
566  // We will read elements until the next line begins with *, since that will be the
567  // next section.
568  while (_in.peek() != '*' && _in.peek() != EOF)
569  {
570  // Read the element ID, it is the first number on each line. It is
571  // followed by a comma, so read that also. We will need this ID later
572  // when we try to assign subdomain IDs
573  dof_id_type abaqus_elem_id = 0;
574  char c;
575  _in >> abaqus_elem_id >> c;
576 
577  // Add an element of the appropriate type to the Mesh.
578  Elem* elem = the_mesh.add_elem(Elem::build(elem_type).release());
579 
580  // Associate the ID returned from libmesh with the abaqus element ID
581  //_libmesh_to_abaqus_elem_mapping[elem->id()] = abaqus_elem_id;
582  _abaqus_to_libmesh_elem_mapping[abaqus_elem_id] = elem->id();
583 
584  // The count of the total number of IDs read for the current element.
585  unsigned id_count=0;
586 
587  // Continue reading line-by-line until we have read enough nodes for this element
588  while (id_count < n_nodes_per_elem)
589  {
590  // Read entire line (up to carriage return) of comma-separated values
591  std::string csv_line;
592  std::getline(_in, csv_line);
593 
594  // Create a stream object out of the current line
595  std::stringstream line_stream(csv_line);
596 
597  // Process the comma-separated values
598  std::string cell;
599  while (std::getline(line_stream, cell, ','))
600  {
601  // FIXME: factor out this strtol stuff into a utility function.
602  char* endptr;
603  dof_id_type abaqus_global_node_id = cast_int<dof_id_type>
604  (std::strtol(cell.c_str(), &endptr, /*base=*/10));
605 
606  if (abaqus_global_node_id!=0 || cell.c_str() != endptr)
607  {
608  // Use the global node number mapping to determine the corresponding libmesh global node id
609  dof_id_type libmesh_global_node_id = _abaqus_to_libmesh_node_mapping[abaqus_global_node_id];
610 
611  // Grab the node pointer from the mesh for this ID
612  Node* node = the_mesh.node_ptr(libmesh_global_node_id);
613 
614  // If node_ptr() returns NULL, it may mean we have not yet read the
615  // *Nodes section, though I assumed that always came before the *Elements section...
616  if (node == NULL)
617  libmesh_error_msg("Error! Mesh returned NULL Node pointer. Either no node exists with ID " \
618  << libmesh_global_node_id \
619  << " or perhaps this input file has *Elements defined before *Nodes?");
620 
621  // Note: id_count is the zero-based abaqus (elem local) node index. We therefore map
622  // it to a libmesh elem local node index using the element definition map
623  unsigned libmesh_elem_local_node_id =
624  eledef.abaqus_zero_based_node_id_to_libmesh_node_id[id_count];
625 
626  // Set this node pointer within the element.
627  elem->set_node(libmesh_elem_local_node_id) = node;
628

653 std::string AbaqusIO::parse_label(std::string line, std::string label_name)
654 {
655  // Handle files which have weird line endings from e.g. windows.
656  // You can check what kind of line endings you have with 'cat -vet'.
657  // For example, some files may have two kinds of line endings like:
658  //
659  // 4997,^I496,^I532,^I487,^I948^M$
660  //
661  // and we don't want to deal with this when extracting a label, so
662  // just remove all the space characters, which should include all
663  // kinds of remaining newlines. (I don't think Abaqus allows
664  // whitespace in label names.)
665  line.erase(std::remove_if(line.begin(), line.end(), isspace), line.end());
666 
667  // Do all string comparisons in upper-case
668  std::string
669  upper_line(line),
670  upper_label_name(label_name);
671  std::transform(upper_line.begin(), upper_line.end(), upper_line.begin(), ::toupper);
672  std::transform(upper_label_name.begin(), upper_label_name.end(), upper_label_name.begin(), ::toupper);
673 
674  // Get index of start of "label="
675  size_t label_index = upper_line.find(upper_label_name + "=");
676 
677  if (label_index != std::string::npos)
678  {
679  // Location of the first comma following "label="
680  size_t comma_index = upper_line.find(",", label_index);
681 
682  // Construct iterators from which to build the sub-string.
683  // Note: The +1 while initializing beg is to skip past the "=" which follows the label name
684  std::string::iterator
685  beg = line.begin() + label_name.size() + 1 + label_index,
686  end = (comma_index == std::string::npos) ? line.end() : line.begin() + comma_index;
687 
688  return std::string(beg, end);
689  }
690 
691  // The label index was not found, return the empty string
692  return std::string("");
693 }

698 void AbaqusIO::read_ids(std::string set_name, container_t& container)
699 {
700  // Grab a reference to a vector that will hold all the IDs
701  std::vector<dof_id_type>& id_storage = container[set_name];
702 
703  // Read until the start of another section is detected, or EOF is encountered
704  while (_in.peek() != '*' && _in.peek() != EOF)
705  {
706  // Read entire comma-separated line into a string
707  std::string csv_line;
708  std::getline(_in, csv_line);
709 
710  // On that line, use std::getline again to parse each
711  // comma-separated entry.
712  std::string cell;
713  std::stringstream line_stream(csv_line);
714  while (std::getline(line_stream, cell, ','))
715  {
716  // If no conversion can be performed by strtol, 0 is returned.
717  //
718  // If endptr is not NULL, strtol() stores the address of the
719  // first invalid character in *endptr. If there were no
720  // digits at all, however, strtol() stores the original
721  // value of str in *endptr.
722  char* endptr;
723 
724  // FIXME - this needs to be updated for 64-bit inputs
725  dof_id_type id = cast_int<dof_id_type>
726  (std::strtol(cell.c_str(), &endptr, /*base=*/10));
727 
728  // Note that lists of comma-separated values in abaqus also
729  // *end* with a comma, so the last call to getline on a given
730  // line will get an empty string, which we must detect.
731  if (id != 0 || cell.c_str() != endptr)
732  {
733  // 'cell' is now a string with an integer id in it
734  id_storage.push_back( id );
735  }
736  }
737  }

776 void AbaqusIO::assign_subdomain_ids()
777 {
778  // Get a reference to the mesh we are reading
779  MeshBase& the_mesh = MeshInput<MeshBase>::mesh();
780 
781  // The number of elemsets we've found while reading
782  std::size_t n_elemsets = _elemset_ids.size();
783 
784  // Fill in a temporary map with (ElemType, index) pairs based on the _elem_types set. This
785  // will allow us to easily look up this index in the loop below.
786  std::map<ElemType, unsigned> elem_types_map;
787  {
788  unsigned ctr=0;
789  for (std::set<ElemType>::iterator it=_elem_types.begin(); it!=_elem_types.end(); ++it)
790  elem_types_map[*it] = ctr++;
791  }
792 
793  // Loop over each Elemset and assign subdomain IDs to Mesh elements
794  {
795  // The maximum element dimension seen while reading the Mesh
796  unsigned char max_dim = this->max_elem_dimension_seen();
797 
798  // The elemset_id counter assigns a logical numbering to the _elemset_ids keys
799  container_t::iterator it = _elemset_ids.begin();
800  for (unsigned elemset_id=0; it != _elemset_ids.end(); ++it, ++elemset_id)
801  {
802  // Grab a reference to the vector of IDs
803  std::vector<dof_id_type>& id_vector = it->second;
804 
805  // Loop over this vector
806  for (std::size_t i=0; i<id_vector.size(); ++i)
807  {
808  // Map the id_vector[i]'th element ID (Abaqus numbering) to LibMesh numbering
809  dof_id_type libmesh_elem_id = _abaqus_to_libmesh_elem_mapping[ id_vector[i] ];
810 
811  // Get pointer to that element
812  Elem* elem = the_mesh.elem(libmesh_elem_id);
813 
814  if (elem == NULL)
815  libmesh_error_msg("Mesh returned NULL pointer for Elem " << libmesh_elem_id);
816 
817  // We won't assign subdomain ids to lower-dimensional
818  // elements, as they are assumed to represent boundary
819  // conditions. Since lower-dimensional elements can
820  // appear in multiple sidesets, it doesn't make sense to
821  // assign them a single subdomain id... only the last one
822  // assigned would actually "stick".
823  if (elem->dim() < max_dim)
824  break;
825 
826  // Compute the proper subdomain ID, based on the formula in the
827  // documentation for this function.
828  subdomain_id_type computed_id = cast_int<subdomain_id_type>
829  (elemset_id + (elem_types_map[elem->type()] * n_elemsets));
830 
831  // Assign this ID to the element in question
832  elem->subdomain_id() = computed_id;
833 
834  // We will also assign a unique name to the computed_id,
835  // which is created by appending the geometric element
836  // name to the elset name provided by the user in the
837  // Abaqus file.
838  std::string computed_name = it->first + "_" + Utility::enum_to_string(elem->type());
839  the_mesh.subdomain_name(computed_id) = computed_name;
840  }
841  }
842  }

884 void AbaqusIO::assign_sideset_ids()
885 {
886  // Get a reference to the mesh we are reading
887  MeshBase& the_mesh = MeshInput<MeshBase>::mesh();
888 
889  // initialize the eletypes map (eletypes is a file-global variable)
890  init_eletypes();
891 
892  // Iterate over the container of sidesets
893  {
894  sideset_container_t::iterator it = _sideset_ids.begin();
895  for (unsigned short current_id=0; it != _sideset_ids.end(); ++it, ++current_id)
896  {
897  // Associate current_id with the name we determined earlier
898  the_mesh.get_boundary_info().sideset_name(current_id) = it->first;
899 
900  // Get a reference to the current vector of nodeset ID values
901  std::vector<std::pair<dof_id_type,unsigned> >& sideset_ids = it->second;
902 
903  for (std::size_t i=0; i<sideset_ids.size(); ++i)
904  {
905  // sideset_ids is a vector of pairs (elem id, side id). Pull them out
906  // now to make the code below more readable.
907  dof_id_type abaqus_elem_id = sideset_ids[i].first;
908  unsigned abaqus_side_number = sideset_ids[i].second;
909 
910  // Map the Abaqus element ID to LibMesh numbering
911  dof_id_type libmesh_elem_id = _abaqus_to_libmesh_elem_mapping[ abaqus_elem_id ];
912 
913  // Get pointer to that element
914  Elem* elem = the_mesh.elem(libmesh_elem_id);
915 
916  // Check that the pointer returned from the Mesh is non-NULL
917  if (elem == NULL)
918  libmesh_error_msg("Mesh returned NULL pointer for Elem " << libmesh_elem_id);
919 
920  // Grab a reference to the element definition for this element type
921  const ElementDefinition& eledef = eletypes[elem->type()];
922 
923  // If the element definition was not found, the call above would have
924  // created one with an uninitialized struct. Check for that here...
925  if (eledef.abaqus_zero_based_side_id_to_libmesh_side_id.size() == 0)
926  libmesh_error_msg("No Abaqus->LibMesh mapping information for ElemType " \
927  << Utility::enum_to_string(elem->type()) \
928  << "!");
929 
930  // Add this node with the current_id (which is determined by the
931  // alphabetical ordering of the map). Side numbers in Abaqus are 1-based,
932  // so we subtract 1 here before passing the abaqus side number to the
933  // mapping array
934  the_mesh.get_boundary_info().add_side
935  (elem,
936  eledef.abaqus_zero_based_side_id_to_libmesh_side_id[abaqus_side_number-1],
937  current_id);
938  }
939  }
940  }
941 
942 
943  // Some elsets (if they contain lower-dimensional elements) also
944  // define sidesets. So loop over them and build a searchable data
945  // structure we can use to assign sidesets.
946  {
947  unsigned char max_dim = this->max_elem_dimension_seen();
948 
949  // multimap from lower-dimensional-element-hash-key to
950  // pair(lower-dimensional-element, boundary_id). The
951  // lower-dimensional element is used to verify the results of the
952  // hash table search. The boundary_id will be used to set a
953  // boundary ID on a higher-dimensional element. We use a multimap
954  // because the lower-dimensional elements can belong to more than
955  // 1 sideset, and multiple lower-dimensional elements can hash to
956  // the same value, but this is very rare.
957  typedef LIBMESH_BEST_UNORDERED_MULTIMAP<dof_id_type,
958  std::pair<Elem*, boundary_id_type> > provide_bcs_t;
959  provide_bcs_t provide_bcs;
960 
961  // The elemset_id counter assigns a logical numbering to the
962  // _elemset_ids keys. We are going to use these ids as boundary
963  // ids, so elemset_id is of type boundary_id_type.
964  container_t::iterator it = _elemset_ids.begin();
965  for (boundary_id_type elemset_id=0; it != _elemset_ids.end(); ++it, ++elemset_id)
966  {
967  // Grab a reference to the vector of IDs
968  std::vector<dof_id_type>& id_vector = it->second;
969 
970  // Loop over this vector
971  for (std::size_t i=0; i<id_vector.size(); ++i)
972  {
973  // Map the id_vector[i]'th element ID (Abaqus numbering) to LibMesh numbering
974  dof_id_type libmesh_elem_id = _abaqus_to_libmesh_elem_mapping[ id_vector[i] ];
975 
976  // Get pointer to that element
977  Elem* elem = the_mesh.elem(libmesh_elem_id);
978 
979  if (elem == NULL)
980  libmesh_error_msg("Mesh returned NULL pointer for Elem " << libmesh_elem_id);
981 
982  // If the element dimension is equal to the maximum
983  // dimension seen, we can break out of this for loop --
984  // this elset does not contain sideset information.
985  if (elem->dim() == max_dim)
986  break;
987 
988  // We can only handle elements that are *exactly*
989  // one dimension lower than the max element
990  // dimension. Not sure if "edge" BCs in 3D
991  // actually make sense/are required...
992  if (elem->dim()+1 != max_dim)
993  libmesh_error_msg("ERROR: Expected boundary element of dimension " << max_dim-1 << " but got " << elem->dim());
994 
995  // Insert the current (key, pair(elem,id)) into the multimap.
996  provide_bcs.insert(std::make_pair(elem->key(),
997  std::make_pair(elem,
998  elemset_id)));
999 
1000  // Associate the name of this sideset with the ID we've
1001  // chosen. It's not necessary to do this for every
1002  // element in the set, but it's convenient to do it here
1003  // since we have all the necessary information...
1004  the_mesh.get_boundary_info().sideset_name(elemset_id) = it->first;
1005  }
1006  }
1007 
1008  // Loop over elements and try to assign boundary information
1009  {
1010  MeshBase::element_iterator e_it = the_mesh.active_elements_begin();
1011  const MeshBase::element_iterator end = the_mesh.active_elements_end();
1012  for ( ; e_it != end; ++e_it)
1013  {
1014  Elem* elem = *e_it;
1015 
1016  if (elem->dim() == max_dim)
1017  {
1018  // This is a max-dimension element that may require BCs.
1019  // For each of its sides, including internal sides, we'll
1020  // see if a lower-dimensional element provides boundary
1021  // information for it. Note that we have not yet called
1022  // find_neighbors(), so we can't use elem->neighbor(sn) in
1023  // this algorithm...
1024  for (unsigned short sn=0; sn<elem->n_sides(); sn++)
1025  {
1026  std::pair<provide_bcs_t::const_iterator,
1027  provide_bcs_t::const_iterator>
1028  range = provide_bcs.equal_range (elem->key(sn));
1029 
1030  // Add boundary information for each side in the range.
1031  for (provide_bcs_t::const_iterator s_it = range.first;
1032  s_it != range.second; ++s_it)
1033  {
1034  // We'll need to compare the lower dimensional element against the current side.
1035  UniquePtr<Elem> side (elem->build_side(sn));
1036 
1037  // Get the value mapped by the iterator.
1038  std::pair<Elem*, boundary_id_type> p = s_it->second;
1039 
1040  // Extract the relevant data from the iterator.
1041  Elem* lower_dim_elem = p.first;
1042  boundary_id_type bid = p.second;
1043 
1044  // This was a hash, so it might not be perfect. Let's verify...
1045  if (*lower_dim_elem == *side)
1046  the_mesh.get_boundary_info().add_side(elem, sn, bid);
1047  }
1048  }
1049  }
1050  }
1051  }
1052  }
1053 }

1057 void AbaqusIO::process_and_discard_comments()
1058 {
1059  std::string dummy;
1060  while (true)
1061  {
1062  // We assume we are at the beginning of a line that may be
1063  // comments or may be data. We need to only discard the line if
1064  // it begins with **, but we must avoid calling std::getline()
1065  // since there's no way to put that back.
1066  if (_in.peek() == '*')
1067  {
1068  // The first character was a star, so actually read it from the stream.
1069  _in.get();
1070 
1071  // Peek at the next character...
1072  if (_in.peek() == '*')
1073  {
1074  // OK, second character was star also, by definition this
1075  // line must be a comment! Read the rest of the line and discard!
1076  std::getline(_in, dummy);
1077  }
1078  else
1079  {
1080  // The second character was _not_ a star, so put back the first star
1081  // we pulled out so that the line can be parsed correctly by somebody
1082  // else!
1083  _in.unget();
1084 
1085  // Finally, break out of the while loop, we are done parsing comments
1086  break;
1087  }
1088  }
1089  else
1090  {
1091  // First character was not *, so this line must be data! Break out of the
1092  // while loop!
1093  break;
1094  }
1095  }
1096 }
