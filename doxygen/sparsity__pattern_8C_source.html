96 void Build::handle_vi_vj(const std::vector<dof_id_type> & element_dofs_i,
97  const std::vector<dof_id_type> & element_dofs_j)
98 {
99  const unsigned int n_dofs_on_element_i =
100  cast_int<unsigned int>(element_dofs_i.size());
101 
102  const processor_id_type proc_id = mesh.processor_id();
103  const dof_id_type first_dof_on_proc = dof_map.first_dof(proc_id);
104  const dof_id_type end_dof_on_proc = dof_map.end_dof(proc_id);
105 
106  std::vector<dof_id_type>
107  dofs_to_add;
108 
109  const unsigned int n_dofs_on_element_j =
110  cast_int<unsigned int>(element_dofs_j.size());
111 
112  // there might be 0 dofs for the other variable on the same element
113  // (when subdomain variables do not overlap) and that's when we do
114  // not do anything
115  if (n_dofs_on_element_j > 0)
116  {
117  for (unsigned int i=0; i<n_dofs_on_element_i; i++)
118  {
119  const dof_id_type ig = element_dofs_i[i];
120 
121  SparsityPattern::Row * row;
122 
123  // We save non-local row components for now so we can
124  // communicate them to other processors later.
125 
126  if ((ig >= first_dof_on_proc) &&
127  (ig < end_dof_on_proc))
128  {
129  // This is what I mean
130  // libmesh_assert_greater_equal ((ig - first_dof_on_proc), 0);
131  // but do the test like this because ig and
132  // first_dof_on_proc are unsigned ints
133  libmesh_assert_greater_equal (ig, first_dof_on_proc);
134  libmesh_assert_less (ig, (sparsity_pattern.size() +
135  first_dof_on_proc));
136 
137  row = &sparsity_pattern[ig - first_dof_on_proc];
138  }
139  else
140  {
141  row = &nonlocal_pattern[ig];
142  }
143 
144  // If the row is empty we will add *all*
145  // the element j DOFs, so just do that.
146  if (row->empty())
147  {
148  row->insert(row->end(),
149  element_dofs_j.begin(),
150  element_dofs_j.end());
151  }
152  else
153  {
154  // Build a list of the DOF indices not found in the
155  // sparsity pattern
156  dofs_to_add.clear();
157 
158  // Cache iterators. Low will move forward, subsequent
159  // searches will be on smaller ranges
160  SparsityPattern::Row::iterator
161  low = std::lower_bound
162  (row->begin(), row->end(), element_dofs_j.front()),
163  high = std::upper_bound
164  (low, row->end(), element_dofs_j.back());
165 
166  for (unsigned int j=0; j<n_dofs_on_element_j; j++)
167  {
168  const dof_id_type jg = element_dofs_j[j];
169 
170  // See if jg is in the sorted range
171  std::pair<SparsityPattern::Row::iterator,
172  SparsityPattern::Row::iterator>
173  pos = std::equal_range (low, high, jg);
174 
175  // Must add jg if it wasn't found
176  if (pos.first == pos.second)
177  dofs_to_add.push_back(jg);
178 
179  // pos.first is now a valid lower bound for any
180  // remaining element j DOFs. (That's why we sorted them.)
181  // Use it for the next search
182  low = pos.first;
183  }
184 
185  // Add to the sparsity pattern
186  if (!dofs_to_add.empty())
187  {
188  const std::size_t old_size = row->size();
189 
190  row->insert (row->end(),
191  dofs_to_add.begin(),
192  dofs_to_add.end());
193 
194  SparsityPattern::sort_row
195  (row->begin(), row->begin()+old_size,
196  row->end());
197  }
198  }
199  } // End dofs-of-var-i loop
200  } // End if-dofs-of-var-j
201 }

205 void Build::operator()(const ConstElemRange & range)
206 {
207  // Compute the sparsity structure of the global matrix. This can be
208  // fed into a PetscMatrix to allocate exactly the number of nonzeros
209  // necessary to store the matrix. This algorithm should be linear
210  // in the (# of elements)*(# nodes per element)
211  const processor_id_type proc_id = mesh.processor_id();
212  const dof_id_type n_dofs_on_proc = dof_map.n_dofs_on_processor(proc_id);
213  const dof_id_type first_dof_on_proc = dof_map.first_dof(proc_id);
214  const dof_id_type end_dof_on_proc = dof_map.end_dof(proc_id);
215 
216  sparsity_pattern.resize(n_dofs_on_proc);
217 
218  // Handle dof coupling specified by library and user coupling functors
219  {
220  const unsigned int n_var = dof_map.n_variables();
221 
222  std::vector<std::vector<dof_id_type> > element_dofs_i(n_var);
223 
224  std::vector<const Elem *> coupled_neighbors;
225  for (ConstElemRange::const_iterator elem_it = range.begin() ; elem_it != range.end(); ++elem_it)
226  {
227  const Elem * const elem = *elem_it;
228 
229  // Make some fake element iterators defining a range
230  // pointing to only this element.
231  Elem * const * elempp = const_cast<Elem * const *>(&elem);
232  Elem * const * elemend = elempp+1;
233 
234  const MeshBase::const_element_iterator fake_elem_it =
235  MeshBase::const_element_iterator(elempp,
236  elemend,
237  Predicates::NotNull<Elem * const *>());
238 
239  const MeshBase::const_element_iterator fake_elem_end =
240  MeshBase::const_element_iterator(elemend,
241  elemend,
242  Predicates::NotNull<Elem * const *>());
243 
244  GhostingFunctor::map_type elements_to_couple;
245 
246  // Man, I wish we had guaranteed unique_ptr availability...
247  std::set<CouplingMatrix *> temporary_coupling_matrices;
248 
249  dof_map.merge_ghost_functor_outputs(elements_to_couple,
250  temporary_coupling_matrices,
251  dof_map.coupling_functors_begin(),
252  dof_map.coupling_functors_end(),
253  fake_elem_it,
254  fake_elem_end,
255  DofObject::invalid_processor_id);
256  for (unsigned int vi=0; vi<n_var; vi++)
257  this->sorted_connected_dofs(elem, element_dofs_i[vi], vi);
258 
259  for (unsigned int vi=0; vi<n_var; vi++)
260  {
261  GhostingFunctor::map_type::iterator etg_it = elements_to_couple.begin();
262  const GhostingFunctor::map_type::iterator etg_end = elements_to_couple.end();
263  for (; etg_it != etg_end; ++etg_it)
264  {
265  const Elem * const partner = etg_it->first;
266  const CouplingMatrix * ghost_coupling = etg_it->second;
267 
268  // Loop over coupling matrix row variables if we have a
269  // coupling matrix, or all variables if not.
270  if (ghost_coupling)
271  {
272  libmesh_assert_equal_to (ghost_coupling->size(), n_var);
273  ConstCouplingRow ccr(vi, *ghost_coupling);
274 
275  for (ConstCouplingRow::const_iterator it = ccr.begin(),
276  end = ccr.end();
277  it != end; ++it)
278  {
279  if (partner == elem)
280  this->handle_vi_vj(element_dofs_i[vi], element_dofs_i[*it]);
281  else
282  {
283  std::vector<dof_id_type> partner_dofs;
284  this->sorted_connected_dofs(partner, partner_dofs, *it);
285  this->handle_vi_vj(element_dofs_i[vi], partner_dofs);
286  }
287  }
288  }
289  else
290  {
291  for (unsigned int vj = 0; vj != n_var; ++vj)
292  {
293  if (partner == elem)
294  this->handle_vi_vj(element_dofs_i[vi], element_dofs_i[vj]);
295  else
296  {
297  std::vector<dof_id_type> partner_dofs;
298  this->sorted_connected_dofs(partner, partner_dofs, vj);
299  this->handle_vi_vj(element_dofs_i[vi], partner_dofs);
300  }
301  }
302  }
303  } // End ghosted element loop
304  } // End vi loop
305 
306  for (std::set<CouplingMatrix *>::iterator
307  it = temporary_coupling_matrices.begin(),
308  end = temporary_coupling_matrices.begin();
309  it != end; ++it)
310  delete *it;
311 
312  } // End range element loop
313  } // End ghosting functor section
314 
315  // Now a new chunk of sparsity structure is built for all of the
316  // DOFs connected to our rows of the matrix.
317 
318  // If we're building a full sparsity pattern, then we've got
319  // complete rows to work with, so we can just count them from
320  // scratch.
321  if (need_full_sparsity_pattern)
322  {
323  n_nz.clear();
324  n_oz.clear();
325  }
326 
327  n_nz.resize (n_dofs_on_proc, 0);
328  n_oz.resize (n_dofs_on_proc, 0);
329 
330  for (dof_id_type i=0; i<n_dofs_on_proc; i++)
331  {
332  // Get the row of the sparsity pattern
333  SparsityPattern::Row & row = sparsity_pattern[i];
334 
335  for (std::size_t j=0; j<row.size(); j++)
336  if ((row[j] < first_dof_on_proc) || (row[j] >= end_dof_on_proc))
337  n_oz[i]++;
338  else
339  n_nz[i]++;
340 
341  // If we're not building a full sparsity pattern, then we want
342  // to avoid overcounting these entries as much as possible.
343  if (!need_full_sparsity_pattern)
344  row.clear();
345  }
346 }

350 void Build::join (const SparsityPattern::Build & other)
351 {
352  const processor_id_type proc_id = mesh.processor_id();
353  const dof_id_type n_global_dofs = dof_map.n_dofs();
354  const dof_id_type n_dofs_on_proc = dof_map.n_dofs_on_processor(proc_id);
355  const dof_id_type first_dof_on_proc = dof_map.first_dof(proc_id);
356  const dof_id_type end_dof_on_proc = dof_map.end_dof(proc_id);
357 
358  libmesh_assert_equal_to (sparsity_pattern.size(), other.sparsity_pattern.size());
359  libmesh_assert_equal_to (n_nz.size(), sparsity_pattern.size());
360  libmesh_assert_equal_to (n_oz.size(), sparsity_pattern.size());
361 
362  for (dof_id_type r=0; r<n_dofs_on_proc; r++)
363  {
364  // increment the number of on and off-processor nonzeros in this row
365  // (note this will be an upper bound unless we need the full sparsity pattern)
366  if (need_full_sparsity_pattern)
367  {
368  SparsityPattern::Row & my_row = sparsity_pattern[r];
369  const SparsityPattern::Row & their_row = other.sparsity_pattern[r];
370 
371  // simple copy if I have no dofs
372  if (my_row.empty())
373  my_row = their_row;
374 
375  // otherwise add their DOFs to mine, resort, and re-unique the row
376  else if (!their_row.empty()) // do nothing for the trivial case where
377  { // their row is empty
378  my_row.insert (my_row.end(),
379  their_row.begin(),
380  their_row.end());
381 
382  // We cannot use SparsityPattern::sort_row() here because it expects
383  // the [begin,middle) [middle,end) to be non-overlapping. This is not
384  // necessarily the case here, so use std::sort()
385  std::sort (my_row.begin(), my_row.end());
386 
387  my_row.erase(std::unique (my_row.begin(), my_row.end()), my_row.end());
388  }
389 
390  // fix the number of on and off-processor nonzeros in this row
391  n_nz[r] = n_oz[r] = 0;
392 
393  for (std::size_t j=0; j<my_row.size(); j++)
394  if ((my_row[j] < first_dof_on_proc) || (my_row[j] >= end_dof_on_proc))
395  n_oz[r]++;
396  else
397  n_nz[r]++;
398  }
399  else
400  {
401  n_nz[r] += other.n_nz[r];
402  n_nz[r] = std::min(n_nz[r], n_dofs_on_proc);
403  n_oz[r] += other.n_oz[r];
404  n_oz[r] =std::min(n_oz[r], static_cast<dof_id_type>(n_global_dofs-n_nz[r]));
405  }
406  }
407 
408  // Move nonlocal row information to ourselves; the other thread
409  // won't need it in the map after that.
410  NonlocalGraph::const_iterator it = other.nonlocal_pattern.begin();
411  for (; it != other.nonlocal_pattern.end(); ++it)
412  {
413 #ifndef NDEBUG
414  const dof_id_type dof_id = it->first;
415 
416  processor_id_type dbg_proc_id = 0;
417  while (dof_id >= dof_map.end_dof(dbg_proc_id))
418  dbg_proc_id++;
419  libmesh_assert (dbg_proc_id != this->processor_id());
420 #endif
421 
422  const SparsityPattern::Row & their_row = it->second;
423 
424  // We should have no empty values in a map
425  libmesh_assert (!their_row.empty());
426 
427  NonlocalGraph::iterator my_it = nonlocal_pattern.find(it->first);
428  if (my_it == nonlocal_pattern.end())
429  {
430  // nonlocal_pattern[it->first].swap(their_row);
431  nonlocal_pattern[it->first] = their_row;
432  }
433  else
434  {
435  SparsityPattern::Row & my_row = my_it->second;
436 
437  my_row.insert (my_row.end(),
438  their_row.begin(),
439  their_row.end());
440 
441  // We cannot use SparsityPattern::sort_row() here because it expects
442  // the [begin,middle) [middle,end) to be non-overlapping. This is not
443  // necessarily the case here, so use std::sort()
444  std::sort (my_row.begin(), my_row.end());
445 
446  my_row.erase(std::unique (my_row.begin(), my_row.end()), my_row.end());
447  }
448  }

453 void Build::parallel_sync ()
454 {
455  parallel_object_only();
456  libmesh_assert(this->comm().verify(need_full_sparsity_pattern));
457 
458  const dof_id_type n_global_dofs = dof_map.n_dofs();
459  const dof_id_type n_dofs_on_proc = dof_map.n_dofs_on_processor(this->processor_id());
460  const dof_id_type local_first_dof = dof_map.first_dof();
461  const dof_id_type local_end_dof = dof_map.end_dof();
462 
463  // Trade sparsity rows with other processors
464  for (processor_id_type p=1; p != this->n_processors(); ++p)
465  {
466  // Push to processor procup while receiving from procdown
467  processor_id_type procup =
468  cast_int<processor_id_type>((this->processor_id() + p) %
469  this->n_processors());
470  processor_id_type procdown =
471  cast_int<processor_id_type>((this->n_processors() + this->processor_id() - p) %
472  this->n_processors());
473 
474  // Pack the sparsity pattern rows to push to procup
475  std::vector<dof_id_type> pushed_row_ids,
476  pushed_row_ids_to_me;
477  std::vector<std::vector<dof_id_type>> pushed_rows,
478  pushed_rows_to_me;
479 
480  // Move nonlocal row information to a structure to send it from;
481  // we don't need it in the map after that.
482  NonlocalGraph::iterator it = nonlocal_pattern.begin();
483  while (it != nonlocal_pattern.end())
484  {
485  const dof_id_type dof_id = it->first;
486  processor_id_type proc_id = 0;
487  while (dof_id >= dof_map.end_dof(proc_id))
488  proc_id++;
489 
490  libmesh_assert (proc_id != this->processor_id());
491 
492  if (proc_id == procup)
493  {
494  pushed_row_ids.push_back(dof_id);
495 
496  // We can't just do the swap trick here, thanks to the
497  // differing vector allocators?
498  pushed_rows.push_back(std::vector<dof_id_type>());
499  pushed_rows.back().assign
500  (it->second.begin(), it->second.end());
501 
502  nonlocal_pattern.erase(it++);
503  }
504  else
505  ++it;
506  }
507 
508  this->comm().send_receive(procup, pushed_row_ids,
509  procdown, pushed_row_ids_to_me);
510  this->comm().send_receive(procup, pushed_rows,
511  procdown, pushed_rows_to_me);
512  pushed_row_ids.clear();
513  pushed_rows.clear();
514 
515  const std::size_t n_rows = pushed_row_ids_to_me.size();
516  for (std::size_t i=0; i != n_rows; ++i)
517  {
518  const dof_id_type r = pushed_row_ids_to_me[i];
519  const dof_id_type my_r = r - local_first_dof;
520 
521  std::vector<dof_id_type> & their_row = pushed_rows_to_me[i];
522 
523  if (need_full_sparsity_pattern)
524  {
525  SparsityPattern::Row & my_row =
526  sparsity_pattern[my_r];
527 
528  // They wouldn't have sent an empty row
529  libmesh_assert(!their_row.empty());
530 
531  // We can end up with an empty row on a dof that touches our
532  // inactive elements but not our active ones
533  if (my_row.empty())
534  {
535  my_row.assign (their_row.begin(),
536  their_row.end());
537  }
538  else
539  {
540  my_row.insert (my_row.end(),
541  their_row.begin(),
542  their_row.end());
543 
544  // We cannot use SparsityPattern::sort_row() here because it expects
545  // the [begin,middle) [middle,end) to be non-overlapping. This is not
546  // necessarily the case here, so use std::sort()
547  std::sort (my_row.begin(), my_row.end());
548 
549  my_row.erase(std::unique (my_row.begin(), my_row.end()), my_row.end());
550  }
551 
552  // fix the number of on and off-processor nonzeros in this row
553  n_nz[my_r] = n_oz[my_r] = 0;
554 
555  for (std::size_t j=0; j<my_row.size(); j++)
556  if ((my_row[j] < local_first_dof) || (my_row[j] >= local_end_dof))
557  n_oz[my_r]++;
558  else
559  n_nz[my_r]++;
560  }
561  else
562  {
563  for (std::size_t j=0; j<their_row.size(); j++)
564  if ((their_row[j] < local_first_dof) || (their_row[j] >= local_end_dof))
565  n_oz[my_r]++;
566  else
567  n_nz[my_r]++;
568 
569  n_nz[my_r] = std::min(n_nz[my_r], n_dofs_on_proc);
570  n_oz[my_r] = std::min(n_oz[my_r],
571  static_cast<dof_id_type>(n_global_dofs-n_nz[my_r]));
572  }
573  }
574  }
575 
576  // We should have sent everything at this point.
577  libmesh_assert (nonlocal_pattern.empty());
578 }
