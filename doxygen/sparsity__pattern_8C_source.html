81 void Build::handle_vi_vj
82  (const Elem * partner,
83  const std::vector<dof_id_type> & element_dofs_i,
84  unsigned int vj)
85 {
86  const unsigned int n_dofs_on_element_i =
87  cast_int<unsigned int>(element_dofs_i.size());
88 
89  const processor_id_type proc_id = mesh.processor_id();
90  const dof_id_type first_dof_on_proc = dof_map.first_dof(proc_id);
91  const dof_id_type end_dof_on_proc = dof_map.end_dof(proc_id);
92 
93  std::vector<dof_id_type> element_dofs_j,
94  dofs_to_add;
95 
96  // Find element dofs for variable vj
97  dof_map.dof_indices (partner, element_dofs_j, vj);
98 #ifdef LIBMESH_ENABLE_CONSTRAINTS
99  dof_map.find_connected_dofs (element_dofs_j);
100 #endif
101 
102  // We can be more efficient if we sort the element DOFs
103  // into increasing order
104  std::sort (element_dofs_j.begin(), element_dofs_j.end());
105 
106  const unsigned int n_dofs_on_element_j =
107  cast_int<unsigned int>(element_dofs_j.size());
108 
109  // there might be 0 dofs for the other variable on the same element
110  // (when subdomain variables do not overlap) and that's when we do
111  // not do anything
112  if (n_dofs_on_element_j > 0)
113  {
114  for (unsigned int i=0; i<n_dofs_on_element_i; i++)
115  {
116  const dof_id_type ig = element_dofs_i[i];
117 
118  SparsityPattern::Row * row;
119 
120  // We save non-local row components for now so we can
121  // communicate them to other processors later.
122 
123  if ((ig >= first_dof_on_proc) &&
124  (ig < end_dof_on_proc))
125  {
126  // This is what I mean
127  // libmesh_assert_greater_equal ((ig - first_dof_on_proc), 0);
128  // but do the test like this because ig and
129  // first_dof_on_proc are unsigned ints
130  libmesh_assert_greater_equal (ig, first_dof_on_proc);
131  libmesh_assert_less (ig, (sparsity_pattern.size() +
132  first_dof_on_proc));
133 
134  row = &sparsity_pattern[ig - first_dof_on_proc];
135  }
136  else
137  {
138  row = &nonlocal_pattern[ig];
139  }
140 
141  // If the row is empty we will add *all*
142  // the element j DOFs, so just do that.
143  if (row->empty())
144  {
145  row->insert(row->end(),
146  element_dofs_j.begin(),
147  element_dofs_j.end());
148  }
149  else
150  {
151  // Build a list of the DOF indices not found in the
152  // sparsity pattern
153  dofs_to_add.clear();
154 
155  // Cache iterators. Low will move forward, subsequent
156  // searches will be on smaller ranges
157  SparsityPattern::Row::iterator
158  low = std::lower_bound
159  (row->begin(), row->end(), element_dofs_j.front()),
160  high = std::upper_bound
161  (low, row->end(), element_dofs_j.back());
162 
163  for (unsigned int j=0; j<n_dofs_on_element_j; j++)
164  {
165  const dof_id_type jg = element_dofs_j[j];
166 
167  // See if jg is in the sorted range
168  std::pair<SparsityPattern::Row::iterator,
169  SparsityPattern::Row::iterator>
170  pos = std::equal_range (low, high, jg);
171 
172  // Must add jg if it wasn't found
173  if (pos.first == pos.second)
174  dofs_to_add.push_back(jg);
175 
176  // pos.first is now a valid lower bound for any
177  // remaining element j DOFs. (That's why we sorted them.)
178  // Use it for the next search
179  low = pos.first;
180  }
181 
182  // Add to the sparsity pattern
183  if (!dofs_to_add.empty())
184  {
185  const std::size_t old_size = row->size();
186 
187  row->insert (row->end(),
188  dofs_to_add.begin(),
189  dofs_to_add.end());
190 
191  SparsityPattern::sort_row
192  (row->begin(), row->begin()+old_size,
193  row->end());
194  }
195  }
196  } // End dofs-of-var-i loop
197  } // End if-dofs-of-var-j
198 }

202 void Build::operator()(const ConstElemRange & range)
203 {
204  // Compute the sparsity structure of the global matrix. This can be
205  // fed into a PetscMatrix to allocate exacly the number of nonzeros
206  // necessary to store the matrix. This algorithm should be linear
207  // in the (# of elements)*(# nodes per element)
208  const processor_id_type proc_id = mesh.processor_id();
209  const dof_id_type n_dofs_on_proc = dof_map.n_dofs_on_processor(proc_id);
210  const dof_id_type first_dof_on_proc = dof_map.first_dof(proc_id);
211  const dof_id_type end_dof_on_proc = dof_map.end_dof(proc_id);
212 
213  sparsity_pattern.resize(n_dofs_on_proc);
214 
215  // Handle dof coupling specified by library and user coupling functors
216  {
217  const unsigned int n_var = dof_map.n_variables();
218 
219  std::vector<dof_id_type> element_dofs_i;
220 
221  std::vector<const Elem *> coupled_neighbors;
222  for (ConstElemRange::const_iterator elem_it = range.begin() ; elem_it != range.end(); ++elem_it)
223  {
224  const Elem * const elem = *elem_it;
225 
226  // Make some fakey element iterators defining a range
227  // pointing to only this element.
228  Elem * const * elempp = const_cast<Elem * const *>(&elem);
229  Elem * const * elemend = elempp+1;
230  const MeshBase::const_element_iterator fake_elem_it =
231  MeshBase::const_element_iterator(elempp, elemend, Predicates::NotNull<Elem * const *>());
232  const MeshBase::const_element_iterator fake_elem_end =
233  MeshBase::const_element_iterator(elemend, elemend, Predicates::NotNull<Elem * const *>());
234 
235  GhostingFunctor::map_type elements_to_couple;
236 
237  // Man, I wish we had guaranteed unique_ptr availability...
238  std::set<CouplingMatrix*> temporary_coupling_matrices;
239 
240  dof_map.merge_ghost_functor_outputs
241  (elements_to_couple,
242  temporary_coupling_matrices,
243  dof_map.coupling_functors_begin(),
244  dof_map.coupling_functors_end(),
245  fake_elem_it, fake_elem_end, DofObject::invalid_processor_id);
246 
247  for (unsigned int vi=0; vi<n_var; vi++)
248  {
249  // Find element dofs for variable vi
250  dof_map.dof_indices (elem, element_dofs_i, vi);
251 #ifdef LIBMESH_ENABLE_CONSTRAINTS
252  dof_map.find_connected_dofs (element_dofs_i);
253 #endif
254 
255  // We can be more efficient if we sort the element DOFs
256  // into increasing order
257  std::sort(element_dofs_i.begin(), element_dofs_i.end());
258 
259  GhostingFunctor::map_type::iterator etg_it = elements_to_couple.begin();
260  const GhostingFunctor::map_type::iterator etg_end = elements_to_couple.end();
261  for (; etg_it != etg_end; ++etg_it)
262  {
263  const Elem * const partner = etg_it->first;
264  const CouplingMatrix *ghost_coupling = etg_it->second;
265 
266  // Loop over coupling matrix row variables if we have a
267  // coupling matrix, or all variables if not.
268  if (ghost_coupling)
269  {
270  libmesh_assert_equal_to (ghost_coupling->size(), n_var);
271  ConstCouplingRow ccr(vi, *ghost_coupling);
272 
273  for (ConstCouplingRow::const_iterator it = ccr.begin(),
274  end = ccr.end();
275  it != end; ++it)
276  this->handle_vi_vj(partner, element_dofs_i, *it);
277  }
278  else
279  {
280  for (unsigned int vj = 0; vj != n_var; ++vj)
281  this->handle_vi_vj(partner, element_dofs_i, vj);
282  }
283  } // End ghosted element loop
284  } // End vi loop
285 
286  for (std::set<CouplingMatrix*>::iterator
287  it = temporary_coupling_matrices.begin(),
288  end = temporary_coupling_matrices.begin();
289  it != end; ++it)
290  delete *it;
291 
292  } // End range element loop
293  } // End ghosting functor section
294 
295  // Now a new chunk of sparsity structure is built for all of the
296  // DOFs connected to our rows of the matrix.
297 
298  // If we're building a full sparsity pattern, then we've got
299  // complete rows to work with, so we can just count them from
300  // scratch.
301  if (need_full_sparsity_pattern)
302  {
303  n_nz.clear();
304  n_oz.clear();
305  }
306 
307  n_nz.resize (n_dofs_on_proc, 0);
308  n_oz.resize (n_dofs_on_proc, 0);
309 
310  for (dof_id_type i=0; i<n_dofs_on_proc; i++)
311  {
312  // Get the row of the sparsity pattern
313  SparsityPattern::Row & row = sparsity_pattern[i];
314 
315  for (dof_id_type j=0; j<row.size(); j++)
316  if ((row[j] < first_dof_on_proc) || (row[j] >= end_dof_on_proc))
317  n_oz[i]++;
318  else
319  n_nz[i]++;
320 
321  // If we're not building a full sparsity pattern, then we want
322  // to avoid overcounting these entries as much as possible.
323  if (!need_full_sparsity_pattern)
324  row.clear();
325  }

330 void Build::join (const SparsityPattern::Build & other)
331 {
332  const processor_id_type proc_id = mesh.processor_id();
333  const dof_id_type n_global_dofs = dof_map.n_dofs();
334  const dof_id_type n_dofs_on_proc = dof_map.n_dofs_on_processor(proc_id);
335  const dof_id_type first_dof_on_proc = dof_map.first_dof(proc_id);
336  const dof_id_type end_dof_on_proc = dof_map.end_dof(proc_id);
337 
338  libmesh_assert_equal_to (sparsity_pattern.size(), other.sparsity_pattern.size());
339  libmesh_assert_equal_to (n_nz.size(), sparsity_pattern.size());
340  libmesh_assert_equal_to (n_oz.size(), sparsity_pattern.size());
341 
342  for (dof_id_type r=0; r<n_dofs_on_proc; r++)
343  {
344  // increment the number of on and off-processor nonzeros in this row
345  // (note this will be an upper bound unless we need the full sparsity pattern)
346  if (need_full_sparsity_pattern)
347  {
348  SparsityPattern::Row & my_row = sparsity_pattern[r];
349  const SparsityPattern::Row & their_row = other.sparsity_pattern[r];
350 
351  // simple copy if I have no dofs
352  if (my_row.empty())
353  my_row = their_row;
354 
355  // otherwise add their DOFs to mine, resort, and re-unique the row
356  else if (!their_row.empty()) // do nothing for the trivial case where
357  { // their row is empty
358  my_row.insert (my_row.end(),
359  their_row.begin(),
360  their_row.end());
361 
362  // We cannot use SparsityPattern::sort_row() here because it expects
363  // the [begin,middle) [middle,end) to be non-overlapping. This is not
364  // necessarily the case here, so use std::sort()
365  std::sort (my_row.begin(), my_row.end());
366 
367  my_row.erase(std::unique (my_row.begin(), my_row.end()), my_row.end());
368  }
369 
370  // fix the number of on and off-processor nonzeros in this row
371  n_nz[r] = n_oz[r] = 0;
372 
373  for (std::size_t j=0; j<my_row.size(); j++)
374  if ((my_row[j] < first_dof_on_proc) || (my_row[j] >= end_dof_on_proc))
375  n_oz[r]++;
376  else
377  n_nz[r]++;
378  }
379  else
380  {
381  n_nz[r] += other.n_nz[r];
382  n_nz[r] = std::min(n_nz[r], n_dofs_on_proc);
383  n_oz[r] += other.n_oz[r];
384  n_oz[r] =std::min(n_oz[r], static_cast<dof_id_type>(n_global_dofs-n_nz[r]));
385  }
386  }
387 
388  // Move nonlocal row information to ourselves; the other thread
389  // won't need it in the map after that.
390  NonlocalGraph::const_iterator it = other.nonlocal_pattern.begin();
391  for (; it != other.nonlocal_pattern.end(); ++it)
392  {
393 #ifndef NDEBUG
394  const dof_id_type dof_id = it->first;
395 
396  processor_id_type dbg_proc_id = 0;
397  while (dof_id >= dof_map.end_dof(dbg_proc_id))
398  dbg_proc_id++;
399  libmesh_assert (dbg_proc_id != this->processor_id());
400 #endif
401 
402  const SparsityPattern::Row & their_row = it->second;
403 
404  // We should have no empty values in a map
405  libmesh_assert (!their_row.empty());
406 
407  NonlocalGraph::iterator my_it = nonlocal_pattern.find(it->first);
408  if (my_it == nonlocal_pattern.end())
409  {
410  // nonlocal_pattern[it->first].swap(their_row);
411  nonlocal_pattern[it->first] = their_row;
412  }
413  else
414  {
415  SparsityPattern::Row & my_row = my_it->second;
416 
417  my_row.insert (my_row.end(),
418  their_row.begin(),
419  their_row.end());
420 
421  // We cannot use SparsityPattern::sort_row() here because it expects
422  // the [begin,middle) [middle,end) to be non-overlapping. This is not
423  // necessarily the case here, so use std::sort()
424  std::sort (my_row.begin(), my_row.end());
425 
426  my_row.erase(std::unique (my_row.begin(), my_row.end()), my_row.end());
427  }
428  }

433 void Build::parallel_sync ()
434 {
435  parallel_object_only();
436  libmesh_assert(this->comm().verify(need_full_sparsity_pattern));
437 
438  const dof_id_type n_global_dofs = dof_map.n_dofs();
439  const dof_id_type n_dofs_on_proc = dof_map.n_dofs_on_processor(this->processor_id());
440  const dof_id_type local_first_dof = dof_map.first_dof();
441  const dof_id_type local_end_dof = dof_map.end_dof();
442 
443  // Trade sparsity rows with other processors
444  for (processor_id_type p=1; p != this->n_processors(); ++p)
445  {
446  // Push to processor procup while receiving from procdown
447  processor_id_type procup =
448  cast_int<processor_id_type>((this->processor_id() + p) %
449  this->n_processors());
450  processor_id_type procdown =
451  cast_int<processor_id_type>((this->n_processors() + this->processor_id() - p) %
452  this->n_processors());
453 
454  // Pack the sparsity pattern rows to push to procup
455  std::vector<dof_id_type> pushed_row_ids,
456  pushed_row_ids_to_me;
457  std::vector<std::vector<dof_id_type> > pushed_rows,
458  pushed_rows_to_me;
459 
460  // Move nonlocal row information to a structure to send it from;
461  // we don't need it in the map after that.
462  NonlocalGraph::iterator it = nonlocal_pattern.begin();
463  while (it != nonlocal_pattern.end())
464  {
465  const dof_id_type dof_id = it->first;
466  processor_id_type proc_id = 0;
467  while (dof_id >= dof_map.end_dof(proc_id))
468  proc_id++;
469 
470  libmesh_assert (proc_id != this->processor_id());
471 
472  if (proc_id == procup)
473  {
474  pushed_row_ids.push_back(dof_id);
475 
476  // We can't just do the swap trick here, thanks to the
477  // differing vector allocators?
478  pushed_rows.push_back(std::vector<dof_id_type>());
479  pushed_rows.back().assign
480  (it->second.begin(), it->second.end());
481 
482  nonlocal_pattern.erase(it++);
483  }
484  else
485  ++it;
486  }
487 
488  this->comm().send_receive(procup, pushed_row_ids,
489  procdown, pushed_row_ids_to_me);
490  this->comm().send_receive(procup, pushed_rows,
491  procdown, pushed_rows_to_me);
492  pushed_row_ids.clear();
493  pushed_rows.clear();
494 
495  const std::size_t n_rows = pushed_row_ids_to_me.size();
496  for (std::size_t i=0; i != n_rows; ++i)
497  {
498  const dof_id_type r = pushed_row_ids_to_me[i];
499  const dof_id_type my_r = r - local_first_dof;
500 
501  std::vector<dof_id_type> & their_row = pushed_rows_to_me[i];
502 
503  if (need_full_sparsity_pattern)
504  {
505  SparsityPattern::Row & my_row =
506  sparsity_pattern[my_r];
507 
508  // They wouldn't have sent an empty row
509  libmesh_assert(!their_row.empty());
510 
511  // We can end up with an empty row on a dof that touches our
512  // inactive elements but not our active ones
513  if (my_row.empty())
514  {
515  my_row.assign (their_row.begin(),
516  their_row.end());
517  }
518  else
519  {
520  my_row.insert (my_row.end(),
521  their_row.begin(),
522  their_row.end());
523 
524  // We cannot use SparsityPattern::sort_row() here because it expects
525  // the [begin,middle) [middle,end) to be non-overlapping. This is not
526  // necessarily the case here, so use std::sort()
527  std::sort (my_row.begin(), my_row.end());
528 
529  my_row.erase(std::unique (my_row.begin(), my_row.end()), my_row.end());
530  }
531 
532  // fix the number of on and off-processor nonzeros in this row
533  n_nz[my_r] = n_oz[my_r] = 0;
534 
535  for (std::size_t j=0; j<my_row.size(); j++)
536  if ((my_row[j] < local_first_dof) || (my_row[j] >= local_end_dof))
537  n_oz[my_r]++;
538  else
539  n_nz[my_r]++;
540  }
541  else
542  {
543  for (std::size_t j=0; j<their_row.size(); j++)
544  if ((their_row[j] < local_first_dof) || (their_row[j] >= local_end_dof))
545  n_oz[my_r]++;
546  else
547  n_nz[my_r]++;
548 
549  n_nz[my_r] = std::min(n_nz[my_r], n_dofs_on_proc);
550  n_oz[my_r] = std::min(n_oz[my_r],
551  static_cast<dof_id_type>(n_global_dofs-n_nz[my_r]));
552  }
553  }
554  }
555 
556  // We should have sent everything at this point.
557  libmesh_assert (nonlocal_pattern.empty());
558 }
