111 bool Hex::is_child_on_side(const unsigned int c,
112  const unsigned int s) const
113 {
114  libmesh_assert_less (c, this->n_children());
115  libmesh_assert_less (s, this->n_sides());
116 
117  // This array maps the Hex8 node numbering to the Hex8 child
118  // numbering. I.e.
119  // node 6 touches child 7, and
120  // node 7 touches child 6, etc.
121  const unsigned int node_child_map[8] = { 0, 1, 3, 2, 4, 5, 7, 6 };
122 
123  for (unsigned int i = 0; i != 4; ++i)
124  if (node_child_map[Hex8::side_nodes_map[s][i]] == c)
125  return true;
126 
127  return false;

186 Real Hex::quality (const ElemQuality q) const
187 {
188  switch (q)
189  {
190 
195  case DIAGONAL:
196  {
197  // Diagonal between node 0 and node 6
198  const Real d06 = this->length(0,6);
199 
200  // Diagonal between node 3 and node 5
201  const Real d35 = this->length(3,5);
202 
203  // Diagonal between node 1 and node 7
204  const Real d17 = this->length(1,7);
205 
206  // Diagonal between node 2 and node 4
207  const Real d24 = this->length(2,4);
208 
209  // Find the biggest and smallest diagonals
210  const Real min = std::min(d06, std::min(d35, std::min(d17, d24)));
211  const Real max = std::max(d06, std::max(d35, std::max(d17, d24)));
212 
213  libmesh_assert_not_equal_to (max, 0.0);
214 
215  return min / max;
216 
217  break;
218  }
219 
224  case TAPER:
225  {
226 
230  const Real d01 = this->length(0,1);
231  const Real d12 = this->length(1,2);
232  const Real d23 = this->length(2,3);
233  const Real d03 = this->length(0,3);
234  const Real d45 = this->length(4,5);
235  const Real d56 = this->length(5,6);
236  const Real d67 = this->length(6,7);
237  const Real d47 = this->length(4,7);
238  const Real d04 = this->length(0,4);
239  const Real d15 = this->length(1,5);
240  const Real d37 = this->length(3,7);
241  const Real d26 = this->length(2,6);
242 
243  std::vector<Real> edge_ratios(12);
244  // Front
245  edge_ratios[0] = std::min(d01, d45) / std::max(d01, d45);
246  edge_ratios[1] = std::min(d04, d15) / std::max(d04, d15);
247 
248  // Right
249  edge_ratios[2] = std::min(d15, d26) / std::max(d15, d26);
250  edge_ratios[3] = std::min(d12, d56) / std::max(d12, d56);
251 
252  // Back
253  edge_ratios[4] = std::min(d67, d23) / std::max(d67, d23);
254  edge_ratios[5] = std::min(d26, d37) / std::max(d26, d37);
255 
256  // Left
257  edge_ratios[6] = std::min(d04, d37) / std::max(d04, d37);
258  edge_ratios[7] = std::min(d03, d47) / std::max(d03, d47);
259 
260  // Bottom
261  edge_ratios[8] = std::min(d01, d23) / std::max(d01, d23);
262  edge_ratios[9] = std::min(d03, d12) / std::max(d03, d12);
263 
264  // Top
265  edge_ratios[10] = std::min(d45, d67) / std::max(d45, d67);
266  edge_ratios[11] = std::min(d56, d47) / std::max(d56, d47);
267 
268  return *(std::min_element(edge_ratios.begin(), edge_ratios.end())) ;
269 
270  break;
271  }
272 
273 
278  case STRETCH:
279  {
280  const Real sqrt3 = 1.73205080756888;
281 
285  const Real d06 = this->length(0,6);
286  const Real d17 = this->length(1,7);
287  const Real d35 = this->length(3,5);
288  const Real d24 = this->length(2,4);
289  const Real max_diag = std::max(d06, std::max(d17, std::max(d35, d24)));
290 
291  libmesh_assert_not_equal_to ( max_diag, 0.0 );
292 
296  std::vector<Real> edges(12);
297  edges[0] = this->length(0,1);
298  edges[1] = this->length(1,2);
299  edges[2] = this->length(2,3);
300  edges[3] = this->length(0,3);
301  edges[4] = this->length(4,5);
302  edges[5] = this->length(5,6);
303  edges[6] = this->length(6,7);
304  edges[7] = this->length(4,7);
305  edges[8] = this->length(0,4);
306  edges[9] = this->length(1,5);
307  edges[10] = this->length(2,6);
308  edges[11] = this->length(3,7);
309 
310  const Real min_edge = *(std::min_element(edges.begin(), edges.end()));
311  return sqrt3 * min_edge / max_diag ;
312  }
313 
314 
319  default:
320  return Elem::quality(q);
321  }

326 std::pair<Real, Real> Hex::qual_bounds (const ElemQuality q) const
327 {
328  std::pair<Real, Real> bounds;
329 
330  switch (q)
331  {
332 
333  case ASPECT_RATIO:
334  bounds.first = 1.;
335  bounds.second = 4.;
336  break;
337 
338  case SKEW:
339  bounds.first = 0.;
340  bounds.second = 0.5;
341  break;
342 
343  case SHEAR:
344  case SHAPE:
345  bounds.first = 0.3;
346  bounds.second = 1.;
347  break;
348 
349  case CONDITION:
350  bounds.first = 1.;
351  bounds.second = 8.;
352  break;
353 
354  case JACOBIAN:
355  bounds.first = 0.5;
356  bounds.second = 1.;
357  break;
358 
359  case DISTORTION:
360  bounds.first = 0.6;
361  bounds.second = 1.;
362  break;
363 
364  case TAPER:
365  bounds.first = 0.;
366  bounds.second = 0.4;
367  break;
368 
369  case STRETCH:
370  bounds.first = 0.25;
371  bounds.second = 1.;
372  break;
373 
374  case DIAGONAL:
375  bounds.first = 0.65;
376  bounds.second = 1.;
377  break;
378 
379  case SIZE:
380  bounds.first = 0.5;
381  bounds.second = 1.;
382  break;
383 
384  default:
385  libMesh::out << "Warning: Invalid quality measure chosen." << std::endl;
386  bounds.first = -1;
387  bounds.second = -1;
388  }
389 
390  return bounds;
391 }
