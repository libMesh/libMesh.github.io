102 {

104  libmesh_assert_less (s, this->n_sides());
105 
106  // This array maps the Hex8 node numbering to the Hex8 child
107  // numbering. I.e.

111 
112  for (unsigned int i = 0; i != 4; ++i)

114  return true;

178 Real Hex::quality (const ElemQuality q) const
179 {
180  switch (q)
181  {
182 
187  case DIAGONAL:
188  {
189  // Diagonal between node 0 and node 6
190  const Real d06 = this->length(0,6);
191 
192  // Diagonal between node 3 and node 5
193  const Real d35 = this->length(3,5);
194 
195  // Diagonal between node 1 and node 7
196  const Real d17 = this->length(1,7);
197 
198  // Diagonal between node 2 and node 4
199  const Real d24 = this->length(2,4);
200 
201  // Find the biggest and smallest diagonals
202  const Real min = std::min(d06, std::min(d35, std::min(d17, d24)));
203  const Real max = std::max(d06, std::max(d35, std::max(d17, d24)));
204 
205  libmesh_assert_not_equal_to (max, 0.0);
206 
207  return min / max;
208 
209  break;
210  }
211 
216  case TAPER:
217  {
218 
222  const Real d01 = this->length(0,1);
223  const Real d12 = this->length(1,2);
224  const Real d23 = this->length(2,3);
225  const Real d03 = this->length(0,3);
226  const Real d45 = this->length(4,5);
227  const Real d56 = this->length(5,6);
228  const Real d67 = this->length(6,7);
229  const Real d47 = this->length(4,7);
230  const Real d04 = this->length(0,4);
231  const Real d15 = this->length(1,5);
232  const Real d37 = this->length(3,7);
233  const Real d26 = this->length(2,6);
234 
235  std::vector<Real> edge_ratios(12);
236  // Front
237  edge_ratios[0] = std::min(d01, d45) / std::max(d01, d45);
238  edge_ratios[1] = std::min(d04, d15) / std::max(d04, d15);
239 
240  // Right
241  edge_ratios[2] = std::min(d15, d26) / std::max(d15, d26);
242  edge_ratios[3] = std::min(d12, d56) / std::max(d12, d56);
243 
244  // Back
245  edge_ratios[4] = std::min(d67, d23) / std::max(d67, d23);
246  edge_ratios[5] = std::min(d26, d37) / std::max(d26, d37);
247 
248  // Left
249  edge_ratios[6] = std::min(d04, d37) / std::max(d04, d37);
250  edge_ratios[7] = std::min(d03, d47) / std::max(d03, d47);
251 
252  // Bottom
253  edge_ratios[8] = std::min(d01, d23) / std::max(d01, d23);
254  edge_ratios[9] = std::min(d03, d12) / std::max(d03, d12);
255 
256  // Top
257  edge_ratios[10] = std::min(d45, d67) / std::max(d45, d67);
258  edge_ratios[11] = std::min(d56, d47) / std::max(d56, d47);
259 
260  return *(std::min_element(edge_ratios.begin(), edge_ratios.end())) ;
261 
262  break;
263  }
264 
265 
270  case STRETCH:
271  {
272  const Real sqrt3 = 1.73205080756888;
273 
277  const Real d06 = this->length(0,6);
278  const Real d17 = this->length(1,7);
279  const Real d35 = this->length(3,5);
280  const Real d24 = this->length(2,4);
281  const Real max_diag = std::max(d06, std::max(d17, std::max(d35, d24)));
282 
283  libmesh_assert_not_equal_to ( max_diag, 0.0 );
284 
288  std::vector<Real> edges(12);
289  edges[0] = this->length(0,1);
290  edges[1] = this->length(1,2);
291  edges[2] = this->length(2,3);
292  edges[3] = this->length(0,3);
293  edges[4] = this->length(4,5);
294  edges[5] = this->length(5,6);
295  edges[6] = this->length(6,7);
296  edges[7] = this->length(4,7);
297  edges[8] = this->length(0,4);
298  edges[9] = this->length(1,5);
299  edges[10] = this->length(2,6);
300  edges[11] = this->length(3,7);
301 
302  const Real min_edge = *(std::min_element(edges.begin(), edges.end()));
303  return sqrt3 * min_edge / max_diag ;
304  }
305 
306 
311  default:
312  return Elem::quality(q);
313  }
314 
315  libmesh_error_msg("We'll never get here!");
316  return 0.;
317 }

321 std::pair<Real, Real> Hex::qual_bounds (const ElemQuality q) const
322 {
323  std::pair<Real, Real> bounds;
324 
325  switch (q)
326  {
327 
328  case ASPECT_RATIO:
329  bounds.first = 1.;
330  bounds.second = 4.;
331  break;
332 
333  case SKEW:
334  bounds.first = 0.;
335  bounds.second = 0.5;
336  break;
337 
338  case SHEAR:
339  case SHAPE:
340  bounds.first = 0.3;
341  bounds.second = 1.;
342  break;
343 
344  case CONDITION:
345  bounds.first = 1.;
346  bounds.second = 8.;
347  break;
348 
349  case JACOBIAN:
350  bounds.first = 0.5;
351  bounds.second = 1.;
352  break;
353 
354  case DISTORTION:
355  bounds.first = 0.6;
356  bounds.second = 1.;
357  break;
358 
359  case TAPER:
360  bounds.first = 0.;
361  bounds.second = 0.4;
362  break;
363 
364  case STRETCH:
365  bounds.first = 0.25;
366  bounds.second = 1.;
367  break;
368 
369  case DIAGONAL:
370  bounds.first = 0.65;
371  bounds.second = 1.;
372  break;
373 
374  case SIZE:
375  bounds.first = 0.5;
376  bounds.second = 1.;
377  break;
378 
379  default:
380  libMesh::out << "Warning: Invalid quality measure chosen." << std::endl;
381  bounds.first = -1;
382  bounds.second = -1;
383  }
384 
385  return bounds;
386 }
